fn statements() {
    let value: u32 = 1u32;
    let (_, pair) = (1u32, 2u32);
    const LIMIT: u32 = 10u32;
    if value == 1u32 {
        assert(true);
    } else if value == 2u32 {
        assert_eq(value, 2u32);
    } else {
        assert_neq(value, 3u32);
    }
    for i: u32 in 0u32..=LIMIT {
        value += i;
    }
    value = value - 1u32;
    return;
}
