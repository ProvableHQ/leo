17//! This module contains a Reconstructor trait for the AST.
18//! It implements default methods for each node to be made
19//! given the information of the old node.
20
21use crate::*;

23/// A Reconstructor trait for expressions in the AST.
24pub trait ExpressionReconstructor {
25    type AdditionalOutput: Default;
26
27    fn reconstruct_expression(&mut self, input: Expression) -> (Expression, Self::AdditionalOutput) {
28        match input {
29            Expression::Access(access) => self.reconstruct_access(access),
30            Expression::AssociatedConstant(associated_constant) => {
31                self.reconstruct_associated_constant(associated_constant)
32            }
33            Expression::AssociatedFunction(associated_function) => {
34                self.reconstruct_associated_function(associated_function)
35            }
36            Expression::Array(array) => self.reconstruct_array(array),
37            Expression::Binary(binary) => self.reconstruct_binary(binary),
38            Expression::Call(call) => self.reconstruct_call(call),
39            Expression::Cast(cast) => self.reconstruct_cast(cast),
40            Expression::Struct(struct_) => self.reconstruct_struct_init(struct_),
41            Expression::Err(err) => self.reconstruct_err(err),
42            Expression::Identifier(identifier) => self.reconstruct_identifier(identifier),
43            Expression::Literal(value) => self.reconstruct_literal(value),
44            Expression::Locator(locator) => self.reconstruct_locator(locator),
45            Expression::Ternary(ternary) => self.reconstruct_ternary(ternary),
46            Expression::Tuple(tuple) => self.reconstruct_tuple(tuple),
47            Expression::Unary(unary) => self.reconstruct_unary(unary),
48            Expression::Unit(unit) => self.reconstruct_unit(unit),
49        }
50    }
51
52    fn reconstruct_access(&mut self, input: AccessExpression) -> (Expression, Self::AdditionalOutput) {
53        match input {
54            AccessExpression::Array(array) => self.reconstruct_array_access(array),
55            AccessExpression::Member(member) => self.reconstruct_member_access(member),
56            AccessExpression::Tuple(tuple) => self.reconstruct_tuple_access(tuple),
57        }
58    }
59
60    fn reconstruct_array_access(&mut self, input: ArrayAccess) -> (Expression, Self::AdditionalOutput) {
61        (
62            Expression::Access(AccessExpression::Array(ArrayAccess {
63                array: Box::new(self.reconstruct_expression(*input.array).0),
64                index: Box::new(self.reconstruct_expression(*input.index).0),
65                span: input.span,
66                id: input.id,
67            })),
68            Default::default(),
69        )
70    }
71
72    fn reconstruct_associated_constant(
73        &mut self,
74        input: AssociatedConstantExpression,
75    ) -> (Expression, Self::AdditionalOutput) {
76        (
77            Expression::AssociatedConstant(AssociatedConstantExpression {
78                ty: input.ty,
79                name: input.name,
80                span: input.span,
81                id: input.id,
82            }),
83            Default::default(),
84        )
85    }
86
87    fn reconstruct_associated_function(
88        &mut self,
89        input: AssociatedFunctionExpression,
90    ) -> (Expression, Self::AdditionalOutput) {
91        (
92            Expression::AssociatedFunction(AssociatedFunctionExpression {
93                variant: input.variant,
94                name: input.name,
95                arguments: input.arguments.into_iter().map(|arg| self.reconstruct_expression(arg).0).collect(),
96                span: input.span,
97                id: input.id,
98            }),
99            Default::default(),
100        )
101    }
102
103    fn reconstruct_member_access(&mut self, input: MemberAccess) -> (Expression, Self::AdditionalOutput) {
104        (
105            Expression::Access(AccessExpression::Member(MemberAccess {
106                inner: Box::new(self.reconstruct_expression(*input.inner).0),
107                name: input.name,
108                span: input.span,
109                id: input.id,
110            })),
111            Default::default(),
112        )
113    }
114
115    fn reconstruct_tuple_access(&mut self, input: TupleAccess) -> (Expression, Self::AdditionalOutput) {
116        (
117            Expression::Access(AccessExpression::Tuple(TupleAccess {
118                tuple: Box::new(self.reconstruct_expression(*input.tuple).0),
119                index: input.index,
120                span: input.span,
121                id: input.id,
122            })),
123            Default::default(),
124        )
125    }
126
127    fn reconstruct_array(&mut self, input: ArrayExpression) -> (Expression, Self::AdditionalOutput) {
128        (
129            Expression::Array(ArrayExpression {
130                elements: input.elements.into_iter().map(|element| self.reconstruct_expression(element).0).collect(),
131                span: input.span,
132                id: input.id,
133            }),
134            Default::default(),
135        )
136    }
137
138    fn reconstruct_binary(&mut self, input: BinaryExpression) -> (Expression, Self::AdditionalOutput) {
139        (
140            Expression::Binary(BinaryExpression {
141                left: Box::new(self.reconstruct_expression(*input.left).0),
142                right: Box::new(self.reconstruct_expression(*input.right).0),
143                op: input.op,
144                span: input.span,
145                id: input.id,
146            }),
147            Default::default(),
148        )
149    }
150
151    fn reconstruct_call(&mut self, input: CallExpression) -> (Expression, Self::AdditionalOutput) {
152        (
153            Expression::Call(CallExpression {
154                function: input.function,
155                arguments: input.arguments.into_iter().map(|arg| self.reconstruct_expression(arg).0).collect(),
156                program: input.program,
157                span: input.span,
158                id: input.id,
159            }),
160            Default::default(),
161        )
162    }
163
164    fn reconstruct_cast(&mut self, input: CastExpression) -> (Expression, Self::AdditionalOutput) {
165        (
166            Expression::Cast(CastExpression {
167                expression: Box::new(self.reconstruct_expression(*input.expression).0),
168                type_: input.type_,
169                span: input.span,
170                id: input.id,
171            }),
172            Default::default(),
173        )
174    }
175
176    fn reconstruct_struct_init(&mut self, input: StructExpression) -> (Expression, Self::AdditionalOutput) {
177        (
178            Expression::Struct(StructExpression {
179                name: input.name,
180                members: input
181                    .members
182                    .into_iter()
183                    .map(|member| StructVariableInitializer {
184                        identifier: member.identifier,
185                        expression: member.expression.map(|expr| self.reconstruct_expression(expr).0),
186                        span: member.span,
187                        id: member.id,
188                    })
189                    .collect(),
190                span: input.span,
191                id: input.id,
192            }),
193            Default::default(),
194        )
195    }
196
197    fn reconstruct_err(&mut self, _input: ErrExpression) -> (Expression, Self::AdditionalOutput) {
198        panic!("`ErrExpression`s should not be in the AST at this phase of compilation.")
199    }
200
201    fn reconstruct_identifier(&mut self, input: Identifier) -> (Expression, Self::AdditionalOutput) {
202        (Expression::Identifier(input), Default::default())
203    }
204
205    fn reconstruct_literal(&mut self, input: Literal) -> (Expression, Self::AdditionalOutput) {
206        (Expression::Literal(input), Default::default())
207    }
208
209    fn reconstruct_locator(&mut self, input: LocatorExpression) -> (Expression, Self::AdditionalOutput) {
210        (Expression::Locator(input), Default::default())
211    }
212
213    fn reconstruct_ternary(&mut self, input: TernaryExpression) -> (Expression, Self::AdditionalOutput) {
214        (
215            Expression::Ternary(TernaryExpression {
216                condition: Box::new(self.reconstruct_expression(*input.condition).0),
217                if_true: Box::new(self.reconstruct_expression(*input.if_true).0),
218                if_false: Box::new(self.reconstruct_expression(*input.if_false).0),
219                span: input.span,
220                id: input.id,
221            }),
222            Default::default(),
223        )
224    }
225
226    fn reconstruct_tuple(&mut self, input: TupleExpression) -> (Expression, Self::AdditionalOutput) {
227        (
228            Expression::Tuple(TupleExpression {
229                elements: input.elements.into_iter().map(|element| self.reconstruct_expression(element).0).collect(),
230                span: input.span,
231                id: input.id,
232            }),
233            Default::default(),
234        )
235    }
236
237    fn reconstruct_unary(&mut self, input: UnaryExpression) -> (Expression, Self::AdditionalOutput) {
238        (
239            Expression::Unary(UnaryExpression {
240                receiver: Box::new(self.reconstruct_expression(*input.receiver).0),
241                op: input.op,
242                span: input.span,
243                id: input.id,
244            }),
245            Default::default(),
246        )
247    }
248
249    fn reconstruct_unit(&mut self, input: UnitExpression) -> (Expression, Self::AdditionalOutput) {
250        (Expression::Unit(input), Default::default())
251    }
252}

254/// A Reconstructor trait for statements in the AST.
255pub trait StatementReconstructor: ExpressionReconstructor {
256    fn reconstruct_statement(&mut self, input: Statement) -> (Statement, Self::AdditionalOutput) {
257        match input {
258            Statement::Assert(assert) => self.reconstruct_assert(assert),
259            Statement::Assign(stmt) => self.reconstruct_assign(*stmt),
260            Statement::Block(stmt) => {
261                let (stmt, output) = self.reconstruct_block(stmt);
262                (Statement::Block(stmt), output)
263            }
264            Statement::Conditional(stmt) => self.reconstruct_conditional(stmt),
265            Statement::Console(stmt) => self.reconstruct_console(stmt),
266            Statement::Const(stmt) => self.reconstruct_const(stmt),
267            Statement::Definition(stmt) => self.reconstruct_definition(stmt),
268            Statement::Expression(stmt) => self.reconstruct_expression_statement(stmt),
269            Statement::Iteration(stmt) => self.reconstruct_iteration(*stmt),
270            Statement::Return(stmt) => self.reconstruct_return(stmt),
271        }
272    }
273
274    fn reconstruct_assert(&mut self, input: AssertStatement) -> (Statement, Self::AdditionalOutput) {
275        (
276            Statement::Assert(AssertStatement {
277                variant: match input.variant {
278                    AssertVariant::Assert(expr) => AssertVariant::Assert(self.reconstruct_expression(expr).0),
279                    AssertVariant::AssertEq(left, right) => AssertVariant::AssertEq(
280                        self.reconstruct_expression(left).0,
281                        self.reconstruct_expression(right).0,
282                    ),
283                    AssertVariant::AssertNeq(left, right) => AssertVariant::AssertNeq(
284                        self.reconstruct_expression(left).0,
285                        self.reconstruct_expression(right).0,
286                    ),
287                },
288                span: input.span,
289                id: input.id,
290            }),
291            Default::default(),
292        )
293    }
294
295    fn reconstruct_assign(&mut self, input: AssignStatement) -> (Statement, Self::AdditionalOutput) {
296        (
297            Statement::Assign(Box::new(AssignStatement {
298                place: input.place,
299                value: self.reconstruct_expression(input.value).0,
300                span: input.span,
301                id: input.id,
302            })),
303            Default::default(),
304        )
305    }
306
307    fn reconstruct_block(&mut self, input: Block) -> (Block, Self::AdditionalOutput) {
308        (
309            Block {
310                statements: input.statements.into_iter().map(|s| self.reconstruct_statement(s).0).collect(),
311                span: input.span,
312                id: input.id,
313            },
314            Default::default(),
315        )
316    }
317
318    fn reconstruct_conditional(&mut self, input: ConditionalStatement) -> (Statement, Self::AdditionalOutput) {
319        (
320            Statement::Conditional(ConditionalStatement {
321                condition: self.reconstruct_expression(input.condition).0,
322                then: self.reconstruct_block(input.then).0,
323                otherwise: input.otherwise.map(|n| Box::new(self.reconstruct_statement(*n).0)),
324                span: input.span,
325                id: input.id,
326            }),
327            Default::default(),
328        )
329    }
330
331    fn reconstruct_console(&mut self, input: ConsoleStatement) -> (Statement, Self::AdditionalOutput) {
332        (
333            Statement::Console(ConsoleStatement {
334                function: match input.function {
335                    ConsoleFunction::Assert(expr) => ConsoleFunction::Assert(self.reconstruct_expression(expr).0),
336                    ConsoleFunction::AssertEq(left, right) => ConsoleFunction::AssertEq(
337                        self.reconstruct_expression(left).0,
338                        self.reconstruct_expression(right).0,
339                    ),
340                    ConsoleFunction::AssertNeq(left, right) => ConsoleFunction::AssertNeq(
341                        self.reconstruct_expression(left).0,
342                        self.reconstruct_expression(right).0,
343                    ),
344                },
345                span: input.span,
346                id: input.id,
347            }),
348            Default::default(),
349        )
350    }
351
352    fn reconstruct_const(&mut self, input: ConstDeclaration) -> (Statement, Self::AdditionalOutput) {
353        (
354            Statement::Const(ConstDeclaration {
355                place: input.place,
356                type_: input.type_,
357                value: self.reconstruct_expression(input.value).0,
358                span: input.span,
359                id: input.id,
360            }),
361            Default::default(),
362        )
363    }
364
365    fn reconstruct_definition(&mut self, input: DefinitionStatement) -> (Statement, Self::AdditionalOutput) {
366        (
367            Statement::Definition(DefinitionStatement {
368                place: input.place,
369                type_: input.type_,
370                value: self.reconstruct_expression(input.value).0,
371                span: input.span,
372                id: input.id,
373            }),
374            Default::default(),
375        )
376    }
377
378    fn reconstruct_expression_statement(&mut self, input: ExpressionStatement) -> (Statement, Self::AdditionalOutput) {
379        (
380            Statement::Expression(ExpressionStatement {
381                expression: self.reconstruct_expression(input.expression).0,
382                span: input.span,
383                id: input.id,
384            }),
385            Default::default(),
386        )
387    }
388
389    fn reconstruct_iteration(&mut self, input: IterationStatement) -> (Statement, Self::AdditionalOutput) {
390        (
391            Statement::Iteration(Box::new(IterationStatement {
392                variable: input.variable,
393                type_: input.type_,
394                start: self.reconstruct_expression(input.start).0,
395                start_value: input.start_value,
396                stop: self.reconstruct_expression(input.stop).0,
397                stop_value: input.stop_value,
398                block: self.reconstruct_block(input.block).0,
399                inclusive: input.inclusive,
400                span: input.span,
401                id: input.id,
402            })),
403            Default::default(),
404        )
405    }
406
407    fn reconstruct_return(&mut self, input: ReturnStatement) -> (Statement, Self::AdditionalOutput) {
408        (
409            Statement::Return(ReturnStatement {
410                expression: self.reconstruct_expression(input.expression).0,
411                span: input.span,
412                id: input.id,
413            }),
414            Default::default(),
415        )
416    }
417}

419/// A Reconstructor trait for the program represented by the AST.
420pub trait ProgramReconstructor: StatementReconstructor {
421    fn reconstruct_program(&mut self, input: Program) -> Program {
422        Program {
423            imports: input
424                .imports
425                .into_iter()
426                .map(|(id, import)| (id, (self.reconstruct_import(import.0), import.1)))
427                .collect(),
428            stubs: input.stubs.into_iter().map(|(id, stub)| (id, self.reconstruct_stub(stub))).collect(),
429            program_scopes: input
430                .program_scopes
431                .into_iter()
432                .map(|(id, scope)| (id, self.reconstruct_program_scope(scope)))
433                .collect(),
434        }
435    }
436
437    fn reconstruct_stub(&mut self, input: Stub) -> Stub {
438        Stub {
439            imports: input.imports,
440            stub_id: input.stub_id,
441            consts: input.consts,
442            structs: input.structs,
443            mappings: input.mappings,
444            span: input.span,
445            functions: input.functions.into_iter().map(|(i, f)| (i, self.reconstruct_function_stub(f))).collect(),
446        }
447    }
448
449    fn reconstruct_program_scope(&mut self, input: ProgramScope) -> ProgramScope {
450        ProgramScope {
451            program_id: input.program_id,
452            structs: input.structs.into_iter().map(|(i, c)| (i, self.reconstruct_struct(c))).collect(),
453            mappings: input.mappings.into_iter().map(|(id, mapping)| (id, self.reconstruct_mapping(mapping))).collect(),
454            functions: input.functions.into_iter().map(|(i, f)| (i, self.reconstruct_function(f))).collect(),
455            consts: input
456                .consts
457                .into_iter()
458                .map(|(i, c)| match self.reconstruct_const(c) {
459                    (Statement::Const(declaration), _) => (i, declaration),
460                    _ => panic!("`reconstruct_const` can only return `Statement::Const`"),
461                })
462                .collect(),
463            span: input.span,
464        }
465    }
466
467    fn reconstruct_function(&mut self, input: Function) -> Function {
468        Function {
469            annotations: input.annotations,
470            variant: input.variant,
471            identifier: input.identifier,
472            input: input.input,
473            output: input.output,
474            output_type: input.output_type,
475            block: self.reconstruct_block(input.block).0,
476            span: input.span,
477            id: input.id,
478        }
479    }
480
481    fn reconstruct_function_stub(&mut self, input: FunctionStub) -> FunctionStub {
482        input
483    }
484
485    fn reconstruct_struct(&mut self, input: Composite) -> Composite {
486        input
487    }
488
489    fn reconstruct_import(&mut self, input: Program) -> Program {
490        self.reconstruct_program(input)
491    }
492
493    fn reconstruct_mapping(&mut self, input: Mapping) -> Mapping {
494        input
495    }
