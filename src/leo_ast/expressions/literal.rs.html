21/// A literal.
22#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
23pub struct Literal {
24    pub span: Span,
25    pub id: NodeID,
26    pub variant: LiteralVariant,
27}
28
29#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
30pub enum LiteralVariant {
31    /// An address literal, e.g., `aleo1qnr4dkkvkgfqph0vzc3y6z2eu975wnpz2925ntjccd5cfqxtyu8s7pyjh9` or `hello.aleo`.
32    Address(String),
33    /// A boolean literal, either `true` or `false`.
34    Boolean(bool),
35    /// A field literal, e.g., `42field`.
36    /// A signed number followed by the keyword `field`.
37    Field(String),
38    /// A group literal, eg `42group`.
39    Group(String),
40    /// An integer literal, e.g., `42u32`.
41    Integer(IntegerType, String),
42    /// A scalar literal, e.g. `1scalar`.
43    /// An unsigned number followed by the keyword `scalar`.
44    Scalar(String),
45    /// An unsuffixed literal, e.g. `42` (without a type suffix)
46    Unsuffixed(String),
47    /// A string literal, e.g., `"foobar"`.
48    String(String),
49}
50
51impl fmt::Display for LiteralVariant {
52    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
53        match &self {
54            Self::Address(address) => write!(f, "{address}"),
55            Self::Boolean(boolean) => write!(f, "{boolean}"),
56            Self::Field(field) => write!(f, "{field}field"),
57            Self::Group(group) => write!(f, "{group}group"),
58            Self::Integer(type_, value) => write!(f, "{value}{type_}"),
59            Self::Scalar(scalar) => write!(f, "{scalar}scalar"),
60            Self::Unsuffixed(value) => write!(f, "{value}"),
61            Self::String(string) => write!(f, "\"{string}\""),
62        }
63    }
64}
65
66impl fmt::Display for Literal {
67    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
68        self.variant.fmt(f)
69    }
70}

74struct DisplayDecimal<'a>(&'a Literal);
75
76impl Literal {
77    pub fn string(s: String, span: Span, id: NodeID) -> Self {
78        Literal { variant: LiteralVariant::String(s), span, id }
79    }
80
81    pub fn field(s: String, span: Span, id: NodeID) -> Self {
82        Literal { variant: LiteralVariant::Field(s), span, id }
83    }
84
85    pub fn group(s: String, span: Span, id: NodeID) -> Self {
86        Literal { variant: LiteralVariant::Group(s), span, id }
87    }
88
89    pub fn address(s: String, span: Span, id: NodeID) -> Self {
90        Literal { variant: LiteralVariant::Address(s), span, id }
91    }
92
93    pub fn scalar(s: String, span: Span, id: NodeID) -> Self {
94        Literal { variant: LiteralVariant::Scalar(s), span, id }
95    }
96
97    pub fn boolean(s: bool, span: Span, id: NodeID) -> Self {
98        Literal { variant: LiteralVariant::Boolean(s), span, id }
99    }
100
101    pub fn integer(integer_type: IntegerType, s: String, span: Span, id: NodeID) -> Self {
102        Literal { variant: LiteralVariant::Integer(integer_type, s), span, id }
103    }
104
105    pub fn unsuffixed(s: String, span: Span, id: NodeID) -> Self {
106        Literal { variant: LiteralVariant::Unsuffixed(s), span, id }
107    }
108
109    /// For displaying a literal as decimal, regardless of the radix in which it was parsed.
110    ///
111    /// In particular this is useful for outputting .aleo files.
112    pub fn display_decimal(&self) -> impl '_ + fmt::Display {
113        DisplayDecimal(self)
114    }
115
116    /// For an integer literal, parse it and cast it to a u32.
117    pub fn as_u32(&self) -> Option<u32> {
118        if let LiteralVariant::Integer(_, s) = &self.variant {
119            u32::from_str_by_radix(&s.replace("_", "")).ok()
120        } else {
121            None
122        }
123    }
124}

126impl fmt::Display for DisplayDecimal<'_> {
127    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
128        // This function is duplicated in `interpreter/src/cursor.rs`,
129        // but there's not really a great place to put a common implementation
130        // right now.
131        fn prepare_snarkvm_string(s: &str, suffix: &str) -> String {
132            // If there's a `-`, separate it from the rest of the string.
133            let (neg, rest) = s.strip_prefix("-").map(|rest| ("-", rest)).unwrap_or(("", s));
134            // Remove leading zeros.
135            let mut rest = rest.trim_start_matches('0');
136            if rest.is_empty() {
137                rest = "0";
138            }
139            format!("{neg}{rest}{suffix}")
140        }
141
142        match &self.0.variant {
143            LiteralVariant::Address(address) => write!(f, "{address}"),
144            LiteralVariant::Boolean(boolean) => write!(f, "{boolean}"),
145            LiteralVariant::Field(field) => write!(f, "{}", prepare_snarkvm_string(field, "field")),
146            LiteralVariant::Group(group) => write!(f, "{}", prepare_snarkvm_string(group, "group")),
147            LiteralVariant::Integer(type_, value) => {
148                let string = value.replace('_', "");
149                if value.starts_with('-') {
150                    let v = i128::from_str_by_radix(&string).expect("Failed to parse integer?");
151                    write!(f, "{v}{type_}")
152                } else {
153                    let v = u128::from_str_by_radix(&string).expect("Failed to parse integer?");
154                    write!(f, "{v}{type_}")
155                }
156            }
157            LiteralVariant::Scalar(scalar) => write!(f, "{}", prepare_snarkvm_string(scalar, "scalar")),
158            LiteralVariant::Unsuffixed(value) => write!(f, "{value}"),
159            LiteralVariant::String(string) => write!(f, "\"{string}\""),
160        }
161    }
162}

170/// This trait allows to parse integer literals of any type generically.
171///
172/// The literal may optionally start with a `-` and/or `0x` or `0o` or 0b`.
173pub trait FromStrRadix: Sized {
174    fn from_str_by_radix(src: &str) -> Result<Self, std::num::ParseIntError>;
175}
176
177macro_rules! implement_from_str_radix {
178    ($($ty:ident)*) => {
179        $(
180            impl FromStrRadix for $ty {
181                fn from_str_by_radix(src: &str) -> Result<Self, std::num::ParseIntError> {
182                    if let Some(stripped) = src.strip_prefix("0x") {
183                        Self::from_str_radix(stripped, 16)
184                    } else if let Some(stripped) = src.strip_prefix("0o") {
185                        Self::from_str_radix(stripped, 8)
186                    } else if let Some(stripped) = src.strip_prefix("0b") {
187                        Self::from_str_radix(stripped, 2)
188                    } else if let Some(stripped) = src.strip_prefix("-0x") {
189                        // We have to remove the 0x prefix and put back in a - to use
190                        // std's parsing. Alternatively we could jump through
191                        // a few hoops to avoid allocating.
192                        let mut s = String::new();
193                        s.push('-');
194                        s.push_str(stripped);
195                        Self::from_str_radix(&s, 16)
196                    } else if let Some(stripped) = src.strip_prefix("-0o") {
197                        // Ditto.
198                        let mut s = String::new();
199                        s.push('-');
200                        s.push_str(stripped);
201                        Self::from_str_radix(&s, 8)
202                    } else if let Some(stripped) = src.strip_prefix("-0b") {
203                        // Ditto.
204                        let mut s = String::new();
205                        s.push('-');
206                        s.push_str(stripped);
207                        Self::from_str_radix(&s, 2)
208                    } else {
209                        Self::from_str_radix(src, 10)
210                    }
211                }
212            }
213        )*
214    };
215}
