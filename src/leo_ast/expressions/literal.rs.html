21/// A literal.
22#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
23pub struct Literal {
24    pub span: Span,
25    pub id: NodeID,
26    pub variant: LiteralVariant,
27}
28
29#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
30pub enum LiteralVariant {
31    /// An address literal, e.g., `aleo1qnr4dkkvkgfqph0vzc3y6z2eu975wnpz2925ntjccd5cfqxtyu8s7pyjh9` or `hello.aleo`.
32    Address(String),
33    /// A boolean literal, either `true` or `false`.
34    Boolean(bool),
35    /// A field literal, e.g., `42field`.
36    /// A signed number followed by the keyword `field`.
37    Field(String),
38    /// A group literal, eg `42group`.
39    Group(String),
40    /// An integer literal, e.g., `42`.
41    Integer(IntegerType, String),
42    /// A scalar literal, e.g. `1scalar`.
43    /// An unsigned number followed by the keyword `scalar`.
44    Scalar(String),
45    /// A string literal, e.g., `"foobar"`.
46    String(String),
47}
48
49impl fmt::Display for LiteralVariant {
50    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
51        match &self {
52            Self::Address(address) => write!(f, "{address}"),
53            Self::Boolean(boolean) => write!(f, "{boolean}"),
54            Self::Field(field) => write!(f, "{field}field"),
55            Self::Group(group) => write!(f, "{group}group"),
56            Self::Integer(type_, value) => write!(f, "{value}{type_}"),
57            Self::Scalar(scalar) => write!(f, "{scalar}scalar"),
58            Self::String(string) => write!(f, "\"{string}\""),
59        }
60    }
61}
62
63impl fmt::Display for Literal {
64    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
65        self.variant.fmt(f)
66    }
67}

71struct DisplayDecimal<'a>(&'a Literal);
72
73impl Literal {
74    pub fn field(s: String, span: Span, id: NodeID) -> Self {
75        Literal { variant: LiteralVariant::Field(s), span, id }
76    }
77
78    pub fn group(s: String, span: Span, id: NodeID) -> Self {
79        Literal { variant: LiteralVariant::Group(s), span, id }
80    }
81
82    pub fn address(s: String, span: Span, id: NodeID) -> Self {
83        Literal { variant: LiteralVariant::Address(s), span, id }
84    }
85
86    pub fn scalar(s: String, span: Span, id: NodeID) -> Self {
87        Literal { variant: LiteralVariant::Scalar(s), span, id }
88    }
89
90    pub fn boolean(s: bool, span: Span, id: NodeID) -> Self {
91        Literal { variant: LiteralVariant::Boolean(s), span, id }
92    }
93
94    pub fn integer(integer_type: IntegerType, s: String, span: Span, id: NodeID) -> Self {
95        Literal { variant: LiteralVariant::Integer(integer_type, s), span, id }
96    }
97
98    /// For displaying a literal as decimal, regardless of the radix in which it was parsed.
99    ///
100    /// In particular this is useful for outputting .aleo files.
101    pub fn display_decimal(&self) -> impl '_ + fmt::Display {
102        DisplayDecimal(self)
103    }
104
105    /// For an integer literal, parse it and cast it to a u32.
106    ///
107    /// Panics if `self` is not an integer literal.
108    pub fn as_u32(&self) -> Option<u32> {
109        if let LiteralVariant::Integer(_, s) = &self.variant {
110            u32::from_str_by_radix(&s.replace("_", "")).ok()
111        } else {
112            panic!("`as_u32` must only be called on integer literals");
113        }
114    }
115}

117impl fmt::Display for DisplayDecimal<'_> {
118    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
119        // This function is duplicated in `interpreter/src/cursor.rs`,
120        // but there's not really a great place to put a common implementation
121        // right now.
122        fn prepare_snarkvm_string(s: &str, suffix: &str) -> String {
123            // If there's a `-`, separate it from the rest of the string.
124            let (neg, rest) = s.strip_prefix("-").map(|rest| ("-", rest)).unwrap_or(("", s));
125            // Remove leading zeros.
126            let mut rest = rest.trim_start_matches('0');
127            if rest.is_empty() {
128                rest = "0";
129            }
130            format!("{neg}{rest}{suffix}")
131        }
132
133        match &self.0.variant {
134            LiteralVariant::Address(address) => write!(f, "{address}"),
135            LiteralVariant::Boolean(boolean) => write!(f, "{boolean}"),
136            LiteralVariant::Field(field) => write!(f, "{}", prepare_snarkvm_string(field, "field")),
137            LiteralVariant::Group(group) => write!(f, "{}", prepare_snarkvm_string(group, "group")),
138            LiteralVariant::Integer(type_, value) => {
139                let string = value.replace('_', "");
140                if value.starts_with('-') {
141                    let v = i128::from_str_by_radix(&string).expect("Failed to parse integer?");
142                    write!(f, "{v}{type_}")
143                } else {
144                    let v = u128::from_str_by_radix(&string).expect("Failed to parse integer?");
145                    write!(f, "{v}{type_}")
146                }
147            }
148            LiteralVariant::Scalar(scalar) => write!(f, "{}", prepare_snarkvm_string(scalar, "scalar")),
149            LiteralVariant::String(string) => write!(f, "\"{string}\""),
150        }
151    }
152}

160/// This trait allows to parse integer literals of any type generically.
161///
162/// The literal may optionally start with a `-` and/or `0x` or `0o` or 0b`.
163pub trait FromStrRadix: Sized {
164    fn from_str_by_radix(src: &str) -> Result<Self, std::num::ParseIntError>;
165}
166
167macro_rules! implement_from_str_radix {
168    ($($ty:ident)*) => {
169        $(
170            impl FromStrRadix for $ty {
171                fn from_str_by_radix(src: &str) -> Result<Self, std::num::ParseIntError> {
172                    if let Some(stripped) = src.strip_prefix("0x") {
173                        Self::from_str_radix(stripped, 16)
174                    } else if let Some(stripped) = src.strip_prefix("0o") {
175                        Self::from_str_radix(stripped, 8)
176                    } else if let Some(stripped) = src.strip_prefix("0b") {
177                        Self::from_str_radix(stripped, 2)
178                    } else if let Some(stripped) = src.strip_prefix("-0x") {
179                        // We have to remove the 0x prefix and put back in a - to use
180                        // std's parsing. Alternatively we could jump through
181                        // a few hoops to avoid allocating.
182                        let mut s = String::new();
183                        s.push('-');
184                        s.push_str(stripped);
185                        Self::from_str_radix(&s, 16)
186                    } else if let Some(stripped) = src.strip_prefix("-0o") {
187                        // Ditto.
188                        let mut s = String::new();
189                        s.push('-');
190                        s.push_str(stripped);
191                        Self::from_str_radix(&s, 8)
192                    } else if let Some(stripped) = src.strip_prefix("-0b") {
193                        // Ditto.
194                        let mut s = String::new();
195                        s.push('-');
196                        s.push_str(stripped);
197                        Self::from_str_radix(&s, 2)
198                    } else {
199                        Self::from_str_radix(src, 10)
200                    }
201                }
202            }
203        )*
204    };
205}
