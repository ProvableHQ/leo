45/// Global values - such as mappings, functions, etc -
46/// are identified by program and name.
47#[derive(Clone, Copy, Debug, Hash, Eq, PartialEq)]
48pub struct GlobalId {
49    pub program: Symbol,
50    pub name: Symbol,
51}
52
53impl fmt::Display for GlobalId {
54    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
55        write!(f, "{}.aleo/{}", self.program, self.name)
56    }
57}
58
59#[derive(Clone, Debug, Eq, PartialEq)]
60pub struct StructContents {
61    pub name: Symbol,
62    pub contents: IndexMap<Symbol, Value>,
63}
64
65impl Hash for StructContents {
66    fn hash<H: Hasher>(&self, state: &mut H) {
67        self.name.hash(state);
68        for (_symbol, value) in self.contents.iter() {
69            value.hash(state);
70        }
71    }
72}

74#[derive(Clone, Debug, Eq, PartialEq, Hash)]
75pub struct AsyncExecution {
76    pub function: GlobalId,
77    pub arguments: Vec<Value>,
78}
79
80#[derive(Clone, Debug, Default, Eq, PartialEq, Hash)]
81pub struct Future(pub Vec<AsyncExecution>);
82
83impl fmt::Display for Future {
84    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
85        write!(f, "Future")?;
86        if !self.0.is_empty() {
87            write!(f, " with calls to ")?;
88            let mut names = self.0.iter().map(|async_ex| async_ex.function).peekable();
89            while let Some(name) = names.next() {
90                write!(f, "{name}")?;
91                if names.peek().is_some() {
92                    write!(f, ", ")?;
93                }
94            }
95        }
96        Ok(())
97    }
98}

100/// A Leo value of any type.
101///
102/// Mappings and functions aren't considered values.
103#[derive(Clone, Debug, Default, Eq, PartialEq, Hash)]
104pub enum Value {
105    #[default]
106    Unit,
107    Bool(bool),
108    U8(u8),
109    U16(u16),
110    U32(u32),
111    U64(u64),
112    U128(u128),
113    I8(i8),
114    I16(i16),
115    I32(i32),
116    I64(i64),
117    I128(i128),
118    Group(SvmGroup),
119    Field(SvmField),
120    Scalar(SvmScalar),
121    Array(Vec<Value>),
122    // Signature(Box<SvmSignature>),
123    Tuple(Vec<Value>),
124    Address(SvmAddress),
125    Future(Future),
126    Struct(StructContents),
127    // String(()),
128}
129
130impl fmt::Display for Value {
131    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
132        use Value::*;
133        match self {
134            Unit => write!(f, "()"),
135
136            Bool(x) => write!(f, "{x}"),
137            U8(x) => write!(f, "{x}u8"),
138            U16(x) => write!(f, "{x}u16"),
139            U32(x) => write!(f, "{x}u32"),
140            U64(x) => write!(f, "{x}u64"),
141            U128(x) => write!(f, "{x}u128"),
142            I8(x) => write!(f, "{x}i8"),
143            I16(x) => write!(f, "{x}i16"),
144            I32(x) => write!(f, "{x}i32"),
145            I64(x) => write!(f, "{x}i64"),
146            I128(x) => write!(f, "{x}i128"),
147            Group(x) => write!(f, "{x}"),
148            Field(x) => write!(f, "{x}"),
149            Scalar(x) => write!(f, "{x}"),
150            Array(x) => {
151                write!(f, "[")?;
152                let mut iter = x.iter().peekable();
153                while let Some(value) = iter.next() {
154                    write!(f, "{value}")?;
155                    if iter.peek().is_some() {
156                        write!(f, ", ")?;
157                    }
158                }
159                write!(f, "]")
160            }
161            Struct(StructContents { name, contents }) => {
162                write!(f, "{name} {{")?;
163                let mut iter = contents.iter().peekable();
164                while let Some((member_name, value)) = iter.next() {
165                    write!(f, "{member_name}: {value}")?;
166                    if iter.peek().is_some() {
167                        write!(f, ", ")?;
168                    }
169                }
170                write!(f, "}}")
171            }
172            Tuple(x) => {
173                write!(f, "(")?;
174                let mut iter = x.iter().peekable();
175                while let Some(value) = iter.next() {
176                    write!(f, "{value}")?;
177                    if iter.peek().is_some() {
178                        write!(f, ", ")?;
179                    }
180                }
181                write!(f, ")")
182            }
183            Address(x) => write!(f, "{x}"),
184            Future(future) => write!(f, "{future}"),
185            // Signature(x) => write!(f, "{x}"),
186            // String(_) => todo!(),
187        }
188    }
189}
