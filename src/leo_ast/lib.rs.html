65/// The abstract syntax tree (AST) for a Leo program.
66///
67/// The [`Ast`] type represents a Leo program as a series of recursive data types.
68/// These data types form a tree that begins from a [`Program`] type root.
69#[derive(Clone, Debug, Default, Eq, PartialEq)]
70pub struct Ast {
71    pub ast: Program,
72}
73
74impl Ast {
75    /// Creates a new AST from a given program tree.
76    pub fn new(program: Program) -> Self {
77        Self { ast: program }
78    }
79
80    /// Returns a reference to the inner program AST representation.
81    pub fn as_repr(&self) -> &Program {
82        &self.ast
83    }
84
85    pub fn into_repr(self) -> Program {
86        self.ast
87    }
88
89    /// Serializes the ast into a JSON string.
90    pub fn to_json_string(&self) -> Result<String> {
91        Ok(serde_json::to_string_pretty(&self.ast).map_err(|e| AstError::failed_to_convert_ast_to_json_string(&e))?)
92    }
93
94    // Converts the ast into a JSON value.
95    // Note that there is no corresponding `from_json_value` function
96    // since we modify JSON values leaving them unable to be converted
97    // back into Programs.
98    pub fn to_json_value(&self) -> Result<serde_json::Value> {
99        Ok(serde_json::to_value(&self.ast).map_err(|e| AstError::failed_to_convert_ast_to_json_value(&e))?)
100    }
101
102    /// Serializes the ast into a JSON file.
103    pub fn to_json_file(&self, mut path: std::path::PathBuf, file_name: &str) -> Result<()> {
104        path.push(file_name);
105        let file = std::fs::File::create(&path).map_err(|e| AstError::failed_to_create_ast_json_file(&path, &e))?;
106        let writer = std::io::BufWriter::new(file);
107        Ok(serde_json::to_writer_pretty(writer, &self.ast)
108            .map_err(|e| AstError::failed_to_write_ast_to_json_file(&path, &e))?)
109    }
110
111    /// Serializes the ast into a JSON value and removes keys from object mappings before writing to a file.
112    pub fn to_json_file_without_keys(
113        &self,
114        mut path: std::path::PathBuf,
115        file_name: &str,
116        excluded_keys: &[&str],
117    ) -> Result<()> {
118        path.push(file_name);
119        let file = std::fs::File::create(&path).map_err(|e| AstError::failed_to_create_ast_json_file(&path, &e))?;
120        let writer = std::io::BufWriter::new(file);
121
122        let mut value = self.to_json_value().unwrap();
123        for key in excluded_keys {
124            value = remove_key_from_json(value, key);
125        }
126        value = normalize_json_value(value);
127
128        Ok(serde_json::to_writer_pretty(writer, &value)
129            .map_err(|e| AstError::failed_to_write_ast_to_json_file(&path, &e))?)
130    }
131
132    /// Deserializes the JSON string into a ast.
133    pub fn from_json_string(json: &str) -> Result<Self> {
134        let ast: Program = serde_json::from_str(json).map_err(|e| AstError::failed_to_read_json_string_to_ast(&e))?;
135        Ok(Self { ast })
136    }
137
138    /// Deserializes the JSON string into a ast from a file.
139    pub fn from_json_file(path: std::path::PathBuf) -> Result<Self> {
140        let data = std::fs::read_to_string(&path).map_err(|e| AstError::failed_to_read_json_file(&path, &e))?;
141        Self::from_json_string(&data)
142    }
143}
144
145impl AsRef<Program> for Ast {
146    fn as_ref(&self) -> &Program {
147        &self.ast
148    }
149}

151/// Helper function to recursively filter keys from AST JSON
152pub fn remove_key_from_json(value: serde_json::Value, key: &str) -> serde_json::Value {
153    match value {
154        serde_json::Value::Object(map) => serde_json::Value::Object(
155            map.into_iter().filter(|(k, _)| k != key).map(|(k, v)| (k, remove_key_from_json(v, key))).collect(),
156        ),
157        serde_json::Value::Array(values) => {
158            serde_json::Value::Array(values.into_iter().map(|v| remove_key_from_json(v, key)).collect())
159        }
160        _ => value,
161    }
162}
163
164/// Helper function to normalize AST JSON into a form compatible with tgc.
165/// This function will traverse the original JSON value and produce a new
166/// one under the following rules:
167/// 1. Remove empty object mappings from JSON arrays
168/// 2. If there are two elements in a JSON array and one is an empty object
169///    mapping and the other is not, then lift up the one that isn't
170pub fn normalize_json_value(value: serde_json::Value) -> serde_json::Value {
171    match value {
172        serde_json::Value::Array(vec) => {
173            let orig_length = vec.len();
174            let mut new_vec: Vec<serde_json::Value> = vec
175                .into_iter()
176                .filter(|v| !matches!(v, serde_json::Value::Object(map) if map.is_empty()))
177                .map(normalize_json_value)
178                .collect();
179
180            if orig_length == 2 && new_vec.len() == 1 {
181                new_vec.pop().unwrap()
182            } else {
183                serde_json::Value::Array(new_vec)
184            }
185        }
186        serde_json::Value::Object(map) => {
187            serde_json::Value::Object(map.into_iter().map(|(k, v)| (k, normalize_json_value(v))).collect())
188        }
189        _ => value,
190    }
191}
