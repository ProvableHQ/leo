107pub fn handle_error<T>(res: Result<T>) -> T {
108    match res {
109        Ok(t) => t,
110        Err(err) => {
111            eprintln!("{err}");
112            exit(err.exit_code());
113        }
114    }
115}

117/// Run command with custom build arguments.
118pub fn run_with_args(cli: CLI) -> Result<()> {
119    if !cli.quiet {
120        // Init logger with optional debug flag.
121        logger::init_logger("leo", match cli.debug {
122            false => 1,
123            true => 2,
124        })?;
125    }
126
127    //  Check for updates. If not forced, it checks once per day.
128    if let Ok(true) = updater::Updater::check_for_updates(false) {
129        let _ = updater::Updater::print_cli();
130    }
131
132    // Get custom root folder and create context for it.
133    // If not specified, default context will be created in cwd.
134    let context = handle_error(Context::new(cli.path, cli.home, false));
135
136    match cli.command {
137        Commands::Add { command } => command.try_execute(context),
138        Commands::Account { command } => command.try_execute(context),
139        Commands::New { command } => command.try_execute(context),
140        Commands::Build { command } => command.try_execute(context),
141        Commands::Debug { command } => command.try_execute(context),
142        Commands::Query { command } => command.try_execute(context),
143        Commands::Clean { command } => command.try_execute(context),
144        Commands::Deploy { command } => command.try_execute(context),
145        Commands::Run { command } => command.try_execute(context),
146        Commands::Execute { command } => command.try_execute(context),
147        Commands::Remove { command } => command.try_execute(context),
148        Commands::Update { command } => command.try_execute(context),
149    }
150}

326    pub(crate) fn sample_nested_package(temp_dir: &Path) {

329        // Remove it if it already exists

337            debug: false,

354        // `nested.aleo` program
355        let program_str = "
356import nested_example_layer_0.aleo;
357program nested.aleo {
358    transition example(public a: u32, b: u32) -> u32 {
359        let c: u32 = nested_example_layer_0.aleo/main(a, b);
360        return c;
361    }
362}
363";
364        // `nested_example_layer_0.aleo` program
365        let nested_example_layer_0 = "
366import nested_example_layer_2.aleo;
367import nested_example_layer_1.aleo;
368
369program nested_example_layer_0.aleo;
370
371function main:
372    input r0 as u32.public;
373    input r1 as u32.private;
374    call nested_example_layer_1.aleo/external_function r0 r1 into r2;
375    output r2 as u32.private;

377
378        // `nested_example_layer_1.aleo` program
379        let nested_example_layer_1 = "
380import nested_example_layer_2.aleo;
381
382program nested_example_layer_1.aleo;
383
384function external_function:
385    input r0 as u32.public;
386    input r1 as u32.private;
387    call nested_example_layer_2.aleo/external_nested_function r0 r1 into r2;
388    output r2 as u32.private;
389";
390
391        // `nested_example_layer_2.aleo` program
392        let nested_example_layer_2 = "
393program nested_example_layer_2.aleo;
394
395function external_nested_function:
396    input r0 as u32.public;
