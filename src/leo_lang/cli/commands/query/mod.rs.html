47///  Query live data from the Aleo network.
48#[derive(Parser, Debug)]
49pub struct LeoQuery {
50    #[clap(short, long, global = true, help = "Endpoint to retrieve network state from. Defaults to entry in `.env`.")]
51    pub endpoint: Option<String>,
52    #[clap(short, long, global = true, help = "Network to use. Defaults to entry in `.env`.")]
53    pub(crate) network: Option<String>,
54    #[clap(subcommand)]
55    pub command: QueryCommands,
56}
57
58impl Command for LeoQuery {
59    type Input = ();
60    type Output = String;
61
62    fn log_span(&self) -> Span {
63        tracing::span!(tracing::Level::INFO, "Leo")
64    }
65
66    fn prelude(&self, _context: Context) -> Result<Self::Input> {
67        Ok(())
68    }
69
70    fn apply(self, context: Context, _: Self::Input) -> Result<Self::Output> {
71        // Parse the network.
72        let network = NetworkName::try_from(context.get_network(&self.network)?)?;
73        let endpoint = context.get_endpoint(&self.endpoint)?;
74        match network {
75            NetworkName::MainnetV0 => handle_query::<MainnetV0>(self, context, &network.to_string(), &endpoint),
76            NetworkName::TestnetV0 => handle_query::<TestnetV0>(self, context, &network.to_string(), &endpoint),
77            NetworkName::CanaryV0 => handle_query::<CanaryV0>(self, context, &network.to_string(), &endpoint),
78        }
79    }
80}
81
82// A helper function to handle the `query` command.
83fn handle_query<N: Network>(
84    query: LeoQuery,
85    context: Context,
86    network: &str,
87    endpoint: &str,
88) -> Result<<LeoQuery as Command>::Output> {
89    let recursive = context.recursive;
90    let (program, output) = match query.command {
91        QueryCommands::Block { command } => (None, command.apply(context, ())?),
92        QueryCommands::Transaction { command } => (None, command.apply(context, ())?),
93        QueryCommands::Program { command } => {
94            // Check if querying for program source code.
95            let program =
96                if command.mappings || command.mapping_value.is_some() { None } else { Some(command.name.clone()) };
97            (program, command.apply(context, ())?)
98        }
99        QueryCommands::Stateroot { command } => (None, command.apply(context, ())?),
100        QueryCommands::Committee { command } => (None, command.apply(context, ())?),
101        QueryCommands::Mempool { command } => {
102            if endpoint == "https://api.explorer.provable.com/v1" {
103                tracing::warn!(
104                    "⚠️  `leo query mempool` is only valid when using a custom endpoint. Specify one using `--endpoint`."
105                );
106            }
107            (None, command.apply(context, ())?)
108        }
109        QueryCommands::Peers { command } => {
110            if endpoint == "https://api.explorer.provable.com/v1" {
111                tracing::warn!(
112                    "⚠️  `leo query peers` is only valid when using a custom endpoint. Specify one using `--endpoint`."
113                );
114            }
115            (None, command.apply(context, ())?)
116        }
117    };
118
119    // Make GET request to retrieve on-chain state.
120    let url = format!("{}/{}/{output}", endpoint, network);
121    let result = fetch_from_network(&url)?;
122    if !recursive {
123        tracing::info!("✅ Successfully retrieved data from '{url}'.\n");
124        println!("{}\n", result);
125    }
126
127    // Verify that the source file parses into a valid Aleo program.
128    if let Some(name) = program {
129        verify_valid_program::<N>(&name, &result)?;
130    }
131
132    Ok(result)
133}
