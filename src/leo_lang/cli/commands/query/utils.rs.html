20// A valid hash is 61 characters long, with preface "ab1" and all characters lowercase or numbers.
21pub fn is_valid_hash(hash: &str) -> Result<(), LeoError> {
22    if hash.len() != 61 {
23        Err(UtilError::invalid_input_id_len(hash, "hash").into())
24    } else if !hash.starts_with("ab1") && hash.chars().all(|c| c.is_ascii_lowercase() || c.is_ascii_digit()) {
25        Err(UtilError::invalid_input_id(hash, "hash", "ab1").into())
26    } else {
27        Ok(())
28    }
29}

56// A valid numerical input is a u32.
57pub fn is_valid_numerical_input(num: &str) -> Result<(), LeoError> {
58    if num.parse::<u32>().is_err() { Err(UtilError::invalid_numerical_input(num).into()) } else { Ok(()) }
59}
60
61// A valid height or hash.
62pub fn is_valid_height_or_hash(input: &str) -> Result<(), LeoError> {
63    match (is_valid_hash(input), is_valid_numerical_input(input)) {
64        (Ok(_), _) | (_, Ok(_)) => Ok(()),
65        _ => Err(UtilError::invalid_height_or_hash(input).into()),
66    }
67}

69// Checks if the string is a valid field, allowing for optional `field` suffix.
70pub fn is_valid_field(field: &str) -> Result<String, LeoError> {
71    let split = field.split("field").collect::<Vec<&str>>();
72
73    if split.len() == 1 && split[0].chars().all(|c| c.is_numeric()) {
74        Ok(format!("{}field", field))
75    } else if split.len() == 2 && split[0].chars().all(|c| c.is_numeric()) && split[1].is_empty() {
76        Ok(field.to_string())
77    } else {
78        Err(UtilError::invalid_field(field).into())
79    }
80}

82// Checks if the string is a valid program name in Aleo.
83pub fn check_valid_program_name(name: String) -> String {
84    if name.ends_with(".aleo") {
85        Package::is_aleo_name_valid(&name[0..name.len() - 5]);
86        name
87    } else {
88        Package::is_aleo_name_valid(&name);
89        format!("{}.aleo", name)
90    }
