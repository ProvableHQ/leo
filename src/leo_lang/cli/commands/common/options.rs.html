125// A helper function to parse amounts, which can either be a `u64` or `default`.
126fn parse_amount(s: &str) -> Result<Option<u64>, String> {
127    let trimmed = s.trim();
128    if trimmed == "default" { Ok(None) } else { trimmed.parse::<u64>().map_err(|e| e.to_string()).map(Some) }
129}
130
131// A helper function to parse record strings, which can either be a string or `default`.
132fn parse_record_string(s: &str) -> Result<Option<String>, String> {
133    let trimmed = s.trim();
134    if trimmed == "default" { Ok(None) } else { Ok(Some(trimmed.to_string())) }
135}

137/// Parses the record string. If the string is a ciphertext, then attempt to decrypt it. Lifted from snarkOS.
138fn parse_record<N: Network>(private_key: &PrivateKey<N>, record: &str) -> Result<Record<N, Plaintext<N>>> {
139    match record.starts_with("record1") {
140        true => {
141            // Parse the ciphertext.
142            let ciphertext = Record::<N, Ciphertext<N>>::from_str(record)?;
143            // Derive the view key.
144            let view_key = ViewKey::try_from(private_key)?;
145            // Decrypt the ciphertext.
146            Ok(ciphertext.decrypt(&view_key)?)
147        }
148        false => Ok(Record::<N, Plaintext<N>>::from_str(record)?),
149    }
150}

152// A helper function to construct fee options for `k` transactions.
153#[allow(clippy::type_complexity)]
154pub fn parse_fee_options<N: Network>(
155    private_key: &PrivateKey<N>,
156    fee_options: &FeeOptions,
157    k: usize,
158) -> Result<Vec<(Option<u64>, Option<u64>, Option<Record<N, Plaintext<N>>>)>> {
159    // Parse the base fees.
160    let base_fees = fee_options.base_fees.clone();
161    // Parse the priority fees.
162    let priority_fees = fee_options.priority_fees.clone();
163    // Parse the fee records.
164    let parse_record = |record: &Option<String>| record.as_ref().map(|r| parse_record::<N>(private_key, r)).transpose();
165    let fee_records = fee_options.fee_records.iter().map(parse_record).collect::<Result<Vec<_>>>()?;
166
167    // Pad the vectors to length `k`.
168    let base_fees = base_fees.into_iter().chain(iter::repeat(None)).take(k);
169    let priority_fees = priority_fees.into_iter().chain(iter::repeat(None)).take(k);
170    let fee_records = fee_records.into_iter().chain(iter::repeat(None)).take(k);
171
172    Ok(base_fees.zip(priority_fees).zip(fee_records).map(|((x, y), z)| (x, y, z)).collect())
173}

200// A helper function to get the consensus version from the fee options.
201// If a consensus version is not provided, then attempt to query the current block height and use it to determine the version.
202pub fn get_consensus_version(
203    consensus_version: &Option<u8>,
204    endpoint: &str,
205    network: NetworkName,
206    heights: &[u32],
207    context: &Context,
208) -> Result<ConsensusVersion> {
209    // Get the consensus version.
210    let result = match consensus_version {
211        Some(1) => Ok(ConsensusVersion::V1),
212        Some(2) => Ok(ConsensusVersion::V2),
213        Some(3) => Ok(ConsensusVersion::V3),
214        Some(4) => Ok(ConsensusVersion::V4),
215        Some(5) => Ok(ConsensusVersion::V5),
216        Some(6) => Ok(ConsensusVersion::V6),
217        Some(7) => Ok(ConsensusVersion::V7),
218        Some(8) => Ok(ConsensusVersion::V8),
219        Some(9) => Ok(ConsensusVersion::V9),
220        Some(10) => Ok(ConsensusVersion::V10),
221        // If none is provided, then attempt to query the current block height and use it to determine the version.
222        None => {
223            println!("Attempting to determine the consensus version from the latest block height at {endpoint}...");
224            // Get the consensus heights for the current network.
225            get_latest_block_height(endpoint, network, context)
226                .and_then(|current_block_height| get_consensus_version_from_height(current_block_height, heights))
227                .map_err(|_| {
228                    CliError::custom(
229                        "Failed to get consensus version. Ensure that your endpoint is valid or provide an explicit version to use via `--consensus-version`",
230                    )
231                        .into()
232                })
233        }
234        Some(version) => Err(CliError::custom(format!("Invalid consensus version: {version}")).into()),
235    };
236
237    // Check `{endpoint}/{network}/consensus_version` endpoint for the consensus version.
238    // If it returns a result and does not match the given version, print a warning.
239    if let Ok(consensus_version) = result {
240        if let Err(e) = check_consensus_version_mismatch(consensus_version, endpoint, network) {
241            println!("⚠️ Warning: {e}");
242        }
243    }
244
245    result
246}

248/// A helper function to check for a consensus version mismatch against the network.
249pub fn check_consensus_version_mismatch(
250    consensus_version: ConsensusVersion,
251    endpoint: &str,
252    network: NetworkName,
253) -> anyhow::Result<()> {
254    // Check the `{endpoint}/{network}/consensus_version` endpoint for the consensus version.
255    if let Ok(response) = fetch_from_network(&format!("{endpoint}/{network}/consensus_version")) {
256        if let Ok(response) = response.parse::<u8>() {
257            let consensus_version = consensus_version as u8;
258            if response != consensus_version {
259                bail!("Expected consensus version {consensus_version} but found {response} at {endpoint}",);
260            }
261        }
262    }
263    Ok(())
264}

266// A helper function to get the consensus version based on the block height.
267// Note. This custom implementation is necessary because we use `snarkVM` with the `test_heights` feature enabled, which does not reflect the actual consensus version heights.
268pub fn get_consensus_version_from_height(seek_height: u32, heights: &[u32]) -> Result<ConsensusVersion> {
269    // Find the consensus version based on the block height.
270    let index = match heights.binary_search_by(|height| height.cmp(&seek_height)) {
271        // If a consensus version was found at this height, return it.
272        Ok(index) => index,
273        // If the specified height was not found, determine whether to return an appropriate version.
274        Err(index) => {
275            if index == 0 {
276                return Err(CliError::custom("Expected consensus version 1 to exist at height 0.").into());
277            } else {
278                // Return the appropriate version belonging to the height *lower* than the sought height.
279                index - 1
280            }
281        }
282    };
283    // Convert the index to a consensus version.
284    Ok(number_to_consensus_version(index + 1))
285}

287// A helper to convert an index to a consensus version.
288pub fn number_to_consensus_version(index: usize) -> ConsensusVersion {
289    match index {
290        1 => ConsensusVersion::V1,
291        2 => ConsensusVersion::V2,
292        3 => ConsensusVersion::V3,
293        4 => ConsensusVersion::V4,
294        5 => ConsensusVersion::V5,
295        6 => ConsensusVersion::V6,
296        7 => ConsensusVersion::V7,
297        8 => ConsensusVersion::V8,
298        9 => ConsensusVersion::V9,
299        10 => ConsensusVersion::V10,
300        _ => panic!("Invalid consensus version: {index}"),
301    }
302}

304/// Get the consensus heights for the current network.
305/// If `is_devnet` is true, first , then return the test consensus heights.
306pub fn get_consensus_heights(network_name: NetworkName, is_devnet: bool) -> Vec<u32> {
307    if is_devnet {
308        TEST_CONSENSUS_VERSION_HEIGHTS.into_iter().map(|(_, v)| v).collect_vec()
309    } else {
310        match network_name {
311            NetworkName::CanaryV0 => CANARY_V0_CONSENSUS_VERSION_HEIGHTS,
312            NetworkName::MainnetV0 => MAINNET_V0_CONSENSUS_VERSION_HEIGHTS,
313            NetworkName::TestnetV0 => TESTNET_V0_CONSENSUS_VERSION_HEIGHTS,
314        }
315        .into_iter()
316        .map(|(_, v)| v)
317        .collect_vec()
318    }
319}

321/// Validates a vector of heights as consensus heights.
322pub fn validate_consensus_heights(heights: &[u32]) -> anyhow::Result<()> {
323    // Assert that the genesis height is 0.
324    ensure!(heights[0] == 0, "Genesis height must be 0.");
325    // Assert that the consensus heights are strictly increasing.
326    for window in heights.windows(2) {
327        if window[0] >= window[1] {
328            bail!("Heights must be strictly increasing, but found: {window:?}");
329        }
330    }
331    Ok(())
332}
