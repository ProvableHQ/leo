97    #[clap(
98        long,

100        value_delimiter = '|',
101        value_parser = parse_amount
102    )]
103    pub(crate) priority_fees: Vec<Option<u64>>,
104    #[clap(
105        short,
106        help = "Records to pay for fees privately, delimited by '|', and used in order. The fees must either be valid plaintext, ciphertext, or `default`. Defaults to public fees.",
107        long,
108        value_delimiter = '|',
109        value_parser = parse_record_string,
110    )]
111    fee_records: Vec<Option<String>>,
112}

114// A helper function to parse amounts, which can either be a `u64` or `default`.
115fn parse_amount(s: &str) -> Result<Option<u64>, String> {
116    let trimmed = s.trim();
117    if trimmed == "default" { Ok(None) } else { trimmed.parse::<u64>().map_err(|e| e.to_string()).map(Some) }
118}
119
120// A helper function to parse record strings, which can either be a string or `default`.
121fn parse_record_string(s: &str) -> Result<Option<String>, String> {
122    let trimmed = s.trim();
123    if trimmed == "default" { Ok(None) } else { Ok(Some(trimmed.to_string())) }
124}

126/// Parses the record string. If the string is a ciphertext, then attempt to decrypt it. Lifted from snarkOS.
127fn parse_record<N: Network>(private_key: &PrivateKey<N>, record: &str) -> Result<Record<N, Plaintext<N>>> {
128    match record.starts_with("record1") {
129        true => {
130            // Parse the ciphertext.
131            let ciphertext = Record::<N, Ciphertext<N>>::from_str(record)?;
132            // Derive the view key.
133            let view_key = ViewKey::try_from(private_key)?;
134            // Decrypt the ciphertext.
135            Ok(ciphertext.decrypt(&view_key)?)
136        }
137        false => Ok(Record::<N, Plaintext<N>>::from_str(record)?),
138    }
139}

141// A helper function to construct fee options for `k` transactions.
142#[allow(clippy::type_complexity)]
143pub fn parse_fee_options<N: Network>(
144    private_key: &PrivateKey<N>,
145    fee_options: &FeeOptions,
146    k: usize,
147) -> Result<Vec<(Option<u64>, Option<u64>, Option<Record<N, Plaintext<N>>>)>> {
148    // Parse the base fees.
149    let base_fees = fee_options.base_fees.clone();
150    // Parse the priority fees.
151    let priority_fees = fee_options.priority_fees.clone();
152    // Parse the fee records.
153    let parse_record = |record: &Option<String>| record.as_ref().map(|r| parse_record::<N>(private_key, r)).transpose();
154    let fee_records = fee_options.fee_records.iter().map(parse_record).collect::<Result<Vec<_>>>()?;
155
156    // Pad the vectors to length `k`.
157    let base_fees = base_fees.into_iter().chain(iter::repeat(None)).take(k);
158    let priority_fees = priority_fees.into_iter().chain(iter::repeat(None)).take(k);
159    let fee_records = fee_records.into_iter().chain(iter::repeat(None)).take(k);
160
161    Ok(base_fees.zip(priority_fees).zip(fee_records).map(|((x, y), z)| (x, y, z)).collect())
162}

189// A helper function to get the consensus version from the fee options.
190// If a consensus version is not provided, then attempt to query the current block height and use it to determine the version.
191pub fn get_consensus_version(
192    consensus_version: &Option<u8>,
193    endpoint: &str,
194    network: NetworkName,
195    context: &Context,
196) -> Result<ConsensusVersion> {
197    // Get the consensus version.
198    match consensus_version {
199        Some(1) => Ok(ConsensusVersion::V1),
200        Some(2) => Ok(ConsensusVersion::V2),
201        Some(3) => Ok(ConsensusVersion::V3),
202        Some(4) => Ok(ConsensusVersion::V4),
203        Some(5) => Ok(ConsensusVersion::V5),
204        Some(6) => Ok(ConsensusVersion::V6),
205        Some(7) => Ok(ConsensusVersion::V7),
206        Some(8) => Ok(ConsensusVersion::V8),
207        Some(9) => Ok(ConsensusVersion::V9),
208        // If none is provided, then attempt to query the current block height and use it to determine the version.
209        None => {
210            println!("Attempting to determine the consensus version from the latest block height at {endpoint}...");
211            get_latest_block_height(endpoint, network, context)
212                .and_then(|current_block_height| get_consensus_version_from_height(current_block_height, network))
213                .map_err(|_| {
214                    CliError::custom(
215                        "Failed to get consensus version. Ensure that your endpoint is valid or provide an explicit version to use via `--consensus-version`",
216                    )
217                        .into()
218                })
219        }
220        Some(version) => Err(CliError::custom(format!("Invalid consensus version: {version}")).into()),
221    }
222}

224// A helper function to get the consensus version based on the block height.
225// Note. This custom implementation is necessary because we use `snarkVM` with the `test_heights` feature enabled, which does not reflect the actual consensus version heights.
226pub fn get_consensus_version_from_height(seek_height: u32, network_name: NetworkName) -> Result<ConsensusVersion> {
227    let heights = if cfg!(feature = "test_network") {
228        TEST_CONSENSUS_VERSION_HEIGHTS
229    } else {
230        match network_name {
231            NetworkName::TestnetV0 => TESTNET_V0_CONSENSUS_VERSION_HEIGHTS,
232            NetworkName::MainnetV0 => MAINNET_V0_CONSENSUS_VERSION_HEIGHTS,
233            NetworkName::CanaryV0 => CANARY_V0_CONSENSUS_VERSION_HEIGHTS,
234        }
235    };
236
237    // Find the consensus version based on the block height.
238    match heights.binary_search_by(|(_, height)| height.cmp(&seek_height)) {
239        // If a consensus version was found at this height, return it.
240        Ok(index) => Ok(heights[index].0),
241        // If the specified height was not found, determine whether to return an appropriate version.
242        Err(index) => {
243            if index == 0 {
244                Err(CliError::custom("Expected consensus version 1 to exist at height 0.").into())
245            } else {
246                // Return the appropriate version belonging to the height *lower* than the sought height.
247                Ok(heights[index - 1].0)
248            }
249        }
250    }
251}
