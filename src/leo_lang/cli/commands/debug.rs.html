30/// Debugs an Aleo program through the interpreter.
31#[derive(Parser, Debug)]
32pub struct LeoDebug {
33    #[arg(long, help = "Use these source files instead of finding source files through the project structure.", num_args = 1..)]
34    pub(crate) paths: Vec<String>,
35
36    #[arg(long, help = "The block height, accessible via block.height.", default_value = "0")]
37    pub(crate) block_height: u32,
38
39    #[arg(long, action, help = "Use the text user interface.")]
40    pub(crate) tui: bool,
41
42    #[clap(flatten)]
43    pub(crate) compiler_options: BuildOptions,
44}
45
46impl Command for LeoDebug {
47    type Input = <LeoBuild as Command>::Output;
48    type Output = ();
49
50    fn log_span(&self) -> Span {
51        tracing::span!(tracing::Level::INFO, "Leo")
52    }
53
54    fn prelude(&self, context: Context) -> Result<Self::Input> {
55        if self.paths.is_empty() {
56            (LeoBuild { options: self.compiler_options.clone() }).execute(context)
57        } else {
58            Ok(())
59        }
60    }
61
62    fn apply(self, context: Context, _: Self::Input) -> Result<Self::Output> {
63        // Parse the network.
64        let network = NetworkName::try_from(context.get_network(&self.compiler_options.network)?)?;
65        match network {
66            NetworkName::TestnetV0 => handle_debug::<TestnetV0>(&self, context),
67            NetworkName::MainnetV0 => {
68                #[cfg(feature = "only_testnet")]
69                panic!("Mainnet chosen with only_testnet feature");
70                #[cfg(not(feature = "only_testnet"))]
71                return handle_debug::<MainnetV0>(&self, context);
72            }
73            NetworkName::CanaryV0 => {
74                #[cfg(feature = "only_testnet")]
75                panic!("Canary chosen with only_testnet feature");
76                #[cfg(not(feature = "only_testnet"))]
77                return handle_debug::<CanaryV0>(&self, context);
78            }
79        }
80    }
81}
82
83fn handle_debug<N: Network>(command: &LeoDebug, context: Context) -> Result<()> {
84    if command.paths.is_empty() {
85        // Get the package path.
86        let package_path = context.dir()?;
87        let home_path = context.home()?;
88
89        // Get the program id.
90        let manifest = Manifest::read_from_dir(&package_path)?;
91        let program_id = ProgramID::<N>::from_str(manifest.program())?;
92
93        // Get the private key.
94        let private_key = context.get_private_key(&None)?;
95        let address = Address::try_from(&private_key)?;
96
97        // Retrieve all local dependencies in post order
98        let main_sym = Symbol::intern(&program_id.name().to_string());
99        let mut retriever = Retriever::<N>::new(
100            main_sym,
101            &package_path,
102            &home_path,
103            context.get_endpoint(&command.compiler_options.endpoint)?.to_string(),
104        )
105        .map_err(|err| UtilError::failed_to_retrieve_dependencies(err, Default::default()))?;
106        let mut local_dependencies =
107            retriever.retrieve().map_err(|err| UtilError::failed_to_retrieve_dependencies(err, Default::default()))?;
108
109        // Push the main program at the end of the list.
110        local_dependencies.push(main_sym);
111
112        let paths: Vec<PathBuf> = local_dependencies
113            .into_iter()
114            .map(|dependency| {
115                let base_path = retriever.get_context(&dependency).full_path();
116                base_path.join("src/main.leo")
117            })
118            .collect();
119
120        let imports_directory = package_path.join("build/imports");
121
122        let aleo_paths: Vec<PathBuf> = if let Ok(dir) = fs::read_dir(imports_directory) {
123            dir.flat_map(|maybe_filename| maybe_filename.ok())
124                .filter(|entry| entry.file_type().ok().map(|filetype| filetype.is_file()).unwrap_or(false))
125                .flat_map(|entry| {
126                    let path = entry.path();
127                    if path.extension().map(|e| e == "aleo").unwrap_or(false) { Some(path) } else { None }
128                })
129                .collect()
130        } else {
131            Vec::new()
132        };
133
134        leo_interpreter::interpret(&paths, &aleo_paths, address, command.block_height, command.tui)
135    } else {
136        let private_key: PrivateKey<TestnetV0> = PrivateKey::from_str(leo_package::TEST_PRIVATE_KEY)?;
137        let address = Address::try_from(&private_key)?;
138
139        let leo_paths: Vec<PathBuf> = command
140            .paths
141            .iter()
142            .filter(|path_str| path_str.ends_with(".leo"))
143            .map(|path_str| path_str.into())
144            .collect();
145        let aleo_paths: Vec<PathBuf> = command
146            .paths
147            .iter()
148            .filter(|path_str| !path_str.ends_with(".leo"))
149            .map(|path_str| path_str.into())
150            .collect();
151
152        leo_interpreter::interpret(&leo_paths, &aleo_paths, address, command.block_height, command.tui)
153    }
154}
