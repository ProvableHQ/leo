44/// Build, Prove and Run Leo program with inputs
45#[derive(Parser, Debug)]
46pub struct LeoExecute {
47    #[clap(
48        name = "NAME",
49        help = "The name of the function to execute, e.g `helloworld.aleo/main` or `main`.",
50        default_value = "main"
51    )]
52    name: String,
53    #[clap(name = "INPUTS", help = "The inputs to the program.")]
54    inputs: Vec<String>,
55    #[clap(flatten)]
56    pub(crate) fee_options: FeeOptions,
57    #[clap(flatten)]
58    pub(crate) action: TransactionAction,
59    #[clap(flatten)]
60    pub(crate) env_override: EnvOptions,
61    #[clap(flatten)]
62    pub(crate) extra: ExtraOptions,
63    #[clap(flatten)]
64    build_options: BuildOptions,
65    #[arg(short, long, help = "The inputs to the program, from a file. Overrides the INPUTS argument.")]
66    file: Option<String>,
67}
68
69impl Command for LeoExecute {
70    type Input = Option<Package>;
71    type Output = ();
72
73    fn log_span(&self) -> Span {
74        tracing::span!(tracing::Level::INFO, "Leo")
75    }
76
77    fn prelude(&self, context: Context) -> Result<Self::Input> {
78        // Get the path to the current directory.
79        let path = context.dir()?;
80        // Get the path to the home directory.
81        let home_path = context.home()?;
82        // If the current directory is a valid Leo package, then build it.
83        if Package::from_directory_no_graph(path, home_path).is_ok() {
84            let package = LeoBuild {
85                env_override: self.env_override.clone(),
86                options: {
87                    let mut options = self.build_options.clone();
88                    options.no_cache = true;
89                    options
90                },
91            }
92            .execute(context)?;
93            // Return the package.
94            Ok(Some(package))
95        } else {
96            Ok(None)
97        }
98    }
99
100    fn apply(self, context: Context, input: Self::Input) -> Result<Self::Output> {
101        // Get the network, accounting for overrides.
102        let network = context.get_network(&self.env_override.network)?.parse()?;
103        // Handle each network with the appropriate parameterization.
104        match network {
105            NetworkName::TestnetV0 => handle_execute::<AleoTestnetV0>(self, context, network, input),
106            NetworkName::MainnetV0 => {
107                #[cfg(feature = "only_testnet")]
108                panic!("Mainnet chosen with only_testnet feature");
109                #[cfg(not(feature = "only_testnet"))]
110                return handle_execute::<AleoV0>(self, context, network, input);
111            }
112            NetworkName::CanaryV0 => {
113                #[cfg(feature = "only_testnet")]
114                panic!("Canary chosen with only_testnet feature");
115                #[cfg(not(feature = "only_testnet"))]
116                return handle_execute::<AleoCanaryV0>(self, context, network, input);
117            }
118        }
119    }
120}

122// A helper function to handle the `execute` command.
123fn handle_execute<A: Aleo>(
124    command: LeoExecute,
125    context: Context,
126    network: NetworkName,
127    package: Option<Package>,
128) -> Result<<LeoExecute as Command>::Output> {
129    // Get the private key and associated address, accounting for overrides.
130    let private_key = context.get_private_key(&command.env_override.private_key)?;
131    let address = Address::<A::Network>::try_from(&private_key)
132        .map_err(|e| CliError::custom(format!("Failed to parse address: {e}")))?;
133
134    // Get the endpoint, accounting for overrides.
135    let endpoint = context.get_endpoint(&command.env_override.endpoint)?;
136
137    // Parse the <NAME> into an optional program name and a function name.
138    // If only a function name is provided, then use the program name from the package.
139    let (program_name, function_name) = match command.name.split_once('/') {
140        Some((program_name, function_name)) => (program_name.to_string(), function_name.to_string()),
141        None => match &package {
142            Some(package) => (
143                package.programs.last().expect("There must be at least one program in a Leo package").name.to_string(),
144                command.name,
145            ),
146            None => {
147                return Err(CliError::custom(format!(
148                    "Running `leo execute {} ...`, without an explicit program name requires that your current working directory is a valid Leo project.",
149                    command.name
150                )).into());
151            }
152        },
153    };
154
155    // Parse the program name as a `ProgramID`.
156    let program_id = ProgramID::<A::Network>::from_str(&program_name)
157        .map_err(|e| CliError::custom(format!("Failed to parse program name: {e}")))?;
158    // Parse the function name as an `Identifier`.
159    let function_id = Identifier::<A::Network>::from_str(&function_name)
160        .map_err(|e| CliError::custom(format!("Failed to parse function name: {e}")))?;
161
162    // Get all the dependencies in the package if it exists.
163    // Get the programs and optional manifests for all programs.
164    let programs = if let Some(package) = &package {
165        // Get the package directories.
166        let build_directory = package.build_directory();
167        let imports_directory = package.imports_directory();
168        let source_directory = package.source_directory();
169        // Get the program names and their bytecode.
170        package
171            .programs
172            .iter()
173            .clone()
174            .map(|program| {
175                let program_id = ProgramID::<A::Network>::from_str(&format!("{}.aleo", program.name))
176                    .map_err(|e| CliError::custom(format!("Failed to parse program ID: {e}")))?;
177                match &program.data {
178                    ProgramData::Bytecode(bytecode) => Ok((program_id, bytecode.to_string())),
179                    ProgramData::SourcePath(path) => {
180                        // Get the path to the built bytecode.
181                        let bytecode_path = if path.as_path() == source_directory.join("main.leo") {
182                            build_directory.join("main.aleo")
183                        } else {
184                            imports_directory.join(format!("{}.aleo", program.name))
185                        };
186                        // Fetch the bytecode.
187                        let bytecode = std::fs::read_to_string(&bytecode_path).map_err(|e| {
188                            CliError::custom(format!("Failed to read bytecode at {}: {e}", bytecode_path.display()))
189                        })?;
190                        // Return the bytecode and the manifest.
191                        Ok((program_id, bytecode))
192                    }
193                }
194            })
195            .collect::<Result<Vec<_>>>()?
196    } else {
197        Vec::new()
198    };
199
200    // Parse the program strings into AVM programs.
201    let mut programs = programs
202        .into_iter()
203        .map(|(name, bytecode)| {
204            // Parse the program.
205            let program = snarkvm::prelude::Program::<A::Network>::from_str(&bytecode)
206                .map_err(|e| CliError::custom(format!("Failed to parse program: {e}")))?;
207            // Return the program and its name.
208            Ok((name, program))
209        })
210        .collect::<Result<Vec<_>>>()?;
211
212    // Determine whether the program is local or remote.
213    let is_local = programs.iter().any(|(name, _)| name == &program_id);
214
215    // If the program is local, then check that the function exists.
216    if is_local {
217        let program =
218            &programs.iter().find(|(name, _)| name == &program_id).expect("Program should exist since it is local").1;
219        if !program.contains_function(&function_id) {
220            return Err(CliError::custom(format!(
221                "Function `{function_name}` does not exist in program `{program_name}`."
222            ))
223            .into());
224        }
225    }
226
227    // Parse the inputs.
228    // If the `file` option is set, read the inputs from the file.
229    let input_strings = if let Some(file) = command.file {
230        // Read the inputs from the file.
231        let file = std::fs::read_to_string(file).map_err(|e| CliError::custom(format!("Failed to read file: {e}")))?;
232        // Parse the inputs from the file.
233        file.lines().map(|line| line.to_string()).collect::<Vec<_>>()
234    } else {
235        // Use the inputs from the command line.
236        command.inputs
237    };
238    let inputs = input_strings
239        .into_iter()
240        .map(|input| {
241            Value::from_str(&input).map_err(|e| CliError::custom(format!("Failed to parse input: {e}")).into())
242        })
243        .collect::<Result<Vec<_>>>()?;
244
245    // Get the first fee option.
246    let (_, priority_fee, record) =
247        parse_fee_options(&private_key, &command.fee_options, 1)?.into_iter().next().unwrap_or((None, None, None));
248
249    // Get the consensus version.
250    let consensus_version =
251        get_consensus_version::<A::Network>(&command.extra.consensus_version, &endpoint, network, &context)?;
252
253    // Print the execution plan.
254    print_execution_plan::<A::Network>(
255        &private_key,
256        &address,
257        &endpoint,
258        &network,
259        &program_name,
260        &function_name,
261        is_local,
262        priority_fee.unwrap_or(0),
263        record.is_some(),
264        &command.action,
265        consensus_version,
266    );
267
268    // Prompt the user to confirm the plan.
269    if !confirm("Do you want to proceed with execution?", command.extra.yes)? {
270        println!("‚ùå Execution aborted.");
271        return Ok(());
272    }
273
274    // Initialize an RNG.
275    let rng = &mut rand::thread_rng();
276
277    // Initialize a new VM.
278    let vm = VM::from(ConsensusStore::<A::Network, ConsensusMemory<A::Network>>::open(StorageMode::Production)?)?;
279
280    // Specify the query
281    let query = SnarkVMQuery::from(&endpoint);
282
283    // If the program is not local, then download it and its dependencies for the network.
284    // Note: The dependencies are downloaded in "post-order" (child before parent).
285    if !is_local {
286        println!("‚¨áÔ∏è Downloading {program_name} and its dependencies from {endpoint}...");
287        programs = load_programs_from_network(&context, program_id, network, &endpoint)?;
288    };
289
290    // Add the programs to the VM.
291    println!("Adding programs to the VM ...");
292    for (_, program) in programs {
293        vm.process().write().add_program(&program)?;
294    }
295
296    // Execute the program and produce a transaction.
297    let transaction = vm.execute(
298        &private_key,
299        (&program_name, &function_name),
300        inputs.iter(),
301        record,
302        priority_fee.unwrap_or(0),
303        Some(query),
304        rng,
305    )?;
306
307    // Print the execution stats.
308    print_execution_stats::<A::Network>(
309        &vm,
310        &program_name,
311        transaction.execution().expect("Expected execution"),
312        priority_fee,
313        consensus_version,
314    )?;
315
316    // Print the transaction.
317    // If the `print` option is set, print the execution transaction to the console.
318    // The transaction is printed in JSON format.
319    if command.action.print {
320        let transaction_json = serde_json::to_string_pretty(&transaction)
321            .map_err(|e| CliError::custom(format!("Failed to serialize transaction: {e}")))?;
322        println!("üñ®Ô∏è Printing execution for {program_name}\n{transaction_json}");
323    }
324
325    // If the `save` option is set, save the execution transaction to a file in the specified directory.
326    // The file format is `program_name.execution.json`.
327    // The directory is created if it doesn't exist.
328    if let Some(path) = &command.action.save {
329        // Create the directory if it doesn't exist.
330        std::fs::create_dir_all(path).map_err(|e| CliError::custom(format!("Failed to create directory: {e}")))?;
331        // Save the transaction to a file.
332        let file_path = PathBuf::from(path).join(format!("{program_name}.execution.json"));
333        println!("üíæ Saving execution for {program_name} at {}", file_path.display());
334        let transaction_json = serde_json::to_string_pretty(&transaction)
335            .map_err(|e| CliError::custom(format!("Failed to serialize transaction: {e}")))?;
336        std::fs::write(file_path, transaction_json)
337            .map_err(|e| CliError::custom(format!("Failed to write transaction to file: {e}")))?;
338    }
339
340    // If the `broadcast` option is set, broadcast each deployment transaction to the network.
341    if command.action.broadcast {
342        println!("üì° Broadcasting execution for {program_name}...");
343        // Get and confirm the fee with the user.
344        let fee = transaction.fee_transition().expect("Expected a fee in the transaction");
345        if !confirm_fee(&fee, &private_key, &address, &endpoint, network, &context, command.extra.yes)? {
346            println!("‚ùå Execution aborted.");
347            return Ok(());
348        }
349        // Broadcast the transaction to the network.
350        let response =
351            handle_broadcast(&format!("{}/{}/transaction/broadcast", endpoint, network), &transaction, &program_name)?;
352        match response.status() {
353            200 => println!(
354                "‚úÖ Successfully broadcast execution with:\n  - transaction ID: '{}'\n  - fee ID: '{}'",
355                transaction.id().to_string().bold().yellow(),
356                fee.id().to_string().bold().yellow()
357            ),
358            _ => {
359                let error_message =
360                    response.into_string().map_err(|e| CliError::custom(format!("Failed to read response: {e}")))?;
361                println!("‚ùå Failed to broadcast execution: {}", error_message);
362            }
363        }
364    }
365
366    Ok(())
367}

401    println!("  {:16}{}", "Fee Record:", if fee_record { "yes" } else { "no (public fee)" });
402
403    println!("\n{}", "‚öôÔ∏è Actions:".bold());
404    if !is_local {
405        println!("  - Program and its dependencies will be downloaded from the network.");
406    }
407    if action.print {
408        println!("  - Transaction will be printed to the console.");
409    } else {
410        println!("  - Transaction will NOT be printed to the console.");
411    }
412    if let Some(path) = &action.save {
413        println!("  - Transaction will be saved to {}", path.bold());
414    } else {
415        println!("  - Transaction will NOT be saved to a file.");
416    }
417    if action.broadcast {
418        println!("  - Transaction will be broadcast to {}", endpoint.bold());
419    } else {
420        println!("  - Transaction will NOT be broadcast to the network.");
421    }
422    println!("{}", "‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ\n".dimmed());
423}

425/// Pretty‚Äëprint execution statistics without a table, using the same UI
426/// conventions as `print_deployment_plan`.
427fn print_execution_stats<N: Network>(
428    vm: &VM<N, ConsensusMemory<N>>,
429    program_name: &str,
430    execution: &Execution<N>,
431    priority_fee: Option<u64>,
432    consensus_version: ConsensusVersion,
433) -> Result<()> {
434    use colored::*;
435
436    // ‚îÄ‚îÄ Gather cost components ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ
437    let (base_fee, (storage_cost, execution_cost)) = if consensus_version == ConsensusVersion::V1 {
438        execution_cost_v1(&vm.process().read(), execution)?
439    } else {
440        execution_cost_v2(&vm.process().read(), execution)?
441    };
442
443    let base_cr = base_fee as f64 / 1_000_000.0;
444    let prio_cr = priority_fee.unwrap_or(0) as f64 / 1_000_000.0;
445    let total_cr = base_cr + prio_cr;
446
447    // ‚îÄ‚îÄ Header ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ
448    println!("\n{} {}", "üìä Execution Summary for".bold(), program_name.bold());
449    println!("{}", "‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ".dimmed());
450
451    // ‚îÄ‚îÄ Cost breakdown ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ
452    println!("{}", "üí∞ Cost Breakdown (credits)".bold());
453    println!("  {:22}{}{:.6}", "Transaction Storage:".cyan(), "".yellow(), storage_cost as f64 / 1_000_000.0);
454    println!("  {:22}{}{:.6}", "On‚Äëchain Execution:".cyan(), "".yellow(), execution_cost as f64 / 1_000_000.0);
455    println!("  {:22}{}{:.6}", "Priority Fee:".cyan(), "".yellow(), prio_cr);
456    println!("  {:22}{}{:.6}", "Total Fee:".cyan(), "".yellow(), total_cr);
457
458    // ‚îÄ‚îÄ Footer rule ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ
459    println!("{}", "‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ".dimmed());
460    Ok(())
