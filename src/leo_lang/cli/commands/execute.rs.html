54/// Build, Prove and Run Leo program with inputs
55#[derive(Parser, Debug)]
56pub struct LeoExecute {
57    #[clap(name = "NAME", help = "The name of the function to execute.", default_value = "main")]
58    name: String,
59    #[clap(name = "INPUTS", help = "The inputs to the program.")]
60    inputs: Vec<String>,
61    #[clap(short, long, help = "Execute the transition on-chain.", default_value = "false")]
62    broadcast: bool,
63    #[clap(short, long, help = "Execute the local program on-chain.", default_value = "false")]
64    local: bool,
65    #[clap(short, long, help = "The program to execute on-chain.")]
66    program: Option<String>,
67    #[clap(flatten)]
68    fee_options: FeeOptions,
69    #[clap(flatten)]
70    compiler_options: BuildOptions,
71    #[arg(short, long, help = "The inputs to the program, from a file. Overrides the INPUTS argument.")]
72    file: Option<String>,
73    #[clap(long, help = "Disables building of the project before execution.", default_value = "false")]
74    pub(crate) no_build: bool,
75}
76
77impl Command for LeoExecute {
78    type Input = <LeoBuild as Command>::Output;
79    type Output = ();
80
81    fn log_span(&self) -> Span {
82        tracing::span!(tracing::Level::INFO, "Leo")
83    }
84
85    fn prelude(&self, context: Context) -> Result<Self::Input> {
86        // No need to build if we are executing an external program.
87        if self.program.is_some() || self.no_build {
88            return Ok(());
89        }
90        (LeoBuild { options: self.compiler_options.clone() }).execute(context)
91    }
92
93    fn apply(self, context: Context, _input: Self::Input) -> Result<Self::Output> {
94        // Parse the network.
95        let network = NetworkName::try_from(context.get_network(&self.compiler_options.network)?)?;
96        let endpoint = context.get_endpoint(&self.compiler_options.endpoint)?;
97        match network {
98            NetworkName::TestnetV0 => handle_execute::<AleoTestnetV0>(self, context, network, &endpoint),
99            NetworkName::MainnetV0 => {
100                #[cfg(feature = "only_testnet")]
101                panic!("Mainnet chosen with only_testnet feature");
102                #[cfg(not(feature = "only_testnet"))]
103                return handle_execute::<AleoV0>(self, context, network, &endpoint);
104            }
105            NetworkName::CanaryV0 => {
106                #[cfg(feature = "only_testnet")]
107                panic!("Canary chosen with only_testnet feature");
108                #[cfg(not(feature = "only_testnet"))]
109                return handle_execute::<AleoCanaryV0>(self, context, network, &endpoint);
110            }
111        }
112    }
113}

115// A helper function to handle the `execute` command.
116fn handle_execute<A: Aleo>(
117    command: LeoExecute,
118    context: Context,
119    network: NetworkName,
120    endpoint: &str,
121) -> Result<<LeoExecute as Command>::Output> {
122    // If input values are provided, then run the program with those inputs.
123    // Otherwise, use the input file.
124    let mut inputs = command.inputs.clone();
125
126    // Add the inputs to the arguments.
127    if let Some(file) = command.file.clone() {
128        // Get the contents from the file.
129        let path = context.dir()?.join(file);
130        let raw_content =
131            std::fs::read_to_string(&path).map_err(|err| PackageError::failed_to_read_file(path.display(), err))?;
132        // Parse the values from the file.
133        let mut content = raw_content.as_str();
134        let mut values = vec![];
135        while let Ok((remaining, value)) = snarkvm::prelude::Value::<A::Network>::parse(content) {
136            content = remaining;
137            values.push(value);
138        }
139        // Check that the remaining content is empty.
140        if !content.trim().is_empty() {
141            return Err(PackageError::failed_to_read_input_file(path.display()).into());
142        }
143        // Convert the values to strings.
144        let mut inputs_from_file = values.into_iter().map(|value| value.to_string()).collect::<Vec<String>>();
145        // Add the inputs from the file to the arguments.
146        inputs.append(&mut inputs_from_file);
147    }
148
149    // Initialize an RNG.
150    let rng = &mut rand::thread_rng();
151
152    // Get the private key.
153    let private_key = match command.fee_options.private_key.clone() {
154        Some(key) => PrivateKey::from_str(&key)?,
155        None => PrivateKey::from_str(
156            &dotenv_private_key().map_err(CliError::failed_to_read_environment_private_key)?.to_string(),
157        )?,
158    };
159    let address = Address::try_from(&private_key)?;
160
161    // If the `broadcast` flag is set, then broadcast the transaction.
162    if command.broadcast {
163        // Get the program name.
164        let program_name = match (command.program.clone(), command.local) {
165            (Some(name), true) => {
166                let local = context.open_manifest::<A::Network>()?.program_id().name().to_string();
167                // Throw error if local name doesn't match the specified name.
168                if name == local {
169                    local
170                } else {
171                    return Err(PackageError::conflicting_on_chain_program_name(local, name).into());
172                }
173            }
174            (Some(name), false) => name.clone(),
175            (None, true) => context.open_manifest::<A::Network>()?.program_id().name().to_string(),
176            (None, false) => return Err(PackageError::missing_on_chain_program_name().into()),
177        };
178
179        // Specify the query
180        let query = SnarkVMQuery::<A::Network, BlockMemory<A::Network>>::from(endpoint);
181
182        // Initialize the storage.
183        let store = ConsensusStore::<A::Network, ConsensusMemory<A::Network>>::open(StorageMode::Production)?;
184
185        // Initialize the VM.
186        let vm = VM::from(store)?;
187
188        // Remove the `.aleo` extension from the program name, if it exists.
189        let program_name = match program_name.strip_suffix(".aleo") {
190            Some(name) => name.to_string(),
191            None => program_name,
192        };
193        // Load the main program, and all of its imports.
194        let program_id = &ProgramID::<A::Network>::from_str(&format!("{program_name}.aleo"))?;
195        load_program_from_network(context.clone(), &mut vm.process().write(), program_id, network, endpoint)?;
196
197        // Compute the authorization.
198        let authorization = vm.authorize(&private_key, program_id, &command.name, inputs, rng)?;
199        // Determine if a fee is required.
200        let is_fee_required = !authorization.is_split();
201        // Determine if a priority fee is declared.
202        let is_priority_fee_declared = command.fee_options.priority_fee > 0;
203        // Compute the execution.
204        let execution = match vm.execute_authorization(authorization, None, Some(query.clone()), rng)? {
205            Transaction::Execute(_, _, execution, _) => execution,
206            _ => unreachable!("VM::execute_authorization should return a Transaction::Execute"),
207        };
208
209        let fee_record = if let Some(record) = command.fee_options.record {
210            Some(parse_record(&private_key, &record)?)
211        } else {
212            None
213        };
214
215        // Check the transaction cost.
216        let base_fee = match command.fee_options.base_fee {
217            Some(base_fee) => base_fee,
218            None => {
219                let (base_fee, (storage_cost, finalize_cost)) =
220                    // Attempt to get the height of the latest block to determine which version of the execution cost to use.
221                    if let Ok(height) = get_latest_block_height(endpoint, &network.to_string(), &context) {
222                        if height < A::Network::CONSENSUS_HEIGHT(ConsensusVersion::V2).unwrap() {
223                            execution_cost_v1(&vm.process().read(), &execution)?
224                        } else {
225                            execution_cost_v2(&vm.process().read(), &execution)?
226                        }
227                    }
228                    // Otherwise, default to the one provided in `fee_options`.
229                    else {
230                        // Get the consensus version from the command.
231                        let version = match command.fee_options.consensus_version {
232                            Some(1) => 1,
233                            None | Some(2) => 2,
234                            Some(version) => {
235                                panic!("Invalid consensus version: {version}. Please specify a valid version.")
236                            }
237                        };
238                        // Print a warning message.
239                        println!("Failed to get the latest block height. Defaulting to V{version}.",);
240                        // Use the provided version.
241                        match version {
242                            1 => execution_cost_v1(&vm.process().read(), &execution)?,
243                            2 => execution_cost_v2(&vm.process().read(), &execution)?,
244                            _ => unreachable!(),
245                        }
246                    };
247
248                // Print the cost breakdown.
249                execution_cost_breakdown(
250                    &program_name,
251                    base_fee as f64 / 1_000_000.0,
252                    storage_cost as f64 / 1_000_000.0,
253                    finalize_cost as f64 / 1_000_000.0,
254                    command.fee_options.priority_fee as f64 / 1_000_000.0,
255                )?;
256                base_fee
257            }
258        };
259
260        // Check if the public balance is sufficient.
261        if fee_record.is_none() {
262            check_balance::<A::Network>(
263                &private_key,
264                endpoint,
265                &network.to_string(),
266                &context,
267                base_fee + command.fee_options.priority_fee,
268            )?;
269        }
270
271        // Compute the fee.
272        let fee = match is_fee_required || is_priority_fee_declared {
273            true => {
274                // Compute the execution ID.
275                let execution_id = execution.to_execution_id()?;
276                // Authorize the fee.
277                let authorization = match fee_record {
278                    Some(record) => vm.authorize_fee_private(
279                        &private_key,
280                        record,
281                        base_fee,
282                        command.fee_options.priority_fee,
283                        execution_id,
284                        rng,
285                    )?,
286                    None => vm.authorize_fee_public(
287                        &private_key,
288                        base_fee,
289                        command.fee_options.priority_fee,
290                        execution_id,
291                        rng,
292                    )?,
293                };
294                // Execute the fee.
295                Some(vm.execute_fee_authorization(authorization, Some(query), rng)?)
296            }
297            false => None,
298        };
299        // Return the execute transaction.
300        let transaction = Transaction::from_execution(*execution, fee)?;
301
302        // Broadcast the execution transaction.
303        if !command.fee_options.dry_run {
304            if !command.fee_options.yes {
305                let prompt = format!(
306                    "Do you want to submit execution of function `{}` on program `{program_name}.aleo` to network {} via endpoint {} using address {}?",
307                    &command.name, network, endpoint, address
308                );
309                // Ask the user for confirmation of the transaction.
310                let confirmation =
311                    Confirm::with_theme(&ColorfulTheme::default()).with_prompt(prompt).default(false).interact();
312
313                // Check if the user confirmed the transaction.
314                if let Ok(confirmation) = confirmation {
315                    if !confirmation {
316                        println!("✅ Successfully aborted the execution transaction for '{}'\n", program_name.bold());
317                        return Ok(());
318                    }
319                } else {
320                    return Err(CliError::confirmation_failed().into());
321                }
322            }
323            println!("✅ Created execution transaction for '{}'\n", program_id.to_string().bold());
324            handle_broadcast(&format!("{}/{}/transaction/broadcast", endpoint, network), transaction, &program_name)?;
325        } else {
326            println!("✅ Successful dry run execution for '{}'\n", program_id.to_string().bold());
327        }
328
329        return Ok(());
330    }
331
332    // Open the Leo build/ directory.
333    let path = context.dir()?.join("build/");
334
335    // Unset the Leo panic hook.
336    let _ = std::panic::take_hook();
337
338    // Conduct the execution locally (code lifted from snarkVM).
339    // Load the package.
340    let package = SnarkVMPackage::open(&path)?;
341    // Convert the inputs.
342    let mut parsed_inputs: Vec<Value<A::Network>> = Vec::new();
343    for input in inputs.iter() {
344        let value = Value::from_str(input)?;
345        parsed_inputs.push(value);
346    }
347    // Execute the request.
348    let (response, execution, metrics) = package
349        .execute::<A, _>(
350            endpoint.to_string(),
351            &private_key,
352            Identifier::try_from(command.name.clone())?,
353            &parsed_inputs,
354            rng,
355        )
356        .map_err(PackageError::execution_error)?;
357
358    let fee = None;
359
360    // Construct the transaction.
361    let transaction = Transaction::from_execution(execution, fee)?;
362
363    // Log the metrics.
364    use num_format::ToFormattedString;
365
366    // Count the number of times a function is called.
367    let mut program_frequency = HashMap::<String, usize>::new();
368    for metric in metrics.iter() {
369        // Prepare the function name string.
370        let function_name_string = format!("'{}/{}'", metric.program_id, metric.function_name).bold();
371
372        // Prepare the function constraints string
373        let function_constraints_string = format!(
374            "{function_name_string} - {} constraints",
375            metric.num_function_constraints.to_formatted_string(LOCALE)
376        );
377
378        // Increment the counter for the function call.
379        match program_frequency.get_mut(&function_constraints_string) {
380            Some(counter) => *counter += 1,
381            None => {
382                let _ = program_frequency.insert(function_constraints_string, 1);
383            }
384        }
385    }
386
387    println!("⛓  Constraints\n");
388    for (function_constraints, counter) in program_frequency {
389        // Log the constraints
390        let counter_string = match counter {
391            1 => "(called 1 time)".to_string().dimmed(),
392            counter => format!("(called {counter} times)").dimmed(),
393        };
394
395        println!(" •  {function_constraints} {counter_string}",)
396    }
397
398    // Log the outputs.
399    match response.outputs().len() {
400        0 => (),
401        1 => println!("\n➡️  Output\n"),
402        _ => println!("\n➡️  Outputs\n"),
403    };
404    for output in response.outputs() {
405        println!(" • {output}");
406    }
407    println!();
408
409    // Print the transaction.
410    println!("{transaction}\n");
411
412    // Prepare the locator.
413    let locator = Locator::<A::Network>::from_str(&format!("{}/{}", package.program_id(), command.name))?;
414    // Prepare the path string.
415    let path_string = format!("(in \"{}\")", path.display());
416
417    println!("✅ Executed '{}' {}", locator.to_string().bold(), path_string.dimmed());
418    Ok(())
419}

421/// A helper function to recursively load the program and all of its imports into the process. Lifted from snarkOS.
422fn load_program_from_network<N: Network>(
423    context: Context,
424    process: &mut Process<N>,
425    program_id: &ProgramID<N>,
426    network: NetworkName,
427    endpoint: &str,
428) -> Result<()> {
429    // Fetch the program.
430    let program_src = LeoQuery {
431        endpoint: Some(endpoint.to_string()),
432        network: Some(network.to_string()),
433        command: QueryCommands::Program {
434            command: query::LeoProgram { name: program_id.to_string(), mappings: false, mapping_value: None },
435        },
436    }
437    .execute(Context::new(context.path.clone(), context.home.clone(), true)?)?;
438    let program = SnarkVMProgram::<N>::from_str(&program_src)?;
439
440    // Return early if the program is already loaded.
441    if process.contains_program(program.id()) {
442        return Ok(());
443    }
444
445    // Iterate through the program imports.
446    for import_program_id in program.imports().keys() {
447        // Add the imports to the process if does not exist yet.
448        if !process.contains_program(import_program_id) {
449            // Recursively load the program and its imports.
450            load_program_from_network(context.clone(), process, import_program_id, network, endpoint)?;
451        }
452    }
453
454    // Add the program to the process if it does not already exist.
455    if !process.contains_program(program.id()) {
456        process.add_program(&program)?;
457    }
458
459    Ok(())
460}

462// A helper function to display a cost breakdown of the execution.
463fn execution_cost_breakdown(
464    name: &str,
465    base_fee: f64,
466    storage_cost: f64,
467    finalize_cost: f64,
468    priority_fee: f64,
469) -> Result<()> {
470    println!("\nBase execution cost for '{}' is {} credits.\n", name.bold(), base_fee);
471    // Display the cost breakdown in a table.
472    let data = [
473        [name, "Cost (credits)"],
474        ["Transaction Storage", &format!("{:.6}", storage_cost)],
475        ["On-chain Execution", &format!("{:.6}", finalize_cost)],
476        ["Priority Fee", &format!("{:.6}", priority_fee)],
477        ["Total", &format!("{:.6}", base_fee + priority_fee)],
478    ];
479    let mut out = Vec::new();
480    text_tables::render(&mut out, data).map_err(CliError::table_render_failed)?;
481    println!("{}", std::str::from_utf8(&out).map_err(CliError::table_render_failed)?);
482    Ok(())
