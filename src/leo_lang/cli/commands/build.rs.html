39impl From<BuildOptions> for CompilerOptions {
40    fn from(options: BuildOptions) -> Self {
41        Self {
42            build: leo_compiler::BuildOptions {
43                dce_enabled: options.enable_dce,
44                conditional_block_max_depth: options.conditional_block_max_depth,
45                disable_conditional_branch_type_checking: options.disable_conditional_branch_type_checking,
46            },
47            output: OutputOptions {
48                ast_spans_enabled: options.enable_ast_spans,
49                ast_snapshots: if options.enable_all_ast_snapshots {
50                    AstSnapshots::All
51                } else {
52                    AstSnapshots::Some(options.ast_snapshots.into_iter().collect())
53                },
54                initial_ast: options.enable_all_ast_snapshots | options.enable_initial_ast_snapshot,
55            },
56        }
57    }
58}

60/// Compile and build program command.
61#[derive(Parser, Debug)]
62pub struct LeoBuild {
63    #[clap(flatten)]
64    pub(crate) options: BuildOptions,
65}
66
67impl Command for LeoBuild {
68    type Input = ();
69    type Output = ();
70
71    fn log_span(&self) -> Span {
72        tracing::span!(tracing::Level::INFO, "Leo")
73    }
74
75    fn prelude(&self, _: Context) -> Result<Self::Input> {
76        Ok(())
77    }
78
79    fn apply(self, context: Context, _: Self::Input) -> Result<Self::Output> {
80        // Parse the network.
81        let network = NetworkName::try_from(context.get_network(&self.options.network)?)?;
82        match network {
83            NetworkName::MainnetV0 => handle_build::<MainnetV0>(&self, context),
84            NetworkName::TestnetV0 => handle_build::<TestnetV0>(&self, context),
85            NetworkName::CanaryV0 => handle_build::<CanaryV0>(&self, context),
86        }
87    }
88}

90// A helper function to handle the build command.
91fn handle_build<N: Network>(command: &LeoBuild, context: Context) -> Result<<LeoBuild as Command>::Output> {
92    // Get the package path.
93    let package_path = context.dir()?;
94    let home_path = context.home()?;
95
96    // Get the program id.
97    let manifest = Manifest::read_from_dir(&package_path)?;
98    let program_id = ProgramID::<N>::from_str(manifest.program())?;
99
100    // Clear and recreate the build directory.
101    let build_directory = package_path.join("build");
102    if build_directory.exists() {
103        std::fs::remove_dir_all(&build_directory).map_err(CliError::build_error)?;
104    }
105    Package::create(&build_directory, &program_id).map_err(CliError::build_error)?;
106
107    // Initialize error handler
108    let handler = Handler::default();
109
110    // Retrieve all local dependencies in post order
111    let main_sym = Symbol::intern(&program_id.name().to_string());
112    let mut retriever = Retriever::<N>::new(
113        main_sym,
114        &package_path,
115        &home_path,
116        context.get_endpoint(&command.options.endpoint)?.to_string(),
117    )
118    .map_err(|err| UtilError::failed_to_retrieve_dependencies(err, Default::default()))?;
119    let mut local_dependencies =
120        retriever.retrieve().map_err(|err| UtilError::failed_to_retrieve_dependencies(err, Default::default()))?;
121
122    // Push the main program at the end of the list to be compiled after all of its dependencies have been processed
123    local_dependencies.push(main_sym);
124
125    // Recursive build will recursively compile all local dependencies. Can disable to save compile time.
126    let recursive_build = !command.options.non_recursive;
127
128    // Loop through all local dependencies and compile them in order
129    for dependency in local_dependencies.into_iter() {
130        if recursive_build || dependency == main_sym {
131            // Get path to the local project
132            let (local_path, stubs) = retriever.prepare_local(dependency)?;
133
134            // Create the outputs directory.
135            let local_outputs_directory = OutputsDirectory::create(&local_path)?;
136
137            // Open the build directory.
138            let local_build_directory = BuildDirectory::create(&local_path)?;
139
140            // Fetch paths to all .leo files in the source directory.
141            let local_source_files = SourceDirectory::files(&local_path)?;
142
143            // Check the source files.
144            SourceDirectory::check_files(&local_source_files)?;
145
146            // Compile all .leo files into .aleo files.
147            for file_path in local_source_files {
148                compile_leo_file(
149                    file_path,
150                    &ProgramID::<N>::try_from(format!("{}.aleo", dependency))
151                        .map_err(|_| UtilError::snarkvm_error_building_program_id(Default::default()))?,
152                    &local_outputs_directory,
153                    &local_build_directory,
154                    &handler,
155                    command.options.clone(),
156                    stubs.clone(),
157                )?;
158            }
159        }
160
161        // Writes `leo.lock` as well as caches objects (when target is an intermediate dependency)
162        retriever.process_local(dependency, recursive_build)?;
163    }
164
165    // `Package::open` checks that the build directory and that `main.aleo` and all imported files are well-formed.
166    Package::<N>::open(&build_directory).map_err(CliError::failed_to_execute_build)?;
167
168    Ok(())
169}

171/// Compiles a Leo file in the `src/` directory.
172#[allow(clippy::too_many_arguments)]
173fn compile_leo_file<N: Network>(
174    file_path: PathBuf,
175    program_id: &ProgramID<N>,
176    outputs: &Path,
177    build: &Path,
178    handler: &Handler,
179    options: BuildOptions,
180    stubs: IndexMap<Symbol, Stub>,
181) -> Result<()> {
182    // Construct program name from the program_id found in `package.json`.
183    let program_name = program_id.name().to_string();
184
185    // Create the path to the Aleo file.
186    let mut aleo_file_path = build.to_path_buf();
187    aleo_file_path.push(format!("main.{}", program_id.network()));
188
189    let enable_dce = options.enable_dce;
190
191    // Create a new instance of the Leo compiler.
192    let mut compiler = Compiler::<N>::new(handler.clone(), outputs.to_path_buf(), Some(options.into()), stubs);
193
194    // Compile the Leo program into Aleo instructions.
195    let instructions = compiler.compile_from_file(file_path)?;
196
197    // Write the instructions.
198    std::fs::File::create(&aleo_file_path)
199        .map_err(CliError::failed_to_load_instructions)?
200        .write_all(instructions.as_bytes())
201        .map_err(CliError::failed_to_load_instructions)?;
202
203    if enable_dce {
204        tracing::info!("    {} statements before dead code elimination.", compiler.statements_before_dce);
205        tracing::info!("    {} statements after dead code elimination.", compiler.statements_after_dce);
206    }
207    tracing::info!("âœ… Compiled '{program_name}.aleo' into Aleo instructions");
208    Ok(())
209}
