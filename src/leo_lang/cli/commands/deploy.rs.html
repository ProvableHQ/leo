44/// Deploys an Aleo program.
45#[derive(Parser, Debug)]
46pub struct LeoDeploy {
47    #[clap(flatten)]
48    pub(crate) fee_options: FeeOptions,
49    #[clap(flatten)]
50    pub(crate) action: TransactionAction,
51    #[clap(flatten)]
52    pub(crate) env_override: EnvOptions,
53    #[clap(flatten)]
54    pub(crate) extra: ExtraOptions,
55    #[clap(long, help = "Skips deployment of any program that contains one of the given substrings.")]
56    pub(crate) skip: Vec<String>,
57    #[clap(flatten)]
58    pub(crate) build_options: BuildOptions,
59}
60
61impl Command for LeoDeploy {
62    type Input = Package;
63    type Output = ();
64
65    fn log_span(&self) -> Span {
66        tracing::span!(tracing::Level::INFO, "Leo")
67    }
68
69    fn prelude(&self, context: Context) -> Result<Self::Input> {
70        LeoBuild {
71            env_override: self.env_override.clone(),
72            options: {
73                let mut options = self.build_options.clone();
74                options.no_cache = true;
75                options
76            },
77        }
78        .execute(context)
79    }
80
81    fn apply(self, context: Context, input: Self::Input) -> Result<Self::Output> {
82        // Get the network, accounting for overrides.
83        let network = context.get_network(&self.env_override.network)?.parse()?;
84        // Handle each network with the appropriate parameterization.
85        match network {
86            NetworkName::TestnetV0 => handle_deploy::<TestnetV0>(&self, context, network, input),
87            NetworkName::MainnetV0 => {
88                #[cfg(feature = "only_testnet")]
89                panic!("Mainnet chosen with only_testnet feature");
90                #[cfg(not(feature = "only_testnet"))]
91                return handle_deploy::<MainnetV0>(&self, context, network, input);
92            }
93            NetworkName::CanaryV0 => {
94                #[cfg(feature = "only_testnet")]
95                panic!("Canary chosen with only_testnet feature");
96                #[cfg(not(feature = "only_testnet"))]
97                return handle_deploy::<CanaryV0>(&self, context, network, input);
98            }
99        }
100    }
101}

103// A helper function to handle deployment logic.
104fn handle_deploy<N: Network>(
105    command: &LeoDeploy,
106    context: Context,
107    network: NetworkName,
108    package: Package,
109) -> Result<<LeoDeploy as Command>::Output> {
110    // Get the private key and associated address, accounting for overrides.
111    let private_key = context.get_private_key(&command.env_override.private_key)?;
112    let address =
113        Address::try_from(&private_key).map_err(|e| CliError::custom(format!("Failed to parse address: {e}")))?;
114
115    // Get the endpoint, accounting for overrides.
116    let endpoint = context.get_endpoint(&command.env_override.endpoint)?;
117
118    // Get the programs and optional manifests for all the programs.
119    let programs_and_manifests = package
120        .get_programs_and_manifests(context.home()?)?
121        .into_iter()
122        .map(|(program_name, program_string, manifest)| {
123            // Parse the program ID from the program name.
124            let program_id = ProgramID::<N>::from_str(&format!("{}.aleo", program_name))
125                .map_err(|e| CliError::custom(format!("Failed to parse program ID: {e}")))?;
126            // Parse the program bytecode.
127            let bytecode = Program::<N>::from_str(&program_string)
128                .map_err(|e| CliError::custom(format!("Failed to parse program: {e}")))?;
129            Ok((program_id, bytecode, manifest))
130        })
131        .collect::<Result<Vec<_>>>()?;
132
133    // Parse the fee options.
134    let fee_options = parse_fee_options(&private_key, &command.fee_options, programs_and_manifests.len())?;
135
136    // Zip up the programs and manifests with the fee options.
137    let tasks = programs_and_manifests
138        .into_iter()
139        .zip(fee_options)
140        .map(|((program, data, manifest), (base_fee, priority_fee, record))| {
141            (program, data, manifest, base_fee, priority_fee, record)
142        })
143        .collect::<Vec<_>>();
144
145    // Split the tasks into local and remote dependencies.
146    let (local, remote) = tasks.into_iter().partition::<Vec<_>, _>(|(_, _, manifest, _, _, _)| manifest.is_some());
147
148    // Split the local tasks into those that should be skipped and those that should not.
149    let (skipped, tasks): (Vec<_>, Vec<_>) = local
150        .into_iter()
151        .partition(|(program_id, _, _, _, _, _)| command.skip.iter().any(|skip| program_id.to_string().contains(skip)));
152
153    // Get the consensus version.
154    let consensus_version = get_consensus_version::<N>(&command.extra.consensus_version, &endpoint, network, &context)?;
155
156    // Print a summary of the deployment plan.
157    print_deployment_plan(
158        &private_key,
159        &address,
160        &endpoint,
161        &network,
162        &tasks,
163        &skipped,
164        &remote,
165        &command.action,
166        consensus_version,
167    );
168
169    // Prompt the user to confirm the plan.
170    if !confirm("Do you want to proceed with deployment?", command.extra.yes)? {
171        println!("âŒ Deployment aborted.");
172        return Ok(());
173    }
174
175    // Initialize an RNG.
176    let rng = &mut rand::thread_rng();
177
178    // Initialize a new VM.
179    let vm = VM::from(ConsensusStore::<N, ConsensusMemory<N>>::open(StorageMode::Production)?)?;
180
181    // Specify the query
182    let query = SnarkVMQuery::from(&endpoint);
183
184    // For each of the programs, generate a deployment transaction.
185    let mut transactions = Vec::new();
186    for (program_id, program, manifest, _, priority_fee, fee_record) in tasks {
187        // If the program is a local dependency, generate a deployment transaction.
188        if manifest.is_some() {
189            println!("ğŸ“¦ Creating deployment transaction for '{}'...\n", program_id.to_string().bold());
190            // Generate the transaction.
191            let transaction = vm
192                .deploy(&private_key, &program, fee_record, priority_fee.unwrap_or(0), Some(query.clone()), rng)
193                .map_err(|e| CliError::custom(format!("Failed to generate deployment transaction: {e}")))?;
194            // Get the deployment.
195            let deployment = transaction.deployment().expect("Expected a deployment in the transaction");
196            // Print the deployment stats.
197            print_deployment_stats(&program_id.to_string(), deployment, priority_fee)?;
198            // Check if the number of variables and constraints are within the limits.
199            if deployment.num_combined_variables()? > N::MAX_DEPLOYMENT_VARIABLES {
200                return Err(CliError::variable_limit_exceeded(program_id, N::MAX_DEPLOYMENT_VARIABLES, network).into());
201            }
202            if deployment.num_combined_constraints()? > N::MAX_DEPLOYMENT_CONSTRAINTS {
203                return Err(
204                    CliError::constraint_limit_exceeded(program_id, N::MAX_DEPLOYMENT_CONSTRAINTS, network).into()
205                );
206            }
207            // Save the transaction.
208            transactions.push((program_id, transaction));
209        }
210        // Add the program to the VM.
211        vm.process().write().add_program(&program)?;
212    }
213
214    // If the `print` option is set, print the deployment transaction to the console.
215    // The transaction is printed in JSON format.
216    if command.action.print {
217        for (program_name, transaction) in transactions.iter() {
218            // Pretty-print the transaction.
219            let transaction_json = serde_json::to_string_pretty(transaction)
220                .map_err(|e| CliError::custom(format!("Failed to serialize transaction: {e}")))?;
221            println!("ğŸ–¨ï¸ Printing deployment for {program_name}\n{transaction_json}")
222        }
223    }
224
225    // If the `save` option is set, save each deployment transaction to a file in the specified directory.
226    // The file format is `program_name.deployment.json`.
227    // The directory is created if it doesn't exist.
228    if let Some(path) = &command.action.save {
229        // Create the directory if it doesn't exist.
230        std::fs::create_dir_all(path).map_err(|e| CliError::custom(format!("Failed to create directory: {e}")))?;
231        for (program_name, transaction) in transactions.iter() {
232            // Save the transaction to a file.
233            let file_path = PathBuf::from(path).join(format!("{program_name}.deployment.json"));
234            println!("ğŸ’¾ Saving deployment for {program_name} at {}", file_path.display());
235            let transaction_json = serde_json::to_string_pretty(transaction)
236                .map_err(|e| CliError::custom(format!("Failed to serialize transaction: {e}")))?;
237            std::fs::write(file_path, transaction_json)
238                .map_err(|e| CliError::custom(format!("Failed to write transaction to file: {e}")))?;
239        }
240    }
241
242    // If the `broadcast` option is set, broadcast each deployment transaction to the network.
243    if command.action.broadcast {
244        for (i, (program_id, transaction)) in transactions.iter().enumerate() {
245            println!("ğŸ“¡ Broadcasting deployment for {program_id}...");
246            // Get and confirm the fee with the user.
247            let fee = transaction.fee_transition().expect("Expected a fee in the transaction");
248            if !confirm_fee(&fee, &private_key, &address, &endpoint, network, &context, command.extra.yes)? {
249                println!("âŒ Deployment aborted.");
250                return Ok(());
251            }
252            let fee_id = fee.id().to_string();
253            let id = transaction.id().to_string();
254            let height_before = crate::cli::check_transaction::current_height(&endpoint, network)?;
255            // Broadcast the transaction to the network.
256            let response = handle_broadcast(
257                &format!("{}/{}/transaction/broadcast", endpoint, network),
258                transaction,
259                &program_id.to_string(),
260            )?;
261
262            let fail = |msg| {
263                println!("âŒ Failed to deploy program {program_id}: {msg}.");
264                let count = transactions.len() - i - 1;
265                match count {
266                    0 => {}
267                    1 => println!("1 expected deployment skipped"),
268                    _ => println!("{count} expected deployments skipped"),
269                }
270                Ok(())
271            };
272
273            match response.status() {
274                200 => {
275                    let status = crate::cli::check_transaction::check_transaction_with_message(
276                        &id,
277                        Some(&fee_id),
278                        &endpoint,
279                        network,
280                        height_before + 1,
281                        command.extra.max_wait,
282                        command.extra.blocks_to_check,
283                    )?;
284                    if status == Some(TransactionStatus::Accepted) {
285                        println!(
286                            "âœ… Successfully broadcast deployment with:\n  - transaction ID: '{}'\n  - fee ID: '{}'",
287                            id.bold().yellow(),
288                            fee_id.bold().yellow()
289                        );
290                    } else {
291                        return fail("Transaction apparently not accepted");
292                    }
293                }
294                _ => {
295                    let error_message = response
296                        .into_string()
297                        .map_err(|e| CliError::custom(format!("Failed to read response: {e}")))?;
298                    return fail(&error_message);
299                }
300            }
301        }
302    }
303
304    Ok(())
305}

307/// Check the tasks to warn the user about any potential issues.
308/// Only local programs are checked.
309/// The following properties are checked:
310/// - If the transaction is to be broadcast:
311///     - The program does not exist on the network.
312/// - The program's external dependencies are the latest version.
313fn check_tasks_for_warnings<N: Network>(
314    endpoint: &str,
315    network: NetworkName,
316    tasks: &[DeploymentTask<N>],
317    action: &TransactionAction,
318) -> Vec<String> {
319    let mut warnings = Vec::new();
320    for (program_id, _, manifest, _, _, _) in tasks {
321        if manifest.is_none() || !action.broadcast {
322            continue;
323        }
324        // Check if the program exists on the network.
325        if fetch_program_from_network(&program_id.to_string(), endpoint, network).is_ok() {
326            warnings.push(format!(
327                "The program '{}' already exists on the network. The deployment will likely fail.",
328                program_id
329            ));
330        }
331    }
332    warnings
333}

425/// Prettyâ€‘print deployment statistics without a table, using the same UI
426/// conventions as `print_deployment_plan`.
427fn print_deployment_stats<N: Network>(
428    program_id: &str,
429    deployment: &Deployment<N>,
430    priority_fee: Option<u64>,
431) -> Result<()> {
432    use colored::*;
433    use num_format::{Locale, ToFormattedString};
434
435    // â”€â”€ Collect statistics â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
436    let variables = deployment.num_combined_variables()?;
437    let constraints = deployment.num_combined_constraints()?;
438    let (base_fee, (storage_cost, synthesis_cost, namespace_cost)) = deployment_cost(deployment)?;
439
440    let base_fee_cr = base_fee as f64 / 1_000_000.0;
441    let prio_fee_cr = priority_fee.unwrap_or(0) as f64 / 1_000_000.0;
442    let total_fee_cr = base_fee_cr + prio_fee_cr;
443
444    // â”€â”€ Header â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
445    println!("\n{} {}", "ğŸ“Š Deployment Summary for".bold(), program_id.bold());
446    println!("{}", "â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€".dimmed());
447
448    // â”€â”€ Highâ€‘level metrics â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
449    println!("  {:22}{}", "Total Variables:".cyan(), variables.to_formatted_string(&Locale::en).yellow());
450    println!("  {:22}{}", "Total Constraints:".cyan(), constraints.to_formatted_string(&Locale::en).yellow());
451
452    // â”€â”€ Cost breakdown â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
