23    /// Returns a [`Program`] AST if all tokens can be consumed and represent a valid Leo program.
24    pub fn parse_program(&mut self) -> Result<Program> {
25        let mut imports = IndexMap::new();
26        let mut program_scopes = IndexMap::new();
27
28        // TODO: Remove restrictions on multiple program scopes
29        let mut parsed_program_scope = false;
30
31        while self.has_next() {
32            match &self.token.token {
33                Token::Import => {
34                    let (id, import) = self.parse_import()?;
35                    imports.insert(id, import);
36                }
37                Token::Program => {
38                    match parsed_program_scope {
39                        // Only one program scope is allowed per file.
40                        true => {
41                            return Err(ParserError::only_one_program_scope_is_allowed(self.token.span).into());
42                        }
43                        false => {
44                            parsed_program_scope = true;
45                            let program_scope = self.parse_program_scope()?;
46                            program_scopes.insert(program_scope.program_id.name.name, program_scope);
47                        }
48                    }
49                }
50                _ => return Err(Self::unexpected_item(&self.token, &[Token::Import, Token::Program]).into()),
51            }
52        }
53
54        // Requires that at least one program scope is present.
55        if !parsed_program_scope {
56            return Err(ParserError::missing_program_scope(self.token.span).into());
57        }
58
59        Ok(Program { imports, stubs: IndexMap::new(), program_scopes })
60    }

62    fn unexpected_item(token: &SpannedToken, expected: &[Token]) -> ParserError {
63        ParserError::unexpected(
64            &token.token,
65            expected.iter().map(|x| format!("'{x}'")).collect::<Vec<_>>().join(", "),
66            token.span,
67        )
68    }

71    /// Parses an import statement `import foo.leo;`.
72    pub(super) fn parse_import(&mut self) -> Result<(Symbol, (Program, Span))> {
73        // Parse `import`.
74        let start = self.expect(&Token::Import)?;
75
76        // Parse `foo`.
77        let import_name = self.expect_identifier()?;
78
79        // Parse `.`.
80        self.expect(&Token::Dot)?;
81
82        // Parse network, which currently must be `aleo`.
83        if !self.eat(&Token::Aleo) {
84            // Throw error for non-aleo networks.
85            return Err(ParserError::invalid_network(self.token.span).into());
86        }
87
88        let end = self.expect(&Token::Semicolon)?;
89
90        // Return the import name and the span.
91        Ok((import_name.name, (Program::default(), start + end)))
92    }

94    /// Parses a program scope `program foo.aleo { ... }`.
95    fn parse_program_scope(&mut self) -> Result<ProgramScope> {
96        // Parse `program` keyword.
97        let start = self.expect(&Token::Program)?;
98
99        // Parse the program name.
100        let name = self.expect_identifier()?;
101
102        // Set the program name in the context.
103        self.program_name = Some(name.name);
104
105        // Parse the `.`.
106        self.expect(&Token::Dot)?;
107
108        // Parse the program network, which must be `aleo`, otherwise throw parser error.
109        let network_span = self.expect(&Token::Aleo).map_err(|_| ParserError::invalid_network(self.token.span))?;
110
111        // Construct the program id.
112        let program_id = ProgramId {
113            name,
114            network: Identifier { name: sym::aleo, id: self.node_builder.next_id(), span: network_span },
115        };
116
117        // Parse `{`.
118        self.expect(&Token::LeftCurly)?;
119
120        // Parse the body of the program scope.
121        let mut consts: Vec<(Symbol, ConstDeclaration)> = Vec::new();
122        let (mut transitions, mut functions) = (Vec::new(), Vec::new());
123        let mut structs: Vec<(Symbol, Composite)> = Vec::new();
124        let mut mappings: Vec<(Symbol, Mapping)> = Vec::new();
125
126        while self.has_next() {
127            match &self.token.token {
128                Token::Const => {
129                    let declaration = self.parse_const_declaration_statement()?;
130                    consts.push((Symbol::intern(&declaration.place.to_string()), declaration));
131                }
132                Token::Struct | Token::Record => {
133                    let (id, struct_) = self.parse_struct()?;
134                    structs.push((id, struct_));
135                }
136                Token::Mapping => {
137                    let (id, mapping) = self.parse_mapping()?;
138                    mappings.push((id, mapping));
139                }
140                Token::At | Token::Async | Token::Function | Token::Transition | Token::Inline | Token::Script => {
141                    let (id, function) = self.parse_function()?;
142
143                    // Partition into transitions and functions so that we don't have to sort later.
144                    if function.variant.is_transition() {
145                        transitions.push((id, function));
146                    } else {
147                        functions.push((id, function));
148                    }
149                }
150                Token::RightCurly => break,
151                _ => {
152                    return Err(Self::unexpected_item(&self.token, &[
153                        Token::Const,
154                        Token::Struct,
155                        Token::Record,
156                        Token::Mapping,
157                        Token::At,
158                        Token::Async,
159                        Token::Function,
160                        Token::Transition,
161                        Token::Inline,
162                        Token::Script,
163                        Token::RightCurly,
164                    ])
165                    .into());
166                }
167            }
168        }
169
170        // Parse `}`.
171        let end = self.expect(&Token::RightCurly)?;
172
173        Ok(ProgramScope {
174            program_id,
175            consts,
176            functions: [transitions, functions].concat(),
177            structs,
178            mappings,
179            span: start + end,
180        })
181    }

183    /// Returns a [`Vec<Member>`] AST node if the next tokens represent a struct member.
184    fn parse_struct_members(&mut self) -> Result<(Vec<Member>, Span)> {
185        let mut members = Vec::new();
186
187        while !self.check(&Token::RightCurly) {
188            let variable = self.parse_member_variable_declaration()?;
189
190            if !self.check(&Token::RightCurly) && !self.eat(&Token::Comma) {
191                self.emit_err(ParserError::comma_expected_after_member(self.token.span));
192            }
193
194            members.push(variable);
195        }
196        let span = self.expect(&Token::RightCurly)?;
197
198        Ok((members, span))
199    }

201    /// Parses `IDENT: TYPE`.
202    pub(super) fn parse_typed_ident(&mut self) -> Result<(Identifier, Type, Span)> {
203        let name = self.expect_identifier()?;
204        self.expect(&Token::Colon)?;
205        let (type_, span) = self.parse_type()?;
206
207        Ok((name, type_, name.span + span))
208    }

210    /// Returns a [`Member`] AST node if the next tokens represent a struct member variable.
211    fn parse_member_variable_declaration(&mut self) -> Result<Member> {
212        let mode = self.parse_mode()?;
213
214        let (identifier, type_, span) = self.parse_typed_ident()?;
215
216        Ok(Member { mode, identifier, type_, span, id: self.node_builder.next_id() })
217    }

219    /// Parses a struct or record definition, e.g., `struct Foo { ... }` or `record Foo { ... }`.
220    pub(super) fn parse_struct(&mut self) -> Result<(Symbol, Composite)> {
221        let is_record = matches!(&self.token.token, Token::Record);
222        let start = self.expect_any(&[Token::Struct, Token::Record])?;
223
224        // Check if using external type
225        let file_type = self.look_ahead(1, |t| &t.token);
226        if self.token.token == Token::Dot && (file_type == &Token::Aleo) {
227            return Err(ParserError::cannot_declare_external_struct(self.token.span).into());
228        }
229
230        let struct_name = self.expect_identifier()?;
231
232        self.expect(&Token::LeftCurly)?;
233        let (members, end) = self.parse_struct_members()?;
234
235        // Only provide a program name for records.
236        let external = if is_record { self.program_name } else { None };
237
238        Ok((struct_name.name, Composite {
239            identifier: struct_name,
240            members,
241            external,
242            is_record,
243            span: start + end,
244            id: self.node_builder.next_id(),
245        }))
246    }

248    /// Parses a mapping declaration, e.g. `mapping balances: address => u128`.
249    pub(super) fn parse_mapping(&mut self) -> Result<(Symbol, Mapping)> {
250        let start = self.expect(&Token::Mapping)?;
251        let identifier = self.expect_identifier()?;
252        self.expect(&Token::Colon)?;
253        let (key_type, _) = self.parse_type()?;
254        self.expect(&Token::BigArrow)?;
255        let (value_type, _) = self.parse_type()?;
256        let end = self.expect(&Token::Semicolon)?;
257        Ok((identifier.name, Mapping {
258            identifier,
259            key_type,
260            value_type,
261            span: start + end,
262            id: self.node_builder.next_id(),
263        }))
264    }

267    /// Returns a [`ParamMode`] AST node if the next tokens represent a function parameter mode.
268    pub(super) fn parse_mode(&mut self) -> Result<Mode> {
269        let private = self.eat(&Token::Private).then_some(self.prev_token.span);
270        let public = self.eat(&Token::Public).then_some(self.prev_token.span);
271        let constant = self.eat(&Token::Constant).then_some(self.prev_token.span);
272
273        match (private, public, constant) {
274            (None, None, None) => Ok(Mode::None),
275            (Some(_), None, None) => Ok(Mode::Private),
276            (None, Some(_), None) => Ok(Mode::Public),
277            (None, None, Some(_)) => Ok(Mode::Constant),
278            _ => {
279                let mut spans = [private, public, constant].into_iter().flatten();
280
281                // There must exist at least one mode, since the none case is handled above.
282                let starting_span = spans.next().unwrap();
283                // Sum the spans.
284                let summed_span = spans.fold(starting_span, |span, next| span + next);
285                // Emit an error.
286                Err(ParserError::inputs_multiple_variable_modes_specified(summed_span).into())
287            }
288        }
289    }

291    /// Returns an [`Input`] AST node if the next tokens represent a function input.
292    fn parse_input(&mut self) -> Result<Input> {
293        let mode = self.parse_mode()?;
294        let name = self.expect_identifier()?;
295        self.expect(&Token::Colon)?;
296
297        let (type_, type_span) = self.parse_type()?;
298        let span = name.span() + type_span;
299
300        Ok(Input { identifier: name, mode, type_, span, id: self.node_builder.next_id() })
301    }

303    /// Returns an [`Output`] AST node if the next tokens represent a function output.
304    fn parse_output(&mut self) -> Result<Output> {
305        let mode = self.parse_mode()?;
306        let (type_, span) = self.parse_type()?;
307        Ok(Output { mode, type_, span, id: self.node_builder.next_id() })
308    }

310    /// Returns an [`Annotation`] AST node if the next tokens represent an annotation.
311    fn parse_annotation(&mut self) -> Result<Annotation> {
312        // Parse the `@` symbol and identifier.
313        let start = self.expect(&Token::At)?;
314        let identifier = match self.token.token {
315            Token::Program => {
316                Identifier { name: sym::program, span: self.expect(&Token::Program)?, id: self.node_builder.next_id() }
317            }
318            _ => self.expect_identifier()?,
319        };
320        let mut span = start + identifier.span;
321
322        // TODO: Verify that this check is sound.
323        // Check that there is no whitespace or comments in between the `@` symbol and identifier.
324        if identifier.span.hi - start.lo > 1 + identifier.name.to_string().len() as u32 {
325            return Err(ParserError::space_in_annotation(span).into());
326        }
327
328        let mut map = IndexMap::new();
329
330        if self.eat(&Token::LeftParen) {
331            loop {
332                let key = self.expect_identifier()?;
333                self.expect(&Token::Assign)?;
334                if let Token::StaticString(s) = &self.token.token {
335                    map.insert(key.name, s.clone());
336                    self.bump();
337                } else {
338                    return Err(ParserError::unexpected(&self.token.token, "a string", self.token.span).into());
339                }
340
341                match &self.token.token {
342                    Token::Comma => {
343                        self.bump();
344                        if let Token::RightParen = &self.token.token {
345                            span = span + self.token.span;
346                            self.bump();
347                            break;
348                        }
349                    }
350                    Token::RightParen => {
351                        span = span + self.token.span;
352                        self.bump();
353                        break;
354                    }
355                    _ => return Err(ParserError::unexpected(&self.token.token, ", or )", self.token.span).into()),
356                }
357            }
358        }
359
360        Ok(Annotation { identifier, span, id: self.node_builder.next_id(), map })
361    }

363    /// Returns an [`(Identifier, Function)`] AST node if the next tokens represent a function name
364    /// and function definition.
365    fn parse_function(&mut self) -> Result<(Symbol, Function)> {
366        // TODO: Handle dangling annotations.
367        // Parse annotations, if they exist.
368        let mut annotations = Vec::new();
369        while self.look_ahead(0, |t| &t.token) == &Token::At {
370            annotations.push(self.parse_annotation()?)
371        }
372        // Parse a potential async signifier.
373        let (is_async, start_async) =
374            if self.token.token == Token::Async { (true, self.expect(&Token::Async)?) } else { (false, Span::dummy()) };
375        // Parse `<variant> IDENT`, where `<variant>` is `function`, `transition`, or `inline`.
376        let (variant, start) = match self.token.token.clone() {
377            Token::Inline => (Variant::Inline, self.expect(&Token::Inline)?),
378            Token::Script => (Variant::Script, self.expect(&Token::Script)?),
379            Token::Function => {
380                (if is_async { Variant::AsyncFunction } else { Variant::Function }, self.expect(&Token::Function)?)
381            }
382            Token::Transition => (
383                if is_async { Variant::AsyncTransition } else { Variant::Transition },
384                self.expect(&Token::Transition)?,
385            ),
386            _ => self.unexpected("'function', 'transition', or 'inline'")?,
387        };
388        let name = self.expect_identifier()?;
389
390        // Parse parameters.
391        let (inputs, ..) = self.parse_paren_comma_list(|p| p.parse_input().map(Some))?;
392
393        // Parse return type.
394        let output = match self.eat(&Token::Arrow) {
395            false => vec![],
396            true => {
397                self.disallow_struct_construction = true;
398                let output = match self.peek_is_left_par() {
399                    true => self.parse_paren_comma_list(|p| p.parse_output().map(Some))?.0,
400                    false => vec![self.parse_output()?],
401                };
402                self.disallow_struct_construction = false;
403                output
404            }
405        };
406
407        // Parse the function body, which must be a block,
408        // but we also allow no body and a semicolon instead (e.g. `fn foo(a:u8);`).
409        let (_has_empty_block, block) = match &self.token.token {
410            Token::LeftCurly => (false, self.parse_block()?),
411            Token::Semicolon => {
412                let semicolon = self.expect(&Token::Semicolon)?;
413                // TODO: make a distinction between empty block and no block (i.e. semicolon)
414                (true, Block { statements: Vec::new(), span: semicolon, id: self.node_builder.next_id() })
415            }
416            _ => self.unexpected("block or semicolon")?,
417        };
418
419        let span = if start_async == Span::dummy() { start + block.span } else { start_async + block.span };
420
421        Ok((
422            name.name,
423            Function::new(annotations, variant, name, inputs, output, block, span, self.node_builder.next_id()),
424        ))
425    }
