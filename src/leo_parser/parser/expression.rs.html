43    /// Returns an [`Expression`] AST node if the next token is an expression.
44    /// Includes struct init expressions.
45    pub(crate) fn parse_expression(&mut self) -> Result<Expression> {
46        // Store current parser state.
47        let prior_fuzzy_state = self.disallow_struct_construction;
48
49        // Allow struct init expressions.
50        self.disallow_struct_construction = false;
51
52        // Parse expression.
53        let result = self.parse_conditional_expression();
54
55        // Restore prior parser state.
56        self.disallow_struct_construction = prior_fuzzy_state;
57
58        result
59    }

61    /// Returns an [`Expression`] AST node if the next tokens represent
62    /// a ternary expression. May or may not include struct init expressions.
63    ///
64    /// Otherwise, tries to parse the next token using [`parse_boolean_or_expression`].
65    pub(super) fn parse_conditional_expression(&mut self) -> Result<Expression> {
66        // Try to parse the next expression. Try BinaryOperation::Or.
67        let mut expr = self.parse_boolean_or_expression()?;
68
69        // Parse the rest of the ternary expression.
70        if self.eat(&Token::Question) {
71            let if_true = self.parse_expression()?;
72            self.expect(&Token::Colon)?;
73            let if_false = self.parse_expression()?;
74            expr = TernaryExpression {
75                span: expr.span() + if_false.span(),
76                condition: expr,
77                if_true,
78                if_false,
79                id: self.node_builder.next_id(),
80            }
81            .into();
82        }
83        Ok(expr)
84    }

86    /// Constructs a binary expression `left op right`.
87    fn bin_expr(node_builder: &NodeBuilder, left: Expression, right: Expression, op: BinaryOperation) -> Expression {
88        BinaryExpression { span: left.span() + right.span(), op, left, right, id: node_builder.next_id() }.into()
89    }

91    /// Parses a left-associative binary expression `<left> token <right>` using `f` for left/right.
92    /// The `token` is translated to `op` in the AST.
93    fn parse_bin_expr(
94        &mut self,
95        tokens: &[Token],
96        mut f: impl FnMut(&mut Self) -> Result<Expression>,
97    ) -> Result<Expression> {
98        let mut expr = f(self)?;
99        while let Some(op) = self.eat_bin_op(tokens) {
100            expr = Self::bin_expr(self.node_builder, expr, f(self)?, op);
101        }
102        Ok(expr)
103    }

121    fn eat_unary_op(&mut self) -> Option<UnaryOperation> {
122        self.eat_any(&[Token::Not, Token::Sub]).then(|| match &self.prev_token.token {
123            Token::Not => UnaryOperation::Not,
124            Token::Sub => UnaryOperation::Negate,
125            _ => panic!("Can't happen."),
126        })
127    }

129    /// Eats one of binary operators matching any in `tokens`.
130    fn eat_bin_op(&mut self, tokens: &[Token]) -> Option<BinaryOperation> {
131        self.eat_any(tokens).then(|| match &self.prev_token.token {
132            Token::Eq => BinaryOperation::Eq,
133            Token::NotEq => BinaryOperation::Neq,
134            Token::Lt => BinaryOperation::Lt,
135            Token::LtEq => BinaryOperation::Lte,
136            Token::Gt => BinaryOperation::Gt,
137            Token::GtEq => BinaryOperation::Gte,
138            Token::Add => BinaryOperation::Add,
139            Token::Sub => BinaryOperation::Sub,
140            Token::Mul => BinaryOperation::Mul,
141            Token::Div => BinaryOperation::Div,
142            Token::Rem => BinaryOperation::Rem,
143            Token::Or => BinaryOperation::Or,
144            Token::And => BinaryOperation::And,
145            Token::BitOr => BinaryOperation::BitwiseOr,
146            Token::BitAnd => BinaryOperation::BitwiseAnd,
147            Token::Pow => BinaryOperation::Pow,
148            Token::Shl => BinaryOperation::Shl,
149            Token::Shr => BinaryOperation::Shr,
150            Token::BitXor => BinaryOperation::Xor,
151            _ => unreachable!("`eat_bin_op` shouldn't produce this"),
152        })
153    }

155    /// Returns an [`Expression`] AST node if the next tokens represent a
156    /// binary relational expression: less than, less than or equals, greater than, greater than or equals.
157    ///
158    /// Otherwise, tries to parse the next token using [`parse_additive_expression`].
159    fn parse_ordering_expression(&mut self) -> Result<Expression> {
160        let mut expr = self.parse_bitwise_exclusive_or_expression()?;
161        if let Some(op) = self.eat_bin_op(&[Token::Lt, Token::LtEq, Token::Gt, Token::GtEq]) {
162            let right = self.parse_bitwise_exclusive_or_expression()?;
163            expr = Self::bin_expr(self.node_builder, expr, right, op);
164        }
165        Ok(expr)
166    }

168    /// Returns an [`Expression`] AST node if the next tokens represent a
169    /// binary equals or not equals expression.
170    ///
171    /// Otherwise, tries to parse the next token using [`parse_ordering_expression`].
172    fn parse_equality_expression(&mut self) -> Result<Expression> {
173        let mut expr = self.parse_ordering_expression()?;
174        if let Some(op) = self.eat_bin_op(&[Token::Eq, Token::NotEq]) {
175            let right = self.parse_ordering_expression()?;
176            expr = Self::bin_expr(self.node_builder, expr, right, op);
177        }
178        Ok(expr)

237    /// Returns an [`Expression`] AST node if the next tokens represent a
238    /// cast expression.
239    ///
240    /// Otherwise, tries to parse the next token using [`parse_unary_expression`].
241    fn parse_cast_expression(&mut self) -> Result<Expression> {
242        let mut expr = self.parse_unary_expression()?;
243        if self.eat(&Token::As) {
244            let (type_, end_span) = self.parse_primitive_type()?;
245            let span = expr.span() + end_span;
246            expr = CastExpression { expression: expr, type_, span, id: self.node_builder.next_id() }.into();
247        }
248
249        Ok(expr)
250    }

252    /// Returns an [`Expression`] AST node if the next tokens represent a
253    /// unary not, negate, or bitwise not expression.
254    ///
255    /// Otherwise, tries to parse the next token using [`parse_postfix_expression`].
256    pub(super) fn parse_unary_expression(&mut self) -> Result<Expression> {
257        let token_span = self.token.span;
258
259        let Some(op) = self.eat_unary_op() else {
260            return self.parse_postfix_expression();
261        };
262
263        let mut inner = self.parse_unary_expression()?;
264
265        // Try to construct a negative literal.
266        if let UnaryOperation::Negate = op {
267            use LiteralVariant::*;
268            if let Expression::Literal(Literal {
269                variant: Integer(_, string) | Field(string) | Group(string) | Scalar(string),
270                span,
271                ..
272            }) = &mut inner
273            {
274                if !string.starts_with('-') {
275                    // The operation was a negation and the literal was not already negative, so fold it in.
276                    string.insert(0, '-');
277                    *span = token_span + *span;
278                    return Ok(inner);
279                }
280            }
281        }
282
283        Ok(UnaryExpression { span: token_span + inner.span(), op, receiver: inner, id: self.node_builder.next_id() }
284            .into())
285    }

288    /// Returns an [`Expression`] AST node if the next tokens represent a
289    /// method call expression.
290    fn parse_method_call_expression(&mut self, receiver: Expression, method: Identifier) -> Result<Expression> {
291        // Parse the argument list.
292        let (mut args, _, span) = self.parse_expr_tuple()?;
293        let span = receiver.span() + span;
294
295        if let (true, Some(op)) = (args.is_empty(), UnaryOperation::from_symbol(method.name)) {
296            // Found an unary operator and the argument list is empty.
297            Ok(UnaryExpression { span, op, receiver, id: self.node_builder.next_id() }.into())
298        } else if let (1, Some(op)) = (args.len(), BinaryOperation::from_symbol(method.name)) {
299            // Found a binary operator and the argument list contains a single argument.
300            Ok(BinaryExpression {
301                span,
302                op,
303                left: receiver,
304                right: args.swap_remove(0),
305                id: self.node_builder.next_id(),
306            }
307            .into())
308        } else if let (2, Some(CoreFunction::SignatureVerify)) =
309            (args.len(), CoreFunction::from_symbols(sym::signature, method.name))
310        {
311            Ok(AssociatedFunctionExpression {
312                variant: Identifier::new(sym::signature, self.node_builder.next_id()),
313                name: method,
314                arguments: std::iter::once(receiver).chain(args).collect(),
315                span,
316                id: self.node_builder.next_id(),
317            }
318            .into())
319        } else if let (0, Some(CoreFunction::FutureAwait)) =
320            (args.len(), CoreFunction::from_symbols(sym::Future, method.name))
321        {
322            Ok(AssociatedFunctionExpression {
323                variant: Identifier::new(sym::Future, self.node_builder.next_id()),
324                name: method,
325                arguments: vec![receiver],
326                span,
327                id: self.node_builder.next_id(),
328            }
329            .into())
330        } else {
331            // Attempt to parse the method call as a mapping operation.
332            match (args.len(), CoreFunction::from_symbols(sym::Mapping, method.name)) {
333                (1, Some(CoreFunction::MappingGet))
334                | (2, Some(CoreFunction::MappingGetOrUse))
335                | (2, Some(CoreFunction::MappingSet))
336                | (1, Some(CoreFunction::MappingRemove))
337                | (1, Some(CoreFunction::MappingContains)) => {
338                    // Found an instance of `<mapping>.get`, `<mapping>.get_or_use`, `<mapping>.set`, `<mapping>.remove`, or `<mapping>.contains`.
339                    Ok(AssociatedFunctionExpression {
340                        variant: Identifier::new(sym::Mapping, self.node_builder.next_id()),
341                        name: method,
342                        arguments: std::iter::once(receiver).chain(args).collect(),
343                        span,
344                        id: self.node_builder.next_id(),
345                    }
346                    .into())
347                }
348                _ => {
349                    // Either an invalid unary/binary operator, or more arguments given.
350                    self.emit_err(ParserError::invalid_method_call(receiver, method, args.len(), span));
351                    Ok(ErrExpression { span, id: self.node_builder.next_id() }.into())
352                }
353            }
354        }
355    }

357    /// Returns an [`Expression`] AST node if the next tokens represent a
358    /// static access expression.
359    fn parse_associated_access_expression(&mut self, module_name: Expression) -> Result<Expression> {
360        // Ensure that the preceding expression is an identifier (a named type).
361        let variant = if let Expression::Identifier(ident) = module_name {
362            ident
363        } else {
364            return Err(ParserError::invalid_associated_access(&module_name, module_name.span()).into());
365        };
366
367        // Parse the constant or function name.
368        let member_name = self.expect_identifier()?;
369
370        // Check if there are arguments.
371        let expression = if self.check(&Token::LeftParen) {
372            // Parse the arguments
373            let (args, _, end) = self.parse_expr_tuple()?;
374
375            // Return the associated function.
376            AssociatedFunctionExpression {
377                span: module_name.span() + end,
378                variant,
379                name: member_name,
380                arguments: args,
381                id: self.node_builder.next_id(),
382            }
383            .into()
384        } else {
385            // Return the associated constant.
386            AssociatedConstantExpression {
387                span: module_name.span() + member_name.span(),
388                ty: Type::Identifier(variant),
389                name: member_name,
390                id: self.node_builder.next_id(),
391            }
392            .into()
393        };
394
395        Ok(expression)
396    }

403    /// Parses an external function call `credits.aleo/transfer()` or locator `token.aleo/accounts`.
404    ///
405    /// In the ABNF grammar,
406    /// an external function call is one of the two kinds of free function calls,
407    /// namely the one that uses a locator to designate the function;
408    /// a locator is a kind of primary expression.
409    fn parse_external_resource(&mut self, expr: Expression, network_span: Span) -> Result<Expression> {
410        // Parse `/`.
411        self.expect(&Token::Div)?;
412
413        // Parse name.
414        let name = self.expect_identifier()?;
415
416        // Ensure the preceding expression is a (program) identifier.
417        let program: Identifier = match expr {
418            Expression::Identifier(identifier) => identifier,
419            _ => unreachable!("Function called must be preceded by a program identifier."),
420        };
421
422        // Parsing a '{' means that user is trying to illegally define an external record.
423        if self.token.token == Token::LeftCurly {
424            return Err(ParserError::cannot_define_external_record(expr.span() + name.span()).into());
425        }
426
427        // If there is no parenthesis, then it is a locator.
428        if self.token.token != Token::LeftParen {
429            // Parse an external resource locator.
430            return Ok(LocatorExpression {
431                program: ProgramId {
432                    name: program,
433                    network: Identifier { name: sym::aleo, span: network_span, id: self.node_builder.next_id() },
434                },
435                name: name.name,
436                span: expr.span() + name.span(),
437                id: self.node_builder.next_id(),
438            }
439            .into());
440        }
441
442        // Parse the function call.
443        let (arguments, _, span) = self.parse_paren_comma_list(|p| p.parse_expression().map(Some))?;
444
445        Ok(CallExpression {
446            span: expr.span() + span,
447            function: name,
448            program: Some(program.name),
449            const_arguments: vec![], // we do not expect const arguments for external calls at this time
450            arguments,
451            id: self.node_builder.next_id(),
452        }
453        .into())
454    }

456    /// Returns an [`Expression`] AST node if the next tokens represent an
457    /// array access, struct member access, tuple access, or method call expression.
458    ///
459    /// Otherwise, tries to parse the next token using [`parse_primary_expression`].
460    /// Note that, as mentioned in [`parse_primary_expression`],
461    /// this function also completes the parsing of some primary expressions
462    /// (as defined in the ABNF grammar),
463    /// which [`parse_primary_expression`] only starts to parse.
464    fn parse_postfix_expression(&mut self) -> Result<Expression> {
465        // We don't directly parse named types and identifiers in associated constants and functions
466        // here as the ABNF states. Rather, those named types and identifiers are parsed
467        // as primary expressions, and combined to form associated constants and functions here.
468        let mut expr = self.parse_primary_expression()?;
469        loop {
470            if self.eat(&Token::Dot) {
471                if self.check_int() {
472                    // Eat a tuple member access.
473                    let (index, span) = self.eat_whole_number()?;
474                    expr = TupleAccess { tuple: expr, index, span, id: self.node_builder.next_id() }.into();
475                } else if self.eat(&Token::Leo) {
476                    return Err(ParserError::only_aleo_external_calls(expr.span()).into());
477                } else if self.eat(&Token::Aleo) {
478                    if self.token.token == Token::Div {
479                        expr = self.parse_external_resource(expr, self.prev_token.span)?;
480                    } else {
481                        // Parse as address literal, e.g. `hello.aleo`.
482                        if !matches!(expr, Expression::Identifier(_)) {
483                            self.emit_err(ParserError::unexpected(expr.to_string(), "an identifier", expr.span()))
484                        }
485
486                        expr =
487                            Literal::address(format!("{expr}.aleo"), expr.span(), self.node_builder.next_id()).into();
488                    }
489                } else {
490                    // Parse instances of `self.address`.
491                    // This needs to be handled as a special case because `address` is a keyword in Leo,
492                    if matches!(expr, Expression::Identifier(id) if id.name == sym::SelfLower)
493                        && self.token.token == Token::Address
494                    {
495                        // Eat the address token.
496                        let span = self.expect(&Token::Address)?;
497                        // Return a member access expression.
498                        expr = MemberAccess {
499                            span: expr.span() + span,
500                            inner: expr,
501                            name: Identifier { name: sym::address, span, id: self.node_builder.next_id() },
502                            id: self.node_builder.next_id(),
503                        }
504                        .into();
505                        continue;
506                    }
507
508                    // Parse identifier name.
509                    let name = self.expect_identifier()?;
510
511                    if self.check(&Token::LeftParen) {
512                        // Eat a method call on a type
513                        expr = self.parse_method_call_expression(expr, name)?
514                    } else {
515                        // Eat a struct member access.
516                        let expr_span = expr.span();
517                        expr = MemberAccess {
518                            inner: expr,
519                            name,
520                            span: expr_span + name.span(),
521                            id: self.node_builder.next_id(),
522                        }
523                        .into();
524                    }
525                }
526            } else if self.eat(&Token::DoubleColon) {
527                // If we see a `::`, then we either expect a core associated expression or a list of const arguments in
528                // square brackets.
529                if self.check(&Token::LeftSquare) {
530                    // Check that the expression is an identifier.
531                    let Expression::Identifier(ident) = &expr else {
532                        return Err(leo_errors::LeoError::ParserError(ParserError::unexpected(
533                            expr.to_string(),
534                            "an identifier",
535                            expr.span(),
536                        )));
537                    };
538
539                    // Parse a list of const arguments in between `[..]`
540                    let const_arguments = self.parse_bracket_comma_list(|p| p.parse_expression().map(Some))?.0;
541
542                    // If the next token is a `(` then we parse a call expression.
543                    // If the next token is a `{`, then we parse a struct init expression.
544                    if self.check(&Token::LeftParen) {
545                        // Parse a list of input arguments in between `(..)`
546                        let (arguments, _, span) = self.parse_paren_comma_list(|p| p.parse_expression().map(Some))?;
547
548                        // Now form a `CallExpression`
549                        expr = CallExpression {
550                            span: expr.span() + span,
551                            function: *ident,
552                            program: self.program_name,
553                            const_arguments,
554                            arguments,
555                            id: self.node_builder.next_id(),
556                        }
557                        .into();
558                    } else if !self.disallow_struct_construction && self.check(&Token::LeftCurly) {
559                        // Parse struct and records inits as struct expressions with const arguments.
560                        // Enforce struct or record type later at type checking.
561                        expr = self.parse_struct_init_expression(*ident, const_arguments)?;
562                    } else {
563                        self.emit_err(ParserError::unexpected(expr.to_string(), "( or {{", expr.span()))
564                    }
565                } else {
566                    // Eat a core associated constant or core associated function call.
567                    expr = self.parse_associated_access_expression(expr)?;
568                }
569            } else if self.eat(&Token::LeftSquare) {
570                // Eat an array access.
571                let index = self.parse_expression()?;
572                // Eat the closing bracket.
573                let span = self.expect(&Token::RightSquare)?;
574                let expr_span = expr.span();
575                expr =
576                    ArrayAccess { array: expr, index, span: expr_span + span, id: self.node_builder.next_id() }.into();
577            } else if self.check(&Token::LeftParen) {
578                // Check that the expression is an identifier.
579                let Expression::Identifier(ident) = &expr else {
580                    return Err(leo_errors::LeoError::ParserError(ParserError::unexpected(
581                        expr.to_string(),
582                        "an identifier",
583                        expr.span(),
584                    )));
585                };
586
587                // Parse a function call that's by itself.
588                let (arguments, _, span) = self.parse_paren_comma_list(|p| p.parse_expression().map(Some))?;
589                expr = CallExpression {
590                    span: expr.span() + span,
591                    function: *ident,
592                    program: self.program_name,
593                    const_arguments: vec![],
594                    arguments,
595                    id: self.node_builder.next_id(),
596                }
597                .into();
598            }
599            // Stop parsing the postfix expression unless a dot or square bracket follows.
600            if !(self.check(&Token::Dot) || self.check(&Token::LeftSquare)) {
601                break;
602            }
603        }
604        Ok(expr)
605    }

607    /// Returns an [`Expression`] AST node if the next tokens represent
608    /// a parenthesized expression or a unit expression
609    /// or a tuple initialization expression or an affine group literal.
610    fn parse_tuple_expression(&mut self) -> Result<Expression> {
611        let (mut elements, trailing, span) = self.parse_expr_tuple()?;
612
613        match (elements.len(), trailing) {
614            (0, _) | (1, true) => {
615                // A tuple with 0 or 1 elements - emit an error since tuples must have at least two elements.
616                Err(ParserError::tuple_must_have_at_least_two_elements("expression", span).into())
617            }
618            (1, false) => {
619                // If there is one element in the tuple but no trailing comma, e.g `(foo)`, return the element.
620                Ok(elements.remove(0))
621            }
622            _ => {
623                // Otherwise, return a tuple expression.
624                // Note: This is the only place where `TupleExpression` is constructed in the parser.
625                Ok(TupleExpression { elements, span, id: self.node_builder.next_id() }.into())
626            }
627        }
628    }
629
630    /// Attempts to parse an array initialization expression and returns an [`Expression`] AST node if successful.
631    fn parse_array_or_repeat_expression(&mut self) -> Result<Expression> {
632        let (open, close) = Delimiter::Bracket.open_close_pair();
633        let open_span = self.expect(&open)?;
634
635        // Attempt to parse the first expression in the array.
636        let Ok(first_expr) = self.parse_expression() else {
637            // If we're unable to parse an expression, just expect a `]` and error out on empty array.
638            let close_span = self.expect(&close)?;
639            return Err(ParserError::array_must_have_at_least_one_element("expression", open_span + close_span).into());
640        };
641
642        // Handle array repetition syntax: [expr; count]
643        if self.eat(&Token::Semicolon) {
644            let count = self.parse_expression()?;
645            let span = open_span + self.expect(&close)?;
646            return Ok(RepeatExpression { expr: first_expr, count, span, id: self.node_builder.next_id() }.into());
647        }
648
649        // Handle array with multiple elements: [expr1, expr2, ...] or single element with or without trailing comma:
650        // [expr,]
651        let mut elements = vec![first_expr];
652        while self.eat(&Token::Comma) && !self.check(&close) {
653            elements.push(self.parse_expression()?);
654        }
655        let span = open_span + self.expect(&close)?;
656        Ok(ArrayExpression { elements, span, id: self.node_builder.next_id() }.into())
657    }
658
659    fn parse_struct_member(&mut self) -> Result<StructVariableInitializer> {
660        let identifier = self.expect_identifier()?;
661
662        let (expression, span) = if self.eat(&Token::Colon) {
663            // Parse individual struct variable declarations.
664            let expression = self.parse_expression()?;
665            let span = identifier.span + expression.span();
666            (Some(expression), span)
667        } else {
668            (None, identifier.span)
669        };
670
671        Ok(StructVariableInitializer { identifier, expression, id: self.node_builder.next_id(), span })
672    }

674    /// Returns an [`Expression`] AST node if the next tokens represent a
675    /// struct initialization expression.
676    /// let foo = Foo { x: 1u8 };
677    pub fn parse_struct_init_expression(
678        &mut self,
679        identifier: Identifier,
680        const_arguments: Vec<Expression>,
681    ) -> Result<Expression> {
682        let (members, _, end) =
683            self.parse_list(Delimiter::Brace, Some(Token::Comma), |p| p.parse_struct_member().map(Some))?;
684
685        Ok(StructExpression {
686            span: identifier.span + end,
687            name: identifier,
688            const_arguments,
689            members,
690            id: self.node_builder.next_id(),
691        }
692        .into())
693    }
694
695    /// Returns an [`Expression`] AST node if the next token is a primary expression:
696    /// - Literals: field, group, unsigned integer, signed integer, boolean, address, string
697    /// - Aggregate type constructors: array, tuple, structs
698    /// - Identifiers: variables, keywords
699    ///
700    /// This function only parses some of the primary expressions defined in the ABNF grammar;
701    /// for the others, it parses their initial parts,
702    /// leaving it to the [self.parse_postfix_expression] function to complete the parsing.
703    /// For example, of the primary expression `u8::c`, this function only parses the `u8` part,
704    /// leaving it to [self.parse_postfix_expression] to parse the `::c` part.
705    /// So technically the expression returned by this function may not quite be
706    /// an expression as defined in the ABNF grammar,
707    /// but it is only a temporary expression that is combined into a larger one
708    /// by [self.parse_postfix_expression], yielding an actual expression according to the grammar.
709    ///
710    /// Returns an expression error if the token cannot be matched.
711    fn parse_primary_expression(&mut self) -> Result<Expression> {
712        if let Token::LeftParen = self.token.token {
713            return self.parse_tuple_expression();
714        } else if let Token::LeftSquare = self.token.token {
715            return self.parse_array_or_repeat_expression();
716        } else if let Token::Async = self.token.token {
717            let async_keyword_span = self.expect(&Token::Async)?;
718            let block = self.parse_block()?;
719            let span = async_keyword_span + block.span;
720            return Ok(AsyncExpression { block, span, id: self.node_builder.next_id() }.into());
721        }
722
723        let SpannedToken { token, span } = self.token.clone();
724        self.bump();
725
726        Ok(match token {
727            Token::Integer(value) => {
728                let suffix_span = self.token.span;
729                let full_span = span + suffix_span;
730                let assert_no_whitespace = |x| assert_no_whitespace(span, suffix_span, &value, x);
731
732                match self.eat_any(INT_TYPES).then_some(&self.prev_token.token) {
733                    // Hex, octal, binary literal on a noninteger is an error.
734                    Some(Token::Field) | Some(Token::Group) | Some(Token::Scalar)
735                        if value.starts_with("0x")
736                            || value.starts_with("0o")
737                            || value.starts_with("0b")
738                            || value.starts_with("-0x")
739                            || value.starts_with("-0o")
740                            || value.starts_with("-0b") =>
741                    {
742                        return Err(ParserError::hexbin_literal_nonintegers(span).into());
743                    }
744                    // Literal followed by `field`, e.g., `42field`.
745                    Some(Token::Field) => {
746                        assert_no_whitespace("field")?;
747                        Literal::field(value, full_span, self.node_builder.next_id()).into()
748                    }
749                    // Literal followed by `group`, e.g., `42group`.
750                    Some(Token::Group) => {
751                        assert_no_whitespace("group")?;
752                        Literal::group(value, full_span, self.node_builder.next_id()).into()
753                    }
754                    // Literal followed by `scalar` e.g., `42scalar`.
755                    Some(Token::Scalar) => {
756                        assert_no_whitespace("scalar")?;
757                        Literal::scalar(value, full_span, self.node_builder.next_id()).into()
758                    }
759                    // Literal followed by other type suffix, e.g., `42u8`.
760                    Some(suffix) => {
761                        assert_no_whitespace(&suffix.to_string())?;
762                        let int_ty = Self::token_to_int_type(suffix).expect("unknown int type token");
763                        Literal::integer(int_ty, value, full_span, self.node_builder.next_id()).into()
764                    }
765                    None => {
766                        // `Integer` tokens with no suffix are `unsuffixed`. We try to infer their
767                        // type in the type inference phase of the type checker.
768                        Literal::unsuffixed(value, span, self.node_builder.next_id()).into()
769                    }
770                }
771            }
772            Token::True => Literal::boolean(true, span, self.node_builder.next_id()).into(),
773            Token::False => Literal::boolean(false, span, self.node_builder.next_id()).into(),
774            Token::AddressLit(address_string) => {
775                if match self.network {
776                    NetworkName::MainnetV0 => address_string.parse::<Address<MainnetV0>>().is_err(),
777                    NetworkName::TestnetV0 => address_string.parse::<Address<TestnetV0>>().is_err(),
778                    NetworkName::CanaryV0 => address_string.parse::<Address<CanaryV0>>().is_err(),
779                } {
780                    self.emit_err(ParserError::invalid_address_lit(&address_string, span));
781                }
782                Literal::address(address_string, span, self.node_builder.next_id()).into()
783            }
784            Token::StaticString(value) => {
785                Literal { span, id: self.node_builder.next_id(), variant: LiteralVariant::String(value) }.into()
786            }
787            Token::Identifier(name) => {
788                let ident = Identifier { name, span, id: self.node_builder.next_id() };
789                if !self.disallow_struct_construction && self.check(&Token::LeftCurly) {
790                    // Parse struct and records inits as struct expressions without const arguments.
791                    // Enforce struct or record type later at type checking.
792                    self.parse_struct_init_expression(ident, Vec::new())?
793                } else {
794                    ident.into()
795                }
796            }
797            Token::SelfLower => Identifier { name: sym::SelfLower, span, id: self.node_builder.next_id() }.into(),
798            Token::Block => Identifier { name: sym::block, span, id: self.node_builder.next_id() }.into(),
799            Token::Future => Identifier { name: sym::Future, span, id: self.node_builder.next_id() }.into(),
800            Token::Network => Identifier { name: sym::network, span, id: self.node_builder.next_id() }.into(),
801            t if crate::type_::TYPE_TOKENS.contains(&t) => Expression::Identifier(Identifier {
802                name: t.keyword_to_symbol().unwrap(),
803                span,
804                id: self.node_builder.next_id(),
805            }),
806            token => {
807                return Err(ParserError::unexpected_str(token, "expression", span).into());
808            }
809        })
810    }
811}
812
813fn assert_no_whitespace(left_span: Span, right_span: Span, left: &str, right: &str) -> Result<()> {
814    if left_span.hi != right_span.lo {
815        let error_span = Span::new(left_span.hi, right_span.lo); // The span between them.
816        return Err(ParserError::unexpected_whitespace(left, right, error_span).into());
817    }
818
819    Ok(())
