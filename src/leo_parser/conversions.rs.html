35fn to_identifier(node: &SyntaxNode<'_>, builder: &NodeBuilder) -> leo_ast::Identifier {
36    let name = Symbol::intern(node.text);
37    leo_ast::Identifier { name, span: node.span, id: builder.next_id() }
38}
39
40fn path_to_parts(node: &SyntaxNode<'_>, builder: &NodeBuilder) -> Vec<leo_ast::Identifier> {
41    let mut identifiers = Vec::new();
42    let mut i = node.span.lo;
43    for text in node.text.split("::") {
44        let end = i + text.len() as u32;
45        let span = leo_span::Span { lo: i, hi: end };
46        let name = Symbol::intern(text);
47        identifiers.push(leo_ast::Identifier { name, span, id: builder.next_id() });
48        // Account for the "::".
49        i = end + 2;
50    }
51    identifiers
52}

54fn to_mode(node: &SyntaxNode<'_>) -> leo_ast::Mode {
55    match node.text {
56        "constant" => leo_ast::Mode::Constant,
57        "private" => leo_ast::Mode::Private,
58        "public" => leo_ast::Mode::Public,
59        _ => leo_ast::Mode::None,
60    }
61}

63fn to_type(node: &SyntaxNode<'_>, builder: &NodeBuilder, handler: &Handler) -> Result<leo_ast::Type> {
64    let SyntaxKind::Type(type_kind) = node.kind else { todo!() };
65
66    let type_ = match type_kind {
67        TypeKind::Address => leo_ast::Type::Address,
68        TypeKind::Array => {
69            let [_l, type_, _s, length, _r] = &node.children[..] else {
70                // This "Can't happen" panic, like others in this file, will not be triggered unless
71                // there is an error in grammar.lalrpop.
72                panic!("Can't happen");
73            };
74            let element_type = to_type(type_, builder, handler)?;
75            let length = to_expression(length, builder, handler)?;
76            leo_ast::ArrayType { element_type: Box::new(element_type), length: Box::new(length) }.into()
77        }
78        TypeKind::Boolean => leo_ast::Type::Boolean,
79        TypeKind::Composite => {
80            let name = &node.children[0];
81            if let Some((program, name_str)) = name.text.split_once(".aleo/") {
82                // This is a locator.
83                let name_id = leo_ast::Identifier {
84                    name: Symbol::intern(name_str),
85                    span: leo_span::Span {
86                        lo: name.span.lo + program.len() as u32 + 5,
87                        hi: name.span.lo + name.text.len() as u32,
88                    },
89                    id: builder.next_id(),
90                };
91                leo_ast::CompositeType {
92                    path: leo_ast::Path::new(Vec::new(), name_id, false, None, name_id.span, builder.next_id()),
93                    const_arguments: Vec::new(),
94                    program: Some(Symbol::intern(program)),
95                }
96                .into()
97            } else {
98                // It's a path.
99                let mut path_components = path_to_parts(name, builder);
100                let mut const_arguments = Vec::new();
101                if let Some(arg_list) = node.children.get(1) {
102                    const_arguments = arg_list
103                        .children
104                        .iter()
105                        .filter(|child| matches!(child.kind, SyntaxKind::Expression(..)))
106                        .map(|child| to_expression(child, builder, handler))
107                        .collect::<Result<Vec<_>>>()?;
108                }
109                let identifier = path_components.pop().unwrap();
110                let path = leo_ast::Path::new(path_components, identifier, false, None, name.span, builder.next_id());
111                leo_ast::CompositeType { path, const_arguments, program: None }.into()
112            }
113        }
114        TypeKind::Field => leo_ast::Type::Field,
115        TypeKind::Future => {
116            if node.children.len() == 1 {
117                leo_ast::FutureType::default().into()
118            } else {
119                let types = node
120                    .children
121                    .iter()
122                    .filter(|child| matches!(child.kind, SyntaxKind::Type(..)))
123                    .map(|child| to_type(child, builder, handler))
124                    .collect::<Result<Vec<_>>>()?;
125                leo_ast::FutureType::new(types, None, true).into()
126            }
127        }
128        TypeKind::Group => leo_ast::Type::Group,
129        TypeKind::Identifier => todo!(),
130        TypeKind::Integer(int_type_kind) => {
131            let int_type = match int_type_kind {
132                IntegerTypeKind::U8 => leo_ast::IntegerType::U8,
133                IntegerTypeKind::U16 => leo_ast::IntegerType::U16,
134                IntegerTypeKind::U32 => leo_ast::IntegerType::U32,
135                IntegerTypeKind::U64 => leo_ast::IntegerType::U64,
136                IntegerTypeKind::U128 => leo_ast::IntegerType::U128,
137                IntegerTypeKind::I8 => leo_ast::IntegerType::I8,
138                IntegerTypeKind::I16 => leo_ast::IntegerType::I16,
139                IntegerTypeKind::I32 => leo_ast::IntegerType::I32,
140                IntegerTypeKind::I64 => leo_ast::IntegerType::I64,
141                IntegerTypeKind::I128 => leo_ast::IntegerType::I128,
142            };
143            leo_ast::Type::Integer(int_type)
144        }
145        TypeKind::Mapping => {
146            todo!()
147        }
148        TypeKind::Optional => {
149            let [inner_type, _q] = &node.children[..] else {
150                // This "Can't happen" panic, like others in this file, will not be triggered unless
151                // there is an error in grammar.lalrpop.
152                panic!("Can't happen");
153            };
154            leo_ast::Type::Optional(leo_ast::OptionalType { inner: Box::new(to_type(inner_type, builder, handler)?) })
155        }
156        TypeKind::Scalar => leo_ast::Type::Scalar,
157        TypeKind::Signature => leo_ast::Type::Signature,
158        TypeKind::String => leo_ast::Type::String,
159        TypeKind::Tuple => {
160            let elements = node
161                .children
162                .iter()
163                .filter(|child| matches!(child.kind, SyntaxKind::Type(..)))
164                .map(|child| to_type(child, builder, handler))
165                .collect::<Result<Vec<_>>>()?;
166            leo_ast::TupleType::new(elements).into()
167        }
168        TypeKind::Numeric => leo_ast::Type::Numeric,
169        TypeKind::Unit => leo_ast::Type::Unit,
170    };
171
172    Ok(type_)
173}

175fn to_block(node: &SyntaxNode<'_>, builder: &NodeBuilder, handler: &Handler) -> Result<leo_ast::Block> {
176    assert_eq!(node.kind, SyntaxKind::Statement(StatementKind::Block));
177
178    let statements = node
179        .children
180        .iter()
181        .filter(|child| matches!(child.kind, SyntaxKind::Statement(..)))
182        .map(|child| to_statement(child, builder, handler))
183        .collect::<Result<Vec<_>>>()?;
184    Ok(leo_ast::Block { statements, span: node.span, id: builder.next_id() })
185}

187pub fn to_statement(node: &SyntaxNode<'_>, builder: &NodeBuilder, handler: &Handler) -> Result<leo_ast::Statement> {
188    let SyntaxKind::Statement(statement_kind) = node.kind else { todo!() };
189
190    let span = node.span;
191    let id = builder.next_id();
192
193    let value = match statement_kind {
194        StatementKind::Assert => {
195            let [_a, _left, expr, _right, _s] = &node.children[..] else {
196                panic!("Can't happen");
197            };
198
199            let expr = to_expression(expr, builder, handler)?;
200            let variant = leo_ast::AssertVariant::Assert(expr);
201
202            leo_ast::AssertStatement { variant, span, id }.into()
203        }
204        StatementKind::AssertEq => {
205            let [_a, _left, e0, _c, e1, _right, _s] = &node.children[..] else {
206                panic!("Can't happen");
207            };
208
209            let e0 = to_expression(e0, builder, handler)?;
210            let e1 = to_expression(e1, builder, handler)?;
211            let variant = leo_ast::AssertVariant::AssertEq(e0, e1);
212
213            leo_ast::AssertStatement { variant, span, id }.into()
214        }
215        StatementKind::AssertNeq => {
216            let [_a, _left, e0, _c, e1, _right, _s] = &node.children[..] else {
217                panic!("Can't happen");
218            };
219
220            let e0 = to_expression(e0, builder, handler)?;
221            let e1 = to_expression(e1, builder, handler)?;
222            let variant = leo_ast::AssertVariant::AssertNeq(e0, e1);
223
224            leo_ast::AssertStatement { variant, span, id }.into()
225        }
226        StatementKind::Assign => {
227            let [lhs, a, rhs, _s] = &node.children[..] else {
228                panic!("Can't happen");
229            };
230
231            let left = to_expression(lhs, builder, handler)?;
232            let right = to_expression(rhs, builder, handler)?;
233            if a.text == "=" {
234                // Just a regular assignment.
235                leo_ast::AssignStatement { place: left, value: right, span, id }.into()
236            } else {
237                // We have to translate it into a binary operation and assignment.
238                let op = match a.text {
239                    "+=" => leo_ast::BinaryOperation::Add,
240                    "-=" => leo_ast::BinaryOperation::Sub,
241                    "*=" => leo_ast::BinaryOperation::Mul,
242                    "/=" => leo_ast::BinaryOperation::Div,
243                    "%=" => leo_ast::BinaryOperation::Rem,
244                    "**=" => leo_ast::BinaryOperation::Pow,
245                    "<<=" => leo_ast::BinaryOperation::Shl,
246                    ">>=" => leo_ast::BinaryOperation::Shr,
247                    "&=" => leo_ast::BinaryOperation::BitwiseAnd,
248                    "|=" => leo_ast::BinaryOperation::BitwiseOr,
249                    "^=" => leo_ast::BinaryOperation::Xor,
250                    "&&=" => leo_ast::BinaryOperation::And,
251                    "||=" => leo_ast::BinaryOperation::Or,
252                    _ => panic!("Can't happen"),
253                };
254
255                let binary_expr = leo_ast::BinaryExpression { left: left.clone(), right, op, span, id };
256
257                leo_ast::AssignStatement { place: left, value: binary_expr.into(), span, id }.into()
258            }
259        }
260        StatementKind::Block => to_block(node, builder, handler)?.into(),
261        StatementKind::Conditional => {
262            match &node.children[..] {
263                [_if, c, block] => {
264                    // No else.
265                    let condition = to_expression(c, builder, handler)?;
266                    let then = to_block(block, builder, handler)?;
267                    leo_ast::ConditionalStatement { condition, then, otherwise: None, span, id }.into()
268                }
269                [_if, c, block, _else, otherwise] => {
270                    // An else clause.
271                    let condition = to_expression(c, builder, handler)?;
272                    let then = to_block(block, builder, handler)?;
273                    let otherwise = to_statement(otherwise, builder, handler)?;
274                    leo_ast::ConditionalStatement { condition, then, otherwise: Some(Box::new(otherwise)), span, id }
275                        .into()
276                }
277
278                _ => panic!("Can't happen"),
279            }
280        }
281        StatementKind::Const => {
282            let [_const, name, _c, type_, _a, rhs, _s] = &node.children[..] else {
283                panic!("Can't happen");
284            };
285
286            let place = to_identifier(name, builder);
287            let type_ = to_type(type_, builder, handler)?;
288            let value = to_expression(rhs, builder, handler)?;
289
290            leo_ast::ConstDeclaration { place, type_, value, span, id }.into()
291        }
292        StatementKind::Definition => {
293            match &node.children[..] {
294                [_let, name, _c, type_, _assign, e, _s] => {
295                    // Singe place, type.
296                    let name = to_identifier(name, builder);
297                    let place = leo_ast::DefinitionPlace::Single(name);
298                    let value = to_expression(e, builder, handler)?;
299                    let type_ = Some(to_type(type_, builder, handler)?);
300                    leo_ast::DefinitionStatement { place, type_, value, span, id }.into()
301                }
302                [_let, name, _assign, e, _s] => {
303                    // Single place, no type.
304                    let name = to_identifier(name, builder);
305                    let place = leo_ast::DefinitionPlace::Single(name);
306                    let value = to_expression(e, builder, handler)?;
307                    leo_ast::DefinitionStatement { place, type_: None, value, span, id }.into()
308                }
309                children => {
310                    // Multiple place, with or without type.
311                    let right_paren_index = children.iter().position(|child| child.text == ")").unwrap();
312
313                    // The items between parens.
314                    let names = children[2..right_paren_index].iter()
315                        // The ones that aren't commas - the identifiers.
316                        .filter(|child| child.text != ",")
317                        .map(|child| to_identifier(child, builder))
318                    .collect();
319
320                    // Get the type, if there is one.
321                    let type_ = children
322                        .iter()
323                        .find(|child| matches!(child.kind, SyntaxKind::Type(..)))
324                        .map(|child| to_type(child, builder, handler))
325                        .transpose()?;
326
327                    let expr = &children[children.len() - 2];
328                    let value = to_expression(expr, builder, handler)?;
329                    let place = leo_ast::DefinitionPlace::Multiple(names);
330                    leo_ast::DefinitionStatement { place, type_, value, span, id }.into()
331                }
332            }
333        }
334        StatementKind::Expression => {
335            let expression = to_expression(&node.children[0], builder, handler)?;
336            leo_ast::ExpressionStatement { expression, span, id }.into()
337        }
338        StatementKind::Iteration => {
339            match &node.children[..] {
340                [_f, i, _n, low, _d, hi, block] => {
341                    // No type.
342                    leo_ast::IterationStatement {
343                        variable: to_identifier(i, builder),
344                        type_: None,
345                        start: to_expression(low, builder, handler)?,
346                        stop: to_expression(hi, builder, handler)?,
347                        inclusive: false,
348                        block: to_block(block, builder, handler)?,
349                        span,
350                        id,
351                    }
352                    .into()
353                }
354                [_f, i, _c, type_, _n, low, _d, hi, block] => {
355                    // With type.
356                    leo_ast::IterationStatement {
357                        variable: to_identifier(i, builder),
358                        type_: Some(to_type(type_, builder, handler)?),
359                        start: to_expression(low, builder, handler)?,
360                        stop: to_expression(hi, builder, handler)?,
361                        inclusive: false,
362                        block: to_block(block, builder, handler)?,
363                        span,
364                        id,
365                    }
366                    .into()
367                }
368                _ => panic!("Can't happen"),
369            }
370        }
371        StatementKind::Return => {
372            match &node.children[..] {
373                [_r, e, _s] => {
374                    // With expression.
375                    let expression = to_expression(e, builder, handler)?;
376                    leo_ast::ReturnStatement { expression, span, id }.into()
377                }
378                [_r, _s] => {
379                    // No expression.
380                    leo_ast::ReturnStatement {
381                        expression: leo_ast::UnitExpression { span, id: builder.next_id() }.into(),
382                        span,
383                        id,
384                    }
385                    .into()
386                }
387                _ => panic!("Can't happen"),
388            }
389        }
390    };
391
392    Ok(value)
393}

395pub fn to_expression(node: &SyntaxNode<'_>, builder: &NodeBuilder, handler: &Handler) -> Result<Expression> {
396    let SyntaxKind::Expression(expression_kind) = node.kind else { panic!("Can't happen") };
397
398    let span = node.span;
399    let id = builder.next_id();
400    let text = || node.text.to_string();
401
402    let value = match expression_kind {
403        ExpressionKind::ArrayAccess => {
404            let [array, _left, index, _right] = &node.children[..] else {
405                panic!("Can't happen");
406            };
407
408            let array = to_expression(array, builder, handler)?;
409            let index = to_expression(index, builder, handler)?;
410
411            leo_ast::ArrayAccess { array, index, span, id }.into()
412        }
413        ExpressionKind::AssociatedConstant => {
414            let mut components = path_to_parts(&node.children[0], builder);
415            let name = components.pop().unwrap();
416            let variant = components.pop().unwrap();
417            leo_ast::AssociatedConstantExpression { ty: leo_ast::Type::Identifier(variant), name, span, id }.into()
418        }
419        ExpressionKind::AssociatedFunctionCall => {
420            let mut components = path_to_parts(&node.children[0], builder);
421            let name = components.pop().unwrap();
422            let variant = components.pop().unwrap();
423            let arguments = node
424                .children
425                .iter()
426                .filter(|child| matches!(child.kind, SyntaxKind::Expression(..)))
427                .map(|child| to_expression(child, builder, handler))
428                .collect::<Result<Vec<_>>>()?;
429
430            leo_ast::AssociatedFunctionExpression { variant, name, arguments, span, id }.into()
431        }
432        ExpressionKind::Async => {
433            let [_a, block] = &node.children[..] else {
434                panic!("Can't happen");
435            };
436            leo_ast::AsyncExpression { block: to_block(block, builder, handler)?, span, id }.into()
437        }
438        ExpressionKind::Array => {
439            let elements = node
440                .children
441                .iter()
442                .filter(|child| matches!(child.kind, SyntaxKind::Expression(..)))
443                .map(|child| to_expression(child, builder, handler))
444                .collect::<Result<Vec<_>>>()?;
445            leo_ast::ArrayExpression { elements, span, id }.into()
446        }
447        ExpressionKind::Binary => {
448            let [lhs, op, rhs] = &node.children[..] else {
449                panic!("Can't happen");
450            };
451            // Matches against strings like this are linear searches
452            // which could potentially be improved to constant time by
453            // storing the logos token in the LALRPOP token and matching
454            // against it here.
455            let op = match op.text {
456                "==" => leo_ast::BinaryOperation::Eq,
457                "!=" => leo_ast::BinaryOperation::Neq,
458                "<" => leo_ast::BinaryOperation::Lt,
459                "<=" => leo_ast::BinaryOperation::Lte,
460                ">" => leo_ast::BinaryOperation::Gt,
461                ">=" => leo_ast::BinaryOperation::Gte,
462                "+" => leo_ast::BinaryOperation::Add,
463                "-" => leo_ast::BinaryOperation::Sub,
464                "*" => leo_ast::BinaryOperation::Mul,
465                "/" => leo_ast::BinaryOperation::Div,
466                "%" => leo_ast::BinaryOperation::Rem,
467                "||" => leo_ast::BinaryOperation::Or,
468                "&&" => leo_ast::BinaryOperation::And,
469                "|" => leo_ast::BinaryOperation::BitwiseOr,
470                "&" => leo_ast::BinaryOperation::BitwiseAnd,
471                "**" => leo_ast::BinaryOperation::Pow,
472                "<<" => leo_ast::BinaryOperation::Shl,
473                ">>" => leo_ast::BinaryOperation::Shr,
474                "^" => leo_ast::BinaryOperation::Xor,
475                _ => panic!("Can't happen"),
476            };
477            let left = to_expression(lhs, builder, handler)?;
478            let right = to_expression(rhs, builder, handler)?;
479
480            leo_ast::BinaryExpression { left, right, op, span, id }.into()
481        }
482        ExpressionKind::Call => {
483            let name = &node.children[0];
484
485            let arguments = node
486                .children
487                .iter()
488                .filter(|child| matches!(child.kind, SyntaxKind::Expression(..)))
489                .map(|child| to_expression(child, builder, handler))
490                .collect::<Result<Vec<_>>>()?;
491
492            let (function, program) = if let Some((first, second)) = name.text.split_once(".aleo/") {
493                // This is a locator.
494                let symbol = Symbol::intern(second);
495                let lo = node.span.lo + first.len() as u32 + ".aleo/".len() as u32;
496                let second_span = Span { lo, hi: lo + second.len() as u32 };
497                let identifier = leo_ast::Identifier { name: symbol, span: second_span, id: builder.next_id() };
498                let function = leo_ast::Path::new(Vec::new(), identifier, false, None, span, builder.next_id());
499                (function, Some(Symbol::intern(first)))
500            } else {
501                // It's a path.
502                let mut components = path_to_parts(name, builder);
503                let identifier = components.pop().unwrap();
504                let function = leo_ast::Path::new(components, identifier, false, None, name.span, builder.next_id());
505                (function, None)
506            };
507
508            let mut const_arguments = Vec::new();
509            if let Some(argument_list) =
510                node.children.iter().find(|child| matches!(child.kind, SyntaxKind::ConstArgumentList))
511            {
512                const_arguments = argument_list
513                    .children
514                    .iter()
515                    .filter(|child| matches!(child.kind, SyntaxKind::Expression(..)))
516                    .map(|child| to_expression(child, builder, handler))
517                    .collect::<Result<Vec<_>>>()?;
518            }
519
520            leo_ast::CallExpression { function, const_arguments, arguments, program, span, id }.into()
521        }
522        ExpressionKind::Cast => {
523            let [expression, _as, type_] = &node.children[..] else {
524                panic!("Can't happen");
525            };
526
527            let expression = to_expression(expression, builder, handler)?;
528            let type_ = to_type(type_, builder, handler)?;
529
530            leo_ast::CastExpression { expression, type_, span, id }.into()
531        }
532        ExpressionKind::Path => {
533            // We need to find the spans of the individual path components, since the
534            // lossless tree just has the span of the entire path.
535            let mut identifiers = path_to_parts(&node.children[0], builder);
536            let identifier = identifiers.pop().unwrap();
537            leo_ast::Path::new(identifiers, identifier, false, None, span, id).into()
538        }
539        ExpressionKind::Literal(literal_kind) => match literal_kind {
540            LiteralKind::Address => {
541                let t = text();
542                if !t.contains(".aleo") && t.parse::<Address<TestnetV0>>().is_err() {
543                    // We do this check here rather than in `leo-parser-lossless` simply
544                    // to avoid a dependency on snarkvm for `leo-parser-lossless`.
545                    handler.emit_err(ParserError::invalid_address_lit(&t, span));
546                }
547                leo_ast::Literal::address(t, span, id).into()
548            }
549            LiteralKind::Boolean => match node.text {
550                "true" => leo_ast::Literal::boolean(true, span, id).into(),
551                "false" => leo_ast::Literal::boolean(false, span, id).into(),
552                _ => panic!("Can't happen"),
553            },
554            LiteralKind::Field => leo_ast::Literal::field(text(), span, id).into(),
555            LiteralKind::Group => leo_ast::Literal::group(text(), span, id).into(),
556            LiteralKind::Integer(integer_literal_kind) => {
557                let integer_type = match integer_literal_kind {
558                    IntegerLiteralKind::U8 => leo_ast::IntegerType::U8,
559                    IntegerLiteralKind::U16 => leo_ast::IntegerType::U16,
560                    IntegerLiteralKind::U32 => leo_ast::IntegerType::U32,
561                    IntegerLiteralKind::U64 => leo_ast::IntegerType::U64,
562                    IntegerLiteralKind::U128 => leo_ast::IntegerType::U128,
563                    IntegerLiteralKind::I8 => leo_ast::IntegerType::I8,
564                    IntegerLiteralKind::I16 => leo_ast::IntegerType::I16,
565                    IntegerLiteralKind::I32 => leo_ast::IntegerType::I32,
566                    IntegerLiteralKind::I64 => leo_ast::IntegerType::I64,
567                    IntegerLiteralKind::I128 => leo_ast::IntegerType::I128,
568                };
569                leo_ast::Literal::integer(integer_type, text(), span, id).into()
570            }
571            LiteralKind::None => leo_ast::Literal::none(span, id).into(),
572            LiteralKind::Scalar => leo_ast::Literal::scalar(text(), span, id).into(),
573            LiteralKind::Unsuffixed => leo_ast::Literal::unsuffixed(text(), span, id).into(),
574            LiteralKind::String => leo_ast::Literal::string(text(), span, id).into(),
575        },
576        ExpressionKind::Locator => {
577            let text = node.children[0].text;
578
579            // Parse the locator string in format "some_program.aleo/some_name"
580            if let Some((program_part, name_part)) = text.split_once(".aleo/") {
581                // Create the program identifier
582                let program_name_symbol = Symbol::intern(program_part);
583                let program_name_span = Span { lo: node.span.lo, hi: node.span.lo + program_part.len() as u32 };
584                let program_name =
585                    leo_ast::Identifier { name: program_name_symbol, span: program_name_span, id: builder.next_id() };
586
587                // Create the network identifier (always "aleo")
588                let network_start = node.span.lo + program_part.len() as u32 + 1; // +1 for the dot
589                let network_span = Span {
590                    lo: network_start,
591                    hi: network_start + 4, // "aleo" is 4 characters
592                };
593                let network =
594                    leo_ast::Identifier { name: Symbol::intern("aleo"), span: network_span, id: builder.next_id() };
595
596                // Create the program ID
597                let program_id = leo_ast::ProgramId { name: program_name, network };
598
599                // Create the resource name
600                let name_symbol = Symbol::intern(name_part);
601
602                leo_ast::LocatorExpression { program: program_id, name: name_symbol, span, id }.into()
603            } else {
604                // Invalid locator format - this should have been caught by the parser
605                handler.emit_err(ParserError::custom("Invalid locator format", span));
606                leo_ast::ErrExpression { span, id }.into()
607            }
608        }
609        ExpressionKind::MemberAccess => {
610            let [struct_, _dot, name] = &node.children[..] else {
611                panic!("Can't happen.");
612            };
613            let inner = to_expression(struct_, builder, handler)?;
614            let name = to_identifier(name, builder);
615
616            leo_ast::MemberAccess { inner, name, span, id }.into()
617        }
618        ExpressionKind::MethodCall => {
619            let [expr, _dot, name, ..] = &node.children[..] else {
620                panic!("Can't happen");
621            };
622
623            let name = to_identifier(name, builder);
624            let receiver = to_expression(expr, builder, handler)?;
625
626            let mut args = node.children[3..]
627                .iter()
628                .filter(|child| matches!(child.kind, SyntaxKind::Expression(..)))
629                .map(|child| to_expression(child, builder, handler))
630                .collect::<Result<Vec<_>>>()?;
631
632            if let (true, Some(op)) = (args.is_empty(), leo_ast::UnaryOperation::from_symbol(name.name)) {
633                // Found an unary operator and the argument list is empty.
634                leo_ast::UnaryExpression { span, op, receiver, id }.into()
635            } else if let (1, Some(op)) = (args.len(), leo_ast::BinaryOperation::from_symbol(name.name)) {
636                // Found a binary operator and the argument list contains a single argument.
637                leo_ast::BinaryExpression { span, op, left: receiver, right: args.pop().unwrap(), id }.into()
638            } else if let (2, Some(leo_ast::CoreFunction::SignatureVerify)) =
639                (args.len(), leo_ast::CoreFunction::from_symbols(sym::signature, name.name))
640            {
641                leo_ast::AssociatedFunctionExpression {
642                    variant: leo_ast::Identifier::new(sym::signature, builder.next_id()),
643                    name,
644                    arguments: std::iter::once(receiver).chain(args).collect(),
645                    span,
646                    id,
647                }
648                .into()
649            } else if let (0, Some(leo_ast::CoreFunction::FutureAwait)) =
650                (args.len(), leo_ast::CoreFunction::from_symbols(sym::Future, name.name))
651            {
652                leo_ast::AssociatedFunctionExpression {
653                    variant: leo_ast::Identifier::new(sym::Future, builder.next_id()),
654                    name,
655                    arguments: vec![receiver],
656                    span,
657                    id: builder.next_id(),
658                }
659                .into()
660            } else if let (0, Some(leo_ast::CoreFunction::OptionalUnwrap)) =
661                (args.len(), leo_ast::CoreFunction::from_symbols(sym::Optional, name.name))
662            {
663                leo_ast::AssociatedFunctionExpression {
664                    variant: leo_ast::Identifier::new(sym::Optional, builder.next_id()),
665                    name,
666                    arguments: vec![receiver],
667                    span,
668                    id: builder.next_id(),
669                }
670                .into()
671            } else if let (1, Some(leo_ast::CoreFunction::OptionalUnwrapOr)) =
672                (args.len(), leo_ast::CoreFunction::from_symbols(sym::Optional, name.name))
673            {
674                leo_ast::AssociatedFunctionExpression {
675                    variant: leo_ast::Identifier::new(sym::Optional, builder.next_id()),
676                    name,
677                    arguments: std::iter::once(receiver).chain(args).collect(),
678                    span,
679                    id: builder.next_id(),
680                }
681                .into()
682            } else {
683                // Attempt to parse the method call as a mapping operation.
684                match (args.len(), leo_ast::CoreFunction::from_symbols(sym::Mapping, name.name)) {
685                    (1, Some(leo_ast::CoreFunction::MappingGet))
686                    | (2, Some(leo_ast::CoreFunction::MappingGetOrUse))
687                    | (2, Some(leo_ast::CoreFunction::MappingSet))
688                    | (1, Some(leo_ast::CoreFunction::MappingRemove))
689                    | (1, Some(leo_ast::CoreFunction::MappingContains)) => {
690                        // Found an instance of `<mapping>.get`, `<mapping>.get_or_use`, `<mapping>.set`, `<mapping>.remove`, or `<mapping>.contains`.
691                        leo_ast::AssociatedFunctionExpression {
692                            variant: leo_ast::Identifier::new(sym::Mapping, builder.next_id()),
693                            name,
694                            arguments: std::iter::once(receiver).chain(args).collect(),
695                            span,
696                            id: builder.next_id(),
697                        }
698                        .into()
699                    }
700                    _ => {
701                        // Either an invalid unary/binary operator, or more arguments given.
702                        handler.emit_err(ParserError::invalid_method_call(receiver, name, args.len(), span));
703                        leo_ast::ErrExpression { span, id: builder.next_id() }.into()
704                    }
705                }
706            }
707        }
708        ExpressionKind::Parenthesized => {
709            let [_left, expr, _right] = &node.children[..] else {
710                panic!("Can't happen");
711            };
712            to_expression(expr, builder, handler)?
713        }
714        ExpressionKind::Repeat => {
715            let [_left, expr, _s, count, _right] = &node.children[..] else {
716                panic!("Can't happen");
717            };
718            let expr = to_expression(expr, builder, handler)?;
719            let count = to_expression(count, builder, handler)?;
720            leo_ast::RepeatExpression { expr, count, span, id }.into()
721        }
722        ExpressionKind::SpecialAccess => {
723            let [qualifier, _dot, name] = &node.children[..] else {
724                panic!("Can't happen");
725            };
726
727            let inner = to_identifier(qualifier, builder).into();
728            let name = to_identifier(name, builder);
729
730            leo_ast::MemberAccess { inner, name, span, id }.into()
731        }
732        ExpressionKind::Struct => {
733            let name = &node.children[0];
734            let mut members = Vec::new();
735            for initializer in node.children.iter().filter(|node| node.kind == SyntaxKind::StructMemberInitializer) {
736                let (init_name, expression) = match &initializer.children[..] {
737                    [init_name] => (init_name, None),
738                    [init_name, _c, expr] => (init_name, Some(to_expression(expr, builder, handler)?)),
739                    _ => panic!("Can't happen"),
740                };
741                let init_name = to_identifier(init_name, builder);
742
743                members.push(leo_ast::StructVariableInitializer {
744                    identifier: init_name,
745                    expression,
746                    span: initializer.span,
747                    id: builder.next_id(),
748                });
749            }
750
751            let mut const_arguments = Vec::new();
752            let maybe_const_params = &node.children[1];
753            if maybe_const_params.kind == SyntaxKind::ConstArgumentList {
754                for argument in &maybe_const_params.children {
755                    if matches!(argument.kind, SyntaxKind::Expression(..)) {
756                        let expr = to_expression(argument, builder, handler)?;
757                        const_arguments.push(expr);
758                    }
759                }
760            }
761
762            let mut identifiers = path_to_parts(name, builder);
763            let identifier = identifiers.pop().unwrap();
764            let path = leo_ast::Path::new(identifiers, identifier, false, None, name.span, builder.next_id());
765
766            leo_ast::StructExpression { path, const_arguments, members, span, id }.into()
767        }
768        ExpressionKind::Ternary => {
769            let [cond, _q, if_, _c, then] = &node.children[..] else {
770                panic!("Can't happen");
771            };
772            let condition = to_expression(cond, builder, handler)?;
773            let if_true = to_expression(if_, builder, handler)?;
774            let if_false = to_expression(then, builder, handler)?;
775            leo_ast::TernaryExpression { condition, if_true, if_false, span, id }.into()
776        }
777        ExpressionKind::Tuple => {
778            let elements = node
779                .children
780                .iter()
781                .filter(|expr| matches!(expr.kind, SyntaxKind::Expression(..)))
782                .map(|expr| to_expression(expr, builder, handler))
783                .collect::<Result<Vec<_>>>()?;
784            leo_ast::TupleExpression { elements, span, id }.into()
785        }
786        ExpressionKind::TupleAccess => {
787            let [expr, _dot, integer] = &node.children[..] else {
788                panic!("Can't happen");
789            };
790
791            let tuple = to_expression(expr, builder, handler)?;
792            let integer_text = integer.text.replace("_", "");
793            let value: usize = integer_text.parse().expect("Integer should parse.");
794            let index = value.into();
795
796            leo_ast::TupleAccess { tuple, index, span, id }.into()
797        }
798        ExpressionKind::Unary => {
799            let [op, operand] = &node.children[..] else {
800                panic!("Can't happen");
801            };
802            let mut operand_expression = to_expression(operand, builder, handler)?;
803            let op_variant = match op.text {
804                "!" => leo_ast::UnaryOperation::Not,
805                "-" => leo_ast::UnaryOperation::Negate,
806                _ => panic!("Can't happen"),
807            };
808            if op_variant == leo_ast::UnaryOperation::Negate {
809                use leo_ast::LiteralVariant::*;
810                if let Expression::Literal(leo_ast::Literal {
811                    variant: Integer(_, string) | Field(string) | Group(string) | Scalar(string),
812                    span,
813                    ..
814                }) = &mut operand_expression
815                {
816                    if !string.starts_with('-') {
817                        // The operation was a negation and the literal was not already negative, so fold it in
818                        // and discard the unary expression.
819                        string.insert(0, '-');
820                        *span = op.span + operand.span;
821                        return Ok(operand_expression);
822                    }
823                }
824            }
825            leo_ast::UnaryExpression { receiver: operand_expression, op: op_variant, span, id }.into()
826        }
827        ExpressionKind::Unit => leo_ast::UnitExpression { span, id }.into(),
828    };
829
830    Ok(value)
831}

833fn to_const_parameters(
834    node: &SyntaxNode<'_>,
835    builder: &NodeBuilder,
836    handler: &Handler,
837) -> Result<Vec<leo_ast::ConstParameter>> {

839

842        .filter(|child| matches!(child.kind, SyntaxKind::ConstParameter))
843        .map(|child| {

845                panic!("Can't happen");
846            };
847
848            Ok(leo_ast::ConstParameter {
849                identifier: to_identifier(id, builder),
850                type_: to_type(type_, builder, handler)?,
851                span: child.span,
852                id: builder.next_id(),
853            })
854        })

856}

858fn to_annotation(node: &SyntaxNode<'_>, builder: &NodeBuilder) -> Result<leo_ast::Annotation> {
859    assert_eq!(node.kind, SyntaxKind::Annotation);
860    let name = to_identifier(&node.children[1], builder);
861
862    let mut map = IndexMap::new();
863    node.children.get(2).inspect(|list| {
864        for member in list.children.iter() {
865            if member.kind != SyntaxKind::AnnotationMember {
866                continue;
867            }
868
869            let [key, _assign, value] = &member.children[..] else {
870                panic!("Can't happen");
871            };
872            let key = Symbol::intern(key.text);
873            // Get rid of the delimiting double quotes on the string.
874            let value = value.text[1..value.text.len() - 1].to_string();
875            map.insert(key, value);
876        }
877    });
878    Ok(leo_ast::Annotation { identifier: name, map, span: node.span, id: builder.next_id() })
879}

881fn to_function(node: &SyntaxNode<'_>, builder: &NodeBuilder, handler: &Handler) -> Result<leo_ast::Function> {
882    assert_eq!(node.kind, SyntaxKind::Function);
883
884    let annotations = node
885        .children
886        .iter()
887        .filter(|child| matches!(child.kind, SyntaxKind::Annotation))
888        .map(|child| to_annotation(child, builder))
889        .collect::<Result<Vec<_>>>()?;
890    let async_index = annotations.len();
891    let is_async = node.children[async_index].text == "async";
892
893    let function_variant_index = if is_async { async_index + 1 } else { async_index };
894
895    // The behavior here matches the old parser - "inline" and "script" may be marked async,
896    // but async is ignored. Presumably we should fix this but it's theoretically a breaking change.
897    let variant = match (is_async, node.children[function_variant_index].text) {
898        (true, "function") => leo_ast::Variant::AsyncFunction,
899        (false, "function") => leo_ast::Variant::Function,
900        (_, "inline") => leo_ast::Variant::Inline,
901        (_, "script") => leo_ast::Variant::Script,
902        (true, "transition") => leo_ast::Variant::AsyncTransition,
903        (false, "transition") => leo_ast::Variant::Transition,
904        _ => panic!("Can't happen"),
905    };
906
907    let name = &node.children[function_variant_index + 1];
908    let id = to_identifier(name, builder);
909
910    let mut const_parameters = Vec::new();
911    if let Some(const_param_list) =
912        node.children.iter().find(|child| matches!(child.kind, SyntaxKind::ConstParameterList))
913    {
914        const_parameters = to_const_parameters(const_param_list, builder, handler)?;
915    }
916
917    let parameter_list = node.children.iter().find(|child| matches!(child.kind, SyntaxKind::ParameterList)).unwrap();
918    let input = parameter_list
919        .children
920        .iter()
921        .filter(|child| matches!(child.kind, SyntaxKind::Parameter))
922        .map(|child| {
923            let mode = to_mode(&child.children[0]);
924            let index = if mode == leo_ast::Mode::None { 0 } else { 1 };
925            let [name, _c, type_] = &child.children[index..] else {
926                panic!("Can't happen");
927            };
928            Ok(leo_ast::Input {
929                identifier: to_identifier(name, builder),
930                mode,
931                type_: to_type(type_, builder, handler)?,
932                span: child.span,
933                id: builder.next_id(),
934            })
935        })
936        .collect::<Result<Vec<_>>>()?;
937
938    let [.., maybe_outputs, block] = &node.children[..] else {
939        panic!("Can't happen");
940    };
941    let block = to_block(block, builder, handler)?;
942
943    let to_output = |node: &SyntaxNode<'_>| -> Result<leo_ast::Output> {
944        let mode = to_mode(&node.children[0]);
945        let type_ = node.children.last().unwrap();
946
947        Ok(leo_ast::Output { mode, type_: to_type(type_, builder, handler)?, span: node.span, id: builder.next_id() })
948    };
949
950    let output = match maybe_outputs.kind {
951        SyntaxKind::FunctionOutput => {
952            let output = to_output(maybe_outputs)?;
953            vec![output]
954        }
955        SyntaxKind::FunctionOutputs => maybe_outputs
956            .children
957            .iter()
958            .filter(|child| matches!(child.kind, SyntaxKind::FunctionOutput))
959            .map(|child| to_output(child))
960            .collect::<Result<Vec<_>>>()?,
961        _ => Vec::new(),
962    };
963
964    Ok(leo_ast::Function::new(
965        annotations,
966        variant,
967        id,
968        const_parameters,
969        input,
970        output,
971        block,
972        node.span,
973        builder.next_id(),
974    ))
975}

977fn to_composite(node: &SyntaxNode<'_>, builder: &NodeBuilder, handler: &Handler) -> Result<leo_ast::Composite> {
978    assert_eq!(node.kind, SyntaxKind::StructDeclaration);
979
980    let [struct_or_record, i, .., members] = &node.children[..] else {
981        panic!("Can't happen");
982    };
983
984    let members = members
985        .children
986        .iter()
987        .filter(|child| matches!(child.kind, SyntaxKind::StructMemberDeclaration))
988        .map(|child| {
989            let (mode, ident, type_) = match &child.children[..] {
990                [ident, _c, type_] => (leo_ast::Mode::None, ident, type_),
991                [privacy, ident, _c, type_] => (to_mode(privacy), ident, type_),
992                _ => panic!("Can't happen"),
993            };
994
995            Ok(leo_ast::Member {
996                mode,
997                identifier: to_identifier(ident, builder),
998                type_: to_type(type_, builder, handler)?,
999                span: child.span,
1000                id: builder.next_id(),
1001            })
1002        })
1003        .collect::<Result<Vec<_>>>()?;
1004
1005    let mut const_parameters = Vec::new();
1006    if let Some(const_param_list) =
1007        node.children.iter().find(|child| matches!(child.kind, SyntaxKind::ConstParameterList))
1008    {
1009        const_parameters = to_const_parameters(const_param_list, builder, handler)?;
1010    }
1011
1012    Ok(leo_ast::Composite {
1013        identifier: to_identifier(i, builder),
1014        const_parameters,
1015        members,
1016        external: None,
1017        is_record: struct_or_record.text == "record",
1018        span: node.span,
1019        id: builder.next_id(),
1020    })
1021}

1072pub fn to_module(
1073    node: &SyntaxNode<'_>,
1074    builder: &NodeBuilder,
1075    program_name: Symbol,
1076    path: Vec<Symbol>,
1077    handler: &Handler,
1078) -> Result<leo_ast::Module> {
1079    assert_eq!(node.kind, SyntaxKind::ModuleContents);
1080
1081    let mut functions = node
1082        .children
1083        .iter()
1084        .filter(|child| matches!(child.kind, SyntaxKind::Function))
1085        .map(|child| {
1086            let function = to_function(child, builder, handler)?;
1087            Ok((function.identifier.name, function))
1088        })
1089        .collect::<Result<Vec<_>>>()?;
1090    // Passes like type checking expect transitions to come first.
1091    // Irrelevant for modules at the moment.
1092    functions.sort_by_key(|func| if func.1.variant.is_transition() { 0u8 } else { 1u8 });
1093
1094    let structs = node
1095        .children
1096        .iter()
1097        .filter(|child| matches!(child.kind, SyntaxKind::StructDeclaration))
1098        .map(|child| {
1099            let composite = to_composite(child, builder, handler)?;
1100            Ok((composite.identifier.name, composite))
1101        })
1102        .collect::<Result<Vec<_>>>()?;
1103
1104    let consts = node
1105        .children
1106        .iter()
1107        .filter(|child| matches!(child.kind, SyntaxKind::GlobalConst))
1108        .map(|child| {
1109            let global_const = to_global_const(child, builder, handler)?;
1110            Ok((global_const.place.name, global_const))
1111        })
1112        .collect::<Result<Vec<_>>>()?;
1113
1114    Ok(leo_ast::Module { program_name, path, consts, structs, functions })
1115}

1117pub fn to_main(node: &SyntaxNode<'_>, builder: &NodeBuilder, handler: &Handler) -> Result<leo_ast::Program> {
1118    assert_eq!(node.kind, SyntaxKind::MainContents);
1119
1120    let imports = node
1121        .children
1122        .iter()
1123        .filter(|child| matches!(child.kind, SyntaxKind::Import))
1124        .map(|child| {
1125            let name = Symbol::intern(child.children[1].text.strip_suffix(".aleo").unwrap());
1126            (name, (leo_ast::Program::default(), child.span))
1127        })
1128        .collect::<IndexMap<_, _>>();
1129
1130    let program_node = node.children.last().unwrap();
1131
1132    let mut functions = program_node
1133        .children
1134        .iter()
1135        .filter(|child| matches!(child.kind, SyntaxKind::Function))
1136        .map(|child| {
1137            let function = to_function(child, builder, handler)?;
1138            Ok((function.identifier.name, function))
1139        })
1140        .collect::<Result<Vec<_>>>()?;
1141    // Passes like type checking expect transitions to come first.
1142    functions.sort_by_key(|func| if func.1.variant.is_transition() { 0u8 } else { 1u8 });
1143
1144    let structs = program_node
1145        .children
1146        .iter()
1147        .filter(|child| matches!(child.kind, SyntaxKind::StructDeclaration))
1148        .map(|child| {
1149            let composite = to_composite(child, builder, handler)?;
1150            Ok((composite.identifier.name, composite))
1151        })
1152        .collect::<Result<Vec<_>>>()?;
1153
1154    let consts = program_node
1155        .children
1156        .iter()
1157        .filter(|child| matches!(child.kind, SyntaxKind::GlobalConst))
1158        .map(|child| {
1159            let global_const = to_global_const(child, builder, handler)?;
1160            Ok((global_const.place.name, global_const))
1161        })
1162        .collect::<Result<Vec<_>>>()?;
1163
1164    let mappings = program_node
1165        .children
1166        .iter()
1167        .filter(|child| matches!(child.kind, SyntaxKind::Mapping))
1168        .map(|child| {
1169            let mapping = to_mapping(child, builder, handler)?;
1170            Ok((mapping.identifier.name, mapping))
1171        })
1172        .collect::<Result<Vec<_>>>()?;
1173
1174    // This follows the behavior of the old parser - if multiple constructors are
1175    // present, we silently throw out all but the last. Probably this should be
1176    // changed but it would theoretically be a breaking change.
1177    let mut constructors = program_node
1178        .children
1179        .iter()
1180        .filter(|child| matches!(child.kind, SyntaxKind::Constructor))
1181        .map(|child| to_constructor(child, builder, handler))
1182        .collect::<Result<Vec<_>>>()?;
1183
1184    if let Some(extra) = constructors.get(1) {
1185        return Err(TypeCheckerError::custom("A program can only have one constructor.", extra.span).into());
1186    }
1187
1188    let program_id_node = &program_node.children[1];
1189    let program_name_text = program_id_node.text.strip_suffix(".aleo").unwrap();
1190    let program_name_symbol = Symbol::intern(program_name_text);
1191    let hi = program_id_node.span.lo + program_name_text.len() as u32;
1192    let program_id = leo_ast::ProgramId {
1193        name: leo_ast::Identifier {
1194            name: program_name_symbol,
1195            span: Span { lo: program_id_node.span.lo, hi },
1196            id: builder.next_id(),
1197        },
1198        network: leo_ast::Identifier { name: sym::aleo, span: Span { lo: hi + 1, hi: hi + 5 }, id: builder.next_id() },
1199    };
1200    let program_scope = leo_ast::ProgramScope {
1201        program_id,
1202        consts,
1203        structs,
1204        mappings,
1205        functions,
1206        constructor: constructors.pop(),
1207        span: node.span,
1208    };
1209    Ok(leo_ast::Program {
1210        modules: Default::default(),
1211        imports,
1212        stubs: Default::default(),
1213        program_scopes: std::iter::once((program_name_symbol, program_scope)).collect(),
1214    })
1215}
