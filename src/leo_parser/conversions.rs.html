35fn to_identifier(node: &SyntaxNode<'_>, builder: &NodeBuilder) -> leo_ast::Identifier {
36    let name = Symbol::intern(node.text);
37    leo_ast::Identifier { name, span: node.span, id: builder.next_id() }
38}
39
40fn path_to_parts(node: &SyntaxNode<'_>, builder: &NodeBuilder) -> Vec<leo_ast::Identifier> {
41    let mut identifiers = Vec::new();
42    let mut i = node.span.lo;
43    for text in node.text.split("::") {
44        let end = i + text.len() as u32;
45        let span = leo_span::Span { lo: i, hi: end };
46        let name = Symbol::intern(text);
47        identifiers.push(leo_ast::Identifier { name, span, id: builder.next_id() });
48        // Account for the "::".
49        i = end + 2;
50    }
51    identifiers
52}

54fn to_mode(node: &SyntaxNode<'_>) -> leo_ast::Mode {
55    match node.text {
56        "constant" => leo_ast::Mode::Constant,
57        "private" => leo_ast::Mode::Private,
58        "public" => leo_ast::Mode::Public,
59        _ => leo_ast::Mode::None,
60    }
61}

63fn to_type(node: &SyntaxNode<'_>, builder: &NodeBuilder, handler: &Handler) -> Result<leo_ast::Type> {
64    let SyntaxKind::Type(type_kind) = node.kind else { todo!() };
65
66    let type_ = match type_kind {
67        TypeKind::Address => leo_ast::Type::Address,
68        TypeKind::Array => {
69            let [_l, type_, _s, length, _r] = &node.children[..] else {
70                // This "Can't happen" panic, like others in this file, will not be triggered unless
71                // there is an error in grammar.lalrpop.
72                panic!("Can't happen");
73            };
74            let element_type = to_type(type_, builder, handler)?;
75            let length = to_expression(length, builder, handler)?;
76            leo_ast::ArrayType { element_type: Box::new(element_type), length: Box::new(length) }.into()
77        }
78        TypeKind::Boolean => leo_ast::Type::Boolean,
79        TypeKind::Composite => {
80            let name = &node.children[0];
81            if let Some((program, name_str)) = name.text.split_once(".aleo/") {
82                // This is a locator.
83                let name_id = leo_ast::Identifier {
84                    name: Symbol::intern(name_str),
85                    span: leo_span::Span {
86                        lo: name.span.lo + program.len() as u32 + 5,
87                        hi: name.span.lo + name.text.len() as u32,
88                    },
89                    id: builder.next_id(),
90                };
91                leo_ast::CompositeType {
92                    path: leo_ast::Path::new(Vec::new(), name_id, None, name_id.span, builder.next_id()),
93                    const_arguments: Vec::new(),
94                    program: Some(Symbol::intern(program)),
95                }
96                .into()
97            } else {
98                // It's a path.
99                let mut path_components = path_to_parts(name, builder);
100                let mut const_arguments = Vec::new();
101                if let Some(arg_list) = node.children.get(1) {
102                    const_arguments = arg_list
103                        .children
104                        .iter()
105                        .filter(|child| matches!(child.kind, SyntaxKind::Expression(..)))
106                        .map(|child| to_expression(child, builder, handler))
107                        .collect::<Result<Vec<_>>>()?;
108                }
109                let identifier = path_components.pop().unwrap();
110                let path = leo_ast::Path::new(path_components, identifier, None, name.span, builder.next_id());
111                leo_ast::CompositeType { path, const_arguments, program: None }.into()
112            }
113        }
114        TypeKind::Field => leo_ast::Type::Field,
115        TypeKind::Future => {
116            if node.children.len() == 1 {
117                leo_ast::FutureType::default().into()
118            } else {
119                let types = node
120                    .children
121                    .iter()
122                    .filter(|child| matches!(child.kind, SyntaxKind::Type(..)))
123                    .map(|child| to_type(child, builder, handler))
124                    .collect::<Result<Vec<_>>>()?;
125                leo_ast::FutureType::new(types, None, true).into()
126            }
127        }
128        TypeKind::Group => leo_ast::Type::Group,
129        TypeKind::Identifier => todo!(),
130        TypeKind::Integer(int_type_kind) => {
131            let int_type = match int_type_kind {
132                IntegerTypeKind::U8 => leo_ast::IntegerType::U8,
133                IntegerTypeKind::U16 => leo_ast::IntegerType::U16,
134                IntegerTypeKind::U32 => leo_ast::IntegerType::U32,
135                IntegerTypeKind::U64 => leo_ast::IntegerType::U64,
136                IntegerTypeKind::U128 => leo_ast::IntegerType::U128,
137                IntegerTypeKind::I8 => leo_ast::IntegerType::I8,
138                IntegerTypeKind::I16 => leo_ast::IntegerType::I16,
139                IntegerTypeKind::I32 => leo_ast::IntegerType::I32,
140                IntegerTypeKind::I64 => leo_ast::IntegerType::I64,
141                IntegerTypeKind::I128 => leo_ast::IntegerType::I128,
142            };
143            leo_ast::Type::Integer(int_type)
144        }
145        TypeKind::Mapping => {
146            todo!()
147        }
148        TypeKind::Scalar => leo_ast::Type::Scalar,
149        TypeKind::Signature => leo_ast::Type::Signature,
150        TypeKind::String => leo_ast::Type::String,
151        TypeKind::Tuple => {
152            let elements = node
153                .children
154                .iter()
155                .filter(|child| matches!(child.kind, SyntaxKind::Type(..)))
156                .map(|child| to_type(child, builder, handler))
157                .collect::<Result<Vec<_>>>()?;
158            leo_ast::TupleType::new(elements).into()
159        }
160        TypeKind::Numeric => leo_ast::Type::Numeric,
161        TypeKind::Unit => leo_ast::Type::Unit,
162    };
163
164    Ok(type_)
165}

167fn to_block(node: &SyntaxNode<'_>, builder: &NodeBuilder, handler: &Handler) -> Result<leo_ast::Block> {
168    assert_eq!(node.kind, SyntaxKind::Statement(StatementKind::Block));
169
170    let statements = node
171        .children
172        .iter()
173        .filter(|child| matches!(child.kind, SyntaxKind::Statement(..)))
174        .map(|child| to_statement(child, builder, handler))
175        .collect::<Result<Vec<_>>>()?;
176    Ok(leo_ast::Block { statements, span: node.span, id: builder.next_id() })
177}

179pub fn to_statement(node: &SyntaxNode<'_>, builder: &NodeBuilder, handler: &Handler) -> Result<leo_ast::Statement> {
180    let SyntaxKind::Statement(statement_kind) = node.kind else { todo!() };
181
182    let span = node.span;
183    let id = builder.next_id();
184
185    let value = match statement_kind {
186        StatementKind::Assert => {
187            let [_a, _left, expr, _right, _s] = &node.children[..] else {
188                panic!("Can't happen");
189            };
190
191            let expr = to_expression(expr, builder, handler)?;
192            let variant = leo_ast::AssertVariant::Assert(expr);
193
194            leo_ast::AssertStatement { variant, span, id }.into()
195        }
196        StatementKind::AssertEq => {
197            let [_a, _left, e0, _c, e1, _right, _s] = &node.children[..] else {
198                panic!("Can't happen");
199            };
200
201            let e0 = to_expression(e0, builder, handler)?;
202            let e1 = to_expression(e1, builder, handler)?;
203            let variant = leo_ast::AssertVariant::AssertEq(e0, e1);
204
205            leo_ast::AssertStatement { variant, span, id }.into()
206        }
207        StatementKind::AssertNeq => {
208            let [_a, _left, e0, _c, e1, _right, _s] = &node.children[..] else {
209                panic!("Can't happen");
210            };
211
212            let e0 = to_expression(e0, builder, handler)?;
213            let e1 = to_expression(e1, builder, handler)?;
214            let variant = leo_ast::AssertVariant::AssertNeq(e0, e1);
215
216            leo_ast::AssertStatement { variant, span, id }.into()
217        }
218        StatementKind::Assign => {
219            let [lhs, a, rhs, _s] = &node.children[..] else {
220                panic!("Can't happen");
221            };
222
223            let left = to_expression(lhs, builder, handler)?;
224            let right = to_expression(rhs, builder, handler)?;
225            if a.text == "=" {
226                // Just a regular assignment.
227                leo_ast::AssignStatement { place: left, value: right, span, id }.into()
228            } else {
229                // We have to translate it into a binary operation and assignment.
230                let op = match a.text {
231                    "+=" => leo_ast::BinaryOperation::Add,
232                    "-=" => leo_ast::BinaryOperation::Sub,
233                    "*=" => leo_ast::BinaryOperation::Mul,
234                    "/=" => leo_ast::BinaryOperation::Div,
235                    "%=" => leo_ast::BinaryOperation::Rem,
236                    "**=" => leo_ast::BinaryOperation::Pow,
237                    "<<=" => leo_ast::BinaryOperation::Shl,
238                    ">>=" => leo_ast::BinaryOperation::Shr,
239                    "&=" => leo_ast::BinaryOperation::BitwiseAnd,
240                    "|=" => leo_ast::BinaryOperation::BitwiseOr,
241                    "^=" => leo_ast::BinaryOperation::Xor,
242                    "&&=" => leo_ast::BinaryOperation::And,
243                    "||=" => leo_ast::BinaryOperation::Or,
244                    _ => panic!("Can't happen"),
245                };
246
247                let binary_expr = leo_ast::BinaryExpression { left: left.clone(), right, op, span, id };
248
249                leo_ast::AssignStatement { place: left, value: binary_expr.into(), span, id }.into()
250            }
251        }
252        StatementKind::Block => to_block(node, builder, handler)?.into(),
253        StatementKind::Conditional => {
254            match &node.children[..] {
255                [_if, c, block] => {
256                    // No else.
257                    let condition = to_expression(c, builder, handler)?;
258                    let then = to_block(block, builder, handler)?;
259                    leo_ast::ConditionalStatement { condition, then, otherwise: None, span, id }.into()
260                }
261                [_if, c, block, _else, otherwise] => {
262                    // An else clause.
263                    let condition = to_expression(c, builder, handler)?;
264                    let then = to_block(block, builder, handler)?;
265                    let otherwise = to_statement(otherwise, builder, handler)?;
266                    leo_ast::ConditionalStatement { condition, then, otherwise: Some(Box::new(otherwise)), span, id }
267                        .into()
268                }
269
270                _ => panic!("Can't happen"),
271            }
272        }
273        StatementKind::Const => {
274            let [_const, name, _c, type_, _a, rhs, _s] = &node.children[..] else {
275                panic!("Can't happen");
276            };
277
278            let place = to_identifier(name, builder);
279            let type_ = to_type(type_, builder, handler)?;
280            let value = to_expression(rhs, builder, handler)?;
281
282            leo_ast::ConstDeclaration { place, type_, value, span, id }.into()
283        }
284        StatementKind::Definition => {
285            match &node.children[..] {
286                [_let, name, _c, type_, _assign, e, _s] => {
287                    // Singe place, type.
288                    let name = to_identifier(name, builder);
289                    let place = leo_ast::DefinitionPlace::Single(name);
290                    let value = to_expression(e, builder, handler)?;
291                    let type_ = Some(to_type(type_, builder, handler)?);
292                    leo_ast::DefinitionStatement { place, type_, value, span, id }.into()
293                }
294                [_let, name, _assign, e, _s] => {
295                    // Single place, no type.
296                    let name = to_identifier(name, builder);
297                    let place = leo_ast::DefinitionPlace::Single(name);
298                    let value = to_expression(e, builder, handler)?;
299                    leo_ast::DefinitionStatement { place, type_: None, value, span, id }.into()
300                }
301                children => {
302                    // Multiple place, with or without type.
303                    let right_paren_index = children.iter().position(|child| child.text == ")").unwrap();
304
305                    // The items between parens.
306                    let names = children[2..right_paren_index].iter()
307                        // The ones that aren't commas - the identifiers.
308                        .filter(|child| child.text != ",")
309                        .map(|child| to_identifier(child, builder))
310                    .collect();
311
312                    // Get the type, if there is one.
313                    let type_ = children
314                        .iter()
315                        .find(|child| matches!(child.kind, SyntaxKind::Type(..)))
316                        .map(|child| to_type(child, builder, handler))
317                        .transpose()?;
318
319                    let expr = &children[children.len() - 2];
320                    let value = to_expression(expr, builder, handler)?;
321                    let place = leo_ast::DefinitionPlace::Multiple(names);
322                    leo_ast::DefinitionStatement { place, type_, value, span, id }.into()
323                }
324            }
325        }
326        StatementKind::Expression => {
327            let expression = to_expression(&node.children[0], builder, handler)?;
328            leo_ast::ExpressionStatement { expression, span, id }.into()
329        }
330        StatementKind::Iteration => {
331            match &node.children[..] {
332                [_f, i, _n, low, _d, hi, block] => {
333                    // No type.
334                    leo_ast::IterationStatement {
335                        variable: to_identifier(i, builder),
336                        type_: None,
337                        start: to_expression(low, builder, handler)?,
338                        stop: to_expression(hi, builder, handler)?,
339                        inclusive: false,
340                        block: to_block(block, builder, handler)?,
341                        span,
342                        id,
343                    }
344                    .into()
345                }
346                [_f, i, _c, type_, _n, low, _d, hi, block] => {
347                    // With type.
348                    leo_ast::IterationStatement {
349                        variable: to_identifier(i, builder),
350                        type_: Some(to_type(type_, builder, handler)?),
351                        start: to_expression(low, builder, handler)?,
352                        stop: to_expression(hi, builder, handler)?,
353                        inclusive: false,
354                        block: to_block(block, builder, handler)?,
355                        span,
356                        id,
357                    }
358                    .into()
359                }
360                _ => panic!("Can't happen"),
361            }
362        }
363        StatementKind::Return => {
364            match &node.children[..] {
365                [_r, e, _s] => {
366                    // With expression.
367                    let expression = to_expression(e, builder, handler)?;
368                    leo_ast::ReturnStatement { expression, span, id }.into()
369                }
370                [_r, _s] => {
371                    // No expression.
372                    leo_ast::ReturnStatement {
373                        expression: leo_ast::UnitExpression { span, id: builder.next_id() }.into(),
374                        span,
375                        id,
376                    }
377                    .into()
378                }
379                _ => panic!("Can't happen"),
380            }
381        }
382    };
383
384    Ok(value)
385}

387pub fn to_expression(node: &SyntaxNode<'_>, builder: &NodeBuilder, handler: &Handler) -> Result<Expression> {
388    let SyntaxKind::Expression(expression_kind) = node.kind else { panic!("Can't happen") };
389
390    let span = node.span;
391    let id = builder.next_id();
392    let text = || node.text.to_string();
393
394    let value = match expression_kind {
395        ExpressionKind::ArrayAccess => {
396            let [array, _left, index, _right] = &node.children[..] else {
397                panic!("Can't happen");
398            };
399
400            let array = to_expression(array, builder, handler)?;
401            let index = to_expression(index, builder, handler)?;
402
403            leo_ast::ArrayAccess { array, index, span, id }.into()
404        }
405        ExpressionKind::AssociatedConstant => {
406            let mut components = path_to_parts(&node.children[0], builder);
407            let name = components.pop().unwrap();
408            let variant = components.pop().unwrap();
409            leo_ast::AssociatedConstantExpression { ty: leo_ast::Type::Identifier(variant), name, span, id }.into()
410        }
411        ExpressionKind::AssociatedFunctionCall => {
412            let mut components = path_to_parts(&node.children[0], builder);
413            let name = components.pop().unwrap();
414            let variant = components.pop().unwrap();
415            let arguments = node
416                .children
417                .iter()
418                .filter(|child| matches!(child.kind, SyntaxKind::Expression(..)))
419                .map(|child| to_expression(child, builder, handler))
420                .collect::<Result<Vec<_>>>()?;
421
422            leo_ast::AssociatedFunctionExpression { variant, name, arguments, span, id }.into()
423        }
424        ExpressionKind::Async => {
425            let [_a, block] = &node.children[..] else {
426                panic!("Can't happen");
427            };
428            leo_ast::AsyncExpression { block: to_block(block, builder, handler)?, span, id }.into()
429        }
430        ExpressionKind::Array => {
431            let elements = node
432                .children
433                .iter()
434                .filter(|child| matches!(child.kind, SyntaxKind::Expression(..)))
435                .map(|child| to_expression(child, builder, handler))
436                .collect::<Result<Vec<_>>>()?;
437            leo_ast::ArrayExpression { elements, span, id }.into()
438        }
439        ExpressionKind::Binary => {
440            let [lhs, op, rhs] = &node.children[..] else {
441                panic!("Can't happen");
442            };
443            // Matches against strings like this are linear searches
444            // which could potentially be improved to constant time by
445            // storing the logos token in the LALRPOP token and matching
446            // against it here.
447            let op = match op.text {
448                "==" => leo_ast::BinaryOperation::Eq,
449                "!=" => leo_ast::BinaryOperation::Neq,
450                "<" => leo_ast::BinaryOperation::Lt,
451                "<=" => leo_ast::BinaryOperation::Lte,
452                ">" => leo_ast::BinaryOperation::Gt,
453                ">=" => leo_ast::BinaryOperation::Gte,
454                "+" => leo_ast::BinaryOperation::Add,
455                "-" => leo_ast::BinaryOperation::Sub,
456                "*" => leo_ast::BinaryOperation::Mul,
457                "/" => leo_ast::BinaryOperation::Div,
458                "%" => leo_ast::BinaryOperation::Rem,
459                "||" => leo_ast::BinaryOperation::Or,
460                "&&" => leo_ast::BinaryOperation::And,
461                "|" => leo_ast::BinaryOperation::BitwiseOr,
462                "&" => leo_ast::BinaryOperation::BitwiseAnd,
463                "**" => leo_ast::BinaryOperation::Pow,
464                "<<" => leo_ast::BinaryOperation::Shl,
465                ">>" => leo_ast::BinaryOperation::Shr,
466                "^" => leo_ast::BinaryOperation::Xor,
467                _ => panic!("Can't happen"),
468            };
469            let left = to_expression(lhs, builder, handler)?;
470            let right = to_expression(rhs, builder, handler)?;
471
472            leo_ast::BinaryExpression { left, right, op, span, id }.into()
473        }
474        ExpressionKind::Call => {
475            let name = &node.children[0];
476
477            let arguments = node
478                .children
479                .iter()
480                .filter(|child| matches!(child.kind, SyntaxKind::Expression(..)))
481                .map(|child| to_expression(child, builder, handler))
482                .collect::<Result<Vec<_>>>()?;
483
484            let (function, program) = if let Some((first, second)) = name.text.split_once(".aleo/") {
485                // This is a locator.
486                let symbol = Symbol::intern(second);
487                let lo = node.span.lo + first.len() as u32 + ".aleo/".len() as u32;
488                let second_span = Span { lo, hi: lo + second.len() as u32 };
489                let identifier = leo_ast::Identifier { name: symbol, span: second_span, id: builder.next_id() };
490                let function = leo_ast::Path::new(Vec::new(), identifier, None, span, builder.next_id());
491                (function, Some(Symbol::intern(first)))
492            } else {
493                // It's a path.
494                let mut components = path_to_parts(name, builder);
495                let identifier = components.pop().unwrap();
496                let function = leo_ast::Path::new(components, identifier, None, name.span, builder.next_id());
497                (function, None)
498            };
499
500            let mut const_arguments = Vec::new();
501            if let Some(argument_list) =
502                node.children.iter().find(|child| matches!(child.kind, SyntaxKind::ConstArgumentList))
503            {
504                const_arguments = argument_list
505                    .children
506                    .iter()
507                    .filter(|child| matches!(child.kind, SyntaxKind::Expression(..)))
508                    .map(|child| to_expression(child, builder, handler))
509                    .collect::<Result<Vec<_>>>()?;
510            }
511
512            leo_ast::CallExpression { function, const_arguments, arguments, program, span, id }.into()
513        }
514        ExpressionKind::Cast => {
515            let [expression, _as, type_] = &node.children[..] else {
516                panic!("Can't happen");
517            };
518
519            let expression = to_expression(expression, builder, handler)?;
520            let type_ = to_type(type_, builder, handler)?;
521
522            leo_ast::CastExpression { expression, type_, span, id }.into()
523        }
524        ExpressionKind::Path => {
525            // We need to find the spans of the individual path components, since the
526            // lossless tree just has the span of the entire path.
527            let mut identifiers = path_to_parts(&node.children[0], builder);
528            let identifier = identifiers.pop().unwrap();
529            leo_ast::Path::new(identifiers, identifier, None, span, id).into()
530        }
531        ExpressionKind::Literal(literal_kind) => match literal_kind {
532            LiteralKind::Address => {
533                let t = text();
534                if !t.contains(".aleo") && t.parse::<Address<TestnetV0>>().is_err() {
535                    // We do this check here rather than in `leo-parser-lossless` simply
536                    // to avoid a dependency on snarkvm for `leo-parser-lossless`.
537                    handler.emit_err(ParserError::invalid_address_lit(&t, span));
538                }
539                leo_ast::Literal::address(t, span, id).into()
540            }
541            LiteralKind::Boolean => match node.text {
542                "true" => leo_ast::Literal::boolean(true, span, id).into(),
543                "false" => leo_ast::Literal::boolean(false, span, id).into(),
544                _ => panic!("Can't happen"),
545            },
546            LiteralKind::Field => leo_ast::Literal::field(text(), span, id).into(),
547            LiteralKind::Group => leo_ast::Literal::group(text(), span, id).into(),
548            LiteralKind::Integer(integer_literal_kind) => {
549                let integer_type = match integer_literal_kind {
550                    IntegerLiteralKind::U8 => leo_ast::IntegerType::U8,
551                    IntegerLiteralKind::U16 => leo_ast::IntegerType::U16,
552                    IntegerLiteralKind::U32 => leo_ast::IntegerType::U32,
553                    IntegerLiteralKind::U64 => leo_ast::IntegerType::U64,
554                    IntegerLiteralKind::U128 => leo_ast::IntegerType::U128,
555                    IntegerLiteralKind::I8 => leo_ast::IntegerType::I8,
556                    IntegerLiteralKind::I16 => leo_ast::IntegerType::I16,
557                    IntegerLiteralKind::I32 => leo_ast::IntegerType::I32,
558                    IntegerLiteralKind::I64 => leo_ast::IntegerType::I64,
559                    IntegerLiteralKind::I128 => leo_ast::IntegerType::I128,
560                };
561                leo_ast::Literal::integer(integer_type, text(), span, id).into()
562            }
563            LiteralKind::Scalar => leo_ast::Literal::scalar(text(), span, id).into(),
564            LiteralKind::Unsuffixed => leo_ast::Literal::unsuffixed(text(), span, id).into(),
565            LiteralKind::String => leo_ast::Literal::string(text(), span, id).into(),
566        },
567        ExpressionKind::Locator => {
568            let text = node.children[0].text;
569
570            // Parse the locator string in format "some_program.aleo/some_name"
571            if let Some((program_part, name_part)) = text.split_once(".aleo/") {
572                // Create the program identifier
573                let program_name_symbol = Symbol::intern(program_part);
574                let program_name_span = Span { lo: node.span.lo, hi: node.span.lo + program_part.len() as u32 };
575                let program_name =
576                    leo_ast::Identifier { name: program_name_symbol, span: program_name_span, id: builder.next_id() };
577
578                // Create the network identifier (always "aleo")
579                let network_start = node.span.lo + program_part.len() as u32 + 1; // +1 for the dot
580                let network_span = Span {
581                    lo: network_start,
582                    hi: network_start + 4, // "aleo" is 4 characters
583                };
584                let network =
585                    leo_ast::Identifier { name: Symbol::intern("aleo"), span: network_span, id: builder.next_id() };
586
587                // Create the program ID
588                let program_id = leo_ast::ProgramId { name: program_name, network };
589
590                // Create the resource name
591                let name_symbol = Symbol::intern(name_part);
592
593                leo_ast::LocatorExpression { program: program_id, name: name_symbol, span, id }.into()
594            } else {
595                // Invalid locator format - this should have been caught by the parser
596                handler.emit_err(ParserError::custom("Invalid locator format", span));
597                leo_ast::ErrExpression { span, id }.into()
598            }
599        }
600        ExpressionKind::MemberAccess => {
601            let [struct_, _dot, name] = &node.children[..] else {
602                panic!("Can't happen.");
603            };
604            let inner = to_expression(struct_, builder, handler)?;
605            let name = to_identifier(name, builder);
606
607            leo_ast::MemberAccess { inner, name, span, id }.into()
608        }
609        ExpressionKind::MethodCall => {
610            let [expr, _dot, name, ..] = &node.children[..] else {
611                panic!("Can't happen");
612            };
613
614            let name = to_identifier(name, builder);
615            let receiver = to_expression(expr, builder, handler)?;
616
617            let mut args = node.children[3..]
618                .iter()
619                .filter(|child| matches!(child.kind, SyntaxKind::Expression(..)))
620                .map(|child| to_expression(child, builder, handler))
621                .collect::<Result<Vec<_>>>()?;
622
623            if let (true, Some(op)) = (args.is_empty(), leo_ast::UnaryOperation::from_symbol(name.name)) {
624                // Found an unary operator and the argument list is empty.
625                leo_ast::UnaryExpression { span, op, receiver, id }.into()
626            } else if let (1, Some(op)) = (args.len(), leo_ast::BinaryOperation::from_symbol(name.name)) {
627                // Found a binary operator and the argument list contains a single argument.
628                leo_ast::BinaryExpression { span, op, left: receiver, right: args.pop().unwrap(), id }.into()
629            } else if let (2, Some(leo_ast::CoreFunction::SignatureVerify)) =
630                (args.len(), leo_ast::CoreFunction::from_symbols(sym::signature, name.name))
631            {
632                leo_ast::AssociatedFunctionExpression {
633                    variant: leo_ast::Identifier::new(sym::signature, builder.next_id()),
634                    name,
635                    arguments: std::iter::once(receiver).chain(args).collect(),
636                    span,
637                    id,
638                }
639                .into()
640            } else if let (0, Some(leo_ast::CoreFunction::FutureAwait)) =
641                (args.len(), leo_ast::CoreFunction::from_symbols(sym::Future, name.name))
642            {
643                leo_ast::AssociatedFunctionExpression {
644                    variant: leo_ast::Identifier::new(sym::Future, builder.next_id()),
645                    name,
646                    arguments: vec![receiver],
647                    span,
648                    id: builder.next_id(),
649                }
650                .into()
651            } else {
652                // Attempt to parse the method call as a mapping operation.
653                match (args.len(), leo_ast::CoreFunction::from_symbols(sym::Mapping, name.name)) {
654                    (1, Some(leo_ast::CoreFunction::MappingGet))
655                    | (2, Some(leo_ast::CoreFunction::MappingGetOrUse))
656                    | (2, Some(leo_ast::CoreFunction::MappingSet))
657                    | (1, Some(leo_ast::CoreFunction::MappingRemove))
658                    | (1, Some(leo_ast::CoreFunction::MappingContains)) => {
659                        // Found an instance of `<mapping>.get`, `<mapping>.get_or_use`, `<mapping>.set`, `<mapping>.remove`, or `<mapping>.contains`.
660                        leo_ast::AssociatedFunctionExpression {
661                            variant: leo_ast::Identifier::new(sym::Mapping, builder.next_id()),
662                            name,
663                            arguments: std::iter::once(receiver).chain(args).collect(),
664                            span,
665                            id: builder.next_id(),
666                        }
667                        .into()
668                    }
669                    _ => {
670                        // Either an invalid unary/binary operator, or more arguments given.
671                        handler.emit_err(ParserError::invalid_method_call(receiver, name, args.len(), span));
672                        leo_ast::ErrExpression { span, id: builder.next_id() }.into()
673                    }
674                }
675            }
676        }
677        ExpressionKind::Parenthesized => {
678            let [_left, expr, _right] = &node.children[..] else {
679                panic!("Can't happen");
680            };
681            to_expression(expr, builder, handler)?
682        }
683        ExpressionKind::Repeat => {
684            let [_left, expr, _s, count, _right] = &node.children[..] else {
685                panic!("Can't happen");
686            };
687            let expr = to_expression(expr, builder, handler)?;
688            let count = to_expression(count, builder, handler)?;
689            leo_ast::RepeatExpression { expr, count, span, id }.into()
690        }
691        ExpressionKind::SpecialAccess => {
692            let [qualifier, _dot, name] = &node.children[..] else {
693                panic!("Can't happen");
694            };
695
696            let inner = to_identifier(qualifier, builder).into();
697            let name = to_identifier(name, builder);
698
699            leo_ast::MemberAccess { inner, name, span, id }.into()
700        }
701        ExpressionKind::Struct => {
702            let name = &node.children[0];
703            let mut members = Vec::new();
704            for initializer in node.children.iter().filter(|node| node.kind == SyntaxKind::StructMemberInitializer) {
705                let (init_name, expression) = match &initializer.children[..] {
706                    [init_name] => (init_name, None),
707                    [init_name, _c, expr] => (init_name, Some(to_expression(expr, builder, handler)?)),
708                    _ => panic!("Can't happen"),
709                };
710                let init_name = to_identifier(init_name, builder);
711
712                members.push(leo_ast::StructVariableInitializer {
713                    identifier: init_name,
714                    expression,
715                    span: initializer.span,
716                    id: builder.next_id(),
717                });
718            }
719
720            let mut const_arguments = Vec::new();
721            let maybe_const_params = &node.children[1];
722            if maybe_const_params.kind == SyntaxKind::ConstArgumentList {
723                for argument in &maybe_const_params.children {
724                    if matches!(argument.kind, SyntaxKind::Expression(..)) {
725                        let expr = to_expression(argument, builder, handler)?;
726                        const_arguments.push(expr);
727                    }
728                }
729            }
730
731            let mut identifiers = path_to_parts(name, builder);
732            let identifier = identifiers.pop().unwrap();
733            let path = leo_ast::Path::new(identifiers, identifier, None, name.span, builder.next_id());
734
735            leo_ast::StructExpression { path, const_arguments, members, span, id }.into()
736        }
737        ExpressionKind::Ternary => {
738            let [cond, _q, if_, _c, then] = &node.children[..] else {
739                panic!("Can't happen");
740            };
741            let condition = to_expression(cond, builder, handler)?;
742            let if_true = to_expression(if_, builder, handler)?;
743            let if_false = to_expression(then, builder, handler)?;
744            leo_ast::TernaryExpression { condition, if_true, if_false, span, id }.into()
745        }
746        ExpressionKind::Tuple => {
747            let elements = node
748                .children
749                .iter()
750                .filter(|expr| matches!(expr.kind, SyntaxKind::Expression(..)))
751                .map(|expr| to_expression(expr, builder, handler))
752                .collect::<Result<Vec<_>>>()?;
753            leo_ast::TupleExpression { elements, span, id }.into()
754        }
755        ExpressionKind::TupleAccess => {
756            let [expr, _dot, integer] = &node.children[..] else {
757                panic!("Can't happen");
758            };
759
760            let tuple = to_expression(expr, builder, handler)?;
761            let integer_text = integer.text.replace("_", "");
762            let value: usize = integer_text.parse().expect("Integer should parse.");
763            let index = value.into();
764
765            leo_ast::TupleAccess { tuple, index, span, id }.into()
766        }
767        ExpressionKind::Unary => {
768            let [op, operand] = &node.children[..] else {
769                panic!("Can't happen");
770            };
771            let mut operand_expression = to_expression(operand, builder, handler)?;
772            let op_variant = match op.text {
773                "!" => leo_ast::UnaryOperation::Not,
774                "-" => leo_ast::UnaryOperation::Negate,
775                _ => panic!("Can't happen"),
776            };
777            if op_variant == leo_ast::UnaryOperation::Negate {
778                use leo_ast::LiteralVariant::*;
779                if let Expression::Literal(leo_ast::Literal {
780                    variant: Integer(_, string) | Field(string) | Group(string) | Scalar(string),
781                    span,
782                    ..
783                }) = &mut operand_expression
784                {
785                    if !string.starts_with('-') {
786                        // The operation was a negation and the literal was not already negative, so fold it in
787                        // and discard the unary expression.
788                        string.insert(0, '-');
789                        *span = op.span + operand.span;
790                        return Ok(operand_expression);
791                    }
792                }
793            }
794            leo_ast::UnaryExpression { receiver: operand_expression, op: op_variant, span, id }.into()
795        }
796        ExpressionKind::Unit => leo_ast::UnitExpression { span, id }.into(),
797    };
798
799    Ok(value)
800}

803    node: &SyntaxNode<'_>,

819                type_: to_type(type_, builder, handler)?,

827fn to_annotation(node: &SyntaxNode<'_>, builder: &NodeBuilder) -> Result<leo_ast::Annotation> {
828    assert_eq!(node.kind, SyntaxKind::Annotation);
829    let name = to_identifier(&node.children[1], builder);
830
831    let mut map = IndexMap::new();
832    node.children.get(2).inspect(|list| {
833        for member in list.children.iter() {
834            if member.kind != SyntaxKind::AnnotationMember {
835                continue;
836            }
837
838            let [key, _assign, value] = &member.children[..] else {
839                panic!("Can't happen");
840            };
841            let key = Symbol::intern(key.text);
842            // Get rid of the delimiting double quotes on the string.
843            let value = value.text[1..value.text.len() - 1].to_string();
844            map.insert(key, value);
845        }
846    });
847    Ok(leo_ast::Annotation { identifier: name, map, span: node.span, id: builder.next_id() })
848}

850fn to_function(node: &SyntaxNode<'_>, builder: &NodeBuilder, handler: &Handler) -> Result<leo_ast::Function> {
851    assert_eq!(node.kind, SyntaxKind::Function);
852
853    let annotations = node
854        .children
855        .iter()
856        .filter(|child| matches!(child.kind, SyntaxKind::Annotation))
857        .map(|child| to_annotation(child, builder))
858        .collect::<Result<Vec<_>>>()?;
859    let async_index = annotations.len();
860    let is_async = node.children[async_index].text == "async";
861
862    let function_variant_index = if is_async { async_index + 1 } else { async_index };
863
864    // The behavior here matches the old parser - "inline" and "script" may be marked async,
865    // but async is ignored. Presumably we should fix this but it's theoretically a breaking change.
866    let variant = match (is_async, node.children[function_variant_index].text) {
867        (true, "function") => leo_ast::Variant::AsyncFunction,
868        (false, "function") => leo_ast::Variant::Function,
869        (_, "inline") => leo_ast::Variant::Inline,
870        (_, "script") => leo_ast::Variant::Script,
871        (true, "transition") => leo_ast::Variant::AsyncTransition,
872        (false, "transition") => leo_ast::Variant::Transition,
873        _ => panic!("Can't happen"),
874    };
875
876    let name = &node.children[function_variant_index + 1];
877    let id = to_identifier(name, builder);
878
879    let mut const_parameters = Vec::new();
880    if let Some(const_param_list) =
881        node.children.iter().find(|child| matches!(child.kind, SyntaxKind::ConstParameterList))
882    {
883        const_parameters = to_const_parameters(const_param_list, builder, handler)?;
884    }
885
886    let parameter_list = node.children.iter().find(|child| matches!(child.kind, SyntaxKind::ParameterList)).unwrap();
887    let input = parameter_list
888        .children
889        .iter()
890        .filter(|child| matches!(child.kind, SyntaxKind::Parameter))
891        .map(|child| {
892            let mode = to_mode(&child.children[0]);
893            let index = if mode == leo_ast::Mode::None { 0 } else { 1 };
894            let [name, _c, type_] = &child.children[index..] else {
895                panic!("Can't happen");
896            };
897            Ok(leo_ast::Input {
898                identifier: to_identifier(name, builder),
899                mode,
900                type_: to_type(type_, builder, handler)?,
901                span: child.span,
902                id: builder.next_id(),
903            })
904        })
905        .collect::<Result<Vec<_>>>()?;
906
907    let [.., maybe_outputs, block] = &node.children[..] else {
908        panic!("Can't happen");
909    };
910    let block = to_block(block, builder, handler)?;
911
912    let to_output = |node: &SyntaxNode<'_>| -> Result<leo_ast::Output> {
913        let mode = to_mode(&node.children[0]);
914        let type_ = node.children.last().unwrap();
915
916        Ok(leo_ast::Output { mode, type_: to_type(type_, builder, handler)?, span: node.span, id: builder.next_id() })
917    };
918
919    let output = match maybe_outputs.kind {
920        SyntaxKind::FunctionOutput => {
921            let output = to_output(maybe_outputs)?;
922            vec![output]
923        }
924        SyntaxKind::FunctionOutputs => maybe_outputs
925            .children
926            .iter()
927            .filter(|child| matches!(child.kind, SyntaxKind::FunctionOutput))
928            .map(|child| to_output(child))
929            .collect::<Result<Vec<_>>>()?,
930        _ => Vec::new(),
931    };
932
933    Ok(leo_ast::Function::new(
934        annotations,
935        variant,
936        id,
937        const_parameters,
938        input,
939        output,
940        block,
941        node.span,
942        builder.next_id(),
943    ))
944}

946fn to_composite(node: &SyntaxNode<'_>, builder: &NodeBuilder, handler: &Handler) -> Result<leo_ast::Composite> {
947    assert_eq!(node.kind, SyntaxKind::StructDeclaration);
948
949    let [struct_or_record, i, .., members] = &node.children[..] else {
950        panic!("Can't happen");
951    };
952
953    let members = members
954        .children
955        .iter()
956        .filter(|child| matches!(child.kind, SyntaxKind::StructMemberDeclaration))
957        .map(|child| {
958            let (mode, ident, type_) = match &child.children[..] {
959                [ident, _c, type_] => (leo_ast::Mode::None, ident, type_),
960                [privacy, ident, _c, type_] => (to_mode(privacy), ident, type_),
961                _ => panic!("Can't happen"),
962            };
963
964            Ok(leo_ast::Member {
965                mode,
966                identifier: to_identifier(ident, builder),
967                type_: to_type(type_, builder, handler)?,
968                span: child.span,
969                id: builder.next_id(),
970            })
971        })
972        .collect::<Result<Vec<_>>>()?;
973
974    let mut const_parameters = Vec::new();
975    if let Some(const_param_list) =
976        node.children.iter().find(|child| matches!(child.kind, SyntaxKind::ConstParameterList))
977    {
978        const_parameters = to_const_parameters(const_param_list, builder, handler)?;
979    }
980
981    Ok(leo_ast::Composite {
982        identifier: to_identifier(i, builder),
983        const_parameters,
984        members,
985        external: None,
986        is_record: struct_or_record.text == "record",
987        span: node.span,
988        id: builder.next_id(),
989    })
990}

1041pub fn to_module(
1042    node: &SyntaxNode<'_>,
1043    builder: &NodeBuilder,
1044    program_name: Symbol,
1045    path: Vec<Symbol>,
1046    handler: &Handler,
1047) -> Result<leo_ast::Module> {
1048    assert_eq!(node.kind, SyntaxKind::ModuleContents);
1049
1050    let mut functions = node
1051        .children
1052        .iter()
1053        .filter(|child| matches!(child.kind, SyntaxKind::Function))
1054        .map(|child| {
1055            let function = to_function(child, builder, handler)?;
1056            Ok((function.identifier.name, function))
1057        })
1058        .collect::<Result<Vec<_>>>()?;
1059    // Passes like type checking expect transitions to come first.
1060    // Irrelevant for modules at the moment.
1061    functions.sort_by_key(|func| if func.1.variant.is_transition() { 0u8 } else { 1u8 });
1062
1063    let structs = node
1064        .children
1065        .iter()
1066        .filter(|child| matches!(child.kind, SyntaxKind::StructDeclaration))
1067        .map(|child| {
1068            let composite = to_composite(child, builder, handler)?;
1069            Ok((composite.identifier.name, composite))
1070        })
1071        .collect::<Result<Vec<_>>>()?;
1072
1073    let consts = node
1074        .children
1075        .iter()
1076        .filter(|child| matches!(child.kind, SyntaxKind::GlobalConst))
1077        .map(|child| {
1078            let global_const = to_global_const(child, builder, handler)?;
1079            Ok((global_const.place.name, global_const))
1080        })
1081        .collect::<Result<Vec<_>>>()?;
1082
1083    Ok(leo_ast::Module { program_name, path, consts, structs, functions })
1084}

1086pub fn to_main(node: &SyntaxNode<'_>, builder: &NodeBuilder, handler: &Handler) -> Result<leo_ast::Program> {
1087    assert_eq!(node.kind, SyntaxKind::MainContents);
1088
1089    let imports = node
1090        .children
1091        .iter()
1092        .filter(|child| matches!(child.kind, SyntaxKind::Import))
1093        .map(|child| {
1094            let name = Symbol::intern(child.children[1].text.strip_suffix(".aleo").unwrap());
1095            (name, (leo_ast::Program::default(), child.span))
1096        })
1097        .collect::<IndexMap<_, _>>();
1098
1099    let program_node = node.children.last().unwrap();
1100
1101    let mut functions = program_node
1102        .children
1103        .iter()
1104        .filter(|child| matches!(child.kind, SyntaxKind::Function))
1105        .map(|child| {
1106            let function = to_function(child, builder, handler)?;
1107            Ok((function.identifier.name, function))
1108        })
1109        .collect::<Result<Vec<_>>>()?;
1110    // Passes like type checking expect transitions to come first.
1111    functions.sort_by_key(|func| if func.1.variant.is_transition() { 0u8 } else { 1u8 });
1112
1113    let structs = program_node
1114        .children
1115        .iter()
1116        .filter(|child| matches!(child.kind, SyntaxKind::StructDeclaration))
1117        .map(|child| {
1118            let composite = to_composite(child, builder, handler)?;
1119            Ok((composite.identifier.name, composite))
1120        })
1121        .collect::<Result<Vec<_>>>()?;
1122
1123    let consts = program_node
1124        .children
1125        .iter()
1126        .filter(|child| matches!(child.kind, SyntaxKind::GlobalConst))
1127        .map(|child| {
1128            let global_const = to_global_const(child, builder, handler)?;
1129            Ok((global_const.place.name, global_const))
1130        })
1131        .collect::<Result<Vec<_>>>()?;
1132
1133    let mappings = program_node
1134        .children
1135        .iter()
1136        .filter(|child| matches!(child.kind, SyntaxKind::Mapping))
1137        .map(|child| {
1138            let mapping = to_mapping(child, builder, handler)?;
1139            Ok((mapping.identifier.name, mapping))
1140        })
1141        .collect::<Result<Vec<_>>>()?;
1142
1143    // This follows the behavior of the old parser - if multiple constructors are
1144    // present, we silently throw out all but the last. Probably this should be
1145    // changed but it would theoretically be a breaking change.
1146    let mut constructors = program_node
1147        .children
1148        .iter()
1149        .filter(|child| matches!(child.kind, SyntaxKind::Constructor))
1150        .map(|child| to_constructor(child, builder, handler))
1151        .collect::<Result<Vec<_>>>()?;
1152
1153    if let Some(extra) = constructors.get(1) {
1154        return Err(TypeCheckerError::custom("A program can only have one constructor.", extra.span).into());
1155    }
1156
1157    let program_id_node = &program_node.children[1];
1158    let program_name_text = program_id_node.text.strip_suffix(".aleo").unwrap();
1159    let program_name_symbol = Symbol::intern(program_name_text);
1160    let hi = program_id_node.span.lo + program_name_text.len() as u32;
1161    let program_id = leo_ast::ProgramId {
1162        name: leo_ast::Identifier {
1163            name: program_name_symbol,
1164            span: Span { lo: program_id_node.span.lo, hi },
1165            id: builder.next_id(),
1166        },
