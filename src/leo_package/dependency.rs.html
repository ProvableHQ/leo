17use crate::{Location, NetworkName};
18
19use serde::{Deserialize, Serialize};
20use std::path::PathBuf;
21
22/// Information about a dependency, as represented in the `program.json` manifest.
23#[derive(Debug, Default, Clone, Eq, PartialEq, Hash, Serialize, Deserialize)]
24pub struct Dependency {
25    /// The name of the program. As this corresponds to what appears in `program.json`,
26    /// it should have the ".aleo" suffix.
27    pub name: String,
28    /// Network, local, or test dependency?
29    pub location: Location,
30    /// For a network dependency, which network?
31    pub network: Option<NetworkName>,
32    /// For a local dependency, where is its package? Or, for a test, where is its source file?
33    pub path: Option<PathBuf>,
34}
35
36impl Dependency {
37    pub fn new(name: String, location: Location, network: Option<NetworkName>, path: Option<PathBuf>) -> Self {
38        Self { name, location, network, path }
39    }
40}
