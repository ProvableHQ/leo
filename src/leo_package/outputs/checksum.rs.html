17//! The build checksum file.
18
19use crate::outputs::OUTPUTS_DIRECTORY_NAME;
20use leo_errors::{PackageError, Result};
21
22use serde::Deserialize;
23use std::{
24    borrow::Cow,
25    fs::{
26        File,
27        {self},
28    },
29    io::Write,
30    path::Path,
31};
32
33pub static CHECKSUM_FILE_EXTENSION: &str = ".sum";
34
35#[derive(Deserialize)]
36pub struct ChecksumFile {
37    pub package_name: String,
38}
39
40impl ChecksumFile {
41    pub fn new(package_name: &str) -> Self {
42        Self { package_name: package_name.to_string() }
43    }
44
45    pub fn exists_at(&self, path: &Path) -> bool {
46        let path = self.setup_file_path(path);
47        path.exists()
48    }
49
50    /// Reads the checksum from the given file path if it exists.
51    pub fn read_from(&self, path: &Path) -> Result<String> {
52        let path = self.setup_file_path(path);
53
54        let string =
55            fs::read_to_string(&path).map_err(|_| PackageError::failed_to_read_checksum_file(path.into_owned()))?;
56        Ok(string)
57    }
58
59    /// Writes the given checksum to a file.
60    pub fn write_to(&self, path: &Path, checksum: String) -> Result<()> {
61        let path = self.setup_file_path(path);
62        let mut file = File::create(path).map_err(PackageError::io_error_checksum_file)?;
63
64        file.write_all(checksum.as_bytes()).map_err(PackageError::io_error_checksum_file)?;
65        Ok(())
66    }
67
68    /// Removes the checksum at the given path if it exists. Returns `true` on success,
69    /// `false` if the file doesn't exist, and `Error` if the file system fails during operation.
70    pub fn remove(&self, path: &Path) -> Result<bool> {
71        let path = self.setup_file_path(path);
72        if !path.exists() {
73            return Ok(false);
74        }
75
76        fs::remove_file(&path).map_err(|_| PackageError::failed_to_remove_checksum_file(path.into_owned()))?;
77        Ok(true)
78    }
79
80    fn setup_file_path<'a>(&self, path: &'a Path) -> Cow<'a, Path> {
81        let mut path = Cow::from(path);
82        if path.is_dir() {
83            if !path.ends_with(OUTPUTS_DIRECTORY_NAME) {
84                path.to_mut().push(OUTPUTS_DIRECTORY_NAME);
85            }
86            path.to_mut().push(format!("{}{CHECKSUM_FILE_EXTENSION}", self.package_name));
87        }
88        path
89    }
90}
