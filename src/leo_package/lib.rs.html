17//! This crate deals with Leo packages on the file system and network.
18//!
19//! The main type is `Package`, which deals with Leo packages on the local filesystem.
20//! A Leo package directory is intended to have a structure like this:
21//! .
22//! ├── .env
23//! ├── program.json
24//! ├── build
25//! │   ├── imports
26//! │   │   └── credits.aleo
27//! │   └── main.aleo
28//! ├── outputs
29//! │   ├── program.TypeChecking.ast
30//! │   └── program.TypeChecking.json
31//! └── src
32//!     └── main.leo
33//!
34//! The file `program.json` is a manifest containing the program name, version, description,
35//! and license, together with information about its dependencies.
36//!
37//! The file `.env` contains definitions for the environment variables NETWORK, PRIVATE_KEY,
38//! and ENDPOINT that may be used when deploying or executing the program.
39//!
40//! Such a directory structure, together with a `.gitignore` file, may be created
41//! on the file system using `Package::initialize`.
42//! ```no_run
43//! # use leo_package::{NetworkName, Package};
44//! let path = Package::initialize("my_package", "path/to/parent", NetworkName::TestnetV0, "http://localhost:3030").unwrap();
45//! ```
46//!
47//! Given an existing directory with such a structure, a `Package` may be created from it with
48//! `Package::from_directory`:
49//! ```no_run
50//! # use leo_package::Package;
51//! let package = Package::from_directory("path/to/package", "/home/me/.aleo").unwrap();
52//! ```
53//! This will read the manifest and env file and keep their data in `package.manifest` and `package.env`.
54//! It will also process dependencies and store them in topological order in `package.programs`. This processing
55//! will involve fetching bytecode from the network for network dependencies.
56//!
57//! If you want to simply read the manifest and env file without processing dependencies, use
58//! `Package::from_directory_no_graph`.
59//!
60//! `Program` generally doesn't need to be created directly, as `Package` will create `Program`s
61//! for the main program and all dependencies. However, if you'd like to fetch bytecode for
62//! a program, you can use `Program::fetch`.

103fn symbol(name: &str) -> Result<Symbol> {
104    name.strip_suffix(".aleo").map(Symbol::intern).ok_or_else(|| PackageError::invalid_network_name(name).into())
105}

107/// Is this a valid name for an Aleo program?
108///
109/// Namely, it must be of the format "xxx.aleo" where `xxx` is nonempty,
110/// consist solely of ASCII alphanumeric characters and underscore, and
111/// begin with a letter.
112pub fn is_valid_aleo_name(name: &str) -> bool {
113    let Some(rest) = name.strip_suffix(".aleo") else {
114        return false;
115    };
116
117    // Check that the name is nonempty.
118    if rest.is_empty() {
119        tracing::error!("Aleo names must be nonempty");
120        return false;
121    }
122
123    let first = rest.chars().next().unwrap();
124
125    // Check that the first character is not an underscore.
126    if first == '_' {
127        tracing::error!("Aleo names cannot begin with an underscore");
128        return false;
129    }
130
131    // Check that the first character is not a number.
132    if first.is_numeric() {
133        tracing::error!("Aleo names cannot begin with a number");
134        return false;
135    }
136
137    // Iterate and check that the name is valid.
138    if rest.chars().any(|c| !c.is_ascii_alphanumeric() && c != '_') {
139        tracing::error!("Aleo names must can only contain ASCII alphanumeric characters and underscores.");
140        return false;
141    }
142
143    true
144}

146// Fetch the given endpoint url and return the sanitized response.
147pub fn fetch_from_network(url: &str) -> Result<String, UtilError> {
148    let response = ureq::AgentBuilder::new()
149        .redirects(0)
150        .build()
151        .get(url)
152        .set("X-Leo-Version", env!("CARGO_PKG_VERSION"))
153        .call()
154        .map_err(|err| UtilError::failed_to_retrieve_from_endpoint(err, Default::default()))?;
155    match response.status() {
156        200 => Ok(response.into_string().unwrap().replace("\\n", "\n").replace('\"', "")),
157        301 => Err(UtilError::endpoint_moved_error(url)),
158        _ => Err(UtilError::network_error(url, response.status(), Default::default())),
159    }
160}

162// Verify that a fetched program is valid aleo instructions.
163pub fn verify_valid_program(name: &str, program: &str) -> Result<(), UtilError> {
164    use snarkvm::prelude::{Program, TestnetV0};
165    use std::str::FromStr as _;
166    match Program::<TestnetV0>::from_str(program) {
167        Ok(_) => Ok(()),
168        Err(_) => Err(UtilError::snarkvm_parsing_error(name)),
169    }
