17//! The compiler for Leo programs.
18//!
19//! The [`Compiler`] type compiles Leo programs into R1CS circuits.

37/// The primary entry point of the Leo compiler.
38pub struct Compiler<N: Network> {
39    /// The path to where the compiler outputs all generated files.
40    output_directory: PathBuf,
41    /// The program name,
42    pub program_name: String,
43    /// Options configuring compilation.
44    compiler_options: CompilerOptions,
45    /// State.
46    state: CompilerState,
47    /// The stubs for imported programs. Produced by `Retriever` module.
48    import_stubs: IndexMap<Symbol, Stub>,
49    /// How many statements were in the AST before DCE?
50    pub statements_before_dce: u32,
51    /// How many statements were in the AST after DCE?
52    pub statements_after_dce: u32,
53    // Allows the compiler to be generic over the network.
54    phantom: std::marker::PhantomData<N>,
55}
56
57impl<N: Network> Compiler<N> {
58    pub fn parse(&mut self, source: &str, filename: FileName) -> Result<()> {
59        // Register the source in the source map.
60        let source_file = with_session_globals(|s| s.source_map.new_source(source, filename));
61
62        // Use the parser to construct the abstract syntax tree (ast).
63        self.state.ast = leo_parser::parse_ast::<N>(
64            self.state.handler.clone(),
65            &self.state.node_builder,
66            &source_file.src,
67            source_file.start_pos,
68        )?;
69
70        // If the program is imported, then check that the name of its program scope matches the file name.
71        // Note that parsing enforces that there is exactly one program scope in a file.
72        // TODO: Clean up check.
73        let program_scope = self.state.ast.ast.program_scopes.values().next().unwrap();
74        self.program_name = program_scope.program_id.name.to_string();
75
76        if self.compiler_options.output.initial_ast {
77            self.write_ast_to_json("initial_ast.json")?;
78        }
79
80        Ok(())
81    }
82
83    pub fn parse_from_file(&mut self, source_file_path: impl AsRef<Path>) -> Result<()> {
84        // Load the program file.
85        let source = fs::read_to_string(&source_file_path)
86            .map_err(|e| CompilerError::file_read_error(source_file_path.as_ref().display().to_string(), e))?;
87        self.parse(&source, FileName::Real(source_file_path.as_ref().into()))
88    }
89
90    /// Returns a new Leo compiler.
91    pub fn new(
92        handler: Handler,
93        output_directory: PathBuf,
94        compiler_options: Option<CompilerOptions>,
95        import_stubs: IndexMap<Symbol, Stub>,
96    ) -> Self {
97        Self {
98            state: CompilerState { handler, ..Default::default() },
99            output_directory,
100            program_name: Default::default(),
101            compiler_options: compiler_options.unwrap_or_default(),
102            import_stubs,
103            statements_before_dce: 0,
104            statements_after_dce: 0,
105            phantom: Default::default(),
106        }
107    }
108
109    /// Runs the compiler stages.
110    pub fn intermediate_passes(&mut self) -> Result<()> {
111        SymbolTableCreation::do_pass((), &mut self.state)?;
112
113        TypeChecking::do_pass(
114            TypeCheckingInput {
115                max_array_elements: N::MAX_ARRAY_ELEMENTS,
116                max_mappings: N::MAX_MAPPINGS,
117                max_functions: N::MAX_FUNCTIONS,
118            },
119            &mut self.state,
120        )?;
121
122        StaticAnalyzing::do_pass(
123            StaticAnalyzingInput {
124                max_depth: self.compiler_options.build.conditional_block_max_depth,
125                conditional_branch_type_checking: !self.compiler_options.build.disable_conditional_branch_type_checking,
126            },
127            &mut self.state,
128        )?;
129
130        ConstPropagationAndUnrolling::do_pass((), &mut self.state)?;
131
132        SsaForming::do_pass(SsaFormingInput { rename_defs: true }, &mut self.state)?;
133
134        Destructuring::do_pass((), &mut self.state)?;
135
136        SsaForming::do_pass(SsaFormingInput { rename_defs: false }, &mut self.state)?;
137
138        Flattening::do_pass((), &mut self.state)?;
139
140        FunctionInlining::do_pass((), &mut self.state)?;
141
142        if self.compiler_options.build.dce_enabled {
143            let output = DeadCodeEliminating::do_pass((), &mut self.state)?;
144            self.statements_before_dce = output.statements_before;
145            self.statements_after_dce = output.statements_after;
146        }
147
148        Ok(())
149    }
150
151    /// Returns a compiled Leo program.
152    pub fn compile(&mut self, source: &str, filename: FileName) -> Result<String> {
153        // Parse the program.
154        self.parse(source, filename)?;
155        // Copy the dependencies specified in `program.json` into the AST.
156        self.add_import_stubs()?;
157        // Run the intermediate compiler stages.
158        self.intermediate_passes()?;
159        // Run code generation.
160        let bytecode = CodeGenerating::do_pass((), &mut self.state)?;
161        Ok(bytecode)
162    }
163
164    pub fn compile_from_file(&mut self, source_file_path: impl AsRef<Path>) -> Result<String> {
165        let source = fs::read_to_string(&source_file_path)
166            .map_err(|e| CompilerError::file_read_error(source_file_path.as_ref().display().to_string(), e))?;
167        self.compile(&source, FileName::Real(source_file_path.as_ref().into()))
168    }
169
170    /// Writes the AST to a JSON file.
171    fn write_ast_to_json(&self, file_suffix: &str) -> Result<()> {
172        // Remove `Span`s if they are not enabled.
173        if self.compiler_options.output.ast_spans_enabled {
174            self.state
175                .ast
176                .to_json_file(self.output_directory.clone(), &format!("{}.{file_suffix}", self.program_name))?;
177        } else {
178            self.state.ast.to_json_file_without_keys(
179                self.output_directory.clone(),
180                &format!("{}.{file_suffix}", self.program_name),
181                &["_span", "span"],
182            )?;
183        }
184        Ok(())
185    }
186
187    /// Merges the dependencies defined in `program.json` with the dependencies imported in `.leo` file
188    pub fn add_import_stubs(&mut self) -> Result<()> {
189        // Create a list of both the explicit dependencies specified in the `.leo` file, as well as the implicit ones derived from those dependencies.
190        let (mut unexplored, mut explored): (IndexSet<Symbol>, IndexSet<Symbol>) =
191            (self.state.ast.ast.imports.keys().cloned().collect(), IndexSet::new());
192        while !unexplored.is_empty() {
193            let mut current_dependencies: IndexSet<Symbol> = IndexSet::new();
194            for program_name in unexplored.iter() {
195                if let Some(stub) = self.import_stubs.get(program_name) {
196                    // Add the program to the explored set
197                    explored.insert(*program_name);
198                    for dependency in stub.imports.iter() {
199                        // If dependency is already explored then don't need to re-explore it
200                        if explored.insert(dependency.name.name) {
201                            current_dependencies.insert(dependency.name.name);
202                        }
203                    }
204                } else {
205                    return Err(CompilerError::imported_program_not_found(
206                        self.program_name.clone(),
207                        *program_name,
208                        self.state.ast.ast.imports[program_name].1,
209                    )
210                    .into());
211                }
212            }
213
214            // Create next batch to explore
215            unexplored = current_dependencies;
216        }
217
218        // Combine the dependencies from `program.json` and `.leo` file while preserving the post-order
219        self.state.ast.ast.stubs =
220            self.import_stubs.clone().into_iter().filter(|(program_name, _)| explored.contains(program_name)).collect();
221        Ok(())
222    }
223}
