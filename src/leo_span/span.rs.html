17//! Defines the `Span` type used to track where code comes from.
18
19use core::ops::{Add, Sub};
20use serde::{Deserialize, Serialize};
21use std::fmt;
22
23use crate::symbol::with_session_globals;
24
25/// The span type which tracks where formatted errors originate from in a Leo file.
26/// This is used in many spots throughout the rest of the Leo crates.
27#[derive(Copy, Clone, Debug, Default, Deserialize, Eq, Hash, PartialEq, Serialize)]
28pub struct Span {
29    /// The start (low) position of the span, inclusive.
30    pub lo: BytePos,
31    /// The end (high) position of the span, exclusive.
32    /// The length of the span is `hi - lo`.
33    pub hi: BytePos,
34}
35
36impl Span {
37    /// Generate a new span from the `start`ing and `end`ing positions.
38    pub fn new(start: BytePos, end: BytePos) -> Self {
39        Self { lo: start, hi: end }
40    }
41
42    /// Generates a dummy span with all defaults.
43    /// Should only be used in temporary situations.
44    pub const fn dummy() -> Self {
45        Self { lo: BytePos(0), hi: BytePos(0) }
46    }
47
48    /// Is the span a dummy?
49    pub fn is_dummy(&self) -> bool {
50        self == &Self::dummy()
51    }
52}
53
54impl fmt::Display for Span {
55    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
56        with_session_globals(|s| write!(f, "{}", s.source_map.span_to_string(*self)))
57    }
58}
59
60impl std::ops::Add for &Span {
61    type Output = Span;
62
63    /// Add two spans (by reference) together.
64    fn add(self, other: &Span) -> Span {
65        *self + *other
66    }
67}
68
69impl std::ops::Add for Span {
70    type Output = Self;
71
72    /// Add two spans together.
73    /// The resulting span is the smallest span that includes both.
74    fn add(self, other: Self) -> Self {
75        let lo = self.lo.min(other.lo);
76        let hi = self.hi.max(other.hi);
77        Self::new(lo, hi)
78    }
79}

85/// Offsets (i.e. positions), in some units (e.g. bytes or characters),
86/// with conversions between unsigned integers.
87pub trait Pos {
88    fn from_usize(n: usize) -> Self;
89    fn to_usize(&self) -> usize;
90    fn from_u32(n: u32) -> Self;
91    fn to_u32(&self) -> u32;
92}
93
94/// Generate one-component tuple structs that implement the [`Pos`] trait.
95macro_rules! impl_pos {
96    (
97        $(
98            $(#[$attr:meta])*
99            $vis:vis struct $ident:ident($inner_vis:vis $inner_ty:ty);
100        )*
101    ) => {
102        $(
103            $(#[$attr])*
104            $vis struct $ident($inner_vis $inner_ty);
105
106            impl Pos for $ident {
107                #[inline(always)]
108                fn from_usize(n: usize) -> $ident {
109                    $ident(n as $inner_ty)
110                }
111
112                #[inline(always)]
113                fn to_usize(&self) -> usize {
114                    self.0 as usize
115                }
116
117                #[inline(always)]
118                fn from_u32(n: u32) -> $ident {
119                    $ident(n as $inner_ty)
120                }
121
122                #[inline(always)]
123                fn to_u32(&self) -> u32 {
124                    self.0 as u32
125                }
126            }
127
128            impl Add for $ident {
129                type Output = $ident;
130
131                #[inline(always)]
132                fn add(self, rhs: $ident) -> $ident {
133                    $ident(self.0 + rhs.0)
134                }
135            }
136
137            impl Sub for $ident {
138                type Output = $ident;
139
140                #[inline(always)]
141                fn sub(self, rhs: $ident) -> $ident {
142                    $ident(self.0 - rhs.0)
143                }
144            }
145        )*
146    };
147}
148
149impl_pos! {
150    /// A byte offset.
151    #[derive(Clone, Copy, PartialEq, Eq, Hash, PartialOrd, Ord, Debug, Serialize, Deserialize, Default)]
152    pub struct BytePos(pub u32);
153
154    /// A character offset.
155    ///
156    /// Because of multibyte UTF-8 characters,
157    /// a byte offset is not equivalent to a character offset.
158    /// The [`SourceMap`] will convert [`BytePos`] values to `CharPos` values as necessary.
159    #[derive(Clone, Copy, PartialEq, Eq, PartialOrd, Ord, Debug)]
160    pub struct CharPos(pub usize);
161}
