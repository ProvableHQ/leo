17//! The source map provides an address space for positions in spans
18//! that is global across the source files that are compiled together.
19//! The source files are organized in a sequence,
20//! with the positions of each source following the ones of the previous source
21//! in the address space of positions
22//! (except for the first source, which starts at the beginning of the address space).
23//! This way, any place in any source is identified by a single position
24//! within the address space covered by the sequence of sources;
25//! the source file is determined from the position.
26
27use crate::span::{BytePos, CharPos, Pos, Span};
28use std::{
29    cell::RefCell,
30    fmt,
31    fs,
32    io,
33    path::{Path, PathBuf},
34    rc::Rc,
35};
36
37/// The source map containing all recorded sources,
38/// methods to register new ones,
39/// and methods to query about spans in relation to recorded sources.
40#[derive(Default)]
41pub struct SourceMap {
42    /// The actual source map data.
43    inner: RefCell<SourceMapInner>,
44}
45
46/// Actual data of the source map.
47/// We use this setup for purposes of interior mutability.
48#[derive(Default)]
49struct SourceMapInner {
50    /// The address space below this value is currently used by the files in the source map.
51    used_address_space: u32,
52
53    /// All the source files recorded thus far.
54    ///
55    /// The list is append-only with mappings from the start byte position
56    /// for fast lookup from a `Span` to its `SourceFile`.
57    source_files: Vec<Rc<SourceFile>>,
58}
59
60impl SourceMap {
61    /// Loads the given `path` and returns a `SourceFile` for it.
62    pub fn load_file(&self, path: &Path) -> io::Result<Rc<SourceFile>> {
63        Ok(self.new_source(&fs::read_to_string(path)?, FileName::Real(path.to_owned())))
64    }
65
66    /// Registers `source` under the given file `name`, returning a `SourceFile` back.
67    pub fn new_source(&self, source: &str, name: FileName) -> Rc<SourceFile> {
68        let len = u32::try_from(source.len()).unwrap();
69        let mut inner = self.inner.borrow_mut();
70        let start_pos = inner.try_allocate_address_space(len).unwrap();
71        let source_file = Rc::new(SourceFile::new(name, source.to_owned(), start_pos));
72        inner.source_files.push(source_file.clone());
73        source_file
74    }
75
76    /// Find the index for the source file containing `pos`.
77    fn find_source_file_index(&self, pos: BytePos) -> Option<usize> {
78        self.inner
79            .borrow()
80            .source_files
81            .binary_search_by_key(&pos, |file| file.start_pos)
82            .map_or_else(|p| p.checked_sub(1), Some)
83    }
84
85    /// Find the source file containing `pos`.
86    pub fn find_source_file(&self, pos: BytePos) -> Option<Rc<SourceFile>> {
87        Some(self.inner.borrow().source_files[self.find_source_file_index(pos)?].clone())
88    }
89
90    /// Finds line column info about a given `pos`.
91    fn find_line_col(&self, pos: BytePos) -> Option<LineCol> {
92        let source_file = self.find_source_file(pos)?;
93        let (line, col) = source_file.lookup_file_pos(pos);
94        Some(LineCol { source_file, line, col })
95    }
96
97    /// Retrieves the location (source file, lines, columns) on the given span.
98    pub fn span_to_location(&self, sp: Span) -> Option<SpanLocation> {
99        let lo = self.find_line_col(sp.lo)?;
100        let hi = self.find_line_col(sp.hi)?;
101        Some(SpanLocation {
102            source_file: lo.source_file,
103            line_start: lo.line,
104            line_stop: hi.line,
105            col_start: lo.col.to_usize() + 1,
106            col_stop: hi.col.to_usize() + 1,
107        })
108    }
109
110    /// Returns a displayable representation of the `span` as a string.
111    pub fn span_to_string(&self, span: Span) -> String {
112        let loc = match self.span_to_location(span) {
113            None => return "no-location".to_string(),
114            Some(l) => l,
115        };
116
117        if loc.line_start == loc.line_stop {
118            format!("{}:{}-{}", loc.line_start, loc.col_start, loc.col_stop)
119        } else {
120            format!("{}:{}-{}:{}", loc.line_start, loc.col_start, loc.line_stop, loc.col_stop)
121        }
122    }
123
124    /// Returns the source contents that is spanned by `span`.
125    pub fn contents_of_span(&self, span: Span) -> Option<String> {
126        let begin = self.find_source_file(span.lo)?;
127        let end = self.find_source_file(span.hi)?;
128        assert_eq!(begin.start_pos, end.start_pos);
129        Some(begin.contents_of_span(span))
130    }
131
132    /// Returns the source contents of the lines that `span` is within.
133    ///
134    /// That is, if the span refers to `x = 4` in the source code:
135    ///
136    /// > ```text
137    /// > // Line 1
138    /// > let x
139    /// >     = 4;
140    /// > // Line 4
141    /// > ```
142    ///
143    /// then the contents on lines 2 and 3 are returned.
144    pub fn line_contents_of_span(&self, span: Span) -> Option<String> {
145        let begin = self.find_source_file(span.lo)?;
146        let end = self.find_source_file(span.hi)?;
147        assert_eq!(begin.start_pos, end.start_pos);
148
149        let idx_lo = begin.lookup_line(span.lo).unwrap_or(0);
150        let idx_hi = begin.lookup_line(span.hi).unwrap_or(0) + 1;
151        let lo_line_pos = begin.lines[idx_lo];
152        let hi_line_pos = if idx_hi < begin.lines.len() { begin.lines[idx_hi] } else { begin.end_pos };
153        Some(begin.contents_of_span(Span::new(lo_line_pos, hi_line_pos)))
154    }
155}
156
157impl SourceMapInner {
158    /// Attempt reserving address space for `size` number of bytes.
159    fn try_allocate_address_space(&mut self, size: u32) -> Option<BytePos> {
160        let current = self.used_address_space;
161        // By adding one, we can distinguish between files, even when they are empty.
162        self.used_address_space = current.checked_add(size)?.checked_add(1)?;
163        Some(BytePos(current))
164    }
165}

167/// A file name.
168///
169/// This is either a wrapper around `PathBuf`,
170/// or a custom string description.
171#[derive(Clone)]
172pub enum FileName {
173    /// A real file.
174    Real(PathBuf),
175    /// Any sort of description for a source.
176    Custom(String),
177}
178
179impl fmt::Display for FileName {
180    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
181        match self {
182            Self::Real(x) if is_color() => x.display().fmt(f),
183            Self::Real(_) => Ok(()),
184            Self::Custom(x) => f.write_str(x),
185        }
186    }
187}

194/// A single source in the [`SourceMap`].
195pub struct SourceFile {
196    /// The name of the file that the source came from.
197    pub name: FileName,
198    /// The complete source code.
199    pub src: String,
200    /// The start position of this source in the `SourceMap`.
201    pub start_pos: BytePos,
202    /// The end position of this source in the `SourceMap`.
203    pub end_pos: BytePos,
204    /// Locations of line beginnings in the source code.
205    lines: Vec<BytePos>,
206    /// Locations of multi-byte characters in the source code.
207    multibyte_chars: Vec<MultiByteChar>,
208}
209
210impl SourceFile {
211    /// Creates a new `SourceFile` given the file `name`,
212    /// source contents, and the `start_pos`ition.
213    ///
214    /// This position is used for analysis purposes.
215    fn new(name: FileName, mut src: String, start_pos: BytePos) -> Self {
216        normalize_src(&mut src);
217        let end_pos = start_pos + BytePos::from_usize(src.len());
218        let (lines, multibyte_chars) = analyze_source_file(&src, start_pos);
219        Self { name, src, start_pos, end_pos, lines, multibyte_chars }
220    }
221
222    /// Converts an absolute `BytePos` to a `CharPos` relative to the `SourceFile`.
223    fn bytepos_to_file_charpos(&self, bpos: BytePos) -> CharPos {
224        // The number of extra bytes due to multibyte chars in the `SourceFile`.
225        let mut total_extra_bytes = 0;
226
227        for mbc in self.multibyte_chars.iter() {
228            if mbc.pos < bpos {
229                // Every character is at least one byte, so we only
230                // count the actual extra bytes.
231                total_extra_bytes += mbc.bytes as u32 - 1;
232                // We should never see a byte position in the middle of a
233                // character.
234                assert!(bpos.to_u32() >= mbc.pos.to_u32() + mbc.bytes as u32);
235            } else {
236                break;
237            }
238        }
239
240        assert!(self.start_pos.to_u32() + total_extra_bytes <= bpos.to_u32());
241        CharPos(bpos.to_usize() - self.start_pos.to_usize() - total_extra_bytes as usize)
242    }
243
244    /// Finds the line containing the given position. The return value is the
245    /// index into the `lines` array of this `SourceFile`, not the 1-based line
246    /// number. If the source file is empty or the position is located before the
247    /// first line, `None` is returned.
248    fn lookup_line(&self, pos: BytePos) -> Option<usize> {
249        match self.lines.binary_search(&pos) {
250            Ok(idx) => Some(idx),
251            Err(0) => None,
252            Err(idx) => Some(idx - 1),
253        }
254    }
255
256    /// Looks up the file's (1-based) line number and (0-based `CharPos`) column offset, for a
257    /// given `BytePos`.
258    fn lookup_file_pos(&self, pos: BytePos) -> (usize, CharPos) {
259        let chpos = self.bytepos_to_file_charpos(pos);
260        match self.lookup_line(pos) {
261            Some(a) => {
262                let line = a + 1; // Line numbers start at 1
263                let linebpos = self.lines[a];
264                let linechpos = self.bytepos_to_file_charpos(linebpos);
265                let col = chpos - linechpos;
266                assert!(chpos >= linechpos);
267                (line, col)
268            }
269            None => (0, chpos),
270        }
271    }
272
273    /// Returns contents of a `span` assumed to be within the given file.
274    fn contents_of_span(&self, span: Span) -> String {
275        let begin_pos = self.bytepos_to_file_charpos(span.lo).to_usize();
276        let end_pos = self.bytepos_to_file_charpos(span.hi).to_usize();
277        String::from_utf8_lossy(&self.src.as_bytes()[begin_pos..end_pos]).into_owned()
278    }
279}

322/// Normalizes the source code and records the normalizations.
323fn normalize_src(src: &mut String) {
324    remove_bom(src);
325    normalize_newlines(src);
326}
327
328/// Removes UTF-8 BOM, if any.
329fn remove_bom(src: &mut String) {
330    if src.starts_with('\u{feff}') {
331        src.drain(..3);
332    }
333}
334
335/// Replaces `\r\n` with `\n` in-place in `src`.
336///
337/// Isolated carriage returns are left alone.
338fn normalize_newlines(src: &mut String) {
339    if !src.as_bytes().contains(&b'\r') {
340        return;
341    }
342
343    // We replace `\r\n` with `\n` in-place, which doesn't break utf-8 encoding.
344    // While we *can* call `as_mut_vec` and do surgery on the live string
345    // directly, let's rather steal the contents of `src`. This makes the code
346    // safe even if a panic occurs.
347
348    let mut buf = std::mem::take(src).into_bytes();
349    let mut gap_len = 0;
350    let mut tail = buf.as_mut_slice();
351    loop {
352        let idx = match find_crlf(&tail[gap_len..]) {
353            None => tail.len(),
354            Some(idx) => idx + gap_len,
355        };
356        tail.copy_within(gap_len..idx, 0);
357        tail = &mut tail[idx - gap_len..];
358        if tail.len() == gap_len {
359            break;
360        }
361        gap_len += 1;
362    }
363
364    // Account for removed `\r`.
365    // After `buf.truncate(..)`, `buf` is guaranteed to contain utf-8 again.
366    let new_len = buf.len() - gap_len;
367    buf.truncate(new_len);
368    *src = String::from_utf8(buf).unwrap();
369
370    fn find_crlf(src: &[u8]) -> Option<usize> {
371        let mut search_idx = 0;
372        while let Some(idx) = find_cr(&src[search_idx..]) {
373            if src[search_idx..].get(idx + 1) != Some(&b'\n') {
374                search_idx += idx + 1;
375                continue;
376            }
377            return Some(search_idx + idx);
378        }
379        None
380    }
381
382    fn find_cr(src: &[u8]) -> Option<usize> {
383        src.iter().position(|&b| b == b'\r')
384    }
385}

387/// Identifies an offset of a multi-byte character in a `SourceFile`.
388#[derive(Copy, Clone, Eq, PartialEq, Debug)]
389struct MultiByteChar {
390    /// The absolute offset of the character in the `SourceMap`.
391    pub pos: BytePos,
392    /// The number of bytes, `>= 2`.
393    pub bytes: u8,
394}
395
396/// Finds all newlines, multi-byte characters, and non-narrow characters in a
397/// SourceFile.
398///
399/// This function will use an SSE2 enhanced implementation if hardware support
400/// is detected at runtime.
401fn analyze_source_file(src: &str, source_file_start_pos: BytePos) -> (Vec<BytePos>, Vec<MultiByteChar>) {
402    let mut lines = vec![source_file_start_pos];
403    let mut multi_byte_chars = vec![];
404
405    let mut i = 0;
406    let src_bytes = src.as_bytes();
407
408    while i < src.len() {
409        let byte = src_bytes[i];
410
411        // How much to advance to get to the next UTF-8 char in the string.
412        let mut char_len = 1;
413
414        let pos = BytePos::from_usize(i) + source_file_start_pos;
415
416        if let b'\n' = byte {
417            lines.push(pos + BytePos(1));
418        } else if byte >= 127 {
419            // The slow path:
420            // This is either ASCII control character "DEL" or the beginning of
421            // a multibyte char. Just decode to `char`.
422            let c = (src[i..]).chars().next().unwrap();
423            char_len = c.len_utf8();
424
425            if char_len > 1 {
426                assert!((2..=4).contains(&char_len));
427                let bytes = char_len as u8;
428                let mbc = MultiByteChar { pos, bytes };
429                multi_byte_chars.push(mbc);
430            }
431        }
432
433        i += char_len;
434    }
435
436    // The code above optimistically registers a new line *after* each \n it encounters.
437    // If that point is already outside the source_file, remove it again.
438    if let Some(&last_line_start) = lines.last() {
439        let source_file_end = source_file_start_pos + BytePos::from_usize(src.len());
440        assert!(source_file_end >= last_line_start);
441        if last_line_start == source_file_end {
442            lines.pop();
443        }
444    }
445
446    (lines, multi_byte_chars)
447}
