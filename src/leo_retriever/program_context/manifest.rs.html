17use crate::Dependency;
18use leo_errors::PackageError;
19use serde::{Deserialize, Serialize};
20use std::path::Path;
21
22// Struct representation of program's `program.json` specification
23#[derive(Debug, Clone, Serialize, Deserialize)]
24pub struct Manifest {
25    program: String,
26    version: String,
27    description: String,
28    license: String,
29    dependencies: Option<Vec<Dependency>>,
30}
31
32impl Manifest {
33    pub fn new(
34        program: &str,
35        version: &str,
36        description: &str,
37        license: &str,
38        dependencies: Option<Vec<Dependency>>,
39    ) -> Self {
40        Self {
41            program: program.to_owned(),
42            version: version.to_owned(),
43            description: description.to_owned(),
44            license: license.to_owned(),
45            dependencies,
46        }
47    }
48
49    pub fn default(program: &str) -> Self {
50        Self {
51            program: format!("{program}.aleo"),
52            version: "0.1.0".to_owned(),
53            description: "".to_owned(),
54            license: "MIT".to_owned(),
55            dependencies: None,
56        }
57    }
58
59    pub fn program(&self) -> &String {
60        &self.program
61    }
62
63    pub fn version(&self) -> &String {
64        &self.version
65    }
66
67    pub fn description(&self) -> &String {
68        &self.description
69    }
70
71    pub fn license(&self) -> &String {
72        &self.license
73    }
74
75    pub fn dependencies(&self) -> &Option<Vec<Dependency>> {
76        &self.dependencies
77    }
78
79    pub fn write_to_dir(&self, path: &Path) -> Result<(), PackageError> {
80        // Serialize the manifest to a JSON string.
81        let contents = serde_json::to_string_pretty(&self)
82            .map_err(|err| PackageError::failed_to_serialize_manifest_file(path.to_str().unwrap(), err))?;
83        // Write the manifest to the file.
84        std::fs::write(path.join("program.json"), contents).map_err(PackageError::failed_to_write_manifest)
85    }
86
87    pub fn read_from_dir(path: &Path) -> Result<Self, PackageError> {
88        // Read the manifest file.
89        let contents = std::fs::read_to_string(path.join("program.json"))
90            .map_err(|_| PackageError::failed_to_load_package(path.to_str().unwrap()))?;
91        // Deserialize the manifest.
92        serde_json::from_str(&contents)
93            .map_err(|err| PackageError::failed_to_deserialize_manifest_file(path.to_str().unwrap(), err))
94    }
95}
