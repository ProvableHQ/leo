use serde::{Deserialize, Serialize};

// Retrievable locations for an external program
#[derive(Debug, Clone, std::cmp::Eq, PartialEq, Hash, Serialize, Deserialize)]
pub enum Location {
    #[serde(rename = "network")]
    Network,
    #[serde(rename = "local")]
    Local,
    #[serde(rename = "git")]
    Git,
}
