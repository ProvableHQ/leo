35/// A pass to perform const propagation and folding.
36///
37/// This pass should be used in conjunction with the Unroller so that
38/// loop bounds and consts in loop bodies can be evaluated.
39///
40/// Any of these expressions:
41/// 1. unary operation,
42/// 2. binary operation,
43/// 3. core function other than cheat codes, mapping ops, or rand functions,
44///
45/// whose arguments are consts or literals will be subject to constant folding.
46/// The ternary conditional operator will also be folded if its condition is
47/// a constant or literal.
48pub struct ConstPropagator<'a> {
49    /// The symbol table associated with the program.
50    pub(crate) symbol_table: &'a mut SymbolTable,
51    /// A mapping between node IDs and their types.
52    pub(crate) type_table: &'a TypeTable,
53    /// A counter used to generate unique node IDs.
54    pub(crate) node_builder: &'a NodeBuilder,
55    /// The error handler.
56    pub(crate) handler: &'a Handler,
57    /// The program name.
58    pub(crate) program: Symbol,
59    /// Have we actually modified the progam at all?
60    pub(crate) changed: bool,
61    /// The RHS of a const declaration we were not able to evaluate.
62    pub(crate) const_not_evaluated: Option<Span>,
63    /// An array index which was not able to be evaluated.
64    pub(crate) array_index_not_evaluated: Option<Span>,
65}
66
67impl<'a> ConstPropagator<'a> {
68    pub(crate) fn new(
69        handler: &'a Handler,
70        symbol_table: &'a mut SymbolTable,
71        type_table: &'a TypeTable,
72        node_builder: &'a NodeBuilder,
73    ) -> Self {
74        Self {
75            handler,
76            symbol_table,
77            type_table,
78            node_builder,
79            program: Symbol::intern(""),
80            changed: false,
81            const_not_evaluated: None,
82            array_index_not_evaluated: None,
83        }
84    }
85
86    /// Enter the symbol table's scope `id`, execute `func`, and then return to the parent scope.
87    pub(crate) fn in_scope<T>(&mut self, id: NodeID, func: impl FnOnce(&mut Self) -> T) -> T {
88        self.symbol_table.enter_scope(Some(id));
89        let result = func(self);
90        self.symbol_table.enter_parent();
91        result
92    }
93
94    /// Emit a `StaticAnalyzerError`.
95    pub(crate) fn emit_err(&self, err: StaticAnalyzerError) {
96        self.handler.emit_err(err);
97    }
98}

100pub(crate) fn value_to_expression(value: &Value, span: Span, node_builder: &NodeBuilder) -> Option<Expression> {
101    use Value::*;
102    let id = node_builder.next_id();
103
104    let result = match value {
105        Unit => Expression::Unit(leo_ast::UnitExpression { span, id }),
106        Bool(x) => Expression::Literal(Literal::Boolean(*x, span, id)),
107        U8(x) => Expression::Literal(Literal::Integer(IntegerType::U8, format!("{x}"), span, id)),
108        U16(x) => Expression::Literal(Literal::Integer(IntegerType::U16, format!("{x}"), span, id)),
109        U32(x) => Expression::Literal(Literal::Integer(IntegerType::U32, format!("{x}"), span, id)),
110        U64(x) => Expression::Literal(Literal::Integer(IntegerType::U64, format!("{x}"), span, id)),
111        U128(x) => Expression::Literal(Literal::Integer(IntegerType::U128, format!("{x}"), span, id)),
112        I8(x) => Expression::Literal(Literal::Integer(IntegerType::I8, format!("{x}"), span, id)),
113        I16(x) => Expression::Literal(Literal::Integer(IntegerType::I16, format!("{x}"), span, id)),
114        I32(x) => Expression::Literal(Literal::Integer(IntegerType::I32, format!("{x}"), span, id)),
115        I64(x) => Expression::Literal(Literal::Integer(IntegerType::I64, format!("{x}"), span, id)),
116        I128(x) => Expression::Literal(Literal::Integer(IntegerType::I128, format!("{x}"), span, id)),
117        Address(x) => Expression::Literal(Literal::Address(format!("{x}"), span, id)),
118        Group(x) => {
119            let mut s = format!("{x}");
120            // Strip off the `group` suffix.
121            s.truncate(s.len() - 5);
122            Expression::Literal(Literal::Group(s, span, id))
123        }
124        Field(x) => {
125            let mut s = format!("{x}");
126            // Strip off the `field` suffix.
127            s.truncate(s.len() - 5);
128            Expression::Literal(Literal::Field(s, span, id))
129        }
130        Scalar(x) => {
131            let mut s = format!("{x}");
132            // Strip off the `scalar` suffix.
133            s.truncate(s.len() - 6);
134            Expression::Literal(Literal::Scalar(s, span, id))
135        }
136        Tuple(x) => {
137            let mut elements = Vec::with_capacity(x.len());
138            for value in x.iter() {
139                elements.push(value_to_expression(value, span, node_builder)?);
140            }
141            Expression::Tuple(TupleExpression { elements, span, id })
142        }
143        Array(x) => {
144            let mut elements = Vec::with_capacity(x.len());
145            for value in x.iter() {
146                elements.push(value_to_expression(value, span, node_builder)?);
147            }
148            Expression::Array(ArrayExpression { elements, span, id })
149        }
150        Struct(x) => Expression::Struct(StructExpression {
151            name: Identifier { name: x.name, id: node_builder.next_id(), span },
152            members: {
153                let mut members = Vec::with_capacity(x.contents.len());
154                for (name, val) in x.contents.iter() {
155                    let initializer = StructVariableInitializer {
156                        identifier: Identifier { name: *name, id: node_builder.next_id(), span },
157                        expression: Some(value_to_expression(val, span, node_builder)?),
158                        span,
159                        id: node_builder.next_id(),
160                    };
161                    members.push(initializer)
162                }
163                members
164            },
165            span,
166            id,
167        }),
168        Future(..) => return None,
169    };
170
171    Some(result)
172}
