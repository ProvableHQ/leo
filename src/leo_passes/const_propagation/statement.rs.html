17use super::ConstPropagationVisitor;
18
19use leo_ast::{
20    AssertStatement,
21    AssertVariant,
22    AssignStatement,
23    Block,
24    ConditionalStatement,
25    ConstDeclaration,
26    DefinitionStatement,
27    Expression,
28    ExpressionReconstructor,
29    ExpressionStatement,
30    IterationStatement,
31    Node,
32    ReturnStatement,
33    Statement,
34    StatementReconstructor,
35};
36
37impl StatementReconstructor for ConstPropagationVisitor<'_> {
38    fn reconstruct_assert(&mut self, mut input: AssertStatement) -> (Statement, Self::AdditionalOutput) {
39        // Catching asserts at compile time is not feasible here due to control flow, but could be done in
40        // a later pass after loops are unrolled and conditionals are flattened.
41        input.variant = match input.variant {
42            AssertVariant::Assert(expr) => AssertVariant::Assert(self.reconstruct_expression(expr).0),
43
44            AssertVariant::AssertEq(lhs, rhs) => {
45                AssertVariant::AssertEq(self.reconstruct_expression(lhs).0, self.reconstruct_expression(rhs).0)
46            }
47
48            AssertVariant::AssertNeq(lhs, rhs) => {
49                AssertVariant::AssertNeq(self.reconstruct_expression(lhs).0, self.reconstruct_expression(rhs).0)
50            }
51        };
52
53        (input.into(), None)
54    }
55
56    fn reconstruct_assign(&mut self, assign: AssignStatement) -> (Statement, Self::AdditionalOutput) {
57        (AssignStatement { value: self.reconstruct_expression(assign.value).0, ..assign }.into(), None)
58    }
59
60    fn reconstruct_block(&mut self, mut block: Block) -> (Block, Self::AdditionalOutput) {
61        self.in_scope(block.id(), |slf| {
62            block.statements.retain_mut(|statement| {
63                let bogus_statement = Statement::dummy();
64                let this_statement = std::mem::replace(statement, bogus_statement);
65                *statement = slf.reconstruct_statement(this_statement).0;
66                !statement.is_empty()
67            });
68            (block, None)
69        })
70    }
71
72    fn reconstruct_conditional(
73        &mut self,
74        mut conditional: ConditionalStatement,
75    ) -> (Statement, Self::AdditionalOutput) {
76        conditional.condition = self.reconstruct_expression(conditional.condition).0;
77        conditional.then = self.reconstruct_block(conditional.then).0;
78        if let Some(mut otherwise) = conditional.otherwise {
79            *otherwise = self.reconstruct_statement(*otherwise).0;
80            conditional.otherwise = Some(otherwise);
81        }
82
83        (Statement::Conditional(conditional), None)
84    }
85
86    fn reconstruct_const(&mut self, mut input: ConstDeclaration) -> (Statement, Self::AdditionalOutput) {
87        let span = input.span();
88
89        let (expr, opt_value) = self.reconstruct_expression(input.value);
90
91        if opt_value.is_some() {
92            self.state.symbol_table.insert_const(self.program, input.place.name, expr.clone());
93        } else {
94            self.const_not_evaluated = Some(span);
95        }
96
97        input.value = expr;
98
99        (Statement::Const(input), None)
100    }
101
102    fn reconstruct_definition(&mut self, definition: DefinitionStatement) -> (Statement, Self::AdditionalOutput) {
103        (DefinitionStatement { value: self.reconstruct_expression(definition.value).0, ..definition }.into(), None)
104    }
105
106    fn reconstruct_expression_statement(
107        &mut self,
108        mut input: ExpressionStatement,
109    ) -> (Statement, Self::AdditionalOutput) {
110        input.expression = self.reconstruct_expression(input.expression).0;
111
112        if matches!(&input.expression, Expression::Unit(..) | Expression::Literal(..)) {
113            // We were able to evaluate this at compile time, but we need to get rid of this statement as
114            // we can't have expression statements that aren't calls.
115            (Statement::dummy(), Default::default())
116        } else {
117            (input.into(), Default::default())
118        }
119    }
120
121    fn reconstruct_iteration(&mut self, iteration: IterationStatement) -> (Statement, Self::AdditionalOutput) {
122        let id = iteration.id();
123        let start = self.reconstruct_expression(iteration.start).0;
124        let stop = self.reconstruct_expression(iteration.stop).0;
125        self.in_scope(id, |slf| {
126            (
127                IterationStatement { start, stop, block: slf.reconstruct_block(iteration.block).0, ..iteration }.into(),
128                None,
129            )
130        })
131    }
132
133    fn reconstruct_return(&mut self, input: ReturnStatement) -> (Statement, Self::AdditionalOutput) {
134        (
135            ReturnStatement { expression: self.reconstruct_expression(input.expression).0, ..input }.into(),
136            Default::default(),
137        )
138    }
139}
