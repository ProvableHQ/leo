32pub struct ConstPropagationOutput {
33    /// Something about the program was actually changed during the pass.
34    pub changed: bool,
35    /// A const declaration whose RHS was not able to be evaluated.
36    pub const_not_evaluated: Option<Span>,
37    /// An array index which was not able to be evaluated.
38    pub array_index_not_evaluated: Option<Span>,
39}
40
41/// A pass to perform const propagation and folding.
42///
43/// This pass should be used in conjunction with the Unroller so that
44/// loop bounds and consts in loop bodies can be evaluated.
45///
46/// Any of these expressions:
47/// 1. unary operation,
48/// 2. binary operation,
49/// 3. core functions other than cheat codes, mapping ops, or rand functions,
50///
51/// whose arguments are consts or literals will be subject to constant folding.
52/// The ternary conditional operator will also be folded if its condition is
53/// a constant or literal.
54///
55/// This includes the LHS of assignment statements which include array indices.
56pub struct ConstPropagation;
57
58impl Pass for ConstPropagation {
59    type Input = ();
60    type Output = ConstPropagationOutput;
61
62    const NAME: &str = "ConstPropagation";
63
64    fn do_pass(_input: Self::Input, state: &mut crate::CompilerState) -> Result<Self::Output> {
65        let mut ast = std::mem::take(&mut state.ast);
66        let mut visitor = ConstPropagationVisitor {
67            state,
68            program: Symbol::intern(""),
69            changed: false,
70            const_not_evaluated: None,
71            array_index_not_evaluated: None,
72        };
73        ast.ast = visitor.reconstruct_program(ast.ast);
74        visitor.state.handler.last_err().map_err(|e| *e)?;
75        visitor.state.ast = ast;
76        Ok(ConstPropagationOutput {
77            changed: visitor.changed,
78            const_not_evaluated: visitor.const_not_evaluated,
79            array_index_not_evaluated: visitor.array_index_not_evaluated,
80        })
81    }
82}
