31    /* Types */
32    fn reconstruct_array_type(&mut self, input: leo_ast::ArrayType) -> (leo_ast::Type, Self::AdditionalOutput) {
33        let (length, opt_value) = self.reconstruct_expression(*input.length);
34
35        // If we can't evaluate this array length, keep track of it for error reporting later
36        if opt_value.is_none() {
37            self.array_length_not_evaluated = Some(length.span());
38        }
39
40        (
41            leo_ast::Type::Array(leo_ast::ArrayType {
42                element_type: Box::new(self.reconstruct_type(*input.element_type).0),
43                length: Box::new(length),
44            }),
45            Default::default(),
46        )
47    }

49    /* Expressions */
50    fn reconstruct_expression(&mut self, input: Expression) -> (Expression, Self::AdditionalOutput) {
51        let old_id = input.id();
52        let (new_expr, opt_value) = match input {
53            Expression::Array(array) => self.reconstruct_array(array),
54            Expression::ArrayAccess(access) => self.reconstruct_array_access(*access),
55            Expression::AssociatedConstant(constant) => self.reconstruct_associated_constant(constant),
56            Expression::AssociatedFunction(function) => self.reconstruct_associated_function(function),
57            Expression::Binary(binary) => self.reconstruct_binary(*binary),
58            Expression::Call(call) => self.reconstruct_call(*call),
59            Expression::Cast(cast) => self.reconstruct_cast(*cast),
60            Expression::Struct(struct_) => self.reconstruct_struct_init(struct_),
61            Expression::Err(err) => self.reconstruct_err(err),
62            Expression::Identifier(identifier) => self.reconstruct_identifier(identifier),
63            Expression::Literal(value) => self.reconstruct_literal(value),
64            Expression::Locator(locator) => self.reconstruct_locator(locator),
65            Expression::MemberAccess(access) => self.reconstruct_member_access(*access),
66            Expression::Repeat(repeat) => self.reconstruct_repeat(*repeat),
67            Expression::Ternary(ternary) => self.reconstruct_ternary(*ternary),
68            Expression::Tuple(tuple) => self.reconstruct_tuple(tuple),
69            Expression::TupleAccess(access) => self.reconstruct_tuple_access(*access),
70            Expression::Unary(unary) => self.reconstruct_unary(*unary),
71            Expression::Unit(unit) => self.reconstruct_unit(unit),
72        };
73
74        if old_id != new_expr.id() {
75            self.changed = true;
76            let old_type =
77                self.state.type_table.get(&old_id).expect("Type checking guarantees that all expressions have a type.");
78            self.state.type_table.insert(new_expr.id(), old_type);
79        }
80
81        (new_expr, opt_value)
82    }

84    fn reconstruct_struct_init(&mut self, mut input: StructExpression) -> (Expression, Self::AdditionalOutput) {
85        let mut values = Vec::new();
86        input.const_arguments.iter_mut().for_each(|arg| {
87            *arg = self.reconstruct_expression(std::mem::take(arg)).0;
88        });
89        for member in input.members.iter_mut() {
90            if let Some(expr) = std::mem::take(&mut member.expression) {
91                let (new_expr, value_opt) = self.reconstruct_expression(expr);
92                member.expression = Some(new_expr);
93                if let Some(value) = value_opt {
94                    values.push(value);
95                }
96            }
97        }
98
99        if values.len() == input.members.len() && input.const_arguments.is_empty() {
100            let value = Value::Struct(StructContents {
101                name: input.name.name,
102                contents: input.members.iter().map(|mem| mem.identifier.name).zip(values).collect(),
103            });
104            (input.into(), Some(value))
105        } else {
106            (input.into(), None)
107        }
108    }

110    fn reconstruct_ternary(&mut self, input: TernaryExpression) -> (Expression, Self::AdditionalOutput) {
111        let (cond, cond_value) = self.reconstruct_expression(input.condition);
112
113        match cond_value {
114            Some(Value::Bool(true)) => self.reconstruct_expression(input.if_true),
115            Some(Value::Bool(false)) => self.reconstruct_expression(input.if_false),
116            _ => (
117                TernaryExpression {
118                    condition: cond,
119                    if_true: self.reconstruct_expression(input.if_true).0,
120                    if_false: self.reconstruct_expression(input.if_false).0,
121                    ..input
122                }
123                .into(),
124                None,
125            ),
126        }
127    }

129    fn reconstruct_array_access(&mut self, input: ArrayAccess) -> (Expression, Self::AdditionalOutput) {
130        let span = input.span();
131        let array_id = input.array.id();
132        let (array, value_opt) = self.reconstruct_expression(input.array);
133        let (index, opt_value) = self.reconstruct_expression(input.index);
134        if let Some(value) = opt_value {
135            // We can perform compile time bounds checking.
136
137            let ty = self.state.type_table.get(&array_id);
138            let Some(Type::Array(array_ty)) = ty else {
139                panic!("Type checking guaranteed that this is an array.");
140            };
141            let len = array_ty.length.as_u32();
142
143            if let Some(len) = len {
144                let index: u32 = match value {
145                    Value::U8(x) => x as u32,
146                    Value::U16(x) => x as u32,
147                    Value::U32(x) => x,
148                    Value::U64(x) => x.try_into().unwrap_or(len),
149                    Value::U128(x) => x.try_into().unwrap_or(len),
150                    Value::I8(x) => x.try_into().unwrap_or(len),
151                    Value::I16(x) => x.try_into().unwrap_or(len),
152                    Value::I32(x) => x.try_into().unwrap_or(len),
153                    Value::I64(x) => x.try_into().unwrap_or(len),
154                    Value::I128(x) => x.try_into().unwrap_or(len),
155                    _ => panic!("Type checking guarantees this is an integer"),
156                };
157
158                if index >= len {
159                    // Only emit a bounds error if we have no other errors yet.
160                    // This prevents a chain of redundant error messages when a loop is unrolled.
161                    if !self.state.handler.had_errors() {
162                        // Get the integer string with no suffix.
163                        let str_index = match value {
164                            Value::U8(x) => format!("{x}"),
165                            Value::U16(x) => format!("{x}"),
166                            Value::U32(x) => format!("{x}"),
167                            Value::U64(x) => format!("{x}"),
168                            Value::U128(x) => format!("{x}"),
169                            Value::I8(x) => format!("{x}"),
170                            Value::I16(x) => format!("{x}"),
171                            Value::I32(x) => format!("{x}"),
172                            Value::I64(x) => format!("{x}"),
173                            Value::I128(x) => format!("{x}"),
174                            _ => unreachable!("We would have panicked above"),
175                        };
176
177                        self.emit_err(StaticAnalyzerError::array_bounds(str_index, len, span));
178                    }
179                } else if let Some(Value::Array(value)) = value_opt {
180                    // We're in bounds and we can evaluate the array at compile time, so just return the value.
181                    let result_value = value.get(index as usize).expect("We already checked bounds.");
182                    return (
183                        value_to_expression(result_value, input.span, &self.state.node_builder).expect(VALUE_ERROR),
184                        Some(result_value.clone()),
185                    );
186                }
187            }
188        } else {
189            self.array_index_not_evaluated = Some(index.span());
190        }
191        (ArrayAccess { array, index, ..input }.into(), None)
192    }

194    fn reconstruct_associated_constant(
195        &mut self,
196        input: leo_ast::AssociatedConstantExpression,
197    ) -> (Expression, Self::AdditionalOutput) {
198        // Currently there is only one associated constant.
199        let generator = Value::generator();
200        let expr = value_to_expression(&generator, input.span(), &self.state.node_builder).expect(VALUE_ERROR);
201        (expr, Some(generator))
202    }

204    fn reconstruct_associated_function(
205        &mut self,
206        mut input: leo_ast::AssociatedFunctionExpression,
207    ) -> (Expression, Self::AdditionalOutput) {
208        let mut values = Vec::new();
209        for argument in input.arguments.iter_mut() {
210            let (new_argument, opt_value) = self.reconstruct_expression(std::mem::take(argument));
211            *argument = new_argument;
212            if let Some(value) = opt_value {
213                values.push(value);
214            }
215        }
216
217        if values.len() == input.arguments.len() && !matches!(input.variant.name, sym::CheatCode | sym::Mapping) {
218            // We've evaluated every argument, and this function isn't a cheat code or mapping
219            // operation, so maybe we can compute the result at compile time.
220            let core_function = CoreFunction::from_symbols(input.variant.name, input.name.name)
221                .expect("Type checking guarantees this is valid.");
222
223            match interpreter_value::evaluate_core_function(&mut values, core_function, &[], input.span()) {
224                Ok(Some(value)) => {
225                    // Successful evaluation.
226                    let expr = value_to_expression(&value, input.span(), &self.state.node_builder).expect(VALUE_ERROR);
227                    return (expr, Some(value));
228                }
229                Ok(None) =>
230                    // No errors, but we were unable to evaluate.
231                    {}
232                Err(err) => {
233                    self.emit_err(StaticAnalyzerError::compile_core_function(err, input.span()));
234                }
235            }
236        }
237
238        (input.into(), Default::default())
239    }

241    fn reconstruct_member_access(&mut self, input: MemberAccess) -> (Expression, Self::AdditionalOutput) {
242        let span = input.span();
243        let (inner, value_opt) = self.reconstruct_expression(input.inner);
244        let member_name = input.name.name;
245        if let Some(Value::Struct(contents)) = value_opt {
246            let value_result =
247                contents.contents.get(&member_name).expect("Type checking guarantees the member exists.");
248
249            (
250                value_to_expression(value_result, span, &self.state.node_builder).expect(VALUE_ERROR),
251                Some(value_result.clone()),
252            )
253        } else {
254            (MemberAccess { inner, ..input }.into(), None)
255        }
256    }

258    fn reconstruct_repeat(&mut self, input: leo_ast::RepeatExpression) -> (Expression, Self::AdditionalOutput) {
259        let (expr, expr_value) = self.reconstruct_expression(input.expr.clone());
260        let (count, count_value) = self.reconstruct_expression(input.count.clone());
261
262        if count_value.is_none() {
263            self.repeat_count_not_evaluated = Some(count.span());
264        }
265
266        match (expr_value, count.as_u32()) {
267            (Some(value), Some(count_u32)) => {
268                (RepeatExpression { expr, count, ..input }.into(), Some(Value::Array(vec![value; count_u32 as usize])))
269            }
270            _ => (RepeatExpression { expr, count, ..input }.into(), None),
271        }
272    }

274    fn reconstruct_tuple_access(&mut self, input: TupleAccess) -> (Expression, Self::AdditionalOutput) {
275        let span = input.span();
276        let (tuple, value_opt) = self.reconstruct_expression(input.tuple);
277        if let Some(Value::Tuple(tuple)) = value_opt {
278            let value_result = tuple.get(input.index.value()).expect("Type checking checked bounds.");
279            (
280                value_to_expression(value_result, span, &self.state.node_builder).expect(VALUE_ERROR),
281                Some(value_result.clone()),
282            )
283        } else {
284            (TupleAccess { tuple, ..input }.into(), None)
285        }
286    }

288    fn reconstruct_array(&mut self, mut input: leo_ast::ArrayExpression) -> (Expression, Self::AdditionalOutput) {
289        let mut values = Vec::new();
290        input.elements.iter_mut().for_each(|element| {
291            let (new_element, value_opt) = self.reconstruct_expression(std::mem::take(element));
292            if let Some(value) = value_opt {
293                values.push(value);
294            }
295            *element = new_element;
296        });
297        if values.len() == input.elements.len() {
298            (input.into(), Some(Value::Array(values)))
299        } else {
300            (input.into(), None)
301        }
302    }

304    fn reconstruct_binary(&mut self, input: leo_ast::BinaryExpression) -> (Expression, Self::AdditionalOutput) {
305        let span = input.span();
306        let input_id = input.id();
307
308        let (left, lhs_opt_value) = self.reconstruct_expression(input.left);
309        let (right, rhs_opt_value) = self.reconstruct_expression(input.right);
310
311        if let (Some(lhs_value), Some(rhs_value)) = (lhs_opt_value, rhs_opt_value) {
312            // We were able to evaluate both operands, so we can evaluate this expression.
313            match interpreter_value::evaluate_binary(
314                span,
315                input.op,
316                &lhs_value,
317                &rhs_value,
318                &self.state.type_table.get(&input_id),
319            ) {
320                Ok(new_value) => {
321                    let new_expr = value_to_expression(&new_value, span, &self.state.node_builder).expect(VALUE_ERROR);
322                    return (new_expr, Some(new_value));
323                }
324                Err(err) => self
325                    .emit_err(StaticAnalyzerError::compile_time_binary_op(lhs_value, rhs_value, input.op, err, span)),
326            }
327        }
328
329        (BinaryExpression { left, right, ..input }.into(), None)
330    }

332    fn reconstruct_call(&mut self, mut input: leo_ast::CallExpression) -> (Expression, Self::AdditionalOutput) {
333        input.const_arguments.iter_mut().for_each(|arg| {
334            *arg = self.reconstruct_expression(std::mem::take(arg)).0;
335        });
336        input.arguments.iter_mut().for_each(|arg| {
337            *arg = self.reconstruct_expression(std::mem::take(arg)).0;
338        });
339        (input.into(), Default::default())
340    }

342    fn reconstruct_cast(&mut self, input: leo_ast::CastExpression) -> (Expression, Self::AdditionalOutput) {
343        let span = input.span();
344
345        let (expr, opt_value) = self.reconstruct_expression(input.expression);
346
347        if let Some(value) = opt_value {
348            if let Some(cast_value) = value.cast(&input.type_) {
349                let expr = value_to_expression(&cast_value, span, &self.state.node_builder).expect(VALUE_ERROR);
350                return (expr, Some(cast_value));
351            } else {
352                self.emit_err(StaticAnalyzerError::compile_time_cast(value, &input.type_, span));
353            }
354        }
355        (CastExpression { expression: expr, ..input }.into(), None)
356    }

362    fn reconstruct_identifier(&mut self, input: leo_ast::Identifier) -> (Expression, Self::AdditionalOutput) {
363        // Substitute the identifier with the constant value if it is a constant that's been evaluated.
364        if let Some(expression) = self.state.symbol_table.lookup_const(self.program, input.name) {
365            let (expression, opt_value) = self.reconstruct_expression(expression);
366            if opt_value.is_some() {
367                return (expression, opt_value);
368            }
369        }
370
371        (input.into(), None)
372    }

374    fn reconstruct_literal(&mut self, mut input: leo_ast::Literal) -> (Expression, Self::AdditionalOutput) {
375        let type_info = self.state.type_table.get(&input.id());
376
377        let value =
378            interpreter_value::literal_to_value(&input, &type_info).expect("Failed to convert literal to value");
379
380        // If we know the type of an unsuffixed literal, might as well change it to a suffixed literal. This way, we
381        // do not have to infer the type again in later passes of type checking.
382        if let LiteralVariant::Unsuffixed(s) = input.variant {
383            match type_info.expect("Expected type information to be available") {
384                Type::Integer(ty) => input.variant = LiteralVariant::Integer(ty, s),
385                Type::Field => input.variant = LiteralVariant::Field(s),
386                Type::Group => input.variant = LiteralVariant::Group(s),
387                Type::Scalar => input.variant = LiteralVariant::Scalar(s),
388                _ => panic!("Type checking should have prevented this."),
389            }
390        }
391        (input.into(), Some(value))
392    }

398    fn reconstruct_tuple(&mut self, mut input: leo_ast::TupleExpression) -> (Expression, Self::AdditionalOutput) {
399        let mut values = Vec::with_capacity(input.elements.len());
400        for expr in input.elements.iter_mut() {
401            let (new_expr, opt_value) = self.reconstruct_expression(std::mem::take(expr));
402            *expr = new_expr;
403            if let Some(value) = opt_value {
404                values.push(value);
405            }
406        }
407
408        let opt_value = if values.len() == input.elements.len() { Some(Value::Tuple(values)) } else { None };
409
410        (input.into(), opt_value)
411    }

413    fn reconstruct_unary(&mut self, input: UnaryExpression) -> (Expression, Self::AdditionalOutput) {
414        let input_id = input.id();
415        let span = input.span;
416        let (receiver, opt_value) = self.reconstruct_expression(input.receiver);
417
418        if let Some(value) = opt_value {
419            // We were able to evaluate the operand, so we can evaluate the expression.
420            match interpreter_value::evaluate_unary(span, input.op, &value, &self.state.type_table.get(&input_id)) {
421                Ok(new_value) => {
422                    let new_expr = value_to_expression(&new_value, span, &self.state.node_builder).expect(VALUE_ERROR);
423                    return (new_expr, Some(new_value));
424                }
425                Err(err) => self.emit_err(StaticAnalyzerError::compile_time_unary_op(value, input.op, err, span)),
426            }
427        }
428        (UnaryExpression { receiver, ..input }.into(), None)
429    }

435    /* Statements */
436    fn reconstruct_assert(&mut self, mut input: AssertStatement) -> (Statement, Self::AdditionalOutput) {
437        // Catching asserts at compile time is not feasible here due to control flow, but could be done in
438        // a later pass after loops are unrolled and conditionals are flattened.
439        input.variant = match input.variant {
440            AssertVariant::Assert(expr) => AssertVariant::Assert(self.reconstruct_expression(expr).0),
441
442            AssertVariant::AssertEq(lhs, rhs) => {
443                AssertVariant::AssertEq(self.reconstruct_expression(lhs).0, self.reconstruct_expression(rhs).0)
444            }
445
446            AssertVariant::AssertNeq(lhs, rhs) => {
447                AssertVariant::AssertNeq(self.reconstruct_expression(lhs).0, self.reconstruct_expression(rhs).0)
448            }
449        };
450
451        (input.into(), None)
452    }

454    fn reconstruct_assign(&mut self, assign: AssignStatement) -> (Statement, Self::AdditionalOutput) {
455        let value = self.reconstruct_expression(assign.value).0;
456        let place = self.reconstruct_expression(assign.place).0;
457        (AssignStatement { value, place, ..assign }.into(), None)

460    fn reconstruct_block(&mut self, mut block: Block) -> (Block, Self::AdditionalOutput) {
461        self.in_scope(block.id(), |slf| {
462            block.statements.retain_mut(|statement| {
463                let bogus_statement = Statement::dummy();
464                let this_statement = std::mem::replace(statement, bogus_statement);
465                *statement = slf.reconstruct_statement(this_statement).0;
466                !statement.is_empty()
467            });
468            (block, None)
469        })
470    }
471
472    fn reconstruct_conditional(
473        &mut self,
474        mut conditional: ConditionalStatement,
475    ) -> (Statement, Self::AdditionalOutput) {
476        conditional.condition = self.reconstruct_expression(conditional.condition).0;
477        conditional.then = self.reconstruct_block(conditional.then).0;
478        if let Some(mut otherwise) = conditional.otherwise {
479            *otherwise = self.reconstruct_statement(*otherwise).0;
480            conditional.otherwise = Some(otherwise);
481        }
482
483        (Statement::Conditional(conditional), None)
484    }
485
486    fn reconstruct_const(&mut self, mut input: ConstDeclaration) -> (Statement, Self::AdditionalOutput) {
487        let span = input.span();
488
489        let type_ = self.reconstruct_type(input.type_).0;
490        let (expr, opt_value) = self.reconstruct_expression(input.value);
491
492        if opt_value.is_some() {
493            self.state.symbol_table.insert_const(self.program, input.place.name, expr.clone());
494        } else {
495            self.const_not_evaluated = Some(span);
496        }
497
498        input.type_ = type_;
499        input.value = expr;
500
501        (Statement::Const(input), None)
502    }

504    fn reconstruct_definition(&mut self, definition: DefinitionStatement) -> (Statement, Self::AdditionalOutput) {
505        (
506            DefinitionStatement {
507                type_: definition.type_.map(|ty| self.reconstruct_type(ty).0),
508                value: self.reconstruct_expression(definition.value).0,
509                ..definition
510            }
511            .into(),
512            None,
513        )
514    }
515
516    fn reconstruct_expression_statement(
517        &mut self,
518        mut input: ExpressionStatement,
519    ) -> (Statement, Self::AdditionalOutput) {
520        input.expression = self.reconstruct_expression(input.expression).0;
521
522        if matches!(&input.expression, Expression::Unit(..) | Expression::Literal(..)) {
523            // We were able to evaluate this at compile time, but we need to get rid of this statement as
524            // we can't have expression statements that aren't calls.
525            (Statement::dummy(), Default::default())
526        } else {
527            (input.into(), Default::default())
528        }
529    }

531    fn reconstruct_iteration(&mut self, iteration: IterationStatement) -> (Statement, Self::AdditionalOutput) {
532        let id = iteration.id();
533        let type_ = iteration.type_.map(|ty| self.reconstruct_type(ty).0);
534        let start = self.reconstruct_expression(iteration.start).0;
535        let stop = self.reconstruct_expression(iteration.stop).0;
536        self.in_scope(id, |slf| {
537            (
538                IterationStatement { type_, start, stop, block: slf.reconstruct_block(iteration.block).0, ..iteration }
539                    .into(),
540                None,
541            )
542        })
543    }
544
545    fn reconstruct_return(&mut self, input: ReturnStatement) -> (Statement, Self::AdditionalOutput) {
546        (
547            ReturnStatement { expression: self.reconstruct_expression(input.expression).0, ..input }.into(),
548            Default::default(),
549        )
550    }
