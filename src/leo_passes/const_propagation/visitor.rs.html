35pub struct ConstPropagationVisitor<'a> {
36    pub state: &'a mut CompilerState,
37    /// The program name.
38    pub program: Symbol,
39    /// Have we actually modified the program at all?
40    pub changed: bool,
41    /// The RHS of a const declaration we were not able to evaluate.
42    pub const_not_evaluated: Option<Span>,
43    /// An array index which was not able to be evaluated.
44    pub array_index_not_evaluated: Option<Span>,
45}
46
47impl ConstPropagationVisitor<'_> {
48    /// Enter the symbol table's scope `id`, execute `func`, and then return to the parent scope.
49    pub fn in_scope<T>(&mut self, id: NodeID, func: impl FnOnce(&mut Self) -> T) -> T {
50        self.state.symbol_table.enter_scope(Some(id));
51        let result = func(self);
52        self.state.symbol_table.enter_parent();
53        result
54    }
55
56    /// Emit a `StaticAnalyzerError`.
57    pub fn emit_err(&self, err: StaticAnalyzerError) {
58        self.state.handler.emit_err(err);
59    }
60}
61
62pub fn value_to_expression(value: &Value, span: Span, node_builder: &NodeBuilder) -> Option<Expression> {
63    use Value::*;
64    let id = node_builder.next_id();
65
66    let result = match value {
67        Unit => leo_ast::UnitExpression { span, id }.into(),
68        Bool(x) => Literal::boolean(*x, span, id).into(),
69        U8(x) => Literal::integer(IntegerType::U8, format!("{x}"), span, id).into(),
70        U16(x) => Literal::integer(IntegerType::U16, format!("{x}"), span, id).into(),
71        U32(x) => Literal::integer(IntegerType::U32, format!("{x}"), span, id).into(),
72        U64(x) => Literal::integer(IntegerType::U64, format!("{x}"), span, id).into(),
73        U128(x) => Literal::integer(IntegerType::U128, format!("{x}"), span, id).into(),
74        I8(x) => Literal::integer(IntegerType::I8, format!("{x}"), span, id).into(),
75        I16(x) => Literal::integer(IntegerType::I16, format!("{x}"), span, id).into(),
76        I32(x) => Literal::integer(IntegerType::I32, format!("{x}"), span, id).into(),
77        I64(x) => Literal::integer(IntegerType::I64, format!("{x}"), span, id).into(),
78        I128(x) => Literal::integer(IntegerType::I128, format!("{x}"), span, id).into(),
79        Address(x) => Literal::address(format!("{x}"), span, id).into(),
80        Group(x) => {
81            let mut s = format!("{x}");
82            // Strip off the `group` suffix.
83            s.truncate(s.len() - 5);
84            Literal::group(s, span, id).into()
85        }
86        Field(x) => {
87            let mut s = format!("{x}");
88            // Strip off the `field` suffix.
89            s.truncate(s.len() - 5);
90            Literal::field(s, span, id).into()
91        }
92        Scalar(x) => {
93            let mut s = format!("{x}");
94            // Strip off the `scalar` suffix.
95            s.truncate(s.len() - 6);
96            Literal::scalar(s, span, id).into()
97        }
98        Tuple(x) => {
99            let mut elements = Vec::with_capacity(x.len());
100            for value in x.iter() {
101                elements.push(value_to_expression(value, span, node_builder)?);
102            }
103            TupleExpression { elements, span, id }.into()
104        }
105        Array(x) => {
106            let mut elements = Vec::with_capacity(x.len());
107            for value in x.iter() {
108                elements.push(value_to_expression(value, span, node_builder)?);
109            }
110            ArrayExpression { elements, span, id }.into()
111        }
112        Struct(x) => StructExpression {
113            name: Identifier { name: x.name, id: node_builder.next_id(), span },
114            members: {
115                let mut members = Vec::with_capacity(x.contents.len());
116                for (name, val) in x.contents.iter() {
117                    let initializer = StructVariableInitializer {
118                        identifier: Identifier { name: *name, id: node_builder.next_id(), span },
119                        expression: Some(value_to_expression(val, span, node_builder)?),
120                        span,
121                        id: node_builder.next_id(),
122                    };
123                    members.push(initializer)
124                }
125                members
126            },
127            span,
128            id,
129        }
130        .into(),
131        Future(..) => return None,
132    };
133
134    Some(result)
135}
