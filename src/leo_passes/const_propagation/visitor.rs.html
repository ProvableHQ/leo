37pub struct ConstPropagationVisitor<'a> {
38    pub state: &'a mut CompilerState,
39    /// The program name.
40    pub program: Symbol,
41    /// The module name.
42    pub module: Vec<Symbol>,
43    /// Have we actually modified the program at all?
44    pub changed: bool,
45    /// The RHS of a const declaration we were not able to evaluate.
46    pub const_not_evaluated: Option<Span>,
47    /// An array index which was not able to be evaluated.
48    pub array_index_not_evaluated: Option<Span>,
49    /// An array length which was not able to be evaluated.
50    pub array_length_not_evaluated: Option<Span>,
51    /// A repeat expression count which was not able to be evaluated.
52    pub repeat_count_not_evaluated: Option<Span>,
53}
54
55impl ConstPropagationVisitor<'_> {
56    /// Enter the symbol table's scope `id`, execute `func`, and then return to the parent scope.
57    pub fn in_scope<T>(&mut self, id: NodeID, func: impl FnOnce(&mut Self) -> T) -> T {
58        self.state.symbol_table.enter_scope(Some(id));
59        let result = func(self);
60        self.state.symbol_table.enter_parent();
61        result
62    }
63
64    /// Enter module scope with path `module`, execute `func`, and then return to the parent module.
65    pub fn in_module_scope<T>(&mut self, module: &[Symbol], func: impl FnOnce(&mut Self) -> T) -> T {
66        let parent_module = self.module.clone();
67        self.module = module.to_vec();
68        let result = func(self);
69        self.module = parent_module;
70        result
71    }
72
73    /// Emit a `StaticAnalyzerError`.
74    pub fn emit_err(&self, err: StaticAnalyzerError) {
75        self.state.handler.emit_err(err);
76    }
77}

79pub fn value_to_expression(value: &Value, span: Span, node_builder: &NodeBuilder) -> Option<Expression> {
80    use Value::*;
81    let id = node_builder.next_id();
82
83    let result = match value {
84        Unit => leo_ast::UnitExpression { span, id }.into(),
85        Bool(x) => Literal::boolean(*x, span, id).into(),
86        U8(x) => Literal::integer(IntegerType::U8, format!("{x}"), span, id).into(),
87        U16(x) => Literal::integer(IntegerType::U16, format!("{x}"), span, id).into(),
88        U32(x) => Literal::integer(IntegerType::U32, format!("{x}"), span, id).into(),
89        U64(x) => Literal::integer(IntegerType::U64, format!("{x}"), span, id).into(),
90        U128(x) => Literal::integer(IntegerType::U128, format!("{x}"), span, id).into(),
91        I8(x) => Literal::integer(IntegerType::I8, format!("{x}"), span, id).into(),
92        I16(x) => Literal::integer(IntegerType::I16, format!("{x}"), span, id).into(),
93        I32(x) => Literal::integer(IntegerType::I32, format!("{x}"), span, id).into(),
94        I64(x) => Literal::integer(IntegerType::I64, format!("{x}"), span, id).into(),
95        I128(x) => Literal::integer(IntegerType::I128, format!("{x}"), span, id).into(),
96        Address(x) => Literal::address(format!("{x}"), span, id).into(),
97        Group(x) => {
98            let mut s = format!("{x}");
99            // Strip off the `group` suffix.
100            s.truncate(s.len() - 5);
101            Literal::group(s, span, id).into()
102        }
103        Field(x) => {
104            let mut s = format!("{x}");
105            // Strip off the `field` suffix.
106            s.truncate(s.len() - 5);
107            Literal::field(s, span, id).into()
108        }
109        Scalar(x) => {
110            let mut s = format!("{x}");
111            // Strip off the `scalar` suffix.
112            s.truncate(s.len() - 6);
113            Literal::scalar(s, span, id).into()
114        }
115        Tuple(x) => {
116            let mut elements = Vec::with_capacity(x.len());
117            for value in x.iter() {
118                elements.push(value_to_expression(value, span, node_builder)?);
119            }
120            TupleExpression { elements, span, id }.into()
121        }
122        Array(x) => {
123            let mut elements = Vec::with_capacity(x.len());
124            for value in x.iter() {
125                elements.push(value_to_expression(value, span, node_builder)?);
126            }
127            ArrayExpression { elements, span, id }.into()
128        }
129        Repeat(expr, count) => RepeatExpression {
130            expr: value_to_expression(expr, span, node_builder)?,
131            count: value_to_expression(count, span, node_builder)?,
132            span,
133            id,
134        }
135        .into(),
136        Struct(x) => StructExpression {
137            path: Path::new(
138                x.path.iter().take(x.path.len() - 1).map(|&sym| Identifier::new(sym, node_builder.next_id())).collect(),
139                Identifier::new(*x.path.last().expect("path must not be empty"), node_builder.next_id()),
140                Some(x.path.clone()),
141                span,
142                node_builder.next_id(),
143            ),
144            const_arguments: Vec::new(), // `Value`s don't have const arguments
145            members: {
146                let mut members = Vec::with_capacity(x.contents.len());
147                for (name, val) in x.contents.iter() {
148                    let initializer = StructVariableInitializer {
149                        identifier: Identifier { name: *name, id: node_builder.next_id(), span },
150                        expression: Some(value_to_expression(val, span, node_builder)?),
151                        span,
152                        id: node_builder.next_id(),
153                    };
154                    members.push(initializer)
155                }
156                members
157            },
158            span,
159            id,
160        }
161        .into(),
162        Future(..) => return None,
163        Unsuffixed(..) => return None,
164    };
165
166    Some(result)
167}
