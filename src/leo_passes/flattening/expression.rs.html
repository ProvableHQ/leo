17use super::FlatteningVisitor;

19use leo_ast::{
20    Expression,
21    ExpressionReconstructor,
22    Identifier,
23    Location,
24    Node,
25    Statement,
26    StructExpression,
27    StructVariableInitializer,
28    TernaryExpression,
29    Type,

32impl ExpressionReconstructor for FlatteningVisitor<'_> {
33    type AdditionalOutput = Vec<Statement>;
34
35    /// Reconstructs a struct init expression, flattening any tuples in the expression.
36    fn reconstruct_struct_init(&mut self, input: StructExpression) -> (Expression, Self::AdditionalOutput) {
37        let mut statements = Vec::new();
38        let mut members = Vec::with_capacity(input.members.len());
39
40        // Reconstruct and flatten the argument expressions.
41        for member in input.members.into_iter() {
42            // Note that this unwrap is safe since SSA guarantees that all struct variable initializers are of the form `<name>: <expr>`.
43            let (expr, stmts) = self.reconstruct_expression(member.expression.unwrap());
44            // Accumulate any statements produced.
45            statements.extend(stmts);
46            // Accumulate the struct members.
47            members.push(StructVariableInitializer {
48                identifier: member.identifier,
49                expression: Some(expr),
50                span: member.span,
51                id: member.id,
52            });
53        }
54
55        (StructExpression { members, ..input }.into(), statements)
56    }
57
58    /// Reconstructs ternary expressions over arrays, structs, and tuples, accumulating any statements that are generated.
59    /// This is necessary because Aleo instructions does not support ternary expressions over composite data types.
60    /// For example, the ternary expression `cond ? (a, b) : (c, d)` is flattened into the following:
61    /// ```leo
62    /// let var$0 = cond ? a : c;
63    /// let var$1 = cond ? b : d;
64    /// (var$0, var$1)
65    /// ```
66    /// For structs, the ternary expression `cond ? a : b`, where `a` and `b` are both structs `Foo { bar: u8, baz: u8 }`, is flattened into the following:
67    /// ```leo
68    /// let var$0 = cond ? a.bar : b.bar;
69    /// let var$1 = cond ? a.baz : b.baz;
70    /// let var$2 = Foo { bar: var$0, baz: var$1 };
71    /// var$2
72    /// ```
73    fn reconstruct_ternary(&mut self, input: TernaryExpression) -> (Expression, Self::AdditionalOutput) {
74        let if_true_type = self
75            .state
76            .type_table
77            .get(&input.if_true.id())
78            .expect("Type checking guarantees that all expressions are typed.");
79        let if_false_type = self
80            .state
81            .type_table
82            .get(&input.if_false.id())
83            .expect("Type checking guarantees that all expressions are typed.");
84
85        // Note that type checking guarantees that both expressions have the same same type. This is a sanity check.
86        assert!(if_true_type.eq_flat_relaxed(&if_false_type));
87
88        fn as_identifier(ident_expr: Expression) -> Identifier {
89            let Expression::Identifier(identifier) = ident_expr else {
90                panic!("SSA form should have guaranteed this is an identifier: {}.", ident_expr);
91            };
92            identifier
93        }
94
95        match &if_true_type {
96            Type::Array(if_true_type) => self.ternary_array(
97                if_true_type,
98                &input.condition,
99                &as_identifier(input.if_true),
100                &as_identifier(input.if_false),
101            ),
102            Type::Composite(if_true_type) => {
103                // Get the struct definitions.
104                let if_true_type = self
105                    .state
106                    .symbol_table
107                    .lookup_struct(if_true_type.id.name)
108                    .or_else(|| {
109                        self.state.symbol_table.lookup_record(Location::new(self.program, if_true_type.id.name))
110                    })
111                    .expect("This definition should exist")
112                    .clone();
113
114                self.ternary_struct(
115                    &if_true_type,
116                    &input.condition,
117                    &as_identifier(input.if_true),
118                    &as_identifier(input.if_false),
119                )
120            }
121            Type::Tuple(if_true_type) => {
122                self.ternary_tuple(if_true_type, &input.condition, &input.if_true, &input.if_false)
123            }
124            _ => {
125                // There's nothing to be done - SSA has guaranteed that `if_true` and `if_false` are identifiers,
126                // so there's not even any point in reconstructing them.
127
128                assert!(matches!(&input.if_true, Expression::Identifier(..)));
129                assert!(matches!(&input.if_false, Expression::Identifier(..)));
130
131                (input.into(), Default::default())
132            }
133        }
134    }
135}
