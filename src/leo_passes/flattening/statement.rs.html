44impl StatementReconstructor for FlatteningVisitor<'_> {
45    /// Rewrites an assert statement into a flattened form.
46    /// Assert statements at the top level only have their arguments flattened.
47    /// Assert statements inside a conditional statement are flattened to such that the check is conditional on
48    /// the execution path being valid.
49    /// For example, the following snippet:
50    /// ```leo
51    /// if condition1 {
52    ///    if condition2 {
53    ///        assert(foo);
54    ///    }
55    /// }
56    /// ```
57    /// is flattened to:
58    /// ```leo
59    /// assert(!(condition1 && condition2) || foo);
60    /// ```
61    /// which is equivalent to the logical formula `(condition1 /\ condition2) ==> foo`.
62    fn reconstruct_assert(&mut self, input: AssertStatement) -> (Statement, Self::AdditionalOutput) {
63        let mut statements = Vec::new();
64
65        // If we are traversing an async function, then we can return the assert as it.
66        if self.is_async {
67            return (input.into(), statements);
68        }
69
70        // Flatten the arguments of the assert statement.
71        let assert = AssertStatement {
72            span: input.span,
73            id: input.id,
74            variant: match input.variant {
75                AssertVariant::Assert(expression) => {
76                    let (expression, additional_statements) = self.reconstruct_expression(expression);
77                    statements.extend(additional_statements);
78                    AssertVariant::Assert(expression)
79                }
80                AssertVariant::AssertEq(left, right) => {
81                    let (left, additional_statements) = self.reconstruct_expression(left);
82                    statements.extend(additional_statements);
83                    let (right, additional_statements) = self.reconstruct_expression(right);
84                    statements.extend(additional_statements);
85                    AssertVariant::AssertEq(left, right)
86                }
87                AssertVariant::AssertNeq(left, right) => {
88                    let (left, additional_statements) = self.reconstruct_expression(left);
89                    statements.extend(additional_statements);
90                    let (right, additional_statements) = self.reconstruct_expression(right);
91                    statements.extend(additional_statements);
92                    AssertVariant::AssertNeq(left, right)
93                }
94            },
95        };
96
97        let mut guards: Vec<Expression> = Vec::new();
98
99        if let Some((guard, guard_statements)) = self.construct_guard() {
100            statements.extend(guard_statements);
101
102            // The not_guard is true if we didn't follow the condition chain
103            // that led to this assertion.
104            let not_guard = UnaryExpression {
105                op: UnaryOperation::Not,
106                receiver: guard.into(),
107                span: Default::default(),
108                id: {
109                    // Create a new node ID for the unary expression.
110                    let id = self.state.node_builder.next_id();
111                    // Update the type table with the type of the unary expression.
112                    self.state.type_table.insert(id, Type::Boolean);
113                    id
114                },
115            }
116            .into();
117            let (identifier, statement) = self.unique_simple_definition(not_guard);
118            statements.push(statement);
119            guards.push(identifier.into());
120        }
121
122        // We also need to guard against early returns.
123        if let Some((guard, guard_statements)) = self.construct_early_return_guard() {
124            guards.push(guard.into());
125            statements.extend(guard_statements);
126        }
127
128        if guards.is_empty() {
129            return (assert.into(), statements);
130        }
131
132        let is_eq = matches!(assert.variant, AssertVariant::AssertEq(..));
133
134        // We need to `or` the asserted expression with the guards,
135        // so extract an appropriate expression.
136        let mut expression = match assert.variant {
137            // If the assert statement is an `assert`, use the expression as is.
138            AssertVariant::Assert(expression) => expression,
139
140            // For `assert_eq` or `assert_neq`, construct a new expression.
141            AssertVariant::AssertEq(left, right) | AssertVariant::AssertNeq(left, right) => {
142                let binary = BinaryExpression {
143                    left,
144                    right,
145                    op: if is_eq { BinaryOperation::Eq } else { BinaryOperation::Neq },
146                    span: Default::default(),
147                    id: self.state.node_builder.next_id(),
148                };
149                self.state.type_table.insert(binary.id, Type::Boolean);
150                let (identifier, statement) = self.unique_simple_definition(binary.into());
151                statements.push(statement);
152                identifier.into()
153            }
154        };
155
156        // The assertion will be that the original assert statement is true or one of the guards is true
157        // (ie, we either didn't follow the condition chain that led to this assert, or else we took an
158        // early return).
159        for guard in guards.into_iter() {
160            let binary = BinaryExpression {
161                left: expression,
162                right: guard,
163                op: BinaryOperation::Or,
164                span: Default::default(),
165                id: self.state.node_builder.next_id(),
166            };
167            self.state.type_table.insert(binary.id(), Type::Boolean);
168            let (identifier, statement) = self.unique_simple_definition(binary.into());
169            statements.push(statement);
170            expression = identifier.into();
171        }
172
173        let assert_statement = AssertStatement { variant: AssertVariant::Assert(expression), ..input }.into();
174
175        (assert_statement, statements)
176    }

182    // TODO: Do we want to flatten nested blocks? They do not affect code generation but it would regularize the AST structure.
183    /// Flattens the statements inside a basic block.
184    /// The resulting block does not contain any conditional statements.
185    fn reconstruct_block(&mut self, block: Block) -> (Block, Self::AdditionalOutput) {
186        let mut statements = Vec::with_capacity(block.statements.len());
187
188        // Flatten each statement, accumulating any new statements produced.
189        for statement in block.statements {
190            let (reconstructed_statement, additional_statements) = self.reconstruct_statement(statement);
191            statements.extend(additional_statements);
192            statements.push(reconstructed_statement);
193        }
194
195        (Block { span: block.span, statements, id: self.state.node_builder.next_id() }, Default::default())
196    }

198    /// Flatten a conditional statement into a list of statements.
199    fn reconstruct_conditional(&mut self, conditional: ConditionalStatement) -> (Statement, Self::AdditionalOutput) {
200        let mut statements = Vec::with_capacity(conditional.then.statements.len());
201
202        // If we are traversing an async function, reconstruct the if and else blocks, but do not flatten them.
203        if self.is_async {
204            let then_block = self.reconstruct_block(conditional.then).0;
205            let otherwise_block = match conditional.otherwise {
206                Some(statement) => match *statement {
207                    Statement::Block(block) => self.reconstruct_block(block).0,
208                    _ => panic!("SSA guarantees that the `otherwise` is always a `Block`"),
209                },
210                None => {
211                    Block { span: Default::default(), statements: Vec::new(), id: self.state.node_builder.next_id() }
212                }
213            };
214
215            return (
216                ConditionalStatement {
217                    then: then_block,
218                    otherwise: Some(Box::new(otherwise_block.into())),
219                    ..conditional
220                }
221                .into(),
222                statements,
223            );
224        }
225
226        // Assign the condition to a variable, as it may be used multiple times.
227        let place = Identifier {
228            name: self.state.assigner.unique_symbol("condition", "$"),
229            span: Default::default(),
230            id: {
231                let id = self.state.node_builder.next_id();
232                self.state.type_table.insert(id, Type::Boolean);
233                id
234            },
235        };
236
237        statements.push(self.simple_definition(place, conditional.condition.clone()));
238
239        // Add condition to the condition stack.
240        self.condition_stack.push(Guard::Unconstructed(place));
241
242        // Reconstruct the then-block and accumulate it constituent statements.
243        statements.extend(self.reconstruct_block(conditional.then).0.statements);
244
245        // Remove condition from the condition stack.
246        self.condition_stack.pop();
247
248        // Consume the otherwise-block and flatten its constituent statements into the current block.
249        if let Some(statement) = conditional.otherwise {
250            // Apply Not to the condition, assign it, and put it on the condition stack.
251            let not_condition = UnaryExpression {
252                op: UnaryOperation::Not,
253                receiver: conditional.condition.clone(),
254                span: conditional.condition.span(),
255                id: conditional.condition.id(),
256            }
257            .into();
258            let not_place = Identifier {
259                name: self.state.assigner.unique_symbol("condition", "$"),
260                span: Default::default(),
261                id: {
262                    let id = self.state.node_builder.next_id();
263                    self.state.type_table.insert(id, Type::Boolean);
264                    id
265                },
266            };
267            statements.push(self.simple_definition(not_place, not_condition));
268            self.condition_stack.push(Guard::Unconstructed(not_place));
269
270            // Reconstruct the otherwise-block and accumulate it constituent statements.
271            match *statement {
272                Statement::Block(block) => statements.extend(self.reconstruct_block(block).0.statements),
273                _ => panic!("SSA guarantees that the `otherwise` is always a `Block`"),
274            }
275
276            // Remove the negated condition from the condition stack.
277            self.condition_stack.pop();
278        };
279
280        (Statement::dummy(), statements)
281    }

283    /// Flattens a definition, if necessary.
284    /// Marks variables as structs as necessary.
285    /// Note that new statements are only produced if the right hand side is a ternary expression over structs.
286    /// Otherwise, the statement is returned as is.
287    fn reconstruct_definition(&mut self, definition: DefinitionStatement) -> (Statement, Self::AdditionalOutput) {
288        // Flatten the rhs of the assignment.
289        let (value, statements) = self.reconstruct_expression(definition.value);
290        match (definition.place, &value) {
291            (DefinitionPlace::Single(identifier), _) => (self.simple_definition(identifier, value), statements),
292            (DefinitionPlace::Multiple(identifiers), expression) => {
293                let output_type = match &self.state.type_table.get(&expression.id()) {
294                    Some(Type::Tuple(tuple_type)) => tuple_type.clone(),
295                    _ => panic!("Type checking guarantees that the output type is a tuple."),
296                };
297
298                for (identifier, type_) in identifiers.iter().zip_eq(output_type.elements().iter()) {
299                    // Add the type of each identifier to the type table.
300                    self.state.type_table.insert(identifier.id, type_.clone());
301                }
302
303                (
304                    DefinitionStatement {
305                        place: DefinitionPlace::Multiple(identifiers),
306                        type_: Type::Err,
307                        value,
308                        span: Default::default(),
309                        id: self.state.node_builder.next_id(),
310                    }
311                    .into(),
312                    statements,
313                )
314            }
315        }
316    }

322    /// Transforms a return statement into an empty block statement.
323    /// Stores the arguments to the return statement, which are later folded into a single return statement at the end of the function.
324    fn reconstruct_return(&mut self, input: ReturnStatement) -> (Statement, Self::AdditionalOutput) {
325        use Expression::*;
326
327        // If we are traversing an async function, return as is.
328        if self.is_async {
329            return (input.into(), Default::default());
330        }
331        // Construct the associated guard.
332        let (guard_identifier, statements) = self.construct_guard().unzip();
333
334        let return_guard = guard_identifier.map_or(ReturnGuard::None, ReturnGuard::Unconstructed);
335
336        let is_tuple_ids = matches!(&input.expression, Tuple(tuple_expr) if tuple_expr .elements.iter() .all(|expr| matches!(expr, Identifier(_))));
337        if !matches!(&input.expression, Unit(_) | Identifier(_) | AssociatedConstant(_)) && !is_tuple_ids {
338            panic!("SSA guarantees that the expression is always an identifier, unit expression, or tuple literal.")
339        }
340
341        self.returns.push((return_guard, input));
342
343        (Statement::dummy(), statements.unwrap_or_default())
344    }
