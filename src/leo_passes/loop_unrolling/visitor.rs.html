17use leo_ast::{AstReconstructor, Block, IterationStatement, Literal, Node, NodeID, Statement, Type, Value};
18
19use leo_errors::LoopUnrollerError;
20use leo_span::{Span, Symbol};
21
22use super::{Clusivity, LoopBound, RangeIterator};
23use crate::CompilerState;
24
25pub struct UnrollingVisitor<'a> {
26    pub state: &'a mut CompilerState,
27    /// The current program name.
28    pub program: Symbol,
29    /// If we've encountered a loop that was not unrolled, here's it's spanned.
30    pub loop_not_unrolled: Option<Span>,
31    /// Have we unrolled any loop?
32    pub loop_unrolled: bool,
33}
34
35impl UnrollingVisitor<'_> {
36    pub fn in_scope<T>(&mut self, id: NodeID, func: impl FnOnce(&mut Self) -> T) -> T {
37        self.state.symbol_table.enter_scope(Some(id));
38        let result = func(self);
39        self.state.symbol_table.enter_parent();
40        result
41    }
42
43    /// Emits a Loop Unrolling Error
44    pub fn emit_err(&self, err: LoopUnrollerError) {
45        self.state.handler.emit_err(err);
46    }
47
48    /// Unrolls an IterationStatement.
49    pub fn unroll_iteration_statement<I: LoopBound>(
50        &mut self,
51        input: IterationStatement,
52        start: Value,
53        stop: Value,
54    ) -> Statement {
55        // Closure to check that the constant values are valid u128.
56        // We already know these are integers since loop unrolling occurs after type checking.
57        let cast_to_number = |v: Value| -> Result<I, Statement> {
58            match v.try_into() {
59                Ok(val_as_u128) => Ok(val_as_u128),
60                Err(err) => {
61                    self.state.handler.emit_err(err);
62                    Err(Statement::dummy())
63                }
64            }
65        };
66
67        // Cast `start` to `I`.
68        let start = match cast_to_number(start) {
69            Ok(v) => v,
70            Err(s) => return s,
71        };
72        // Cast `stop` to `I`.
73        let stop = match cast_to_number(stop) {
74            Ok(v) => v,
75            Err(s) => return s,
76        };
77
78        let new_block_id = self.state.node_builder.next_id();
79
80        let iter =
81            RangeIterator::new(start, stop, if input.inclusive { Clusivity::Inclusive } else { Clusivity::Exclusive });
82
83        // Create a block statement to replace the iteration statement.
84        self.in_scope(new_block_id, |slf| {
85            Block {
86                span: input.span,
87                statements: iter.map(|iteration_count| slf.unroll_single_iteration(&input, iteration_count)).collect(),
88                id: new_block_id,
89            }
90            .into()
91        })
92    }
93
94    /// A helper function to unroll a single iteration an IterationStatement.
95    fn unroll_single_iteration<I: LoopBound>(&mut self, input: &IterationStatement, iteration_count: I) -> Statement {
96        // Construct a new node ID.
97        let const_id = self.state.node_builder.next_id();
98
99        let iterator_type =
100            self.state.type_table.get(&input.variable.id()).expect("guaranteed to have a type after type checking");
101
102        // Update the type table.
103        self.state.type_table.insert(const_id, iterator_type.clone());
104
105        let outer_block_id = self.state.node_builder.next_id();
106
107        // Reconstruct `iteration_count` as a `Literal`.
108        let Type::Integer(integer_type) = &iterator_type else {
109            unreachable!("Type checking enforces that the iteration variable is of integer type");
110        };
111
112        self.in_scope(outer_block_id, |slf| {
113            let value = Literal::integer(*integer_type, iteration_count.to_string(), Default::default(), const_id);
114
115            // Add the loop variable as a constant for the current scope.
116            slf.state.symbol_table.insert_const(slf.program, &[input.variable.name], value.into());
117
118            let duplicated_body =
119                super::duplicate::duplicate(input.block.clone(), &mut slf.state.symbol_table, &slf.state.node_builder);
120
121            let result = slf.reconstruct_block(duplicated_body).0.into();
122
123            Block { statements: vec![result], span: input.span(), id: outer_block_id }.into()
124        })
125    }
126}
