17use leo_ast::{Expression::Literal, *};
18
19use leo_errors::LoopUnrollerError;
20
21use super::UnrollingVisitor;
22
23impl AstReconstructor for UnrollingVisitor<'_> {
24    type AdditionalOutput = ();
25
26    /* Expressions */
27    fn reconstruct_repeat(&mut self, input: RepeatExpression) -> (Expression, Self::AdditionalOutput) {
28        // Because the value of `count` affects the type of a repeat expression, we need to assign a new ID to the
29        // reconstructed `RepeatExpression` and update the type table accordingly.
30        let new_id = self.state.node_builder.next_id();
31        let new_count = self.reconstruct_expression(input.count).0;
32        let el_ty = self.state.type_table.get(&input.expr.id()).expect("guaranteed by type checking");
33        self.state.type_table.insert(new_id, Type::Array(ArrayType::new(el_ty, new_count.clone())));
34        (
35            RepeatExpression { expr: self.reconstruct_expression(input.expr).0, count: new_count, id: new_id, ..input }
36                .into(),
37            Default::default(),
38        )
39    }
40
41    fn reconstruct_block(&mut self, mut input: Block) -> (Block, Self::AdditionalOutput) {
42        self.in_scope(input.id(), |slf| {
43            input.statements = input.statements.into_iter().map(|stmt| slf.reconstruct_statement(stmt).0).collect();
44
45            (input, Default::default())
46        })
47    }
48
49    fn reconstruct_const(&mut self, input: ConstDeclaration) -> (Statement, Self::AdditionalOutput) {
50        (input.into(), Default::default())
51    }
52
53    fn reconstruct_definition(&mut self, input: DefinitionStatement) -> (Statement, Self::AdditionalOutput) {
54        (
55            DefinitionStatement {
56                type_: input.type_.map(|ty| self.reconstruct_type(ty).0),
57                value: self.reconstruct_expression(input.value).0,
58                ..input
59            }
60            .into(),
61            Default::default(),
62        )
63    }
64
65    fn reconstruct_iteration(&mut self, input: IterationStatement) -> (Statement, Self::AdditionalOutput) {
66        // There's no need to reconstruct the bound expressions - they must be constants
67        // which can be evaluated through constant propagation.
68
69        let Literal(start_lit_ref) = &input.start else {
70            self.loop_not_unrolled = Some(input.start.span());
71            return (Statement::Iteration(Box::new(input)), Default::default());
72        };
73
74        let Literal(stop_lit_ref) = &input.stop else {
75            self.loop_not_unrolled = Some(input.stop.span());
76            return (Statement::Iteration(Box::new(input)), Default::default());
77        };
78
79        // Helper to clone and resolve Unsuffixed -> Integer literal based on type table
80        let resolve_unsuffixed = |lit: &leo_ast::Literal, expr_id| {
81            let mut resolved = lit.clone();
82            if let LiteralVariant::Unsuffixed(s) = &resolved.variant {
83                if let Some(Type::Integer(integer_type)) = self.state.type_table.get(&expr_id) {
84                    resolved.variant = LiteralVariant::Integer(integer_type, s.clone());
85                }
86            }
87            resolved
88        };
89
90        // Clone and resolve both literals
91        let resolved_start_lit = resolve_unsuffixed(start_lit_ref, input.start.id());
92        let resolved_stop_lit = resolve_unsuffixed(stop_lit_ref, input.stop.id());
93
94        // Convert resolved literals into constant values
95        let start_value = Value::try_from(&resolved_start_lit).unwrap();
96        let stop_value = Value::try_from(&resolved_stop_lit).unwrap();
97
98        // Ensure loop bounds are strictly increasing
99        let bounds_invalid = match (&start_value, &stop_value) {
100            (Value::I8(a, _), Value::I8(b, _)) => a >= b,
101            (Value::I16(a, _), Value::I16(b, _)) => a >= b,
102            (Value::I32(a, _), Value::I32(b, _)) => a >= b,
103            (Value::I64(a, _), Value::I64(b, _)) => a >= b,
104            (Value::I128(a, _), Value::I128(b, _)) => a >= b,
105            (Value::U8(a, _), Value::U8(b, _)) => a >= b,
106            (Value::U16(a, _), Value::U16(b, _)) => a >= b,
107            (Value::U32(a, _), Value::U32(b, _)) => a >= b,
108            (Value::U64(a, _), Value::U64(b, _)) => a >= b,
109            (Value::U128(a, _), Value::U128(b, _)) => a >= b,
110            _ => panic!("Type checking guarantees that loop bounds are integers of the same type."),
111        };
112
113        if bounds_invalid {
114            self.emit_err(LoopUnrollerError::loop_range_decreasing(input.stop.span()));
115        }
116
117        self.loop_unrolled = true;
118
119        // Perform loop unrolling using i128 â€” all numeric bounds are converted to this internally
120        (self.unroll_iteration_statement::<i128>(input, start_value, stop_value), Default::default())
121    }
122}
