40impl StatementReconstructor for Destructurer<'_> {
41    fn reconstruct_assign(&mut self, _assign: AssignStatement) -> (Statement, Self::AdditionalOutput) {
42        panic!("`AssignStatement`s should not exist in the AST at this phase of compilation.")
43    }
44
45    fn reconstruct_block(&mut self, block: Block) -> (Block, Self::AdditionalOutput) {
46        let mut statements = Vec::with_capacity(block.statements.len());
47
48        // Reconstruct the statements in the block, accumulating any additional statements.
49        for statement in block.statements {
50            let (reconstructed_statement, additional_statements) = self.reconstruct_statement(statement);
51            statements.extend(additional_statements);
52            statements.push(reconstructed_statement);
53        }
54
55        (Block { span: block.span, statements, id: self.node_builder.next_id() }, Default::default())
56    }
57
58    fn reconstruct_conditional(&mut self, input: ConditionalStatement) -> (Statement, Self::AdditionalOutput) {
59        // Conditional statements can only exist in finalize blocks.
60        if !self.is_async {
61            unreachable!("`ConditionalStatement`s should not be in the AST at this phase of compilation.")
62        } else {
63            (
64                Statement::Conditional(ConditionalStatement {
65                    condition: self.reconstruct_expression(input.condition).0,
66                    then: self.reconstruct_block(input.then).0,
67                    otherwise: input.otherwise.map(|n| Box::new(self.reconstruct_statement(*n).0)),
68                    span: input.span,
69                    id: input.id,
70                }),
71                Default::default(),
72            )
73        }
74    }
75
76    fn reconstruct_console(&mut self, _: ConsoleStatement) -> (Statement, Self::AdditionalOutput) {
77        panic!("`ConsoleStatement`s should not be in the AST at this phase of compilation.")
78    }
79
80    fn reconstruct_definition(&mut self, definition: DefinitionStatement) -> (Statement, Self::AdditionalOutput) {
81        use DefinitionPlace::*;
82
83        let (value, mut statements) = self.reconstruct_expression(definition.value);
84        let ty = self.type_table.get(&value.id()).expect("Expressions should have a type.");
85        match (definition.place, value, ty) {
86            (Single(identifier), Expression::Tuple(tuple), Type::Tuple(..)) => {
87                // Just put the identifier in `self.tuples`. We don't need to keep our definition.
88                self.tuples.insert(identifier.name, tuple);
89                (Statement::dummy(), statements)
90            }
91            (Single(identifier), Expression::Identifier(rhs), Type::Tuple(..)) => {
92                // Again, just put the identifier in `self.tuples` and we don't need to keep our definition.
93                let tuple_expr = self.tuples.get(&rhs.name).expect("We should have encountered this tuple by now");
94                self.tuples.insert(identifier.name, tuple_expr.clone());
95                (Statement::dummy(), statements)
96            }
97            (Single(identifier), Expression::Call(rhs), Type::Tuple(tuple_type)) => {
98                // Make new identifiers for each member of the tuple.
99                let identifiers: Vec<Identifier> = (0..tuple_type.elements().len())
100                    .map(|i| {
101                        Identifier::new(
102                            self.assigner.unique_symbol(identifier.name, format!("$index${i}$")),
103                            self.node_builder.next_id(),
104                        )
105                    })
106                    .collect();
107
108                // Make expressions corresponding to those identifiers.
109                let expressions: Vec<Expression> = identifiers
110                    .iter()
111                    .zip_eq(tuple_type.elements().iter())
112                    .map(|(identifier, type_)| {
113                        let expr = Expression::Identifier(*identifier);
114                        self.type_table.insert(expr.id(), type_.clone());
115                        expr
116                    })
117                    .collect();
118
119                // Make a tuple expression.
120                let expr = TupleExpression {
121                    elements: expressions,
122                    span: Default::default(),
123                    id: self.node_builder.next_id(),
124                };
125
126                // Put it in the type table.
127                self.type_table.insert(expr.id(), Type::Tuple(tuple_type));
128
129                // Put it into `self.tuples`.
130                self.tuples.insert(identifier.name, expr);
131
132                // Define the new variables. We don't need to keep the old definition.
133                let stmt = Statement::Definition(DefinitionStatement {
134                    place: Multiple(identifiers),
135                    type_: Type::Err,
136                    value: Expression::Call(rhs),
137                    span: Default::default(),
138                    id: self.node_builder.next_id(),
139                });
140
141                (stmt, statements)
142            }
143            (Multiple(identifiers), Expression::Tuple(tuple), Type::Tuple(..)) => {
144                // Just make a definition for each tuple element.
145                for (identifier, expr) in identifiers.into_iter().zip_eq(tuple.elements) {
146                    let stmt = Statement::Definition(DefinitionStatement {
147                        place: Single(identifier),
148                        type_: Type::Err,
149                        value: expr,
150                        span: Default::default(),
151                        id: self.node_builder.next_id(),
152                    });
153                    statements.push(stmt);
154                }
155
156                // We don't need to keep the original definition.
157                (Statement::dummy(), statements)
158            }
159            (Multiple(identifiers), Expression::Identifier(identifier), Type::Tuple(..)) => {
160                // Again, make a definition for each tuple element.
161                let tuple = self.tuples.get(&identifier.name).expect("We should have encountered this tuple by now");
162                for (identifier, expr) in identifiers.into_iter().zip_eq(tuple.elements.iter()) {
163                    let stmt = Statement::Definition(DefinitionStatement {
164                        place: Single(identifier),
165                        type_: Type::Err,
166                        value: expr.clone(),
167                        span: Default::default(),
168                        id: self.node_builder.next_id(),
169                    });
170                    statements.push(stmt);
171                }
172
173                // We don't need to keep the original definition.
174                (Statement::dummy(), statements)
175            }
176            (m @ Multiple(..), value @ Expression::Call(..), Type::Tuple(..)) => {
177                // Just reconstruct the statement.
178                let stmt = Statement::Definition(DefinitionStatement {
179                    place: m,
180                    type_: Type::Err,
181                    value,
182                    span: definition.span,
183                    id: definition.id,
184                });
185                (stmt, statements)
186            }
187            (_, Expression::Ternary(..), Type::Tuple(..)) => {
188                panic!("Ternary conditionals of tuple type should have been removed by flattening");
189            }
190            (_, _, Type::Tuple(..)) => {
191                panic!("Expressions of tuple type can only be tuple literals, identifiers, or calls.");
192            }
193            (Single(identifier), rhs, _) => {
194                // This isn't a tuple. Just build the definition again.
195                (
196                    Statement::Definition(DefinitionStatement {
197                        place: Single(identifier),
198                        type_: Type::Err,
199                        value: rhs,
200                        span: Default::default(),
201                        id: definition.id,
202                    }),
203                    statements,
204                )
205            }
206            (Multiple(_), _, _) => panic!("A definition with multiple identifiers must have tuple type"),
207        }
208    }
209
210    fn reconstruct_iteration(&mut self, _: IterationStatement) -> (Statement, Self::AdditionalOutput) {
211        panic!("`IterationStatement`s should not be in the AST at this phase of compilation.");
212    }
213
214    fn reconstruct_return(&mut self, input: ReturnStatement) -> (Statement, Self::AdditionalOutput) {
215        // Note that SSA guarantees that `input.expression` is either a literal, identifier, or unit expression.
216        let expression = match input.expression {
217            // If the input is an identifier that maps to a tuple, use the tuple expression.
218            Expression::Identifier(identifier) if self.tuples.contains_key(&identifier.name) => {
219                // Note that the `unwrap` is safe since the match arm checks that the entry exists in `self.tuples`.
220                let tuple = self.tuples.get(&identifier.name).unwrap().clone();
221                Expression::Tuple(tuple)
222            }
223            // Otherwise, use the original expression.
224            _ => input.expression,
225        };
226
227        (Statement::Return(ReturnStatement { expression, span: input.span, id: input.id }), Default::default())
228    }
