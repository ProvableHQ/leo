37impl ExpressionReconstructor for DestructuringVisitor<'_> {
38    type AdditionalOutput = Vec<Statement>;
39
40    /// Replaces a tuple access expression with the appropriate expression.
41    fn reconstruct_tuple_access(&mut self, input: TupleAccess) -> (Expression, Self::AdditionalOutput) {
42        let Expression::Identifier(identifier) = input.tuple.as_ref() else {
43            panic!("SSA guarantees that subexpressions are identifiers or literals.");
44        };
45
46        // Look up the expression in the tuple map.
47        match self.tuples.get(&identifier.name).and_then(|tuple_names| tuple_names.get(input.index.value())) {
48            Some(id) => (Expression::Identifier(*id), Default::default()),
49            None => {
50                if !matches!(self.state.type_table.get(&identifier.id), Some(Type::Future(_))) {
51                    panic!("Type checking guarantees that all tuple accesses are declared and indices are valid.");
52                }
53
54                let expr = Expression::Access(AccessExpression::Array(ArrayAccess {
55                    array: Box::new(Expression::Identifier(*identifier)),
56                    index: Box::new(Expression::Literal(Literal::Integer(
57                        IntegerType::U32,
58                        input.index.to_string(),
59                        input.span,
60                        Default::default(),
61                    ))),
62                    span: input.span,
63                    id: input.id,
64                }));
65
66                (expr, Default::default())
67            }
68        }
69    }
70
71    /// If this is a ternary expression on tuples of length `n`, we'll need to change it into
72    /// `n` ternary expressions on the members.
73    fn reconstruct_ternary(&mut self, mut input: TernaryExpression) -> (Expression, Self::AdditionalOutput) {
74        let (if_true, mut statements) = self.reconstruct_expression_tuple(std::mem::take(&mut input.if_true));
75        let (if_false, statements2) = self.reconstruct_expression_tuple(std::mem::take(&mut input.if_false));
76        statements.extend(statements2);
77
78        match (if_true, if_false) {
79            (Expression::Tuple(tuple_true), Expression::Tuple(tuple_false)) => {
80                // Aleo's `ternary` opcode doesn't know about tuples, so we have to handle this.
81                let Some(Type::Tuple(tuple_type)) = self.state.type_table.get(&tuple_true.id()) else {
82                    panic!("Should have tuple type");
83                };
84
85                // We'll be reusing `input.condition`, so assign it to a variable.
86                let cond = if let Expression::Identifier(..) = *input.condition {
87                    *input.condition
88                } else {
89                    let place = Identifier::new(
90                        self.state.assigner.unique_symbol("cond", "$$"),
91                        self.state.node_builder.next_id(),
92                    );
93
94                    let definition = self.state.assigner.simple_definition(
95                        place,
96                        *input.condition,
97                        self.state.node_builder.next_id(),
98                    );
99
100                    statements.push(definition);
101
102                    self.state.type_table.insert(place.id(), Type::Boolean);
103
104                    Expression::Identifier(place)
105                };
106
107                // These will be the `elements` of our resulting tuple.
108                let mut elements = Vec::with_capacity(tuple_true.elements.len());
109
110                // Create an individual `ternary` for each tuple member and assign the
111                // result to a new variable.
112                for (i, (lhs, rhs, ty)) in
113                    izip!(tuple_true.elements, tuple_false.elements, tuple_type.elements()).enumerate()
114                {
115                    let identifier = Identifier::new(
116                        self.state.assigner.unique_symbol(format_args!("ternary_{i}"), "$$"),
117                        self.state.node_builder.next_id(),
118                    );
119
120                    let expression = Expression::Ternary(TernaryExpression {
121                        condition: Box::new(cond.clone()),
122                        if_true: Box::new(lhs),
123                        if_false: Box::new(rhs),
124                        span: Default::default(),
125                        id: self.state.node_builder.next_id(),
126                    });
127
128                    self.state.type_table.insert(identifier.id(), ty.clone());
129                    self.state.type_table.insert(expression.id(), ty.clone());
130
131                    let definition = self.state.assigner.simple_definition(
132                        identifier,
133                        expression,
134                        self.state.node_builder.next_id(),
135                    );
136
137                    statements.push(definition);
138                    elements.push(Expression::Identifier(identifier));
139                }
140
141                // Now rebuild the tuple from the variables we just created.
142                let expr = Expression::Tuple(TupleExpression {
143                    elements,
144                    span: Default::default(),
145                    id: self.state.node_builder.next_id(),
146                });
147
148                self.state.type_table.insert(expr.id(), Type::Tuple(tuple_type.clone()));
149
150                (expr, statements)
151            }
152            (if_true, if_false) => {
153                // This isn't a tuple. Just rebuild it and otherwise leave it alone.
154                *input.if_true = if_true;
155                *input.if_false = if_false;
156                (Expression::Ternary(input), statements)
157            }
158        }
159    }
160}
