17use crate::{CallGraph, StructGraph, SymbolTable, TypeTable};
18
19use leo_ast::{Function, Program, ProgramId, Variant};
20use leo_span::Symbol;
21
22use indexmap::IndexMap;
23
24pub struct CodeGenerator<'a> {
25    /// The symbol table for the program.
26    pub(crate) symbol_table: &'a SymbolTable,
27    /// A mapping between expressions and their types.
28    pub(crate) type_table: &'a TypeTable,
29    /// The struct dependency graph for the program.
30    pub(crate) struct_graph: &'a StructGraph,
31    /// The call graph for the program.
32    pub(crate) _call_graph: &'a CallGraph,
33    /// A counter to track the next available register.
34    pub(crate) next_register: u64,
35    /// Reference to the current function.
36    pub(crate) current_function: Option<&'a Function>,
37    /// Mapping of variables to registers.
38    pub(crate) variable_mapping: IndexMap<&'a Symbol, String>,
39    /// Mapping of composite names to a tuple containing metadata associated with the name.
40    /// The first element of the tuple indicate whether the composite is a record or not.
41    /// The second element of the tuple is a string modifier used for code generation.
42    pub(crate) composite_mapping: IndexMap<&'a Symbol, (bool, String)>,
43    /// Mapping of global identifiers to their associated names.
44    pub(crate) global_mapping: IndexMap<&'a Symbol, String>,
45    /// The variant of the function we are currently traversing.
46    pub(crate) variant: Option<Variant>,
47    /// A reference to program. This is needed to look up external programs.
48    pub(crate) program: &'a Program,
49    /// The program ID of the current program.
50    pub(crate) program_id: Option<ProgramId>,
51    /// A reference to the finalize caller.
52    pub(crate) finalize_caller: Option<Symbol>,
53    /// A counter to track the next available label.
54    pub(crate) next_label: u64,
55    /// The depth of the current conditional block.
56    pub(crate) conditional_depth: u64,
57}
58
59impl<'a> CodeGenerator<'a> {
60    /// Initializes a new `CodeGenerator`.
61    pub fn new(
62        symbol_table: &'a SymbolTable,
63        type_table: &'a TypeTable,
64        struct_graph: &'a StructGraph,
65        _call_graph: &'a CallGraph,
66        program: &'a Program,
67    ) -> Self {
68        // Initialize variable mapping.
69        Self {
70            symbol_table,
71            type_table,
72            struct_graph,
73            _call_graph,
74            next_register: 0,
75            current_function: None,
76            variable_mapping: IndexMap::new(),
77            composite_mapping: IndexMap::new(),
78            global_mapping: IndexMap::new(),
79            variant: None,
80            program,
81            program_id: None,
82            finalize_caller: None,
83            next_label: 0u64,
84            conditional_depth: 0u64,
85        }
86    }
87}
