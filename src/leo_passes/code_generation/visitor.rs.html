17use crate::CompilerState;
18
19use leo_ast::{Function, Program, ProgramId, Variant};
20use leo_span::Symbol;
21
22use indexmap::{IndexMap, IndexSet};
23
24pub struct CodeGeneratingVisitor<'a> {
25    pub state: &'a CompilerState,
26    /// A counter to track the next available register.
27    pub next_register: u64,
28    /// Reference to the current function.
29    pub current_function: Option<&'a Function>,
30    /// Mapping of variables to registers.
31    pub variable_mapping: IndexMap<Symbol, String>,
32    /// Mapping of composite names to a tuple containing metadata associated with the name.
33    /// The first element of the tuple indicate whether the composite is a record or not.
34    /// The second element of the tuple is a string modifier used for code generation.
35    pub composite_mapping: IndexMap<Symbol, (bool, String)>,
36    /// Mapping of global identifiers to their associated names.
37    pub global_mapping: IndexMap<Symbol, String>,
38    /// The variant of the function we are currently traversing.
39    pub variant: Option<Variant>,
40    /// A reference to program. This is needed to look up external programs.
41    pub program: &'a Program,
42    /// The program ID of the current program.
43    pub program_id: Option<ProgramId>,
44    /// A reference to the finalize caller.
45    pub finalize_caller: Option<Symbol>,
46    /// A counter to track the next available label.
47    pub next_label: u64,
48    /// The depth of the current conditional block.
49    pub conditional_depth: u64,
50    /// Internal record input registers of the current function.
51    /// This is necessary as if we output them, we need to clone them.
52    pub internal_record_inputs: IndexSet<String>,
53}
54
55impl CodeGeneratingVisitor<'_> {
56    pub fn next_register(&mut self) -> String {
57        self.next_register += 1;
58        format!("r{}", self.next_register - 1)
59    }
60}
