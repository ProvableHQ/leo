17use crate::CompilerState;
18
19use leo_ast::{Function, Program, ProgramId, Variant};
20use leo_span::Symbol;
21
22use indexmap::{IndexMap, IndexSet};
23
24pub struct CodeGeneratingVisitor<'a> {
25    pub state: &'a CompilerState,
26    /// A counter to track the next available register.
27    pub next_register: u64,
28    /// Reference to the current function.
29    pub current_function: Option<&'a Function>,
30    /// Mapping of variables to registers.
31    pub variable_mapping: IndexMap<Symbol, String>,
32    /// Mapping of composite names to a tuple containing metadata associated with the name.
33    /// The first element of the tuple indicate whether the composite is a record or not.
34    /// The second element of the tuple is a string modifier used for code generation.
35    pub composite_mapping: IndexMap<Symbol, (bool, String)>,
36    /// Mapping of global identifiers to their associated names.
37    pub global_mapping: IndexMap<Symbol, String>,
38    /// The variant of the function we are currently traversing.
39    pub variant: Option<Variant>,
40    /// A reference to program. This is needed to look up external programs.
41    pub program: &'a Program,
42    /// The program ID of the current program.
43    pub program_id: Option<ProgramId>,
44    /// A reference to the finalize caller.
45    pub finalize_caller: Option<Symbol>,
46    /// A counter to track the next available label.
47    pub next_label: u64,
48    /// The depth of the current conditional block.
49    pub conditional_depth: u64,
50    /// Internal record input registers of the current function.
51    /// This is necessary as if we output them, we need to clone them.
52    pub internal_record_inputs: IndexSet<String>,
53}
54
55impl CodeGeneratingVisitor<'_> {
56    pub(crate) fn next_register(&mut self) -> String {
57        self.next_register += 1;
58        format!("r{}", self.next_register - 1)
59    }
60
61    /// Legalize a struct name. If it's already legal, then just keep it as is. For now, this
62    /// expects two possible struct name formats:
63    /// - Names that are generated for const generic structs during monomorphization such as: `Foo::[1u32, 2u32]`. These
64    ///   names are modified according to `transform_generic_struct_name`.
65    /// - All other names which are assumed to be legal (this is not really checked but probably should be).
66    pub(crate) fn legalize_struct_name(input: String) -> String {
67        Self::transform_generic_struct_name(&input).unwrap_or(input)
68    }
69
70    /// Given a struct name as a `&str`, transform it into a legal name if the it happens to be a const generic struct.
71    /// For example, if the name of the struct is `Foo::[1u32, 2u32]`, then transform it to `Foo__90ViPfqSIPb`. The
72    /// suffix is computed using the hash of the string `Foo::[1u32, 2u32]`
73    fn transform_generic_struct_name(input: &str) -> Option<String> {
74        use base62::encode;
75        use regex::Regex;
76        use sha2::{Digest, Sha256};
77
78        // Match format like: foo::[1, 2]
79        let re = Regex::new(r#"^([a-zA-Z_][\w]*)::\[(.*?)\]$"#).unwrap();
80        let captures = re.captures(input)?;
81
82        let ident = captures.get(1)?.as_str();
83
84        // Compute SHA-256 hash of the entire input
85        let mut hasher = Sha256::new();
86        hasher.update(input.as_bytes());
87        let hash = hasher.finalize();
88
89        // Take first 8 bytes and encode as base62
90        let hash_prefix = &hash[..8]; // 4 bytes = 32 bits
91        let hash_number = u64::from_be_bytes([
92            hash_prefix[0],
93            hash_prefix[1],
94            hash_prefix[2],
95            hash_prefix[3],
96            hash_prefix[4],
97            hash_prefix[5],
98            hash_prefix[6],
99            hash_prefix[7],
100        ]);
101        let hash_base62 = encode(hash_number);
102
103        // Format: <truncated_ident>__<hash>
104        let fixed_suffix_len = 2 + hash_base62.len(); // __ + hash
105        let max_ident_len = 31 - fixed_suffix_len;
106
107        Some(format!("{ident:.max_ident_len$}__{hash_base62}"))
108    }
109}
