50/// Implement the necessary methods to visit nodes in the AST.
51impl CodeGeneratingVisitor<'_> {
52    pub fn visit_expression(&mut self, input: &Expression) -> (String, String) {
53        match input {
54            Expression::Array(expr) => self.visit_array(expr),
55            Expression::ArrayAccess(expr) => self.visit_array_access(expr),
56            Expression::AssociatedConstant(expr) => self.visit_associated_constant(expr),
57            Expression::AssociatedFunction(expr) => self.visit_associated_function(expr),
58            Expression::Binary(expr) => self.visit_binary(expr),
59            Expression::Call(expr) => self.visit_call(expr),
60            Expression::Cast(expr) => self.visit_cast(expr),
61            Expression::Struct(expr) => self.visit_struct_init(expr),
62            Expression::Err(expr) => self.visit_err(expr),
63            Expression::Identifier(expr) => self.visit_identifier(expr),
64            Expression::Literal(expr) => self.visit_value(expr),
65            Expression::Locator(expr) => self.visit_locator(expr),
66            Expression::MemberAccess(expr) => self.visit_member_access(expr),
67            Expression::Ternary(expr) => self.visit_ternary(expr),
68            Expression::Tuple(expr) => self.visit_tuple(expr),
69            Expression::TupleAccess(_) => panic!("Tuple accesses should not appear in the AST at this point."),
70            Expression::Unary(expr) => self.visit_unary(expr),
71            Expression::Unit(expr) => self.visit_unit(expr),
72        }
73    }
74
75    fn visit_identifier(&mut self, input: &Identifier) -> (String, String) {
76        (
77            self.variable_mapping.get(&input.name).or_else(|| self.global_mapping.get(&input.name)).unwrap().clone(),
78            String::new(),
79        )
80    }

86    fn visit_value(&mut self, input: &Literal) -> (String, String) {
87        // AVM can only parse decimal numbers.
88        let literal = if let LiteralVariant::Unsuffixed(value) = &input.variant {
89            // For unsuffixed lierals, consult the `type_table` for their types. The type checker
90            // ensures that their type can only be `Integer`, `Field`, `Group`, or `Scalar`.
91            match self.state.type_table.get(&input.id) {
92                Some(Type::Integer(int_ty)) => Literal {
93                    variant: LiteralVariant::Integer(int_ty, value.clone()),
94                    id: self.state.node_builder.next_id(),
95                    span: input.span,
96                },
97                Some(Type::Field) => Literal {
98                    variant: LiteralVariant::Field(value.clone()),
99                    id: self.state.node_builder.next_id(),
100                    span: input.span,
101                },
102                Some(Type::Group) => Literal {
103                    variant: LiteralVariant::Group(value.clone()),
104                    id: self.state.node_builder.next_id(),
105                    span: input.span,
106                },
107                Some(Type::Scalar) => Literal {
108                    variant: LiteralVariant::Scalar(value.clone()),
109                    id: self.state.node_builder.next_id(),
110                    span: input.span,
111                },
112                _ => panic!(
113                    "Unexpected type for unsuffixed integer literal. This should have been caught by the type checker"
114                ),
115            }
116        } else {
117            input.clone()
118        };
119        (format!("{}", literal.display_decimal()), String::new())
120    }

122    fn visit_locator(&mut self, input: &LocatorExpression) -> (String, String) {
123        if input.program.name.name == self.program_id.expect("Locators only appear within programs.").name.name {
124            // This locator refers to the current program, so we only output the name, not the program.
125            (format!("{}", input.name), String::new())
126        } else {
127            (format!("{input}"), String::new())
128        }
129    }

131    fn visit_binary(&mut self, input: &BinaryExpression) -> (String, String) {
132        let (left_operand, left_instructions) = self.visit_expression(&input.left);
133        let (right_operand, right_instructions) = self.visit_expression(&input.right);
134
135        let opcode = match input.op {
136            BinaryOperation::Add => String::from("add"),
137            BinaryOperation::AddWrapped => String::from("add.w"),
138            BinaryOperation::And => String::from("and"),
139            BinaryOperation::BitwiseAnd => String::from("and"),
140            BinaryOperation::Div => String::from("div"),
141            BinaryOperation::DivWrapped => String::from("div.w"),
142            BinaryOperation::Eq => String::from("is.eq"),
143            BinaryOperation::Gte => String::from("gte"),
144            BinaryOperation::Gt => String::from("gt"),
145            BinaryOperation::Lte => String::from("lte"),
146            BinaryOperation::Lt => String::from("lt"),
147            BinaryOperation::Mod => String::from("mod"),
148            BinaryOperation::Mul => String::from("mul"),
149            BinaryOperation::MulWrapped => String::from("mul.w"),
150            BinaryOperation::Nand => String::from("nand"),
151            BinaryOperation::Neq => String::from("is.neq"),
152            BinaryOperation::Nor => String::from("nor"),
153            BinaryOperation::Or => String::from("or"),
154            BinaryOperation::BitwiseOr => String::from("or"),
155            BinaryOperation::Pow => String::from("pow"),
156            BinaryOperation::PowWrapped => String::from("pow.w"),
157            BinaryOperation::Rem => String::from("rem"),
158            BinaryOperation::RemWrapped => String::from("rem.w"),
159            BinaryOperation::Shl => String::from("shl"),
160            BinaryOperation::ShlWrapped => String::from("shl.w"),
161            BinaryOperation::Shr => String::from("shr"),
162            BinaryOperation::ShrWrapped => String::from("shr.w"),
163            BinaryOperation::Sub => String::from("sub"),
164            BinaryOperation::SubWrapped => String::from("sub.w"),
165            BinaryOperation::Xor => String::from("xor"),
166        };
167
168        let destination_register = self.next_register();
169        let binary_instruction = format!("    {opcode} {left_operand} {right_operand} into {destination_register};\n",);
170
171        // Concatenate the instructions.
172        let mut instructions = left_instructions;
173        instructions.push_str(&right_instructions);
174        instructions.push_str(&binary_instruction);
175
176        (destination_register, instructions)
177    }

179    fn visit_cast(&mut self, input: &CastExpression) -> (String, String) {
180        let (expression_operand, mut instructions) = self.visit_expression(&input.expression);
181
182        // Construct the destination register.
183        let destination_register = self.next_register();
184
185        let cast_instruction = format!(
186            "    cast {expression_operand} into {destination_register} as {};\n",
187            Self::visit_type(&input.type_)
188        );
189
190        // Concatenate the instructions.
191        instructions.push_str(&cast_instruction);
192
193        (destination_register, instructions)
194    }

196    fn visit_array(&mut self, input: &ArrayExpression) -> (String, String) {
197        let mut expression_operands = String::new();
198        let mut instructions = String::new();
199        for (operand, operand_instructions) in input.elements.iter().map(|expr| self.visit_expression(expr)) {
200            let space = if expression_operands.is_empty() { "" } else { " " };
201            write!(&mut expression_operands, "{space}{operand}").unwrap();
202            instructions.push_str(&operand_instructions);
203        }
204
205        // Construct the destination register.
206        let destination_register = self.next_register();
207
208        // Get the array type.
209        let Some(array_type @ Type::Array(..)) = self.state.type_table.get(&input.id) else {
210            panic!("All types should be known at this phase of compilation");
211        };
212        let array_type: String = Self::visit_type(&array_type);
213
214        let array_instruction =
215            format!("    cast {expression_operands} into {destination_register} as {};\n", array_type);
216
217        // Concatenate the instructions.
218        instructions.push_str(&array_instruction);
219
220        (destination_register, instructions)
221    }

223    fn visit_unary(&mut self, input: &UnaryExpression) -> (String, String) {
224        let (expression_operand, expression_instructions) = self.visit_expression(&input.receiver);
225
226        // Note that non-empty suffixes must be preceded by a space.
227        let (opcode, suffix) = match input.op {
228            UnaryOperation::Abs => ("abs", ""),
229            UnaryOperation::AbsWrapped => ("abs.w", ""),
230            UnaryOperation::Double => ("double", ""),
231            UnaryOperation::Inverse => ("inv", ""),
232            UnaryOperation::Not => ("not", ""),
233            UnaryOperation::Negate => ("neg", ""),
234            UnaryOperation::Square => ("square", ""),
235            UnaryOperation::SquareRoot => ("sqrt", ""),
236            UnaryOperation::ToXCoordinate => ("cast", " as group.x"),
237            UnaryOperation::ToYCoordinate => ("cast", " as group.y"),
238        };
239
240        let destination_register = self.next_register();
241        let unary_instruction = format!("    {opcode} {expression_operand} into {destination_register}{suffix};\n");
242
243        // Concatenate the instructions.
244        let mut instructions = expression_instructions;
245        instructions.push_str(&unary_instruction);
246
247        (destination_register, instructions)
248    }
249
250    fn visit_ternary(&mut self, input: &TernaryExpression) -> (String, String) {
251        let (condition_operand, condition_instructions) = self.visit_expression(&input.condition);
252        let (if_true_operand, if_true_instructions) = self.visit_expression(&input.if_true);
253        let (if_false_operand, if_false_instructions) = self.visit_expression(&input.if_false);
254
255        let destination_register = self.next_register();
256        let ternary_instruction = format!(
257            "    ternary {condition_operand} {if_true_operand} {if_false_operand} into {destination_register};\n",
258        );
259
260        // Concatenate the instructions.
261        let mut instructions = condition_instructions;
262        instructions.push_str(&if_true_instructions);
263        instructions.push_str(&if_false_instructions);
264        instructions.push_str(&ternary_instruction);
265
266        (destination_register, instructions)
267    }

269    fn visit_struct_init(&mut self, input: &StructExpression) -> (String, String) {
270        // Lookup struct or record.
271        let name = if let Some((is_record, type_)) = self.composite_mapping.get(&input.name.name) {
272            if *is_record {
273                // record.private;
274                format!("{}.{type_}", input.name)
275            } else {
276                // foo; // no visibility for structs
277                input.name.to_string()
278            }
279        } else {
280            panic!("All composite types should be known at this phase of compilation")
281        };
282
283        // Initialize instruction builder strings.
284        let mut instructions = String::new();
285        let mut struct_init_instruction = String::from("    cast ");
286
287        // Visit each struct member and accumulate instructions from expressions.
288        for member in input.members.iter() {
289            let operand = if let Some(expr) = member.expression.as_ref() {
290                // Visit variable expression.
291                let (variable_operand, variable_instructions) = self.visit_expression(expr);
292                instructions.push_str(&variable_instructions);
293
294                variable_operand
295            } else {
296                // Push operand identifier.
297                let (ident_operand, ident_instructions) = self.visit_identifier(&member.identifier);
298                instructions.push_str(&ident_instructions);
299
300                ident_operand
301            };
302
303            // Push operand name to struct init instruction.
304            write!(struct_init_instruction, "{operand} ").expect("failed to write to string");
305        }
306
307        // Push destination register to struct init instruction.
308        let destination_register = self.next_register();
309        writeln!(struct_init_instruction, "into {destination_register} as {name};",)
310            .expect("failed to write to string");
311
312        instructions.push_str(&struct_init_instruction);
313
314        (destination_register, instructions)
315    }

317    fn visit_array_access(&mut self, input: &ArrayAccess) -> (String, String) {
318        let (array_operand, _) = self.visit_expression(&input.array);
319
320        assert!(
321            matches!(self.state.type_table.get(&input.index.id()), Some(Type::Integer(_))),
322            "unexpected type for for array index. This should have been caught by the type checker."
323        );
324
325        let index_operand = match &input.index {
326            Expression::Literal(Literal {
327                variant: LiteralVariant::Integer(_, s) | LiteralVariant::Unsuffixed(s),
328                ..
329            }) => format!("{s}u32"),
330            _ => panic!("Array indices must be integer literals"),
331        };
332
333        (format!("{array_operand}[{index_operand}]"), String::new())
334    }

336    fn visit_member_access(&mut self, input: &MemberAccess) -> (String, String) {

338        let member_access = format!("{}.{}", inner_expr, input.name);

348    // Pedersen64::hash() -> hash.ped64
349    fn visit_associated_function(&mut self, input: &AssociatedFunctionExpression) -> (String, String) {
350        let mut instructions = String::new();
351
352        // Visit each function argument and accumulate instructions from expressions.
353        let arguments = input
354            .arguments
355            .iter()
356            .map(|argument| {
357                let (arg_string, arg_instructions) = self.visit_expression(argument);
358                instructions.push_str(&arg_instructions);
359                arg_string
360            })
361            .collect::<Vec<_>>();
362
363        // Helper function to construct the instruction associated with a simple function call.
364        // This assumes that the function call has one output.
365        let mut construct_simple_function_call = |function: &Identifier, variant: &str, arguments: Vec<String>| {
366            // Split function into [opcode, return type] e.g. hash_to_field -> [hash, field]
367            let function_name = function.name.to_string();
368            let mut names = function_name.split("_to_");
369            let opcode = names.next().expect("failed to get opcode");
370            let return_type = names.next().expect("failed to get type");
371
372            let mut instruction = format!("    {opcode}.{variant}");
373            for argument in arguments {
374                write!(instruction, " {argument}").expect("failed to write to string");
375            }
376            let destination_register = self.next_register();
377            writeln!(instruction, " into {destination_register} as {return_type};").expect("failed to write to string");
378            (destination_register, instruction)
379        };
380
381        // Construct the instruction.
382        let (destination, instruction) = match input.variant.name {
383            sym::BHP256 => construct_simple_function_call(&input.name, "bhp256", arguments),
384            sym::BHP512 => construct_simple_function_call(&input.name, "bhp512", arguments),
385            sym::BHP768 => construct_simple_function_call(&input.name, "bhp768", arguments),
386            sym::BHP1024 => construct_simple_function_call(&input.name, "bhp1024", arguments),
387            sym::Keccak256 => construct_simple_function_call(&input.name, "keccak256", arguments),
388            sym::Keccak384 => construct_simple_function_call(&input.name, "keccak384", arguments),
389            sym::Keccak512 => construct_simple_function_call(&input.name, "keccak512", arguments),
390            sym::Pedersen64 => construct_simple_function_call(&input.name, "ped64", arguments),
391            sym::Pedersen128 => construct_simple_function_call(&input.name, "ped128", arguments),
392            sym::Poseidon2 => construct_simple_function_call(&input.name, "psd2", arguments),
393            sym::Poseidon4 => construct_simple_function_call(&input.name, "psd4", arguments),
394            sym::Poseidon8 => construct_simple_function_call(&input.name, "psd8", arguments),
395            sym::SHA3_256 => construct_simple_function_call(&input.name, "sha3_256", arguments),
396            sym::SHA3_384 => construct_simple_function_call(&input.name, "sha3_384", arguments),
397            sym::SHA3_512 => construct_simple_function_call(&input.name, "sha3_512", arguments),
398            sym::Mapping => match input.name.name {
399                sym::get => {
400                    let mut instruction = "    get".to_string();
401                    let destination_register = self.next_register();
402                    // Write the mapping name and the key.
403                    writeln!(instruction, " {}[{}] into {destination_register};", arguments[0], arguments[1])
404                        .expect("failed to write to string");
405                    (destination_register, instruction)
406                }
407                sym::get_or_use => {
408                    let mut instruction = "    get.or_use".to_string();
409                    let destination_register = self.next_register();
410                    // Write the mapping name, the key, and the default value.
411                    writeln!(
412                        instruction,
413                        " {}[{}] {} into {destination_register};",
414                        arguments[0], arguments[1], arguments[2]
415                    )
416                    .expect("failed to write to string");
417                    (destination_register, instruction)
418                }
419                sym::set => {
420                    let mut instruction = "    set".to_string();
421                    // Write the value, mapping name, and the key.
422                    writeln!(instruction, " {} into {}[{}];", arguments[2], arguments[0], arguments[1])
423                        .expect("failed to write to string");
424                    (String::new(), instruction)
425                }
426                sym::remove => {
427                    let mut instruction = "    remove".to_string();
428                    // Write the mapping name and the key.
429                    writeln!(instruction, " {}[{}];", arguments[0], arguments[1]).expect("failed to write to string");
430                    (String::new(), instruction)
431                }
432                sym::contains => {
433                    let mut instruction = "    contains".to_string();
434                    let destination_register = self.next_register();
435                    // Write the mapping name and the key.
436                    writeln!(instruction, " {}[{}] into {destination_register};", arguments[0], arguments[1])
437                        .expect("failed to write to string");
438                    (destination_register, instruction)
439                }
440                _ => panic!("The only variants of Mapping are get, get_or, and set"),
441            },
442            sym::group => {
443                match input.name {
444                    Identifier { name: sym::to_x_coordinate, .. } => {
445                        let mut instruction = "    cast".to_string();
446                        let destination_register = self.next_register();
447                        // Write the argument and the destination register.
448                        writeln!(instruction, " {} into {destination_register} as group.x;", arguments[0],)
449                            .expect("failed to write to string");
450                        (destination_register, instruction)
451                    }
452                    Identifier { name: sym::to_y_coordinate, .. } => {
453                        let mut instruction = "    cast".to_string();
454                        let destination_register = self.next_register();
455                        // Write the argument and the destination register.
456                        writeln!(instruction, " {} into {destination_register} as group.y;", arguments[0],)
457                            .expect("failed to write to string");
458                        (destination_register, instruction)
459                    }
460                    _ => panic!("The only associated methods of group are to_x_coordinate and to_y_coordinate"),
461                }
462            }
463            sym::ChaCha => {
464                // Get the destination register.
465                let destination_register = self.next_register();
466                // Construct the instruction template.
467                let mut instruction = format!("    rand.chacha into {destination_register} as ");
468                // Write the return type.
469                match input.name {
470                    Identifier { name: sym::rand_address, .. } => writeln!(instruction, "address;"),
471                    Identifier { name: sym::rand_bool, .. } => writeln!(instruction, "boolean;"),
472                    Identifier { name: sym::rand_field, .. } => writeln!(instruction, "field;"),
473                    Identifier { name: sym::rand_group, .. } => writeln!(instruction, "group;"),
474                    Identifier { name: sym::rand_i8, .. } => writeln!(instruction, "i8;"),
475                    Identifier { name: sym::rand_i16, .. } => writeln!(instruction, "i16;"),
476                    Identifier { name: sym::rand_i32, .. } => writeln!(instruction, "i32;"),
477                    Identifier { name: sym::rand_i64, .. } => writeln!(instruction, "i64;"),
478                    Identifier { name: sym::rand_i128, .. } => writeln!(instruction, "i128;"),
479                    Identifier { name: sym::rand_scalar, .. } => writeln!(instruction, "scalar;"),
480                    Identifier { name: sym::rand_u8, .. } => writeln!(instruction, "u8;"),
481                    Identifier { name: sym::rand_u16, .. } => writeln!(instruction, "u16;"),
482                    Identifier { name: sym::rand_u32, .. } => writeln!(instruction, "u32;"),
483                    Identifier { name: sym::rand_u64, .. } => writeln!(instruction, "u64;"),
484                    Identifier { name: sym::rand_u128, .. } => writeln!(instruction, "u128;"),
485                    _ => panic!("The only associated methods of ChaCha are `rand_*`"),
486                }
487                .expect("failed to write to string");
488                (destination_register, instruction)
489            }
490            sym::signature => {
491                let mut instruction = "    sign.verify".to_string();
492                let destination_register = self.next_register();
493                // Write the arguments and the destination register.
494                writeln!(
495                    instruction,
496                    " {} {} {} into {destination_register};",
497                    arguments[0], arguments[1], arguments[2]
498                )
499                .expect("failed to write to string");
500                (destination_register, instruction)
501            }
502            sym::Future => {
503                let mut instruction = "    await".to_string();
504                writeln!(instruction, " {};", arguments[0]).expect("failed to write to string");
505                (String::new(), instruction)
506            }
507            sym::CheatCode => {
508                (String::new(), String::new())
509                // Do nothing. Cheat codes do not generate instructions.
510            }
511            _ => {
512                panic!("All core functions should be known at this phase of compilation")
513            }
514        };
515        // Add the instruction to the list of instructions.
516        instructions.push_str(&instruction);
517
518        (destination, instructions)
519    }

521    fn visit_call(&mut self, input: &CallExpression) -> (String, String) {
522        // Lookup the function return type.
523        let function_name = match input.function.borrow() {
524            Expression::Identifier(identifier) => identifier.name,
525            _ => panic!("Parsing guarantees that a function name is always an identifier."),
526        };
527
528        let caller_program = self.program_id.expect("Calls only appear within programs.").name.name;
529        let callee_program = input.program.unwrap_or(caller_program);
530
531        let func_symbol = self
532            .state
533            .symbol_table
534            .lookup_function(Location::new(callee_program, function_name))
535            .expect("Type checking guarantees functions exist");
536
537        // Need to determine the program the function originated from as well as if the function has a finalize block.
538        let mut call_instruction = if caller_program != callee_program {
539            // All external functions must be defined as stubs.
540            assert!(
541                self.program.stubs.get(&callee_program).is_some(),
542                "Type checking guarantees that imported and stub programs are present."
543            );
544            format!("    call {}.aleo/{}", callee_program, input.function)
545        } else if func_symbol.function.variant.is_async() {
546            format!("    async {}", self.current_function.unwrap().identifier)
547        } else {
548            format!("    call {}", input.function)
549        };
550
551        let mut instructions = String::new();
552
553        for argument in input.arguments.iter() {
554            let (argument, argument_instructions) = self.visit_expression(argument);
555            write!(call_instruction, " {argument}").expect("failed to write to string");
556            instructions.push_str(&argument_instructions);
557        }
558
559        // Initialize storage for the destination registers.
560        let mut destinations = Vec::new();
561
562        // Create operands for the output registers.
563        match func_symbol.function.output_type.clone() {
564            Type::Unit => {} // Do nothing
565            Type::Tuple(tuple) => match tuple.length() {
566                0 | 1 => panic!("Parsing guarantees that a tuple type has at least two elements"),
567                len => {
568                    for _ in 0..len {
569                        destinations.push(self.next_register());
570                    }
571                }
572            },
573            _ => {
574                destinations.push(self.next_register());
575            }
576        }
577
578        // Add a register for async functions to represent the future created.
579        if func_symbol.function.variant == Variant::AsyncFunction {
580            destinations.push(self.next_register());
581        }
582
583        // Construct the output operands. These are the destination registers **without** the future.
584        let output_operands = destinations.join(" ");
585
586        // If destination registers were created, write them to the call instruction.
587        if !destinations.is_empty() {
588            write!(call_instruction, " into").expect("failed to write to string");
589            for destination in &destinations {
590                write!(call_instruction, " {}", destination).expect("failed to write to string");
591            }
592        }
593
594        // Write the closing semicolon.
595        writeln!(call_instruction, ";").expect("failed to write to string");
596
597        // Push the call instruction to the list of instructions.
598        instructions.push_str(&call_instruction);
599
600        // Return the output operands and the instructions.
601        (output_operands, instructions)
602    }
603
604    fn visit_tuple(&mut self, input: &TupleExpression) -> (String, String) {
605        // Need to return a single string here so we will join the tuple elements with ' '
606        // and split them after this method is called.
607        let mut tuple_elements = Vec::with_capacity(input.elements.len());
608        let mut instructions = String::new();
609
610        // Visit each tuple element and accumulate instructions from expressions.
611        for element in input.elements.iter() {
612            let (element, element_instructions) = self.visit_expression(element);
613            tuple_elements.push(element);
614            instructions.push_str(&element_instructions);
615        }
616
617        // CAUTION: does not return the destination_register.
618        (tuple_elements.join(" "), instructions)
619    }
620
621    fn visit_unit(&mut self, _input: &UnitExpression) -> (String, String) {
622        panic!("`UnitExpression`s should not be visited during code generation.")
623    }

625    pub fn clone_register(&mut self, register: &str, typ: &Type) -> (String, String) {
626        let new_reg = self.next_register();
627        match typ {
628            Type::Address
629            | Type::Boolean
630            | Type::Field
631            | Type::Group
632            | Type::Scalar
633            | Type::Signature
634            | Type::Integer(_) => {
635                // These types can be cloned just by casting them to themselves.
636                let instruction = format!("    cast {register} into {new_reg} as {typ};\n");
637                (new_reg, instruction)
638            }
639
640            Type::Array(array_type) => {
641                // We need to cast the old array's members into the new array.
642                let mut instruction = "    cast ".to_string();
643                for i in 0..array_type.length() {
644                    write!(&mut instruction, "{register}[{i}u32] ").unwrap();
645                }
646                writeln!(&mut instruction, "into {new_reg} as {};", Self::visit_type(typ)).unwrap();
647                (new_reg, instruction)
648            }
649
650            Type::Composite(comp_ty) => {
651                // We need to cast the old struct or record's members into the new one.
652                let program = comp_ty.program.unwrap_or(self.program_id.unwrap().name.name);
653                let location = Location::new(program, comp_ty.id.name);
654                let comp = self
655                    .state
656                    .symbol_table
657                    .lookup_record(location)
658                    .or_else(|| self.state.symbol_table.lookup_struct(comp_ty.id.name))
659                    .unwrap();
660                let mut instruction = "    cast ".to_string();
661                for member in &comp.members {
662                    write!(&mut instruction, "{register}.{} ", member.identifier.name).unwrap();
663                }
664                writeln!(
665                    &mut instruction,
666                    "into {new_reg} as {};",
667                    // We call `..with_visibility` just so we get the `.record` appended if it's a record.
668                    self.visit_type_with_visibility(typ, leo_ast::Mode::None)
669                )
670                .unwrap();
671                (new_reg, instruction)
672            }
673
674            Type::Mapping(..)
675            | Type::Future(..)
676            | Type::Tuple(..)
677            | Type::Identifier(..)
678            | Type::String
679            | Type::Unit
680            | Type::Numeric
681            | Type::Err => panic!("Objects of type {typ} cannot be cloned."),
682        }
683    }
