60/// Implement the necessary methods to visit nodes in the AST.
61impl CodeGeneratingVisitor<'_> {
62    pub fn visit_expression(&mut self, input: &Expression) -> (String, String) {
63        let is_empty_type = self.state.type_table.get(&input.id()).map(|ty| ty.is_empty()).unwrap_or(false);
64        let is_pure = input.is_pure();
65
66        if is_empty_type && is_pure {
67            // ignore expresssion
68            return (String::new(), String::new());
69        }
70
71        match input {
72            Expression::Array(expr) => self.visit_array(expr),
73            Expression::ArrayAccess(expr) => self.visit_array_access(expr),
74            Expression::AssociatedConstant(expr) => self.visit_associated_constant(expr),
75            Expression::AssociatedFunction(expr) => self.visit_associated_function(expr),
76            Expression::Async(expr) => self.visit_async(expr),
77            Expression::Binary(expr) => self.visit_binary(expr),
78            Expression::Call(expr) => self.visit_call(expr),
79            Expression::Cast(expr) => self.visit_cast(expr),
80            Expression::Struct(expr) => self.visit_struct_init(expr),
81            Expression::Err(expr) => self.visit_err(expr),
82            Expression::Path(expr) => self.visit_path(expr),
83            Expression::Literal(expr) => self.visit_value(expr),
84            Expression::Locator(expr) => self.visit_locator(expr),
85            Expression::MemberAccess(expr) => self.visit_member_access(expr),
86            Expression::Repeat(expr) => self.visit_repeat(expr),
87            Expression::Ternary(expr) => self.visit_ternary(expr),
88            Expression::Tuple(expr) => self.visit_tuple(expr),
89            Expression::TupleAccess(_) => panic!("Tuple accesses should not appear in the AST at this point."),
90            Expression::Unary(expr) => self.visit_unary(expr),
91            Expression::Unit(expr) => self.visit_unit(expr),
92        }
93    }

95    fn visit_path(&mut self, input: &Path) -> (String, String) {
96        // The only relevant paths here are paths to local variable or to mappings, so we really only care about their
97        // names since mappings are only allowed in the top level program scope
98        let var_name = input.identifier().name;
99        (
100            self.variable_mapping.get(&var_name).or_else(|| self.global_mapping.get(&var_name)).unwrap().clone(),
101            String::new(),
102        )
103    }

109    fn visit_value(&mut self, input: &Literal) -> (String, String) {
110        // AVM can only parse decimal numbers.
111        let literal = if let LiteralVariant::Unsuffixed(value) = &input.variant {
112            // For unsuffixed lierals, consult the `type_table` for their types. The type checker
113            // ensures that their type can only be `Integer`, `Field`, `Group`, or `Scalar`.
114            match self.state.type_table.get(&input.id) {
115                Some(Type::Integer(int_ty)) => Literal {
116                    variant: LiteralVariant::Integer(int_ty, value.clone()),
117                    id: self.state.node_builder.next_id(),
118                    span: input.span,
119                },
120                Some(Type::Field) => Literal {
121                    variant: LiteralVariant::Field(value.clone()),
122                    id: self.state.node_builder.next_id(),
123                    span: input.span,
124                },
125                Some(Type::Group) => Literal {
126                    variant: LiteralVariant::Group(value.clone()),
127                    id: self.state.node_builder.next_id(),
128                    span: input.span,
129                },
130                Some(Type::Scalar) => Literal {
131                    variant: LiteralVariant::Scalar(value.clone()),
132                    id: self.state.node_builder.next_id(),
133                    span: input.span,
134                },
135                _ => panic!(
136                    "Unexpected type for unsuffixed integer literal. This should have been caught by the type checker"
137                ),
138            }
139        } else {
140            input.clone()
141        };
142        (format!("{}", literal.display_decimal()), String::new())
143    }

145    fn visit_locator(&mut self, input: &LocatorExpression) -> (String, String) {
146        if input.program.name.name == self.program_id.expect("Locators only appear within programs.").name.name {
147            // This locator refers to the current program, so we only output the name, not the program.
148            (format!("{}", input.name), String::new())
149        } else {
150            (format!("{input}"), String::new())
151        }
152    }

154    fn visit_binary(&mut self, input: &BinaryExpression) -> (String, String) {
155        let (left_operand, left_instructions) = self.visit_expression(&input.left);
156        let (right_operand, right_instructions) = self.visit_expression(&input.right);
157
158        let opcode = match input.op {
159            BinaryOperation::Add => String::from("add"),
160            BinaryOperation::AddWrapped => String::from("add.w"),
161            BinaryOperation::And => String::from("and"),
162            BinaryOperation::BitwiseAnd => String::from("and"),
163            BinaryOperation::Div => String::from("div"),
164            BinaryOperation::DivWrapped => String::from("div.w"),
165            BinaryOperation::Eq => String::from("is.eq"),
166            BinaryOperation::Gte => String::from("gte"),
167            BinaryOperation::Gt => String::from("gt"),
168            BinaryOperation::Lte => String::from("lte"),
169            BinaryOperation::Lt => String::from("lt"),
170            BinaryOperation::Mod => String::from("mod"),
171            BinaryOperation::Mul => String::from("mul"),
172            BinaryOperation::MulWrapped => String::from("mul.w"),
173            BinaryOperation::Nand => String::from("nand"),
174            BinaryOperation::Neq => String::from("is.neq"),
175            BinaryOperation::Nor => String::from("nor"),
176            BinaryOperation::Or => String::from("or"),
177            BinaryOperation::BitwiseOr => String::from("or"),
178            BinaryOperation::Pow => String::from("pow"),
179            BinaryOperation::PowWrapped => String::from("pow.w"),
180            BinaryOperation::Rem => String::from("rem"),
181            BinaryOperation::RemWrapped => String::from("rem.w"),
182            BinaryOperation::Shl => String::from("shl"),
183            BinaryOperation::ShlWrapped => String::from("shl.w"),
184            BinaryOperation::Shr => String::from("shr"),
185            BinaryOperation::ShrWrapped => String::from("shr.w"),
186            BinaryOperation::Sub => String::from("sub"),
187            BinaryOperation::SubWrapped => String::from("sub.w"),
188            BinaryOperation::Xor => String::from("xor"),
189        };
190
191        let destination_register = self.next_register();
192        let binary_instruction = format!("    {opcode} {left_operand} {right_operand} into {destination_register};\n",);
193
194        // Concatenate the instructions.
195        let mut instructions = left_instructions;
196        instructions.push_str(&right_instructions);
197        instructions.push_str(&binary_instruction);
198
199        (destination_register, instructions)
200    }

202    fn visit_cast(&mut self, input: &CastExpression) -> (String, String) {
203        let (expression_operand, mut instructions) = self.visit_expression(&input.expression);
204
205        // Construct the destination register.
206        let destination_register = self.next_register();
207
208        let cast_instruction = format!(
209            "    cast {expression_operand} into {destination_register} as {};\n",
210            Self::visit_type(&input.type_)
211        );
212
213        // Concatenate the instructions.
214        instructions.push_str(&cast_instruction);
215
216        (destination_register, instructions)
217    }

219    fn visit_array(&mut self, input: &ArrayExpression) -> (String, String) {
220        let mut expression_operands = String::new();
221        let mut instructions = String::new();
222
223        for (operand, operand_instructions) in input.elements.iter().map(|expr| self.visit_expression(expr)) {
224            let space = if expression_operands.is_empty() { "" } else { " " };
225            write!(&mut expression_operands, "{space}{operand}").unwrap();
226            instructions.push_str(&operand_instructions);
227        }
228
229        // Construct the destination register.
230        let destination_register = self.next_register();
231
232        // Get the array type.
233        let Some(array_type @ Type::Array(..)) = self.state.type_table.get(&input.id) else {
234            panic!("All types should be known at this phase of compilation");
235        };
236        let array_type: String = Self::visit_type(&array_type);
237
238        let array_instruction =
239            format!("    cast {expression_operands} into {destination_register} as {array_type};\n");
240
241        // Concatenate the instructions.
242        instructions.push_str(&array_instruction);
243
244        (destination_register, instructions)
245    }

247    fn visit_unary(&mut self, input: &UnaryExpression) -> (String, String) {
248        let (expression_operand, expression_instructions) = self.visit_expression(&input.receiver);
249
250        // Note that non-empty suffixes must be preceded by a space.
251        let (opcode, suffix) = match input.op {
252            UnaryOperation::Abs => ("abs", ""),
253            UnaryOperation::AbsWrapped => ("abs.w", ""),
254            UnaryOperation::Double => ("double", ""),
255            UnaryOperation::Inverse => ("inv", ""),
256            UnaryOperation::Not => ("not", ""),
257            UnaryOperation::Negate => ("neg", ""),
258            UnaryOperation::Square => ("square", ""),
259            UnaryOperation::SquareRoot => ("sqrt", ""),
260            UnaryOperation::ToXCoordinate => ("cast", " as group.x"),
261            UnaryOperation::ToYCoordinate => ("cast", " as group.y"),
262        };
263
264        let destination_register = self.next_register();
265        let unary_instruction = format!("    {opcode} {expression_operand} into {destination_register}{suffix};\n");
266
267        // Concatenate the instructions.
268        let mut instructions = expression_instructions;
269        instructions.push_str(&unary_instruction);
270
271        (destination_register, instructions)
272    }

274    fn visit_ternary(&mut self, input: &TernaryExpression) -> (String, String) {
275        let (condition_operand, condition_instructions) = self.visit_expression(&input.condition);
276        let (if_true_operand, if_true_instructions) = self.visit_expression(&input.if_true);
277        let (if_false_operand, if_false_instructions) = self.visit_expression(&input.if_false);
278
279        let destination_register = self.next_register();
280        let ternary_instruction = format!(
281            "    ternary {condition_operand} {if_true_operand} {if_false_operand} into {destination_register};\n",
282        );
283
284        // Concatenate the instructions.
285        let mut instructions = condition_instructions;
286        instructions.push_str(&if_true_instructions);
287        instructions.push_str(&if_false_instructions);
288        instructions.push_str(&ternary_instruction);
289
290        (destination_register, instructions)
291    }

293    fn visit_struct_init(&mut self, input: &StructExpression) -> (String, String) {
294        // Lookup struct or record.
295        let name = if let Some((is_record, type_)) = self.composite_mapping.get(&input.path.absolute_path()) {
296            if *is_record {
297                // record.private;
298                let [record_name] = &input.path.absolute_path()[..] else {
299                    panic!("Absolute paths to records can only have a single segment at this stage.")
300                };
301                format!("{record_name}.{type_}")
302            } else {
303                // foo; // no visibility for structs
304                Self::legalize_path(&input.path.absolute_path()).expect("path format cannot be legalized at this point")
305            }
306        } else {
307            panic!("All composite types should be known at this phase of compilation")
308        };
309
310        // Initialize instruction builder strings.
311        let mut instructions = String::new();
312        let mut struct_init_instruction = String::from("    cast ");
313
314        // Visit each struct member and accumulate instructions from expressions.
315        for member in input.members.iter() {
316            let operand = if let Some(expr) = member.expression.as_ref() {
317                // Visit variable expression.
318                let (variable_operand, variable_instructions) = self.visit_expression(expr);
319                instructions.push_str(&variable_instructions);
320
321                variable_operand
322            } else {
323                // Push operand identifier.
324                let (ident_operand, ident_instructions) =
325                    self.visit_path(&Path::from(member.identifier).into_absolute());
326                instructions.push_str(&ident_instructions);
327
328                ident_operand
329            };
330
331            // Push operand name to struct init instruction.
332            write!(struct_init_instruction, "{operand} ").expect("failed to write to string");
333        }
334
335        // Push destination register to struct init instruction.
336        let destination_register = self.next_register();
337        writeln!(struct_init_instruction, "into {destination_register} as {name};",)
338            .expect("failed to write to string");
339
340        instructions.push_str(&struct_init_instruction);
341
342        (destination_register, instructions)
343    }

345    fn visit_array_access(&mut self, input: &ArrayAccess) -> (String, String) {
346        let (array_operand, _) = self.visit_expression(&input.array);
347
348        assert!(
349            matches!(self.state.type_table.get(&input.index.id()), Some(Type::Integer(_))),
350            "unexpected type for for array index. This should have been caught by the type checker."
351        );
352
353        let index_operand = match &input.index {
354            Expression::Literal(Literal {
355                variant: LiteralVariant::Integer(_, s) | LiteralVariant::Unsuffixed(s),
356                ..
357            }) => format!("{s}u32"),
358            _ => panic!("Array indices must be integer literals"),
359        };
360
361        (format!("{array_operand}[{index_operand}]"), String::new())
362    }

364    fn visit_member_access(&mut self, input: &MemberAccess) -> (String, String) {
365        // Handle `self.address`, `self.caller`, `self.checksum`, `self.edition`, `self.id`, `self.program_owner`, `self.signer`.
366        if let Expression::Path(path) = input.inner.borrow()
367            && matches!(path.try_absolute_path().as_deref(), Some([sym::SelfLower]))
368        {
369            // Get the current program ID.
370            let program_id = self.program_id.expect("Program ID should be set before traversing the program");
371
372            match input.name.name {
373                // Return the program ID directly.
374                sym::address | sym::id => {
375                    return (program_id.to_string(), String::new());
376                }
377                // Return the appropriate snarkVM operand.
378                name @ (sym::checksum | sym::edition | sym::program_owner) => {
379                    return (name.to_string(), String::new());
380                }
381                _ => {} // Do nothing as `self.signer` and `self.caller` are handled below.
382            }
383        }
384
385        let (inner_expr, _) = self.visit_expression(&input.inner);
386        let member_access = format!("{}.{}", inner_expr, input.name);
387
388        (member_access, String::new())
389    }

391    fn visit_repeat(&mut self, input: &RepeatExpression) -> (String, String) {
392        let (operand, mut operand_instructions) = self.visit_expression(&input.expr);
393        let count = input.count.as_u32().expect("repeat count should be known at this point");
394
395        let expression_operands = std::iter::repeat_n(operand, count as usize).collect::<Vec<_>>().join(" ");
396
397        // Construct the destination register.
398        let destination_register = self.next_register();
399
400        // Get the array type.
401        let Some(array_type @ Type::Array(..)) = self.state.type_table.get(&input.id) else {
402            panic!("All types should be known at this phase of compilation");
403        };
404        let array_type: String = Self::visit_type(&array_type);
405
406        let array_instruction =
407            format!("    cast {expression_operands} into {destination_register} as {array_type};\n");
408
409        // Concatenate the instructions.
410        operand_instructions.push_str(&array_instruction);
411
412        (destination_register, operand_instructions)
413    }

420    // Pedersen64::hash() -> hash.ped64
421    fn visit_associated_function(&mut self, input: &AssociatedFunctionExpression) -> (String, String) {
422        let mut instructions = String::new();
423
424        // Visit each function argument and accumulate instructions from expressions.
425        let arguments = input
426            .arguments
427            .iter()
428            .map(|argument| {
429                let (arg_string, arg_instructions) = self.visit_expression(argument);
430                instructions.push_str(&arg_instructions);
431                arg_string
432            })
433            .collect::<Vec<_>>();
434
435        // A helper function to help with `Program::checksum`, `Program::edition`, and `Program::program_owner`.
436        let generate_program_core = |program: &str, name: &str| {
437            // Get the program ID from the first argument.
438            let program_id = ProgramId::from_str_with_network(&program.replace("\"", ""), self.state.network)
439                .expect("Type checking guarantees that the program name is valid");
440            // If the program name matches the current program ID, then use the operand directly, otherwise fully qualify the operand.
441            let operand = match program_id.to_string()
442                == self.program_id.expect("The program ID is set before traversing the program").to_string()
443            {
444                true => name.to_string(),
445                false => format!("{program_id}/{name}"),
446            };
447            (operand, String::new())
448        };
449
450        // Construct the instruction.
451        let (destination, instruction) = match CoreFunction::try_from(input).ok() {
452            Some(CoreFunction::Commit(variant, ref type_)) => {
453                let mut instruction = format!("    {}", CommitVariant::opcode(variant as u8));
454                let destination_register = self.next_register();
455                // Write the arguments and the destination register.
456                writeln!(instruction, " {} {} into {destination_register} as {type_};", arguments[0], arguments[1])
457                    .expect("failed to write to string");
458                (destination_register, instruction)
459            }
460            Some(CoreFunction::Hash(variant, ref type_)) => {
461                let mut instruction = format!("    {}", variant.opcode());
462                let destination_register = self.next_register();
463                let type_ = match self.state.network {
464                    NetworkName::TestnetV0 => {
465                        type_.to_snarkvm::<TestnetV0>().expect("TYC guarantees that the type is valid").to_string()
466                    }
467                    NetworkName::CanaryV0 => {
468                        type_.to_snarkvm::<CanaryV0>().expect("TYC guarantees that the type is valid").to_string()
469                    }
470                    NetworkName::MainnetV0 => {
471                        type_.to_snarkvm::<MainnetV0>().expect("TYC guarantees that the type is valid").to_string()
472                    }
473                };
474                // Write the arguments and the destination register.
475                writeln!(instruction, " {} into {destination_register} as {type_};", arguments[0])
476                    .expect("failed to write to string");
477                (destination_register, instruction)
478            }
479            Some(CoreFunction::Get) => {
480                let mut instruction = "    get".to_string();
481                let destination_register = self.next_register();
482                // Write the mapping name and the key.
483                writeln!(instruction, " {}[{}] into {destination_register};", arguments[0], arguments[1])
484                    .expect("failed to write to string");
485                (destination_register, instruction)
486            }
487            Some(CoreFunction::MappingGetOrUse) => {
488                let mut instruction = "    get.or_use".to_string();
489                let destination_register = self.next_register();
490                // Write the mapping name, the key, and the default value.
491                writeln!(
492                    instruction,
493                    " {}[{}] {} into {destination_register};",
494                    arguments[0], arguments[1], arguments[2]
495                )
496                .expect("failed to write to string");
497                (destination_register, instruction)
498            }
499            Some(CoreFunction::Set) => {
500                let mut instruction = "    set".to_string();
501                // Write the value, mapping name, and the key.
502                writeln!(instruction, " {} into {}[{}];", arguments[2], arguments[0], arguments[1])
503                    .expect("failed to write to string");
504                (String::new(), instruction)
505            }
506            Some(CoreFunction::MappingRemove) => {
507                let mut instruction = "    remove".to_string();
508                // Write the mapping name and the key.
509                writeln!(instruction, " {}[{}];", arguments[0], arguments[1]).expect("failed to write to string");
510                (String::new(), instruction)
511            }
512            Some(CoreFunction::MappingContains) => {
513                let mut instruction = "    contains".to_string();
514                let destination_register = self.next_register();
515                // Write the mapping name and the key.
516                writeln!(instruction, " {}[{}] into {destination_register};", arguments[0], arguments[1])
517                    .expect("failed to write to string");
518                (destination_register, instruction)
519            }
520            Some(CoreFunction::GroupToXCoordinate) => {
521                let mut instruction = "    cast".to_string();
522                let destination_register = self.next_register();
523                // Write the argument and the destination register.
524                writeln!(instruction, " {} into {destination_register} as group.x;", arguments[0],)
525                    .expect("failed to write to string");
526                (destination_register, instruction)
527            }
528            Some(CoreFunction::GroupToYCoordinate) => {
529                let mut instruction = "    cast".to_string();
530                let destination_register = self.next_register();
531                // Write the argument and the destination register.
532                writeln!(instruction, " {} into {destination_register} as group.y;", arguments[0],)
533                    .expect("failed to write to string");
534                (destination_register, instruction)
535            }
536            Some(CoreFunction::ChaChaRand(type_)) => {
537                // Get the destination register.
538                let destination_register = self.next_register();
539                // Construct the instruction template.
540                let instruction = format!("    rand.chacha into {destination_register} as {type_};\n");
541
542                (destination_register, instruction)
543            }
544            Some(CoreFunction::SignatureVerify) => {
545                let mut instruction = "    sign.verify".to_string();
546                let destination_register = self.next_register();
547                // Write the arguments and the destination register.
548                writeln!(
549                    instruction,
550                    " {} {} {} into {destination_register};",
551                    arguments[0], arguments[1], arguments[2]
552                )
553                .expect("failed to write to string");
554                (destination_register, instruction)
555            }
556            Some(CoreFunction::ECDSAVerify(variant)) => {
557                let mut instruction = format!("    {}", variant.opcode());
558                let destination_register = self.next_register();
559                // Write the arguments and the destination register.
560                writeln!(
561                    instruction,
562                    " {} {} {} into {destination_register};",
563                    arguments[0], arguments[1], arguments[2]
564                )
565                .expect("failed to write to string");
566                (destination_register, instruction)
567            }
568            Some(CoreFunction::FutureAwait) => {
569                let mut instruction = "    await".to_string();
570                writeln!(instruction, " {};", arguments[0]).expect("failed to write to string");
571                (String::new(), instruction)
572            }
573            Some(CoreFunction::ProgramChecksum) => generate_program_core(&arguments[0], "checksum"),
574            Some(CoreFunction::ProgramEdition) => generate_program_core(&arguments[0], "edition"),
575            Some(CoreFunction::ProgramOwner) => generate_program_core(&arguments[0], "program_owner"),
576            Some(CoreFunction::CheatCodePrintMapping)
577            | Some(CoreFunction::CheatCodeSetBlockHeight)
578            | Some(CoreFunction::CheatCodeSetSigner) => {
579                (String::new(), String::new())
580                // Do nothing. Cheat codes do not generate instructions.
581            }
582            Some(CoreFunction::Serialize(variant)) => {
583                // Get the input type.
584                let Some(input_type) = self.state.type_table.get(&input.arguments[0].id()) else {
585                    panic!("All types should be known at this phase of compilation");
586                };
587                // Get the instruction variant.
588                let (is_raw, variant) = match variant {
589                    SerializeVariant::ToBits => (false, "bits"),
590                    SerializeVariant::ToBitsRaw => (true, "bits.raw"),
591                };
592                // Get the size in bits of the input type.
593                let size_in_bits = match self.state.network {
594                    NetworkName::TestnetV0 => {
595                        input_type.size_in_bits::<TestnetV0, _>(is_raw, |_| bail!("structs are not supported"))
596                    }
597                    NetworkName::MainnetV0 => {
598                        input_type.size_in_bits::<MainnetV0, _>(is_raw, |_| bail!("structs are not supported"))
599                    }
600                    NetworkName::CanaryV0 => {
601                        input_type.size_in_bits::<CanaryV0, _>(is_raw, |_| bail!("structs are not supported"))
602                    }
603                }
604                .expect("TYC guarantees that all types have a valid size in bits");
605
606                // Construct the output array type.
607                let output_array_type = format!("[boolean; {size_in_bits}u32]");
608                // Construct the destination register.
609                let destination_register = self.next_register();
610                // Construct the instruction template.
611                let instruction = format!(
612                    "    serialize.{variant} {} ({}) into {destination_register} ({output_array_type});\n",
613                    arguments[0],
614                    Self::visit_type(&input_type)
615                );
616
617                (destination_register, instruction)
618            }
619            Some(CoreFunction::Deserialize(variant, output_type)) => {
620                // Get the instruction variant.
621                let variant = match variant {
622                    DeserializeVariant::FromBits => "bits",
623                    DeserializeVariant::FromBitsRaw => "bits.raw",
624                };
625                // Get the input type.
626                let Some(input_type) = self.state.type_table.get(&input.arguments[0].id()) else {
627                    panic!("All types should be known at this phase of compilation");
628                };
629                // Construct the destination register.
630                let destination_register = self.next_register();
631                // Construct the instruction template.
632                let instruction = format!(
633                    "    deserialize.{variant} {} ({}) into {destination_register} ({});\n",
634                    arguments[0],
635                    Self::visit_type(&input_type),
636                    Self::visit_type(&output_type)
637                );
638
639                (destination_register, instruction)
640            }
641            Some(CoreFunction::OptionalUnwrap) | Some(CoreFunction::OptionalUnwrapOr) => {
642                panic!("`Optional` core functions should have been lowered before code generation")
643            }
644            Some(CoreFunction::VectorPush)
645            | Some(CoreFunction::VectorPop)
646            | Some(CoreFunction::VectorLen)
647            | Some(CoreFunction::VectorClear)
648            | Some(CoreFunction::VectorSwapRemove) => {
649                panic!("`Vector` core functions should have been lowered before code generation")
650            }
651            None => {
652                panic!("All core functions should be known at this phase of compilation")
653            }
654        };
655        // Add the instruction to the list of instructions.
656        instructions.push_str(&instruction);
657
658        (destination, instructions)
659    }

665    fn visit_call(&mut self, input: &CallExpression) -> (String, String) {
666        let caller_program = self.program_id.expect("Calls only appear within programs.").name.name;
667        let callee_program = input.program.unwrap_or(caller_program);
668        let func_symbol = self
669            .state
670            .symbol_table
671            .lookup_function(&Location::new(callee_program, input.function.absolute_path().to_vec()))
672            .expect("Type checking guarantees functions exist");
673
674        // Need to determine the program the function originated from as well as if the function has a finalize block.
675        let mut call_instruction = if caller_program != callee_program {
676            // All external functions must be defined as stubs.
677            assert!(
678                self.program.stubs.get(&callee_program).is_some(),
679                "Type checking guarantees that imported and stub programs are present."
680            );
681            format!("    call {}.aleo/{}", callee_program, input.function)
682        } else if func_symbol.function.variant.is_async() {
683            format!("    async {}", self.current_function.unwrap().identifier)
684        } else {
685            format!("    call {}", input.function)
686        };
687
688        let mut instructions = String::new();
689
690        for argument in input.arguments.iter() {
691            let (argument, argument_instructions) = self.visit_expression(argument);
692            write!(call_instruction, " {argument}").expect("failed to write to string");
693            instructions.push_str(&argument_instructions);
694        }
695
696        // Initialize storage for the destination registers.
697        let mut destinations = Vec::new();
698
699        // Create operands for the output registers.
700        match func_symbol.function.output_type.clone() {
701            t if t.is_empty() => {} // Do nothing
702            Type::Tuple(tuple) => match tuple.length() {
703                0 | 1 => panic!("Parsing guarantees that a tuple type has at least two elements"),
704                len => {
705                    for _ in 0..len {
706                        destinations.push(self.next_register());
707                    }
708                }
709            },
710            _ => {
711                destinations.push(self.next_register());
712            }
713        }
714
715        // Add a register for async functions to represent the future created.
716        if func_symbol.function.variant == Variant::AsyncFunction {
717            destinations.push(self.next_register());
718        }
719
720        // Construct the output operands. These are the destination registers **without** the future.
721        let output_operands = destinations.join(" ");
722
723        // If destination registers were created, write them to the call instruction.
724        if !destinations.is_empty() {
725            write!(call_instruction, " into").expect("failed to write to string");
726            for destination in &destinations {
727                write!(call_instruction, " {destination}").expect("failed to write to string");
728            }
729        }
730
731        // Write the closing semicolon.
732        writeln!(call_instruction, ";").expect("failed to write to string");
733
734        // Push the call instruction to the list of instructions.
735        instructions.push_str(&call_instruction);
736
737        // Return the output operands and the instructions.
738        (output_operands, instructions)
739    }
740
741    fn visit_tuple(&mut self, input: &TupleExpression) -> (String, String) {
742        // Need to return a single string here so we will join the tuple elements with ' '
743        // and split them after this method is called.
744        let mut tuple_elements = Vec::with_capacity(input.elements.len());
745        let mut instructions = String::new();
746
747        // Visit each tuple element and accumulate instructions from expressions.
748        for element in input.elements.iter() {
749            let (element, element_instructions) = self.visit_expression(element);
750            tuple_elements.push(element);
751            instructions.push_str(&element_instructions);
752        }
753
754        // CAUTION: does not return the destination_register.
755        (tuple_elements.join(" "), instructions)
756    }

762    pub fn clone_register(&mut self, register: &str, typ: &Type) -> (String, String) {
763        if typ.is_empty() {
764            return (String::new(), String::new());
765        }
766        let new_reg = self.next_register();
767        match typ {
768            Type::Address
769            | Type::Boolean
770            | Type::Field
771            | Type::Group
772            | Type::Scalar
773            | Type::Signature
774            | Type::Integer(_) => {
775                // These types can be cloned just by casting them to themselves.
776                let instruction = format!("    cast {register} into {new_reg} as {typ};\n");
777                (new_reg, instruction)
778            }
779
780            Type::Array(array_type) => {
781                // We need to cast the old array's members into the new array.
782                let mut instruction = "    cast ".to_string();
783                for i in 0..array_type.length.as_u32().expect("length should be known at this point") as usize {
784                    write!(&mut instruction, "{register}[{i}u32] ").unwrap();
785                }
786                writeln!(&mut instruction, "into {new_reg} as {};", Self::visit_type(typ)).unwrap();
787                (new_reg, instruction)
788            }
789
790            Type::Composite(comp_ty) => {
791                // We need to cast the old struct or record's members into the new one.
792                let program = comp_ty.program.unwrap_or(self.program_id.unwrap().name.name);
793                let location = Location::new(program, comp_ty.path.absolute_path().to_vec());
794                let comp = self
795                    .state
796                    .symbol_table
797                    .lookup_record(&location)
798                    .or_else(|| self.state.symbol_table.lookup_struct(&comp_ty.path.absolute_path()))
799                    .unwrap();
800                let mut instruction = "    cast ".to_string();
801                for member in &comp.members {
802                    write!(&mut instruction, "{register}.{} ", member.identifier.name).unwrap();
803                }
804                writeln!(
805                    &mut instruction,
806                    "into {new_reg} as {};",
807                    // We call `..with_visibility` just so we get the `.record` appended if it's a record.
808                    self.visit_type_with_visibility(typ, leo_ast::Mode::None)
809                )
810                .unwrap();
811                (new_reg, instruction)
812            }
813
814            Type::Optional(_) => panic!("All optional types should have been lowered by now."),
815
816            Type::Vector(_) => panic!("All vector types should have been lowered by now."),
817
818            Type::Mapping(..)
819            | Type::Future(..)
820            | Type::Tuple(..)
821            | Type::Identifier(..)
822            | Type::String
823            | Type::Unit
824            | Type::Numeric
825            | Type::Err => panic!("Objects of type {typ} cannot be cloned."),
826        }
827    }
