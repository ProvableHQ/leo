39impl CodeGeneratingVisitor<'_> {
40    fn visit_statement(&mut self, input: &Statement) -> String {
41        match input {
42            Statement::Assert(stmt) => self.visit_assert(stmt),
43            Statement::Assign(stmt) => self.visit_assign(stmt),
44            Statement::Block(stmt) => self.visit_block(stmt),
45            Statement::Conditional(stmt) => self.visit_conditional(stmt),
46            Statement::Console(stmt) => self.visit_console(stmt),
47            Statement::Const(_) => {
48                panic!("`ConstStatement`s should not be in the AST at this phase of compilation.")
49            }
50            Statement::Definition(stmt) => self.visit_definition(stmt),
51            Statement::Expression(stmt) => self.visit_expression_statement(stmt),
52            Statement::Iteration(stmt) => self.visit_iteration(stmt),
53            Statement::Return(stmt) => self.visit_return(stmt),
54        }
55    }
56
57    fn visit_assert(&mut self, input: &AssertStatement) -> String {
58        let mut generate_assert_instruction = |name: &str, left: &Expression, right: &Expression| {
59            let (left_operand, left_instructions) = self.visit_expression(left);
60            let (right_operand, right_instructions) = self.visit_expression(right);
61            let assert_instruction = format!("    {name} {left_operand} {right_operand};\n");
62
63            // Concatenate the instructions.
64            let mut instructions = left_instructions;
65            instructions.push_str(&right_instructions);
66            instructions.push_str(&assert_instruction);
67
68            instructions
69        };
70        match &input.variant {
71            AssertVariant::Assert(expr) => {
72                let (operand, mut instructions) = self.visit_expression(expr);
73                let assert_instruction = format!("    assert.eq {operand} true;\n");
74
75                instructions.push_str(&assert_instruction);
76                instructions
77            }
78            AssertVariant::AssertEq(left, right) => generate_assert_instruction("assert.eq", left, right),
79            AssertVariant::AssertNeq(left, right) => generate_assert_instruction("assert.neq", left, right),
80        }
81    }
82
83    fn visit_return(&mut self, input: &ReturnStatement) -> String {
84        if let Expression::Unit(..) = input.expression {
85            // Skip empty return statements.
86            return String::new();
87        }
88
89        let (operand, mut expression_instructions) = self.visit_expression(&input.expression);
90        // Get the output type of the function.
91        let output = self.current_function.unwrap().output.iter();
92        // If the operand string is empty, initialize an empty vector.
93        let operand_strings = match operand.is_empty() {
94            true => vec![],
95            false => operand.split(' ').collect_vec(),
96        };
97
98        let mut future_output = String::new();
99        let mut instructions = operand_strings
100            .iter()
101            .zip_eq(output)
102            .map(|(operand, output)| {
103                // Transitions outputs with no mode are private.
104                // Note that this unwrap is safe because we set the variant before traversing the function.
105                let visibility = match (self.variant.unwrap().is_transition(), output.mode) {
106                    (true, Mode::None) => Mode::Private,
107                    (_, mode) => mode,
108                };
109
110                if let Type::Future(_) = output.type_ {
111                    future_output = format!(
112                        "    output {} as {}.aleo/{}.future;\n",
113                        operand,
114                        self.program_id.unwrap().name,
115                        self.current_function.unwrap().identifier,
116                    );
117                    String::new()
118                } else {
119                    format!(
120                        "    output {} as {};\n",
121                        operand,
122                        self.visit_type_with_visibility(&output.type_, visibility)
123                    )
124                }
125            })
126            .join("");
127
128        // Insert future output at the end.
129        instructions.push_str(&future_output);
130
131        expression_instructions.push_str(&instructions);
132
133        expression_instructions
134    }
135
136    fn visit_definition(&mut self, input: &DefinitionStatement) -> String {
137        match (&input.place, &input.value) {
138            (DefinitionPlace::Single(identifier), _) => {
139                let (operand, expression_instructions) = self.visit_expression(&input.value);
140                self.variable_mapping.insert(identifier.name, operand);
141                expression_instructions
142            }
143            (DefinitionPlace::Multiple(identifiers), Expression::Call(_)) => {
144                let (operand, expression_instructions) = self.visit_expression(&input.value);
145                // Add the destinations to the variable mapping.
146                for (identifier, operand) in identifiers.iter().zip_eq(operand.split(' ')) {
147                    self.variable_mapping.insert(identifier.name, operand.to_string());
148                }
149                expression_instructions
150            }
151            _ => panic!("Previous passes should have ensured that a definition with multiple identifiers is a `Call`."),
152        }
153    }
154
155    fn visit_expression_statement(&mut self, input: &ExpressionStatement) -> String {
156        self.visit_expression(&input.expression).1
157    }
158
159    fn visit_assign(&mut self, _input: &AssignStatement) -> String {
160        panic!("AssignStatement's should not exist in SSA form.")
161    }
162
163    fn visit_conditional(&mut self, _input: &ConditionalStatement) -> String {
164        // Note that this unwrap is safe because we set the variant before traversing the function.
165        if !self.variant.unwrap().is_async_function() {
166            panic!("`ConditionalStatement`s should not be in the AST at this phase of compilation.")
167        } else {
168            // Construct a label for the end of the `then` block.
169            let end_then_label = format!("end_then_{}_{}", self.conditional_depth, self.next_label);
170            self.next_label += 1;
171            // Construct a label for the end of the `otherwise` block if it exists.
172            let (has_otherwise, end_otherwise_label) = {
173                match _input.otherwise.is_some() {
174                    true => {
175                        // Construct a label for the end of the `otherwise` block.
176                        let end_otherwise_label =
177                            { format!("end_otherwise_{}_{}", self.conditional_depth, self.next_label) };
178                        self.next_label += 1;
179                        (true, end_otherwise_label)
180                    }
181                    false => (false, String::new()),
182                }
183            };
184
185            // Increment the conditional depth.
186            self.conditional_depth += 1;
187
188            // Create a `branch` instruction.
189            let (condition, mut instructions) = self.visit_expression(&_input.condition);
190            instructions.push_str(&format!("    branch.eq {condition} false to {end_then_label};\n"));
191
192            // Visit the `then` block.
193            instructions.push_str(&self.visit_block(&_input.then));
194            // If the `otherwise` block is present, add a branch instruction to jump to the end of the `otherwise` block.
195            if has_otherwise {
196                instructions.push_str(&format!("    branch.eq true true to {end_otherwise_label};\n"));
197            }
198
199            // Add a label for the end of the `then` block.
200            instructions.push_str(&format!("    position {};\n", end_then_label));
201
202            // Visit the `otherwise` block.
203            if let Some(else_block) = &_input.otherwise {
204                // Visit the `otherwise` block.
205                instructions.push_str(&self.visit_statement(else_block));
206                // Add a label for the end of the `otherwise` block.
207                instructions.push_str(&format!("    position {end_otherwise_label};\n"));
208            }
209
210            // Decrement the conditional depth.
211            self.conditional_depth -= 1;
212
213            instructions
214        }
215    }
216
217    fn visit_iteration(&mut self, _input: &IterationStatement) -> String {
218        panic!("`IterationStatement`s should not be in the AST at this phase of compilation.");
219    }
220
221    fn visit_console(&mut self, _: &ConsoleStatement) -> String {
222        panic!("Parsing guarantees that `ConsoleStatement`s are not present in the AST.")
223    }
224
225    pub(crate) fn visit_block(&mut self, input: &Block) -> String {
226        // For each statement in the block, visit it and add its instructions to the list.
227        input.statements.iter().map(|stmt| self.visit_statement(stmt)).join("")
228    }
