22/// A struct dependency graph.
23pub type StructGraph = DiGraph<Symbol>;

25/// A call graph.
26pub type CallGraph = DiGraph<Symbol>;

28/// An import dependency graph.
29pub type ImportGraph = DiGraph<Symbol>;

31/// A node in a graph.
32pub trait Node: Copy + 'static + Eq + PartialEq + Debug + Hash {}
33
34impl Node for Symbol {}
35
36/// Errors in directed graph operations.
37#[derive(Debug)]
38pub enum DiGraphError<N: Node> {
39    /// An error that is emitted when a cycle is detected in the directed graph. Contains the path of the cycle.
40    CycleDetected(Vec<N>),
41}
42
43/// A directed graph.
44#[derive(Debug, PartialEq, Eq)]
45pub struct DiGraph<N: Node> {
46    /// The set of nodes in the graph.
47    nodes: IndexSet<N>,
48    /// The directed edges in the graph.
49    /// Each entry in the map is a node in the graph, and the set of nodes that it points to.
50    edges: IndexMap<N, IndexSet<N>>,
51}
52
53impl<N: Node> DiGraph<N> {
54    /// Initializes a new `DiGraph` from a vector of source nodes.
55    pub fn new(nodes: IndexSet<N>) -> Self {
56        Self { nodes, edges: IndexMap::new() }
57    }
58
59    /// Adds an edge to the graph.
60    pub fn add_edge(&mut self, from: N, to: N) {
61        // Add `from` and `to` to the set of nodes if they are not already in the set.
62        self.nodes.insert(from);
63        self.nodes.insert(to);
64
65        // Add the edge to the adjacency list.
66        let entry = self.edges.entry(from).or_default();
67        entry.insert(to);
68    }
69
70    /// Returns `true` if the graph contains the given node.
71    pub fn contains_node(&self, node: N) -> bool {
72        self.nodes.contains(&node)
73    }
74
75    /// Returns the post-order ordering of the graph.
76    /// Detects if there is a cycle in the graph.
77    pub fn post_order(&self) -> Result<IndexSet<N>, DiGraphError<N>> {
78        // The set of nodes that do not need to be visited again.
79        let mut finished: IndexSet<N> = IndexSet::with_capacity(self.nodes.len());
80
81        // Perform a depth-first search of the graph, starting from `node`, for each node in the graph.
82        for node in self.nodes.iter() {
83            // If the node has not been explored, explore it.
84            if !finished.contains(node) {
85                // The set of nodes that are on the path to the current node in the search.
86                let mut discovered: IndexSet<N> = IndexSet::new();
87                // Check if there is a cycle in the graph starting from `node`.
88                if let Some(node) = self.contains_cycle_from(*node, &mut discovered, &mut finished) {
89                    let mut path = vec![node];
90                    // Backtrack through the discovered nodes to find the cycle.
91                    while let Some(next) = discovered.pop() {
92                        // Add the node to the path.
93                        path.push(next);
94                        // If the node is the same as the first node in the path, we have found the cycle.
95                        if next == node {
96                            break;
97                        }
98                    }
99                    // Reverse the path to get the cycle in the correct order.
100                    path.reverse();
101                    // A cycle was detected. Return the path of the cycle.
102                    return Err(DiGraphError::CycleDetected(path));
103                }
104            }
105        }
106        // No cycle was found. Return the set of nodes in topological order.
107        Ok(finished)
108    }
109
110    /// Retains a subset of the nodes, and removes all edges in which the source or destination is not in the subset.
111    pub fn retain_nodes(&mut self, nodes: &IndexSet<N>) {
112        // Remove the nodes from the set of nodes.
113        self.nodes.retain(|node| nodes.contains(node));
114        self.edges.retain(|node, _| nodes.contains(node));
115        // Remove the edges that reference the nodes.
116        for (_, children) in self.edges.iter_mut() {
117            children.retain(|child| nodes.contains(child));
118        }
119    }
120
121    // Detects if there is a cycle in the graph starting from the given node, via a recursive depth-first search.
122    // If there is no cycle, returns `None`.
123    // If there is a cycle, returns the node that was most recently discovered.
124    // Nodes are added to `finished` in post-order order.
125    fn contains_cycle_from(&self, node: N, discovered: &mut IndexSet<N>, finished: &mut IndexSet<N>) -> Option<N> {
126        // Add the node to the set of discovered nodes.
127        discovered.insert(node);
128
129        // Check each outgoing edge of the node.
130        if let Some(children) = self.edges.get(&node) {
131            for child in children.iter() {
132                // If the node already been discovered, there is a cycle.
133                if discovered.contains(child) {
134                    // Insert the child node into the set of discovered nodes; this is used to reconstruct the cycle.
135                    // Note that this case is always hit when there is a cycle.
136                    return Some(*child);
137                }
138                // If the node has not been explored, explore it.
139                if !finished.contains(child) {
140                    if let Some(child) = self.contains_cycle_from(*child, discovered, finished) {
141                        return Some(child);
142                    }
143                }
144            }
145        }
146
147        // Remove the node from the set of discovered nodes.
148        discovered.pop();
149        // Add the node to the set of finished nodes.
150        finished.insert(node);
151
152        None
153    }
154}

156#[cfg(test)]
157mod test {
158    use super::*;
159
160    impl Node for u32 {}
161
162    fn check_post_order<N: Node>(graph: &DiGraph<N>, expected: &[N]) {
163        let result = graph.post_order();
164        assert!(result.is_ok());
165
166        let order: Vec<N> = result.unwrap().into_iter().collect();
167        assert_eq!(order, expected);
168    }
169
170    #[test]
171    fn test_post_order() {
172        let mut graph = DiGraph::<u32>::new(IndexSet::new());
173
174        graph.add_edge(1, 2);
175        graph.add_edge(1, 3);
176        graph.add_edge(2, 4);
177        graph.add_edge(3, 4);
178        graph.add_edge(4, 5);
179
180        check_post_order(&graph, &[5, 4, 2, 3, 1]);
181
182        let mut graph = DiGraph::<u32>::new(IndexSet::new());
183
184        // F -> B
185        graph.add_edge(6, 2);
186        // B -> A
187        graph.add_edge(2, 1);
188        // B -> D
189        graph.add_edge(2, 4);
190        // D -> C
191        graph.add_edge(4, 3);
192        // D -> E
193        graph.add_edge(4, 5);
194        // F -> G
195        graph.add_edge(6, 7);
196        // G -> I
197        graph.add_edge(7, 9);
198        // I -> H
199        graph.add_edge(9, 8);
200
201        // A, C, E, D, B, H, I, G, F.
202        check_post_order(&graph, &[1, 3, 5, 4, 2, 8, 9, 7, 6]);
203    }
204
205    #[test]
206    fn test_cycle() {
207        let mut graph = DiGraph::<u32>::new(IndexSet::new());
208
209        graph.add_edge(1, 2);
210        graph.add_edge(2, 3);
211        graph.add_edge(2, 4);
212        graph.add_edge(4, 1);
213
214        let result = graph.post_order();
215        assert!(result.is_err());
216
217        let DiGraphError::CycleDetected(cycle) = result.unwrap_err();
218        let expected = Vec::from([1u32, 2, 4, 1]);
219        assert_eq!(cycle, expected);
220    }

223    fn test_unconnected_graph() {
224        let graph = DiGraph::<u32>::new(IndexSet::from([1, 2, 3, 4, 5]));

227    }
