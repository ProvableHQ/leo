27/// Maps global and local symbols to information about them.
28///
29/// Scopes are indexed by the NodeID of the function, block, or iteration.
30#[derive(Debug, Default)]
31pub struct SymbolTable {
32    /// Functions indexed by location.
33    functions: IndexMap<Location, FunctionSymbol>,
34
35    /// Records indexed by location.
36    records: IndexMap<Location, Composite>,
37
38    /// Structs indexed by location.
39    structs: IndexMap<Symbol, Composite>,
40
41    /// Consts that have been successfully evaluated.
42    global_consts: IndexMap<Location, Expression>,
43
44    /// Global variables indexed by location.
45    globals: IndexMap<Location, VariableSymbol>,
46
47    /// Local tables index by the NodeID of the function, iteration, or block they're contained in.
48    all_locals: HashMap<NodeID, LocalTable>,
49
50    /// The current LocalTable we're looking at.
51    local: Option<LocalTable>,
52}
53
54#[derive(Clone, Default, Debug)]
55struct LocalTable {
56    inner: Rc<RefCell<LocalTableInner>>,
57}
58
59#[derive(Clone, Default, Debug)]
60struct LocalTableInner {
61    /// The `NodeID` of the function, iteration, or block this table indexes.
62    id: NodeID,
63
64    /// The parent `NodeID` of this scope, if it exists.
65    parent: Option<NodeID>,
66
67    /// The children of `NodeID` of this scope
68    children: Vec<NodeID>,
69
70    /// The consts we've evaluated in this scope.
71    consts: HashMap<Symbol, Expression>,
72
73    /// Variables in this scope, indexed by name.
74    variables: HashMap<Symbol, VariableSymbol>,
75}
76
77impl LocalTable {
78    fn new(symbol_table: &mut SymbolTable, id: NodeID, parent: Option<NodeID>) -> Self {
79        // If parent exists, register this scope as its child
80        if let Some(parent_id) = parent {
81            if let Some(parent_table) = symbol_table.all_locals.get_mut(&parent_id) {
82                parent_table.inner.borrow_mut().children.push(id);
83            }
84        }
85
86        LocalTable {
87            inner: Rc::new(RefCell::new(LocalTableInner {
88                id,
89                parent,
90                consts: HashMap::new(),
91                variables: HashMap::new(),
92                children: vec![], // Must still initialize our own children list
93            })),
94        }
95    }
96
97    /// Creates a duplicate of this local scope with a new `NodeID`.
98    ///
99    /// TODO: This currently clones the `children` list, which is incorrect. The new scope may incorrectly
100    /// appear to have descendants that still belong to the original scope. This breaks the structure of
101    /// the scope tree and may cause symbol resolution to behave incorrectly.
102    fn dup(&self, new_id: NodeID) -> Self {
103        let mut inner = self.inner.borrow().clone();
104        inner.id = new_id;
105        LocalTable { inner: Rc::new(RefCell::new(inner)) }
106    }
107}

109impl SymbolTable {
110    /// Iterator over all the structs (not records) in this program.
111    pub fn iter_structs(&self) -> impl Iterator<Item = (Symbol, &Composite)> {
112        self.structs.iter().map(|(name, comp)| (*name, comp))
113    }
114
115    /// Iterator over all the records in this program.
116    pub fn iter_records(&self) -> impl Iterator<Item = (Location, &Composite)> {
117        self.records.iter().map(|(loc, comp)| (*loc, comp))
118    }
119
120    /// Iterator over all the functions in this program.
121    pub fn iter_functions(&self) -> impl Iterator<Item = (Location, &FunctionSymbol)> {
122        self.functions.iter().map(|(loc, func_symbol)| (*loc, func_symbol))
123    }
124
125    /// Access the struct by this name if it exists.
126    pub fn lookup_struct(&self, name: Symbol) -> Option<&Composite> {
127        self.structs.get(&name)
128    }
129
130    /// Access the record at this location if it exists.
131    pub fn lookup_record(&self, location: Location) -> Option<&Composite> {
132        self.records.get(&location)
133    }
134
135    /// Access the function at this location if it exists.
136    pub fn lookup_function(&self, location: Location) -> Option<&FunctionSymbol> {
137        self.functions.get(&location)
138    }
139
140    /// Access the variable accessible by this name in the current scope.
141    pub fn lookup_variable(&self, program: Symbol, name: Symbol) -> Option<VariableSymbol> {
142        let mut current = self.local.as_ref();
143
144        while let Some(table) = current {
145            let borrowed = table.inner.borrow();
146            let value = borrowed.variables.get(&name);
147            if value.is_some() {
148                return value.cloned();
149            }
150
151            current = borrowed.parent.and_then(|id| self.all_locals.get(&id));
152        }
153
154        self.globals.get(&Location::new(program, name)).cloned()
155    }
156
157    /// Enter the scope of this `NodeID`, creating a table if it doesn't exist yet.
158    ///
159    /// Passing `None` means to enter the global scope.
160    pub fn enter_scope(&mut self, id: Option<NodeID>) {
161        self.local = id.map(|id| {
162            let parent = self.local.as_ref().map(|table| table.inner.borrow().id);
163            let new_local_table = if let Some(existing) = self.all_locals.get(&id) {
164                existing.clone()
165            } else {
166                let new_table = LocalTable::new(self, id, parent);
167                self.all_locals.insert(id, new_table.clone());
168                new_table
169            };
170
171            assert_eq!(parent, new_local_table.inner.borrow().parent, "Entered scopes out of order.");
172            new_local_table.clone()
173        });
174    }
175
176    /// Enter the new scope with id `new_id`, duplicating its local symbol table from the scope at `old_id`.
177    ///
178    /// This is useful for a pass like loop unrolling, in which the loop body must be duplicated multiple times.
179    pub fn enter_scope_duped(&mut self, new_id: NodeID, old_id: NodeID) {
180        let old_local_table = self.all_locals.get(&old_id).expect("Must have an old scope to dup from.");
181        let new_local_table = old_local_table.dup(new_id);
182        let parent = self.local.as_ref().map(|table| table.inner.borrow().id);
183        new_local_table.inner.borrow_mut().parent = parent;
184        self.all_locals.insert(new_id, new_local_table.clone());
185        self.local = Some(new_local_table);
186    }
187
188    /// Enter the parent scope of the current scope (or the global scope if there is no local parent scope).
189    pub fn enter_parent(&mut self) {
190        let parent: Option<NodeID> = self.local.as_ref().and_then(|table| table.inner.borrow().parent);
191        self.local = parent.map(|id| self.all_locals.get(&id).expect("Parent should exist.")).cloned();
192    }
193
194    /// Checks if a `symbol` is local to `scope`.
195    pub fn is_local_to(&self, scope: NodeID, symbol: Symbol) -> bool {
196        self.all_locals.get(&scope).map(|locals| locals.inner.borrow().variables.contains_key(&symbol)).unwrap_or(false)
197    }
198
199    /// Checks whether `symbol` is defined in the current scope (self.local) or any of its
200    /// ancestor scopes, up to and including `scope`.
201    ///
202    /// Returns `false` if the current scope is not a descendant of `scope`.
203    pub fn is_defined_in_scope_or_ancestor_until(&self, scope: NodeID, symbol: Symbol) -> bool {
204        let mut current = self.local.as_ref();
205
206        while let Some(table) = current {
207            let inner = table.inner.borrow();
208
209            // Check if symbol is defined in this scope
210            if inner.variables.contains_key(&symbol) {
211                return true;
212            }
213
214            // Stop when we reach the given upper-bound scope
215            if inner.id == scope {
216                break;
217            }
218
219            // Move to parent
220            current = inner.parent.and_then(|parent_id| self.all_locals.get(&parent_id));
221        }
222
223        false
224    }
225
226    /// Checks if a `symbol` is local to `scope` or any of its child scopes.
227    pub fn is_local_to_or_in_child_scope(&self, scope: NodeID, symbol: Symbol) -> bool {
228        let mut stack = vec![scope];
229
230        while let Some(current_id) = stack.pop() {
231            if let Some(table) = self.all_locals.get(&current_id) {
232                let inner = table.inner.borrow();
233
234                if inner.variables.contains_key(&symbol) {
235                    return true;
236                }
237
238                stack.extend(&inner.children);
239            }
240        }
241
242        false
243    }

245    /// Insert an evaluated const into the current scope.
246    pub fn insert_const(&mut self, program: Symbol, name: Symbol, value: Expression) {
247        if let Some(table) = self.local.as_mut() {
248            table.inner.borrow_mut().consts.insert(name, value);
249        } else {
250            self.global_consts.insert(Location::new(program, name), value);
251        }
252    }
253
254    /// Find the evaluated const accessible by the given name in the current scope.
255    pub fn lookup_const(&self, program: Symbol, name: Symbol) -> Option<Expression> {
256        let mut current = self.local.as_ref();
257
258        while let Some(table) = current {
259            let borrowed = table.inner.borrow();
260            let value = borrowed.consts.get(&name);
261            if value.is_some() {
262                return value.cloned();
263            }
264
265            current = borrowed.parent.and_then(|id| self.all_locals.get(&id));
266        }
267
268        self.global_consts.get(&Location::new(program, name)).cloned()
269    }

271    /// Insert a struct at this name.
272    ///
273    /// Since structs are indexed only by name, the program is used only to check shadowing.
274    pub fn insert_struct(&mut self, program: Symbol, name: Symbol, composite: Composite) -> Result<()> {
275        if let Some(old_composite) = self.structs.get(&name) {
276            if eq_struct(&composite, old_composite) {
277                Ok(())
278            } else {
279                Err(AstError::redefining_external_struct(name, composite.span).into())
280            }
281        } else {
282            let location = Location::new(program, name);
283            self.check_shadow_global(location, composite.span)?;
284            self.structs.insert(name, composite);
285            Ok(())
286        }
287    }

289    /// Insert a record at this location.
290    pub fn insert_record(&mut self, location: Location, composite: Composite) -> Result<()> {
291        self.check_shadow_global(location, composite.span)?;
292        self.records.insert(location, composite);
293        Ok(())
294    }

296    /// Insert a function at this location.
297    pub fn insert_function(&mut self, location: Location, function: Function) -> Result<()> {
298        self.check_shadow_global(location, function.span)?;
299        self.functions.insert(location, FunctionSymbol { function, finalizer: None });
300        Ok(())
301    }
302
303    /// Insert a global at this location.
304    pub fn insert_global(&mut self, location: Location, var: VariableSymbol) -> Result<()> {
305        self.check_shadow_global(location, var.span)?;
306        self.globals.insert(location, var);
307        Ok(())
308    }
309
310    /// Access the global at this location if it exists.
311    pub fn lookup_global(&self, location: Location) -> Option<&VariableSymbol> {
312        self.globals.get(&location)
313    }
314
315    fn check_shadow_global(&self, location: Location, span: Span) -> Result<()> {
316        if self.functions.contains_key(&location) {
317            Err(AstError::shadowed_function(location.name, span).into())
318        } else if self.records.contains_key(&location) {
319            Err(AstError::shadowed_record(location.name, span).into())
320        } else if self.structs.contains_key(&location.name) {
321            Err(AstError::shadowed_struct(location.name, span).into())
322        } else if self.globals.contains_key(&location) {
323            Err(AstError::shadowed_variable(location.name, span).into())
324        } else {
325            Ok(())
326        }
327    }
328
329    fn check_shadow_variable(&self, program: Symbol, name: Symbol, span: Span) -> Result<()> {
330        let mut current = self.local.as_ref();
331
332        while let Some(table) = current {
333            if table.inner.borrow().variables.contains_key(&name) {
334                return Err(AstError::shadowed_variable(name, span).into());
335            }
336            current = table.inner.borrow().parent.map(|id| self.all_locals.get(&id).expect("Parent should exist."));
337        }
338
339        self.check_shadow_global(Location::new(program, name), span)?;
340
341        Ok(())
342    }
343
344    /// Insert a variable into the current scope.
345    pub fn insert_variable(&mut self, program: Symbol, name: Symbol, var: VariableSymbol) -> Result<()> {
346        self.check_shadow_variable(program, name, var.span)?;
347
348        if let Some(table) = self.local.as_mut() {
349            table.inner.borrow_mut().variables.insert(name, var);
350        } else {
351            self.globals.insert(Location::new(program, name), var);
352        }
353
354        Ok(())
355    }

357    /// Attach a finalizer to a function.
358    pub fn attach_finalizer(
359        &mut self,
360        caller: Location,
361        callee: Location,
362        future_inputs: Vec<Location>,
363        inferred_inputs: Vec<Type>,
364    ) -> Result<()> {
365        let callee_location = Location::new(callee.program, callee.name);
366
367        if let Some(func) = self.functions.get_mut(&caller) {
368            func.finalizer = Some(Finalizer { location: callee_location, future_inputs, inferred_inputs });
369            Ok(())
370        } else {
371            Err(AstError::function_not_found(caller.name).into())
372        }
373    }

376fn eq_struct(new: &Composite, old: &Composite) -> bool {
377    if new.members.len() != old.members.len() {
378        return false;
379    }
380
381    new.members
382        .iter()
383        .zip(old.members.iter())
384        .all(|(member1, member2)| member1.name() == member2.name() && member1.type_.eq_flat_relaxed(&member2.type_))
385}
