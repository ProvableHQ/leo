28/// Maps global and local symbols to information about them.
29///
30/// Scopes are indexed by the NodeID of the function, block, or iteration.
31#[derive(Debug, Default)]
32pub struct SymbolTable {
33    /// Functions indexed by location.
34    functions: IndexMap<Location, FunctionSymbol>,
35
36    /// Records indexed by location.
37    records: IndexMap<Location, Composite>,
38
39    /// Structs indexed by a path.
40    structs: IndexMap<Vec<Symbol>, Composite>,
41
42    /// Consts that have been successfully evaluated.
43    global_consts: IndexMap<Location, Expression>,
44
45    /// Global variables indexed by location.
46    globals: IndexMap<Location, VariableSymbol>,
47
48    /// Local tables index by the NodeID of the function, iteration, or block they're contained in.
49    all_locals: HashMap<NodeID, LocalTable>,
50
51    /// The current LocalTable we're looking at.
52    local: Option<LocalTable>,
53}
54
55#[derive(Clone, Default, Debug)]
56struct LocalTable {
57    inner: Rc<RefCell<LocalTableInner>>,
58}
59
60#[derive(Clone, Default, Debug)]
61struct LocalTableInner {
62    /// The `NodeID` of the function, iteration, or block this table indexes.
63    id: NodeID,
64
65    /// The parent `NodeID` of this scope, if it exists.
66    parent: Option<NodeID>,
67
68    /// The children of `NodeID` of this scope
69    children: Vec<NodeID>,
70
71    /// The consts we've evaluated in this scope.
72    consts: HashMap<Symbol, Expression>,
73
74    /// Variables in this scope, indexed by name.
75    variables: HashMap<Symbol, VariableSymbol>,
76}
77
78impl LocalTable {
79    fn new(symbol_table: &mut SymbolTable, id: NodeID, parent: Option<NodeID>) -> Self {
80        // If parent exists, register this scope as its child
81        if let Some(parent_id) = parent
82            && let Some(parent_table) = symbol_table.all_locals.get_mut(&parent_id)
83        {
84            parent_table.inner.borrow_mut().children.push(id);
85        }
86
87        LocalTable {
88            inner: Rc::new(RefCell::new(LocalTableInner {
89                id,
90                parent,
91                consts: HashMap::new(),
92                variables: HashMap::new(),
93                children: vec![], // Must still initialize our own children list
94            })),
95        }
96    }
97
98    /// Recursively duplicates this table and all children.
99    /// `new_parent` is the NodeID of the parent in the new tree (None for root).
100    pub fn dup(
101        &self,
102        node_builder: &NodeBuilder,
103        all_locals: &mut HashMap<NodeID, LocalTable>,
104        new_parent: Option<NodeID>,
105    ) -> LocalTable {
106        let inner = self.inner.borrow();
107
108        // Generate a new ID for this table
109        let new_id = node_builder.next_id();
110
111        // Recursively duplicate children with new_id as their parent
112        let new_children: Vec<NodeID> = inner
113            .children
114            .iter()
115            .map(|child_id| {
116                let child_table = all_locals.get(child_id).expect("Child must exist").clone();
117                let duped_child = child_table.dup(node_builder, all_locals, Some(new_id));
118                let child_new_id = duped_child.inner.borrow().id;
119                all_locals.insert(child_new_id, duped_child);
120                child_new_id
121            })
122            .collect();
123
124        // Duplicate this table with correct parent
125        let new_table = LocalTable {
126            inner: Rc::new(RefCell::new(LocalTableInner {
127                id: new_id,
128                parent: new_parent,
129                children: new_children,
130                consts: inner.consts.clone(),
131                variables: inner.variables.clone(),
132            })),
133        };
134
135        // Register in all_locals
136        all_locals.insert(new_id, new_table.clone());
137
138        new_table
139    }
140}

188    /// Attempts to look up a variable by a path.
189    ///
190    /// First, it tries to resolve the symbol as a global using the full path under the given program.
191    /// If that fails and the path is non-empty, it falls back to resolving the last component
192    /// of the path as a local symbol.
193    ///
194    /// # Arguments
195    ///
196    /// * `program` - The root symbol representing the program or module context.
197    /// * `path` - A slice of symbols representing the absolute path to the variable.
198    ///
199    /// # Returns
200    ///
201    /// An `Option<VariableSymbol>` containing the resolved symbol if found, otherwise `None`.
202    pub fn lookup_path(&self, program: Symbol, path: &[Symbol]) -> Option<VariableSymbol> {
203        self.lookup_global(&Location::new(program, path.to_vec()))
204            .cloned()
205            .or_else(|| path.last().copied().and_then(|name| self.lookup_local(name)))
206    }

208    /// Access the variable accessible by this name in the current scope.
209    pub fn lookup_local(&self, name: Symbol) -> Option<VariableSymbol> {
210        let mut current = self.local.as_ref();
211
212        while let Some(table) = current {
213            let borrowed = table.inner.borrow();
214            let value = borrowed.variables.get(&name);
215            if value.is_some() {
216                return value.cloned();
217            }
218
219            current = borrowed.parent.and_then(|id| self.all_locals.get(&id));
220        }
221
222        None
223    }

225    /// Enter the scope of this `NodeID`, creating a table if it doesn't exist yet.
226    ///
227    /// Passing `None` means to enter the global scope.
228    pub fn enter_scope(&mut self, id: Option<NodeID>) {
229        self.local = id.map(|id| {
230            let parent = self.local.as_ref().map(|table| table.inner.borrow().id);
231            let new_local_table = if let Some(existing) = self.all_locals.get(&id) {
232                existing.clone()
233            } else {
234                let new_table = LocalTable::new(self, id, parent);
235                self.all_locals.insert(id, new_table.clone());
236                new_table
237            };
238
239            assert_eq!(parent, new_local_table.inner.borrow().parent, "Entered scopes out of order.");
240            new_local_table.clone()
241        });
242    }
243
244    /// Enter a new scope by duplicating the local table at `old_id` recursively.
245    ///
246    /// Each scope in the subtree receives a fresh NodeID from `node_builder`.
247    /// The new root scope's parent is set to the current scope (if any).
248    /// Returns the NodeID of the new duplicated root scope.
249    pub fn enter_scope_duped(&mut self, old_id: NodeID, node_builder: &NodeBuilder) -> usize {
250        let old_local_table = self.all_locals.get(&old_id).expect("Must have an old scope to dup from.").clone();
251
252        // Recursively duplicate the table and all its children
253        let new_local_table =
254            old_local_table.dup(node_builder, &mut self.all_locals, self.local.as_ref().map(|t| t.inner.borrow().id));
255
256        let new_id = new_local_table.inner.borrow().id;
257
258        // Update current scope
259        self.local = Some(new_local_table);
260
261        new_id
262    }

264    /// Enter the parent scope of the current scope (or the global scope if there is no local parent scope).
265    pub fn enter_parent(&mut self) {
266        let parent: Option<NodeID> = self.local.as_ref().and_then(|table| table.inner.borrow().parent);
267        self.local = parent.map(|id| self.all_locals.get(&id).expect("Parent should exist.")).cloned();
268    }

270    /// Checks if a `symbol` is local to `scope`.
271    pub fn is_local_to(&self, scope: NodeID, symbol: Symbol) -> bool {
272        self.all_locals.get(&scope).map(|locals| locals.inner.borrow().variables.contains_key(&symbol)).unwrap_or(false)
273    }

275    /// Checks whether `symbol` is defined in the current scope (self.local) or any of its
276    /// ancestor scopes, up to and including `scope`.
277    ///
278    /// Returns `false` if the current scope is not a descendant of `scope`.
279    pub fn is_defined_in_scope_or_ancestor_until(&self, scope: NodeID, symbol: Symbol) -> bool {
280        let mut current = self.local.as_ref();
281
282        while let Some(table) = current {
283            let inner = table.inner.borrow();
284
285            // Check if symbol is defined in this scope
286            if inner.variables.contains_key(&symbol) {
287                return true;
288            }
289
290            // Stop when we reach the given upper-bound scope
291            if inner.id == scope {
292                break;
293            }
294
295            // Move to parent
296            current = inner.parent.and_then(|parent_id| self.all_locals.get(&parent_id));
297        }
298
299        false
300    }
301
302    /// Checks if a `symbol` is local to `scope` or any of its child scopes.
303    pub fn is_local_to_or_in_child_scope(&self, scope: NodeID, symbol: Symbol) -> bool {
304        let mut stack = vec![scope];
305
306        while let Some(current_id) = stack.pop() {
307            if let Some(table) = self.all_locals.get(&current_id) {
308                let inner = table.inner.borrow();
309
310                if inner.variables.contains_key(&symbol) {
311                    return true;
312                }
313
314                stack.extend(&inner.children);
315            }
316        }
317
318        false
319    }

321    /// Insert an evaluated const into the current scope.
322    pub fn insert_const(&mut self, program: Symbol, path: &[Symbol], value: Expression) {
323        if let Some(table) = self.local.as_mut() {
324            let [const_name] = &path else { panic!("Local consts cannot have paths with more than 1 segment.") };
325            table.inner.borrow_mut().consts.insert(*const_name, value);
326        } else {
327            self.global_consts.insert(Location::new(program, path.to_vec()), value);
328        }
329    }

331    /// Find the evaluated const accessible by the given name in the current scope.
332    pub fn lookup_const(&self, program: Symbol, path: &[Symbol]) -> Option<Expression> {
333        let mut current = self.local.as_ref();
334
335        while let Some(table) = current {
336            let borrowed = table.inner.borrow();
337            let value = borrowed.consts.get(path.last().expect("all paths must have at least 1 segment"));
338            if value.is_some() {
339                return value.cloned();
340            }
341
342            current = borrowed.parent.and_then(|id| self.all_locals.get(&id));
343        }
344
345        self.global_consts.get(&Location::new(program, path.to_vec())).cloned()
346    }

348    /// Insert a struct at this name.
349    ///
350    /// Since structs are indexed only by name, the program is used only to check shadowing.
351    pub fn insert_struct(&mut self, program: Symbol, path: &[Symbol], composite: Composite) -> Result<()> {
352        if let Some(old_composite) = self.structs.get(path) {
353            if eq_struct(&composite, old_composite) {
354                Ok(())
355            } else {
356                Err(AstError::redefining_external_struct(path.iter().format("::"), old_composite.span).into())
357            }
358        } else {
359            let location = Location::new(program, path.to_vec());
360            self.check_shadow_global(&location, composite.identifier.span)?;
361            self.structs.insert(path.to_vec(), composite);
362            Ok(())
363        }
364    }

392    pub fn emit_shadow_error(name: Symbol, span: Span, prev_span: Span) -> LeoError {
393        AstError::name_defined_multiple_times(name, span)
394            .with_labels(vec![
395                Label::new(format!("previous definition of `{name}` here"), prev_span).with_color(Color::Blue),
396                Label::new(format!("`{name}` redefined here"), span),
397            ])
398            .into()
399    }
400
401    fn check_shadow_global(&self, location: &Location, span: Span) -> Result<()> {
402        let name = location.path.last().expect("location path must have at least one segment.");
403
404        self.functions
405            .get(location)
406            .map(|f| f.function.identifier.span)
407            .or_else(|| self.records.get(location).map(|r| r.identifier.span))
408            .or_else(|| self.structs.get(&location.path).map(|s| s.identifier.span))
409            .or_else(|| self.globals.get(location).map(|g| g.span))
410            .map_or_else(|| Ok(()), |prev_span| Err(Self::emit_shadow_error(*name, span, prev_span)))
411    }
412
413    fn check_shadow_variable(&self, program: Symbol, path: &[Symbol], span: Span) -> Result<()> {
414        let mut current = self.local.as_ref();
415
416        while let Some(table) = current {
417            if let [name] = &path
418                && let Some(var_symbol) = table.inner.borrow().variables.get(name)
419            {
420                return Err(Self::emit_shadow_error(*name, span, var_symbol.span));
421            }
422            current = table.inner.borrow().parent.map(|id| self.all_locals.get(&id).expect("Parent should exist."));
423        }
424
425        self.check_shadow_global(&Location::new(program, path.to_vec()), span)?;
426
427        Ok(())
428    }

430    /// Insert a variable into the current scope.
431    pub fn insert_variable(&mut self, program: Symbol, path: &[Symbol], var: VariableSymbol) -> Result<()> {
432        self.check_shadow_variable(program, path, var.span)?;
433
434        if let Some(table) = self.local.as_mut() {
435            let [name] = &path else { panic!("Local variables cannot have paths with more than 1 segment.") };
436            table.inner.borrow_mut().variables.insert(*name, var);
437        } else {
438            self.globals.insert(Location::new(program, path.to_vec()), var);
439        }
440
441        Ok(())
442    }

444    /// Attach a finalizer to a function.
445    pub fn attach_finalizer(
446        &mut self,
447        caller: Location,
448        callee: Location,
449        future_inputs: Vec<Location>,
450        inferred_inputs: Vec<Type>,
451    ) -> Result<()> {
452        let callee_location = Location::new(callee.program, callee.path);
453
454        if let Some(func) = self.functions.get_mut(&caller) {
455            func.finalizer = Some(Finalizer { location: callee_location, future_inputs, inferred_inputs });
456            Ok(())
457        } else {
458            Err(AstError::function_not_found(caller.path.iter().format("::")).into())
459        }
460    }

463fn eq_struct(new: &Composite, old: &Composite) -> bool {
464    if new.members.len() != old.members.len() {
465        return false;
466    }
467
468    new.members
469        .iter()
470        .zip(old.members.iter())
471        .all(|(member1, member2)| member1.name() == member2.name() && member1.type_.eq_flat_relaxed(&member2.type_))
