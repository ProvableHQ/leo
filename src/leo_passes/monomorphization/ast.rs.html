36impl AstReconstructor for MonomorphizationVisitor<'_> {
37    type AdditionalOutput = ();
38
39    /* Types */
40    fn reconstruct_composite_type(&mut self, input: leo_ast::CompositeType) -> (leo_ast::Type, Self::AdditionalOutput) {
41        // Proceed only if there are some const arguments.
42        if input.const_arguments.is_empty() {
43            return (Type::Composite(input), Default::default());
44        }
45
46        // Ensure all const arguments are literals; if not, we skip this struct type instantiation for now and mark it
47        // as unresolved.
48        //
49        // The types of the const arguments are checked in the type checking pass.
50        if input.const_arguments.iter().any(|arg| !matches!(arg, Expression::Literal(_))) {
51            self.unresolved_struct_types.push(input.clone());
52            return (Type::Composite(input), Default::default());
53        }
54
55        // At this stage, we know that we're going to modify the program
56        self.changed = true;
57        (
58            Type::Composite(CompositeType {
59                id: Identifier {
60                    name: self.monomorphize_struct(&input.id.name, &input.const_arguments), // use the new name
61                    span: input.id.span,
62                    id: self.state.node_builder.next_id(),
63                },
64                const_arguments: vec![], // remove const arguments
65                program: input.program,
66            }),
67            Default::default(),
68        )
69    }
70
71    /* Expressions */
72    fn reconstruct_call(&mut self, input_call: CallExpression) -> (Expression, Self::AdditionalOutput) {
73        // Skip calls to functions from other programs.
74        if input_call.program.unwrap() != self.program {
75            return (input_call.into(), Default::default());
76        }
77
78        // Look up the already reconstructed function by name.
79        let callee_fn = self
80            .reconstructed_functions
81            .get(&input_call.function.name)
82            .expect("Callee should already be reconstructed (post-order traversal).");
83
84        // Proceed only if the function variant is `inline` and if there are some const arguments.
85        if !matches!(callee_fn.variant, Variant::Inline) || input_call.const_arguments.is_empty() {
86            return (input_call.into(), Default::default());
87        }
88
89        // Ensure all const arguments are literals; if not, we skip this call for now and mark it as unresolved.
90        //
91        // The types of the const arguments are checked in the type checking pass.
92        if input_call.const_arguments.iter().any(|arg| !matches!(arg, Expression::Literal(_))) {
93            self.unresolved_calls.push(input_call.clone());
94            return (input_call.into(), Default::default());
95        }
96
97        // Generate a unique name for the monomorphized function based on const arguments.
98        //
99        // For a function `fn foo::[x: u32, y: u32](..)`, the generated name would be `"foo::[1u32, 2u32]"` for a call
100        // that sets `x` to `1u32` and `y` to `2u32`. We know this name is safe to use because it's not a valid
101        // identifier in the user code.
102        let new_callee_name = leo_span::Symbol::intern(&format!(
103            "\"{}::[{}]\"",
104            input_call.function.name,
105            input_call.const_arguments.iter().format(", ")
106        ));
107
108        // Check if the new callee name is not already present in `reconstructed_functions`. This ensures that we do not
109        // add a duplicate definition for the same function.
110        if self.reconstructed_functions.get(&new_callee_name).is_none() {
111            // Build mapping from const parameters to const argument values.
112            let const_param_map: IndexMap<_, _> = callee_fn
113                .const_parameters
114                .iter()
115                .map(|param| param.identifier().name)
116                .zip_eq(&input_call.const_arguments)
117                .collect();
118
119            // Function to replace identifier expressions with their corresponding const argument or keep them unchanged.
120            let replace_identifier = |expr: &Expression| match expr {
121                Expression::Identifier(ident) => {
122                    const_param_map.get(&ident.name).map_or(Expression::Identifier(*ident), |&expr| expr.clone())
123                }
124                _ => expr.clone(),
125            };
126
127            let mut replacer = Replacer::new(replace_identifier, &self.state.node_builder);
128
129            // Create a new version of `callee_fn` that has a new name, no const parameters, and a new function ID.
130
131            // First, reconstruct the function by changing all instances of const generic parameters to literals
132            // according to `const_param_map`.
133            let mut function = replacer.reconstruct_function(callee_fn.clone());
134
135            // Now, reconstruct the function to actually monomorphize its content such as generic struct expressions.
136            function = self.reconstruct_function(function);
137            function.identifier = Identifier {
138                name: new_callee_name,
139                span: leo_span::Span::default(),
140                id: self.state.node_builder.next_id(),
141            };
142            function.const_parameters = vec![];
143            function.id = self.state.node_builder.next_id();
144
145            // Keep track of the new function in case other functions need it.
146            self.reconstructed_functions.insert(new_callee_name, function);
147
148            // Now keep track of the function we just monomorphized
149            self.monomorphized_functions.insert(input_call.function.name);
150        }
151        // At this stage, we know that we're going to modify the program
152        self.changed = true;
153
154        // Finally, construct the updated call expression that points to a monomorphized version and return it.
155        (
156            CallExpression {
157                function: Identifier {
158                    name: new_callee_name, // use the new name
159                    span: leo_span::Span::default(),
160                    id: self.state.node_builder.next_id(),
161                },
162                const_arguments: vec![], // remove const arguments
163                arguments: input_call.arguments,
164                program: input_call.program,
165                span: input_call.span, // Keep pointing to the original call expression
166                id: input_call.id,
167            }
168            .into(),
169            Default::default(),
170        )
171    }
172
173    fn reconstruct_struct_init(&mut self, mut input: StructExpression) -> (Expression, Self::AdditionalOutput) {
174        // Handle all the struct members first
175        let members = input
176            .members
177            .clone()
178            .into_iter()
179            .map(|member| StructVariableInitializer {
180                identifier: member.identifier,
181                expression: member.expression.map(|expr| self.reconstruct_expression(expr).0),
182                span: member.span,
183                id: member.id,
184            })
185            .collect();
186
187        // Proceed only if there are some const arguments.
188        if input.const_arguments.is_empty() {
189            input.members = members;
190            return (input.into(), Default::default());
191        }
192
193        // Ensure all const arguments are literals; if not, we skip this struct expression for now and mark it as
194        // unresolved.
195        //
196        // The types of the const arguments are checked in the type checking pass.
197        if input.const_arguments.iter().any(|arg| !matches!(arg, Expression::Literal(_))) {
198            self.unresolved_struct_exprs.push(input.clone());
199            input.members = members;
200            return (input.into(), Default::default());
201        }
202
203        // At this stage, we know that we're going to modify the program
204        self.changed = true;
205
206        // Finally, construct the updated struct expression that points to a monomorphized version and return it.
207        (
208            StructExpression {
209                name: Identifier {
210                    name: self.monomorphize_struct(&input.name.name, &input.const_arguments),
211                    span: input.name.span,
212                    id: self.state.node_builder.next_id(),
213                },
214                members,
215                const_arguments: vec![], // remove const arguments
216                span: input.span,        // Keep pointing to the original struct expression
217                id: input.id,
218            }
219            .into(),
220            Default::default(),
221        )
222    }
