18use indexmap::{IndexMap, IndexSet};
19use itertools::Itertools;
20use leo_ast::{
21    CallExpression,
22    Composite,
23    CompositeType,

25    Function,
26    Identifier,

30use leo_span::Symbol;
31
32pub struct MonomorphizationVisitor<'a> {
33    pub state: &'a mut CompilerState,
34    /// The main program.
35    pub program: Symbol,
36    /// A map of reconstructed functions in the current program scope.
37    pub reconstructed_functions: IndexMap<Symbol, Function>,
38    /// A set of all functions that have been monomorphized at least once. This keeps track of the _original_ names of
39    /// the functions not the names of the monomorphized versions.
40    pub monomorphized_functions: IndexSet<Symbol>,
41    /// A map of reconstructed functions in the current program scope.
42    pub reconstructed_structs: IndexMap<Symbol, Composite>,
43    /// A set of all functions that have been monomorphized at least once. This keeps track of the _original_ names of
44    /// the functions not the names of the monomorphized versions.
45    pub monomorphized_structs: IndexSet<Symbol>,
46    /// A vector of all the calls to const generic functions that have not been resolved.
47    pub unresolved_calls: Vec<CallExpression>,
48    /// A vector of all the struct expressions of const generic structs that have not been resolved.
49    pub unresolved_struct_exprs: Vec<StructExpression>,
50    /// A vector of all the struct type instantiations of const generic structs that have not been resolved.
51    pub unresolved_struct_types: Vec<CompositeType>,
52    /// Have we actually modified the program at all?
53    pub changed: bool,
54}
55
56impl MonomorphizationVisitor<'_> {
57    /// Monomorphizes a generic struct by substituting const parameters with concrete arguments and caching result.
58    /// Generates a unique name like `Foo::[1u32, 2u32]` based on the original name and the provided const arguments.
59    /// Replaces all const parameter references in the struct body with values, then resolves nested generics.
60    /// Assigns a new identifier and struct ID, clears const params, and stores the result to avoid reprocessing.
61    /// Panics if the original struct is not found in `reconstructed_structs` (should already be reconstructed).
62    ///
63    /// # Arguments
64    /// * `name` - Symbol of the original generic struct.
65    /// * `const_arguments` - Const values to substitute into the struct.
66    /// * Returns a `Symbol` for the newly monomorphized struct.
67    ///
68    /// Note: this functions already assumes that all provided const arguments are literals.
69    pub(crate) fn monomorphize_struct(&mut self, name: &Symbol, const_arguments: &Vec<Expression>) -> Symbol {
70        // Generate a unique name for the monomorphized struct based on const arguments.
71        //
72        // For `struct Foo::[x: u32, y: u32](..)`, the generated name would be `Foo::[1u32, 2u32]` for a struct
73        // expression that sets `x` to `1u32` and `y` to `2u32`. We know this name is safe to use because it's not a
74        // valid identifier in the user code.
75        //
76        // Later, we have to legalize these names because they are not valid Aleo identifiers. We do this in codegen.
77        let new_struct_name = leo_span::Symbol::intern(&format!("{}::[{}]", name, const_arguments.iter().format(", ")));
78
79        // Check if the new struct name is not already present in `reconstructed_structs`. This ensures that we do not
80        // add a duplicate definition for the same struct.
81        if self.reconstructed_structs.get(&new_struct_name).is_none() {
82            // Look up the already reconstructed struct by name.
83            let struct_ = self
84                .reconstructed_structs
85                .get(name)
86                .expect("Struct should already be reconstructed (post-order traversal).");
87
88            // Build mapping from const parameters to const argument values.
89            let const_param_map: IndexMap<_, _> =
90                struct_.const_parameters.iter().map(|param| param.identifier().name).zip_eq(const_arguments).collect();
91
92            // Function to replace identifier expressions with their corresponding const argument or keep them unchanged.
93            let replace_identifier = |expr: &Expression| match expr {
94                Expression::Identifier(ident) => {
95                    const_param_map.get(&ident.name).map_or(Expression::Identifier(*ident), |&expr| expr.clone())
96                }
97                _ => expr.clone(),
98            };
99
100            let mut replacer = Replacer::new(replace_identifier, &self.state.node_builder);
101
102            // Create a new version of `struct_` that has a new name, no const parameters, and a new struct ID.
103            //
104            // First, reconstruct the struct by changing all instances of const generic parameters to literals
105            // according to `const_param_map`.
106            let mut struct_ = replacer.reconstruct_struct(struct_.clone());
107
108            // Now, reconstruct the struct to actually monomorphize its content such as generic struct type
109            // instantiations.
110            struct_ = self.reconstruct_struct(struct_);
111            struct_.identifier = Identifier {
112                name: new_struct_name,
113                span: leo_span::Span::default(),
114                id: self.state.node_builder.next_id(),
115            };
116            struct_.const_parameters = vec![];
117            struct_.id = self.state.node_builder.next_id();
118
119            // Keep track of the new struct in case other structs need it.
120            self.reconstructed_structs.insert(new_struct_name, struct_);
121
122            // Now keep track of the struct we just monomorphized
123            self.monomorphized_structs.insert(*name);
124        }
125
126        new_struct_name
127    }
128}
