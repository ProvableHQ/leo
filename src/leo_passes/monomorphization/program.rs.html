17use super::MonomorphizationVisitor;
18use leo_ast::{Function, ProgramReconstructor, ProgramScope, Variant};
19use leo_span::Symbol;
20
21use indexmap::IndexMap;
22
23impl ProgramReconstructor for MonomorphizationVisitor<'_> {
24    fn reconstruct_program_scope(&mut self, input: ProgramScope) -> ProgramScope {
25        // Set the current program name from the input.
26        self.program = input.program_id.name.name;
27
28        // Create a map of function names to their definitions for fast access.
29        let mut function_map: IndexMap<Symbol, Function> = input.functions.into_iter().collect();
30
31        // Compute a post-order traversal of the call graph. This ensures that functions are processed after all their
32        // callees. Make sure to only to compute the post order by considering the entry points of the program, which
33        // are `async transition`, `transition` and `function`.
34        let order = self
35            .state
36            .call_graph
37            .post_order_from_entry_points(|node| {
38                function_map
39                    .get(node)
40                    .map(|f| {
41                        matches!(
42                            f.variant,
43                            Variant::AsyncTransition | Variant::Transition | Variant::Function | Variant::Script
44                        )
45                    })
46                    .unwrap_or(false)
47            })
48            .unwrap(); // This unwrap is safe because the type checker guarantees an acyclic graph.
49
50        // Reconstruct functions in post-order.
51        for function_name in &order {
52            // Skip external functions (i.e., not in the input map).
53            if let Some(function) = function_map.swap_remove(function_name) {
54                // Perform monomorphization or other reconstruction logic.
55                self.function = function.identifier.name;
56                let reconstructed_function = self.reconstruct_function(function);
57                // Store the reconstructed function for inclusion in the output scope.
58                self.reconstructed_functions.insert(*function_name, reconstructed_function);
59            }
60        }
61
62        // Retain only functions that are either not yet monomorphized or are still referenced by calls.
63        self.reconstructed_functions.retain(|f, _| {
64            let is_monomorphized = self.monomorphized_functions.contains(f);
65
66            let is_still_called = self.unresolved_calls.iter().any(|c| c.function.name == *f);
67
68            if is_monomorphized && !is_still_called {
69                // It's monomorphized and there are no unresolved calls to it - remove it.
70                self.state.call_graph.remove_node(f);
71                false
72            } else {
73                true
74            }
75        });
76
77        // Move reconstructed functions into the final `ProgramScope`, clearing the temporary storage for the next scope.
78        // Make sure to place transitions before all the other functions.
79        let (transitions, mut non_transitions): (Vec<_>, Vec<_>) = core::mem::take(&mut self.reconstructed_functions)
80            .into_iter()
81            .partition(|(_, f)| f.variant.is_transition());
82
83        let mut all_functions = transitions;
84        all_functions.append(&mut non_transitions);
85
86        // Return the fully reconstructed scope with updated functions.
87        ProgramScope {
88            program_id: input.program_id,
89            structs: input.structs,
90            mappings: input.mappings,
91            functions: all_functions,
92            consts: input.consts,
93            span: input.span,
94        }
95    }
96}
