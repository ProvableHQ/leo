17use super::MonomorphizationVisitor;
18use crate::Replacer;
19
20use leo_ast::{CallExpression, Expression, ExpressionReconstructor, Identifier, ProgramReconstructor, Variant};
21
22use indexmap::IndexMap;
23use itertools::Itertools;
24
25impl ExpressionReconstructor for MonomorphizationVisitor<'_> {
26    type AdditionalOutput = ();
27
28    fn reconstruct_call(&mut self, input_call: CallExpression) -> (Expression, Self::AdditionalOutput) {
29        // Skip calls to functions from other programs.
30        if input_call.program.unwrap() != self.program {
31            return (input_call.into(), Default::default());
32        }
33
34        // Look up the already reconstructed function by name.
35        let callee_fn = self
36            .reconstructed_functions
37            .get(&input_call.function.name)
38            .expect("Callee should already be reconstructed (post-order traversal).");
39
40        // Proceed only if the function variant is `inline` and if there are some const arguments.
41        if !matches!(callee_fn.variant, Variant::Inline) || input_call.const_arguments.is_empty() {
42            return (input_call.into(), Default::default());
43        }
44
45        // Ensure all const arguments are literals; if not, we skip this call for now and mark it as unresolved.
46        //
47        // The types of the const arguments are checked in the type checking pass.
48        if input_call.const_arguments.iter().any(|arg| !matches!(arg, Expression::Literal(_))) {
49            self.unresolved_calls.push(input_call.clone());
50            return (input_call.into(), Default::default());
51        }
52
53        // Generate a unique name for the monomorphized function based on const arguments.
54        //
55        // For a function `fn foo::[x: u32, y: u32](..)`, the generated name would be `foo::[1u32, 2u32]` for a call
56        // that sets `x` to `1u32` and `y` to `2u32`. We know this name is safe to use because it's not a valid
57        // identifier in the user code.
58        let new_callee_name = leo_span::Symbol::intern(&format!(
59            "\"{}::[{}]\"",
60            input_call.function.name,
61            input_call.const_arguments.iter().format(", ")
62        ));
63
64        // Check if the new callee name is not already present in `reconstructed_functions`. This ensures that we do not
65        // add a duplicate entry for the same function and only insert a new version with a unique name.
66        if self.reconstructed_functions.get(&new_callee_name).is_none() {
67            // Build mapping from const parameters to const argument values.
68            let const_param_map: IndexMap<_, _> = callee_fn
69                .const_parameters
70                .iter()
71                .map(|param| param.identifier().name)
72                .zip_eq(&input_call.const_arguments)
73                .collect();
74
75            // Function to replace identifiers with their corresponding const argument or keep them unchanged.
76            let replace_identifier = |ident: &Identifier| {
77                const_param_map.get(&ident.name).map_or(Expression::Identifier(*ident), |&expr| expr.clone())
78            };
79
80            let mut replacer = Replacer::new(replace_identifier, &self.state.node_builder);
81
82            // Create a new version of `callee_fn` that has a new name, no const parameters, and a new function ID.
83            self.function = callee_fn.identifier.name;
84            let mut function = replacer.reconstruct_function(callee_fn.clone());
85            function.identifier = Identifier {
86                name: new_callee_name,
87                span: leo_span::Span::default(),
88                id: self.state.node_builder.next_id(),
89            };
90            function.const_parameters = vec![];
91            function.id = self.state.node_builder.next_id();
92
93            // Keep track of the new function in case other functions need it.
94            self.reconstructed_functions.insert(new_callee_name, function);
95
96            // Now keep track of the function we just monomorphized
97            self.monomorphized_functions.insert(input_call.function.name);
98        }
99
100        // Update call graph with edges for the monomorphized function. We do this by basically cloning the edges in
101        // and out of `callee_name` and replicating them for a new node that contains `new_callee_name`.
102        if let Some(neighbors) = self.state.call_graph.neighbors(&input_call.function.name) {
103            for neighbor in neighbors {
104                if neighbor != input_call.function.name {
105                    self.state.call_graph.add_edge(new_callee_name, neighbor);
106                }
107            }
108        }
109        self.state.call_graph.add_edge(self.function, new_callee_name);
110
111        // Finally, construct the updated call expression that points to the monomorphized version and return it.
112        (
113            CallExpression {
114                function: Identifier {
115                    name: new_callee_name, // use the new name
116                    span: leo_span::Span::default(),
117                    id: self.state.node_builder.next_id(),
118                },
119                const_arguments: vec![], // remove const arguments
120                arguments: input_call.arguments,
121                program: input_call.program,
122                span: input_call.span, // Keep pointing to the original call expression
123                id: input_call.id,
124            }
125            .into(),
126            Default::default(),
127        )
128    }
129}
