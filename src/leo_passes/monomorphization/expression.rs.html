31use indexmap::IndexMap;

34impl ExpressionReconstructor for MonomorphizationVisitor<'_> {
35    type AdditionalOutput = ();
36
37    fn reconstruct_call(&mut self, input_call: CallExpression) -> (Expression, Self::AdditionalOutput) {
38        // Skip calls to functions from other programs.
39        if input_call.program.unwrap() != self.program {
40            return (input_call.into(), Default::default());
41        }
42
43        // Look up the already reconstructed function by name.
44        let callee_fn = self
45            .reconstructed_functions
46            .get(&input_call.function.name)
47            .expect("Callee should already be reconstructed (post-order traversal).");
48
49        // Proceed only if the function variant is `inline` and if there are some const arguments.
50        if !matches!(callee_fn.variant, Variant::Inline) || input_call.const_arguments.is_empty() {
51            return (input_call.into(), Default::default());
52        }
53
54        // Ensure all const arguments are literals; if not, we skip this call for now and mark it as unresolved.
55        //
56        // The types of the const arguments are checked in the type checking pass.
57        if input_call.const_arguments.iter().any(|arg| !matches!(arg, Expression::Literal(_))) {
58            self.unresolved_calls.push(input_call.clone());
59            return (input_call.into(), Default::default());
60        }
61
62        // Generate a unique name for the monomorphized function based on const arguments.
63        //
64        // For a function `fn foo::[x: u32, y: u32](..)`, the generated name would be `"foo::[1u32, 2u32]"` for a call
65        // that sets `x` to `1u32` and `y` to `2u32`. We know this name is safe to use because it's not a valid
66        // identifier in the user code.
67        let new_callee_name = leo_span::Symbol::intern(&format!(
68            "\"{}::[{}]\"",
69            input_call.function.name,
70            input_call.const_arguments.iter().format(", ")
71        ));
72
73        // Check if the new callee name is not already present in `reconstructed_functions`. This ensures that we do not
74        // add a duplicate definition for the same function.
75        if self.reconstructed_functions.get(&new_callee_name).is_none() {
76            // Build mapping from const parameters to const argument values.
77            let const_param_map: IndexMap<_, _> = callee_fn
78                .const_parameters
79                .iter()
80                .map(|param| param.identifier().name)
81                .zip_eq(&input_call.const_arguments)
82                .collect();
83
84            // Function to replace identifiers with their corresponding const argument or keep them unchanged.
85            let replace_identifier = |ident: &Identifier| {
86                const_param_map.get(&ident.name).map_or(Expression::Identifier(*ident), |&expr| expr.clone())
87            };
88
89            let mut replacer = Replacer::new(replace_identifier, &self.state.node_builder);
90
91            // Create a new version of `callee_fn` that has a new name, no const parameters, and a new function ID.
92
93            // First, reconstruct the function by changing all instances of const generic parameters to literals
94            // according to `const_param_map`.
95            let mut function = replacer.reconstruct_function(callee_fn.clone());
96
97            // Now, reconstruct the function to actually monomorphize its content such as generic struct expressions.
98            function = self.reconstruct_function(function);
99            function.identifier = Identifier {
100                name: new_callee_name,
101                span: leo_span::Span::default(),
102                id: self.state.node_builder.next_id(),
103            };
104            function.const_parameters = vec![];
105            function.id = self.state.node_builder.next_id();
106
107            // Keep track of the new function in case other functions need it.
108            self.reconstructed_functions.insert(new_callee_name, function);
109
110            // Now keep track of the function we just monomorphized
111            self.monomorphized_functions.insert(input_call.function.name);
112        }
113
114        // At this stage, we know that we're going to modify the program
115        self.changed = true;
116
117        // Finally, construct the updated call expression that points to a monomorphized version and return it.
118        (
119            CallExpression {
120                function: Identifier {
121                    name: new_callee_name, // use the new name
122                    span: leo_span::Span::default(),
123                    id: self.state.node_builder.next_id(),
124                },
125                const_arguments: vec![], // remove const arguments
126                arguments: input_call.arguments,
127                program: input_call.program,
128                span: input_call.span, // Keep pointing to the original call expression
129                id: input_call.id,
130            }
131            .into(),
132            Default::default(),
133        )
134    }
135
136    fn reconstruct_struct_init(&mut self, mut input: StructExpression) -> (Expression, Self::AdditionalOutput) {
137        // Handle all the struct members first
138        let members = input
139            .members
140            .clone()
141            .into_iter()
142            .map(|member| StructVariableInitializer {
143                identifier: member.identifier,
144                expression: member.expression.map(|expr| self.reconstruct_expression(expr).0),
145                span: member.span,
146                id: member.id,
147            })
148            .collect();
149
150        // Proceed only if there are some const arguments.
151        if input.const_arguments.is_empty() {
152            input.members = members;
153            return (input.into(), Default::default());
154        }
155
156        // Ensure all const arguments are literals; if not, we skip this struct expression for now and mark it as
157        // unresolved.
158        //
159        // The types of the const arguments are checked in the type checking pass.
160        if input.const_arguments.iter().any(|arg| !matches!(arg, Expression::Literal(_))) {
161            self.unresolved_struct_exprs.push(input.clone());
162            input.members = members;
163            return (input.into(), Default::default());
164        }
165
166        // At this stage, we know that we're going to modify the program
167        self.changed = true;
168
169        // Finally, construct the updated struct expression that points to a monomorphized version and return it.
170        (
171            StructExpression {
172                name: Identifier {
173                    name: self.monomorphize_struct(&input.name.name, &input.const_arguments),
174                    span: input.name.span,
175                    id: self.state.node_builder.next_id(),
176                },
177                members,
178                const_arguments: vec![], // remove const arguments
179                span: input.span,        // Keep pointing to the original struct expression
180                id: input.id,
181            }
182            .into(),
183            Default::default(),
184        )
185    }
186}
