17use crate::{CompilerState, ConstPropagation, Pass, Unrolling};
18
19use leo_errors::{CompilerError, Result};
20
21/// Pass that runs const propagation and loop unrolling until a fixed point.
22pub struct ConstPropagationAndUnrolling;
23
24impl Pass for ConstPropagationAndUnrolling {
25    type Input = ();
26    type Output = ();
27
28    const NAME: &str = "ConstPropagationAndUnrolling";
29
30    fn do_pass(_input: Self::Input, state: &mut CompilerState) -> Result<Self::Output> {
31        const LARGE_LOOP_BOUND: usize = 1024usize;
32
33        for _ in 0..LARGE_LOOP_BOUND {
34            let loop_unroll_output = Unrolling::do_pass((), state)?;
35
36            let const_prop_output = ConstPropagation::do_pass((), state)?;
37
38            if !const_prop_output.changed && !loop_unroll_output.loop_unrolled {
39                // We've got a fixed point, so see if we have any errors.
40                if let Some(not_evaluated_span) = const_prop_output.const_not_evaluated {
41                    return Err(CompilerError::const_not_evaluated(not_evaluated_span).into());
42                }
43
44                if let Some(not_evaluated_span) = const_prop_output.array_index_not_evaluated {
45                    return Err(CompilerError::array_index_not_evaluated(not_evaluated_span).into());
46                }
47
48                if let Some(not_unrolled_span) = loop_unroll_output.loop_not_unrolled {
49                    return Err(CompilerError::loop_bounds_not_evaluated(not_unrolled_span).into());
50                }
51
52                return Ok(());
53            }
54        }
55
56        // Note that it's challenging to write code in practice that demonstrates this error, because Leo code
57        // with many nested loops or operations will blow the stack in the compiler before this bound is hit.
58        Err(CompilerError::const_prop_unroll_many_loops(LARGE_LOOP_BOUND, Default::default()).into())
59    }
60}
