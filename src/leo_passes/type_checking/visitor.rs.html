29pub struct TypeCheckingVisitor<'a> {
30    pub state: &'a mut CompilerState,
31    /// The state of the current scope being traversed.
32    pub scope_state: ScopeState,
33    /// Mapping from async function stub name to the inferred input types.
34    pub async_function_input_types: IndexMap<Location, Vec<Type>>,
35    /// Mapping from async function name to the names of async transition callers.
36    pub async_function_callers: IndexMap<Location, IndexSet<Location>>,
37    /// The set of used composites.
38    pub used_structs: IndexSet<Symbol>,
39    /// So we can check if we exceed limits on array size, number of mappings, or number of functions.
40    pub limits: TypeCheckingInput,
41    /// For detecting the error `TypeCheckerError::async_cannot_assign_outside_conditional`.
42    pub conditional_scopes: Vec<IndexSet<Symbol>>,
43}

45impl TypeCheckingVisitor<'_> {
46    pub fn in_scope<T>(&mut self, id: NodeID, func: impl FnOnce(&mut Self) -> T) -> T {
47        self.state.symbol_table.enter_scope(Some(id));
48        let result = func(self);
49        self.state.symbol_table.enter_parent();
50        result
51    }
52
53    pub fn in_conditional_scope<T>(&mut self, func: impl FnOnce(&mut Self) -> T) -> T {
54        self.conditional_scopes.push(Default::default());
55        let result = func(self);
56        self.conditional_scopes.pop();
57        result
58    }
59
60    pub fn insert_symbol_conditional_scope(&mut self, symbol: Symbol) {
61        self.conditional_scopes.last_mut().expect("A conditional scope must be present.").insert(symbol);
62    }
63
64    pub fn symbol_in_conditional_scope(&mut self, symbol: Symbol) -> bool {
65        self.conditional_scopes.last().map(|set| set.contains(&symbol)).unwrap_or(false)
66    }

68    /// Emits a type checker error.
69    pub fn emit_err(&self, err: TypeCheckerError) {
70        self.state.handler.emit_err(err);
71    }

73    /// Emits a type checker warning
74    pub fn emit_warning(&self, warning: TypeCheckerWarning) {
75        self.state.handler.emit_warning(warning.into());

78    /// Emits an error if the two given types are not equal.
79    pub fn check_eq_types(&self, t1: &Option<Type>, t2: &Option<Type>, span: Span) {
80        match (t1, t2) {
81            (Some(t1), Some(t2)) if !Type::eq_flat_relaxed(t1, t2) => {
82                // If both types are futures, print them out.
83                if let (Type::Future(f1), Type::Future(f2)) = (t1, t2) {
84                    println!("Future 1: {:?}", f1);
85                    println!("Future 2: {:?}", f2);
86                }
87                self.emit_err(TypeCheckerError::type_should_be(t1, t2, span))
88            }
89            (Some(type_), None) | (None, Some(type_)) => {
90                self.emit_err(TypeCheckerError::type_should_be("no type", type_, span))
91            }
92            _ => {}
93        }
94    }

96    /// Use this method when you know the actual type.
97    /// Emits an error to the handler if the `actual` type is not equal to the `expected` type.
98    pub fn assert_and_return_type(&mut self, actual: Type, expected: &Option<Type>, span: Span) -> Type {
99        // If expected is `Type::Err`, we don't want to actually report a redundant error.
100        if expected.is_some() && !matches!(expected, Some(Type::Err)) {
101            self.check_eq_types(&Some(actual.clone()), expected, span);
102        }
103        actual
104    }
105
106    pub fn maybe_assert_type(&mut self, actual: &Type, expected: &Option<Type>, span: Span) {
107        if let Some(expected) = expected {
108            self.assert_type(actual, expected, span);
109        }
110    }
111
112    pub fn assert_type(&mut self, actual: &Type, expected: &Type, span: Span) {
113        if actual != &Type::Err && !self.eq_user(actual, expected) {
114            // If `actual` is Err, we will have already reported an error.
115            self.emit_err(TypeCheckerError::type_should_be2(actual, format!("type `{expected}`"), span));
116        }
117    }
118
119    pub fn assert_int_type(&self, type_: &Type, span: Span) {
120        if !matches!(type_, Type::Err | Type::Integer(_)) {
121            self.emit_err(TypeCheckerError::type_should_be2(type_, "an integer", span));
122        }
123    }

185    /// Type checks the inputs to an associated constant and returns the expected output type.
186    pub fn get_core_constant(&self, type_: &Type, constant: &Identifier) -> Option<CoreConstant> {
187        if let Type::Identifier(ident) = type_ {
188            // Lookup core constant
189            match CoreConstant::from_symbols(ident.name, constant.name) {
190                None => {
191                    // Not a core constant.
192                    self.emit_err(TypeCheckerError::invalid_core_constant(ident.name, constant.name, ident.span()));
193                }
194                Some(core_constant) => return Some(core_constant),
195            }
196        }
197        None
198    }

200    /// Emits an error if the `struct` is not a core library struct.
201    /// Emits an error if the `function` is not supported by the struct.
202    pub fn get_core_function_call(&self, struct_: &Identifier, function: &Identifier) -> Option<CoreFunction> {
203        // Lookup core struct
204        match CoreFunction::from_symbols(struct_.name, function.name) {
205            None => {
206                // Not a core library struct.
207                self.emit_err(TypeCheckerError::invalid_core_function(struct_.name, function.name, struct_.span()));
208                None
209            }
210            Some(core_instruction) => Some(core_instruction),
211        }
212    }

214    /// Type checks the inputs to a core function call and returns the expected output type.
215    /// Emits an error if the correct number of arguments are not provided.
216    /// Emits an error if the arguments are not of the correct type.
217    pub fn check_core_function_call(
218        &mut self,
219        core_function: CoreFunction,
220        arguments: &[(Type, Span)],
221        function_span: Span,
222    ) -> Type {
223        // Check that the number of arguments is correct.
224        if arguments.len() != core_function.num_args() {
225            self.emit_err(TypeCheckerError::incorrect_num_args_to_call(
226                core_function.num_args(),
227                arguments.len(),
228                function_span,
229            ));
230            return Type::Err;
231        }
232
233        let assert_not_mapping_tuple_unit = |type_: &Type, span: Span| {
234            if matches!(type_, Type::Mapping(_) | Type::Tuple(_) | Type::Unit) {
235                self.emit_err(TypeCheckerError::type_should_be2(type_, "anything but a mapping, tuple, or unit", span));
236            }
237        };
238
239        // Make sure the input is no bigger than 64 bits.
240        // Due to overhead in the bitwise representations of types in SnarkVM, 64 bit integers
241        // input more than 64 bits to a hash function, as do all structs and arrays.
242        let assert_pedersen_64_bit_input = |type_: &Type, span: Span| {
243            if !matches!(
244                type_,
245                Type::Integer(IntegerType::U8)
246                    | Type::Integer(IntegerType::U16)
247                    | Type::Integer(IntegerType::U32)
248                    | Type::Integer(IntegerType::I8)
249                    | Type::Integer(IntegerType::I16)
250                    | Type::Integer(IntegerType::I32)
251                    | Type::Boolean
252                    | Type::Err
253            ) {
254                self.emit_err(TypeCheckerError::type_should_be2(
255                    type_,
256                    "an integer of less than 64 bits or a bool",
257                    span,
258                ));
259            }
260        };
261
262        // Make sure the input is no bigger than 128 bits.
263        //
264        // Due to overhead in the bitwise representations of types in SnarkVM, 128 bit integers
265        // input more than 128 bits to a hash function, as do most structs and arrays. We could
266        // actually allow arrays with a single element of type smaller than 64 bits, but it
267        // seems most understandable to the user to simply disallow composite types entirely.
268        let assert_pedersen_128_bit_input = |type_: &Type, span: Span| {
269            if !matches!(
270                type_,
271                Type::Integer(IntegerType::U8)
272                    | Type::Integer(IntegerType::U16)
273                    | Type::Integer(IntegerType::U32)
274                    | Type::Integer(IntegerType::U64)
275                    | Type::Integer(IntegerType::I8)
276                    | Type::Integer(IntegerType::I16)
277                    | Type::Integer(IntegerType::I32)
278                    | Type::Integer(IntegerType::I64)
279                    | Type::Boolean
280                    | Type::Err
281            ) {
282                self.emit_err(TypeCheckerError::type_should_be2(
283                    type_,
284                    "an integer of less than 128 bits or a bool",
285                    span,
286                ));
287            }
288        };
289
290        // Check that the arguments are of the correct type.
291        match core_function {
292            CoreFunction::BHP256CommitToAddress
293            | CoreFunction::BHP512CommitToAddress
294            | CoreFunction::BHP768CommitToAddress
295            | CoreFunction::BHP1024CommitToAddress => {
296                assert_not_mapping_tuple_unit(&arguments[0].0, arguments[0].1);
297                self.assert_type(&arguments[1].0, &Type::Scalar, arguments[1].1);
298                Type::Address
299            }
300            CoreFunction::BHP256CommitToField
301            | CoreFunction::BHP512CommitToField
302            | CoreFunction::BHP768CommitToField
303            | CoreFunction::BHP1024CommitToField => {
304                assert_not_mapping_tuple_unit(&arguments[0].0, arguments[0].1);
305                self.assert_type(&arguments[1].0, &Type::Scalar, arguments[1].1);
306                Type::Field
307            }
308            CoreFunction::BHP256CommitToGroup
309            | CoreFunction::BHP512CommitToGroup
310            | CoreFunction::BHP768CommitToGroup
311            | CoreFunction::BHP1024CommitToGroup => {
312                assert_not_mapping_tuple_unit(&arguments[0].0, arguments[0].1);
313                self.assert_type(&arguments[1].0, &Type::Scalar, arguments[1].1);
314                Type::Group
315            }
316            CoreFunction::BHP256HashToAddress
317            | CoreFunction::BHP512HashToAddress
318            | CoreFunction::BHP768HashToAddress
319            | CoreFunction::BHP1024HashToAddress
320            | CoreFunction::Keccak256HashToAddress
321            | CoreFunction::Keccak384HashToAddress
322            | CoreFunction::Keccak512HashToAddress
323            | CoreFunction::Poseidon2HashToAddress
324            | CoreFunction::Poseidon4HashToAddress
325            | CoreFunction::Poseidon8HashToAddress
326            | CoreFunction::SHA3_256HashToAddress
327            | CoreFunction::SHA3_384HashToAddress
328            | CoreFunction::SHA3_512HashToAddress => {
329                assert_not_mapping_tuple_unit(&arguments[0].0, arguments[0].1);
330                Type::Address
331            }
332            CoreFunction::BHP256HashToField
333            | CoreFunction::BHP512HashToField
334            | CoreFunction::BHP768HashToField
335            | CoreFunction::BHP1024HashToField
336            | CoreFunction::Keccak256HashToField
337            | CoreFunction::Keccak384HashToField
338            | CoreFunction::Keccak512HashToField
339            | CoreFunction::Poseidon2HashToField
340            | CoreFunction::Poseidon4HashToField

837    /// Emits an error if the type or its constituent types is not valid.
838    pub fn assert_type_is_valid(&mut self, type_: &Type, span: Span) {
839        match type_ {
840            // Unit types may only appear as the return type of a function.
841            Type::Unit => {
842                self.emit_err(TypeCheckerError::unit_type_only_return(span));
843            }
844            // String types are temporarily disabled.
845            Type::String => {
846                self.emit_err(TypeCheckerError::strings_are_not_supported(span));
847            }
848            // Check that the named composite type has been defined.
849            Type::Composite(struct_)
850                if self.lookup_struct(struct_.program.or(self.scope_state.program_name), struct_.id.name).is_none() =>
851            {
852                self.emit_err(TypeCheckerError::undefined_type(struct_.id.name, span));
853            }
854            // Check that the constituent types of the tuple are valid.
855            Type::Tuple(tuple_type) => {
856                for type_ in tuple_type.elements().iter() {
857                    self.assert_type_is_valid(type_, span);
858                }
859            }
860            // Check that the constituent types of mapping are valid.
861            Type::Mapping(mapping_type) => {
862                self.assert_type_is_valid(&mapping_type.key, span);
863                self.assert_type_is_valid(&mapping_type.value, span);
864            }
865            // Check that the array element types are valid.
866            Type::Array(array_type) => {
867                // Check that the array length is valid.
868                match array_type.length() {
869                    0 => self.emit_err(TypeCheckerError::array_empty(span)),
870                    length => {
871                        if length > self.limits.max_array_elements {
872                            self.emit_err(TypeCheckerError::array_too_large(
873                                length,
874                                self.limits.max_array_elements,
875                                span,
876                            ))
877                        }
878                    }
879                }
880                // Check that the array element type is valid.
881                match array_type.element_type() {
882                    // Array elements cannot be futures.
883                    Type::Future(_) => self.emit_err(TypeCheckerError::array_element_cannot_be_future(span)),
884                    // Array elements cannot be tuples.
885                    Type::Tuple(_) => self.emit_err(TypeCheckerError::array_element_cannot_be_tuple(span)),
886                    // Array elements cannot be records.
887                    Type::Composite(struct_type) => {
888                        // Look up the type.
889                        if let Some(struct_) = self
890                            .lookup_struct(struct_type.program.or(self.scope_state.program_name), struct_type.id.name)
891                        {
892                            // Check that the type is not a record.
893                            if struct_.is_record {
894                                self.emit_err(TypeCheckerError::array_element_cannot_be_record(span));
895                            }
896                        }
897                    }
898                    _ => {} // Do nothing.
899                }
900                self.assert_type_is_valid(array_type.element_type(), span);
901            }
902            _ => {} // Do nothing.
903        }
904    }

919    /// Helper function to check that the input and output of function are valid
920    pub fn check_function_signature(&mut self, function: &Function, is_stub: bool) {
921        self.scope_state.variant = Some(function.variant);
922
923        let mut inferred_inputs: Vec<Type> = Vec::new();
924
925        if self.scope_state.variant == Some(Variant::AsyncFunction) && !self.scope_state.is_stub {
926            // Async functions are not allowed to return values.
927            if !function.output.is_empty() {
928                self.emit_err(TypeCheckerError::async_function_cannot_return_value(function.span()));
929            }
930
931            // Iterator over the `finalize` member (type Finalizer) of each async transition that calls
932            // this async function.
933            let mut caller_finalizers = self
934                .async_function_callers
935                .get(&Location::new(self.scope_state.program_name.unwrap(), function.identifier.name))
936                .map(|callers| {
937                    callers
938                        .iter()
939                        .flat_map(|caller| {
940                            let caller = Location::new(caller.program, caller.name);
941                            self.state.symbol_table.lookup_function(caller)
942                        })
943                        .flat_map(|fn_symbol| fn_symbol.finalizer.clone())
944                })
945                .into_iter()
946                .flatten();
947
948            if let Some(first) = caller_finalizers.next() {
949                inferred_inputs = first.inferred_inputs.clone();
950            } else {
951                self.emit_warning(TypeCheckerWarning::async_function_is_never_called_by_transition_function(
952                    function.identifier.name,
953                    function.span(),
954                ));
955            }
956
957            // If any input is a future that doesn't have the same member type for all
958            // finalizers, set that member to `Type::Err`.
959            for finalizer in caller_finalizers {
960                assert_eq!(inferred_inputs.len(), finalizer.inferred_inputs.len());
961                for (t1, t2) in inferred_inputs.iter_mut().zip(finalizer.inferred_inputs.iter()) {
962                    self.merge_types(t1, t2);
963                }
964            }
965        }
966
967        for (i, input) in function.input.iter().enumerate() {
968            // No need to check compatibility of these types; that's already been done
969            let table_type = inferred_inputs.get(i).unwrap_or_else(|| input.type_());
970
971            // Check that the type of input parameter is defined.
972            self.assert_type_is_valid(table_type, input.span());
973
974            // Check that the type of the input parameter is not a tuple.
975            if matches!(table_type, Type::Tuple(_)) {
976                self.emit_err(TypeCheckerError::function_cannot_take_tuple_as_input(input.span()))
977            }
978
979            // Make sure only transitions can take a record as an input.
980            if let Type::Composite(struct_) = table_type {
981                // Throw error for undefined type.
982                if !function.variant.is_transition() {
983                    if let Some(elem) =
984                        self.lookup_struct(struct_.program.or(self.scope_state.program_name), struct_.id.name)
985                    {
986                        if elem.is_record {
987                            self.emit_err(TypeCheckerError::function_cannot_input_or_output_a_record(input.span()))
988                        }
989                    } else {
990                        self.emit_err(TypeCheckerError::undefined_type(struct_.id, input.span()));
991                    }
992                }
993            }
994
995            // This unwrap works since we assign to `variant` above.
996            match self.scope_state.variant.unwrap() {
997                // If the function is a transition function, then check that the parameter mode is not a constant.
998                Variant::Transition | Variant::AsyncTransition if input.mode() == Mode::Constant => {
999                    self.emit_err(TypeCheckerError::transition_function_inputs_cannot_be_const(input.span()))
1000                }
1001                // If the function is standard function or inline, then check that the parameters do not have an associated mode.
1002                Variant::Function | Variant::Inline if input.mode() != Mode::None => {
1003                    self.emit_err(TypeCheckerError::regular_function_inputs_cannot_have_modes(input.span()))
1004                }
1005                // If the function is an async function, then check that the input parameter is not constant or private.
1006                Variant::AsyncFunction if matches!(input.mode(), Mode::Constant | Mode::Private) => {
1007                    self.emit_err(TypeCheckerError::async_function_input_must_be_public(input.span()));
1008                }
1009                _ => {} // Do nothing.
1010            }
1011
1012            if matches!(table_type, Type::Future(..)) {
1013                // Future parameters may only appear in async functions.
1014                if !matches!(self.scope_state.variant, Some(Variant::AsyncFunction)) {
1015                    self.emit_err(TypeCheckerError::no_future_parameters(input.span()));
1016                }
1017            }
1018
1019            if !is_stub {
1020                // Add the input to the symbol table.
1021                if let Err(err) = self.state.symbol_table.insert_variable(
1022                    self.scope_state.program_name.unwrap(),
1023                    input.identifier().name,
1024                    VariableSymbol {
1025                        type_: table_type.clone(),
1026                        span: input.identifier.span(),
1027                        declaration: VariableType::Input(input.mode()),
1028                    },
1029                ) {
1030                    self.state.handler.emit_err(err);
1031                }
1032
1033                // Add the input to the type table.
1034                self.state.type_table.insert(input.identifier().id(), table_type.clone());
1035            }
1036        }
1037
1038        // Type check the function's return type.
1039        // Note that checking that each of the component types are defined is sufficient to check that `output_type` is defined.
1040        function.output.iter().enumerate().for_each(|(index, function_output)| {
1041            // If the function is not a transition function, then it cannot output a record.
1042            // Note that an external output must always be a record.
1043            if let Type::Composite(struct_) = function_output.type_.clone() {
1044                if let Some(val) =
1045                    self.lookup_struct(struct_.program.or(self.scope_state.program_name), struct_.id.name)
1046                {
1047                    if val.is_record && !function.variant.is_transition() {
1048                        self.emit_err(TypeCheckerError::function_cannot_input_or_output_a_record(function_output.span));
1049                    }
1050                }
1051            }
1052
1053            // Check that the output type is valid.
1054            self.assert_type_is_valid(&function_output.type_, function_output.span);
1055
1056            // Check that the type of the output is not a tuple. This is necessary to forbid nested tuples.
1057            if matches!(&function_output.type_, Type::Tuple(_)) {
1058                self.emit_err(TypeCheckerError::nested_tuple_type(function_output.span))
1059            }
1060            // Check that the mode of the output is valid.
1061            // For functions, only public and private outputs are allowed
1062            if function_output.mode == Mode::Constant {
1063                self.emit_err(TypeCheckerError::cannot_have_constant_output_mode(function_output.span));
1064            }
1065            // Async transitions must return exactly one future, and it must be in the last position.
1066            if self.scope_state.variant == Some(Variant::AsyncTransition)
1067                && ((index < function.output.len() - 1 && matches!(function_output.type_, Type::Future(_)))
1068                    || (index == function.output.len() - 1 && !matches!(function_output.type_, Type::Future(_))))
1069            {
1070                self.emit_err(TypeCheckerError::async_transition_invalid_output(function_output.span));
1071            }
1072            // If the function is not an async transition, then it cannot have a future as output.
1073            if !matches!(self.scope_state.variant, Some(Variant::AsyncTransition) | Some(Variant::Script))
1074                && matches!(function_output.type_, Type::Future(_))
1075            {
1076                self.emit_err(TypeCheckerError::only_async_transition_can_return_future(function_output.span));
1077            }
1078        });
1079    }

1081    /// Merge inferred types into `lhs`.
1082    ///
1083    /// That is, if `lhs` and `rhs` aren't equal, set `lhs` to Type::Err;
1084    /// or, if they're both futures, set any member of `lhs` that isn't
1085    /// equal to the equivalent member of `rhs` to `Type::Err`.
1086    fn merge_types(&self, lhs: &mut Type, rhs: &Type) {
1087        if let Type::Future(f1) = lhs {
1088            if let Type::Future(f2) = rhs {
1089                for (i, type_) in f2.inputs.iter().enumerate() {
1090                    if let Some(lhs_type) = f1.inputs.get_mut(i) {
1091                        self.merge_types(lhs_type, type_);
1092                    } else {
1093                        f1.inputs.push(Type::Err);
1094                    }
1095                }
1096            } else {
1097                *lhs = Type::Err;
1098            }
1099        } else if !self.eq_user(lhs, rhs) {
1100            *lhs = Type::Err;
1101        }
1102    }

1104    /// Are the types considered equal as far as the Leo user is concerned?
1105    ///
1106    /// In particular, any comparison involving an `Err` is `true`,
1107    /// composite types are resolved to the current program if not specified,
1108    /// and Futures which aren't explicit compare equal to other Futures.
1109    pub fn eq_user(&self, t1: &Type, t2: &Type) -> bool {
1110        match (t1, t2) {
1111            (Type::Err, _)
1112            | (_, Type::Err)
1113            | (Type::Address, Type::Address)
1114            | (Type::Boolean, Type::Boolean)
1115            | (Type::Field, Type::Field)
1116            | (Type::Group, Type::Group)
1117            | (Type::Scalar, Type::Scalar)
1118            | (Type::Signature, Type::Signature)
1119            | (Type::String, Type::String)
1120            | (Type::Unit, Type::Unit) => true,
1121            (Type::Array(left), Type::Array(right)) => {
1122                left.length() == right.length() && self.eq_user(left.element_type(), right.element_type())
1123            }
1124            (Type::Identifier(left), Type::Identifier(right)) => left.name == right.name,
1125            (Type::Integer(left), Type::Integer(right)) => left == right,
1126            (Type::Mapping(left), Type::Mapping(right)) => {
1127                self.eq_user(&left.key, &right.key) && self.eq_user(&left.value, &right.value)
1128            }
1129            (Type::Tuple(left), Type::Tuple(right)) if left.length() == right.length() => left
1130                .elements()
1131                .iter()
1132                .zip_eq(right.elements().iter())
1133                .all(|(left_type, right_type)| self.eq_user(left_type, right_type)),
1134            (Type::Composite(left), Type::Composite(right)) => {
1135                let left_program = left.program.or(self.scope_state.program_name);
1136                let right_program = right.program.or(self.scope_state.program_name);
1137
1138                left.id.name == right.id.name && left_program == right_program
1139            }
1140            (Type::Future(left), Type::Future(right)) if !left.is_explicit || !right.is_explicit => true,
1141            (Type::Future(left), Type::Future(right)) if left.inputs.len() == right.inputs.len() => left
1142                .inputs()
1143                .iter()
1144                .zip_eq(right.inputs().iter())
1145                .all(|(left_type, right_type)| self.eq_user(left_type, right_type)),
1146            _ => false,
1147        }
1148    }

1150    /// Wrapper around lookup_struct that additionally records all structs that are used in the program.
1151    pub fn lookup_struct(&mut self, program: Option<Symbol>, name: Symbol) -> Option<Composite> {
1152        let record_comp = program.and_then(|prog| self.state.symbol_table.lookup_record(Location::new(prog, name)));
1153        let comp = record_comp.or_else(|| self.state.symbol_table.lookup_struct(name));
1154        // Record the usage.
1155        if let Some(s) = comp {
1156            // If it's a struct or internal record, mark it used.
1157            if !s.is_record || program == self.scope_state.program_name {
1158                self.used_structs.insert(s.identifier.name);
1159            }
1160        }
1161        comp.cloned()
1162    }

1164    /// Inserts variable to symbol table.
1165    pub fn insert_variable(&mut self, inferred_type: Option<Type>, name: &Identifier, type_: Type, span: Span) {
1166        self.insert_symbol_conditional_scope(name.name);
1167
1168        let is_future = match &type_ {
1169            Type::Future(..) => true,
1170            Type::Tuple(tuple_type) if matches!(tuple_type.elements().last(), Some(Type::Future(..))) => true,
1171            _ => false,
1172        };
1173
1174        if is_future {
1175            // It can happen that the call location has not been set if there was an error
1176            // in the call that produced the Future.
1177            if let Some(call_location) = self.scope_state.call_location {
1178                self.scope_state.futures.insert(name.name, call_location);
1179            }
1180        }
1181
1182        let ty = match (is_future, inferred_type) {
1183            (false, _) => type_,
1184            (true, Some(inferred)) => inferred,
1185            (true, None) => unreachable!("Type checking guarantees the inferred type is present"),
1186        };
1187
1188        // Insert the variable into the symbol table.
1189        if let Err(err) =
1190            self.state.symbol_table.insert_variable(self.scope_state.program_name.unwrap(), name.name, VariableSymbol {
1191                type_: ty.clone(),
1192                span,
1193                declaration: VariableType::Mut,
1194            })
1195        {
1196            self.state.handler.emit_err(err);
1197        }
1198    }

1200    // Checks if the access operation is valid inside the current function variant.
1201    pub fn check_access_allowed(&mut self, name: &str, finalize_op: bool, span: Span) {
1202        // Check that the function context matches.
1203        if self.scope_state.variant == Some(Variant::AsyncFunction) && !finalize_op {
1204            self.state.handler.emit_err(TypeCheckerError::invalid_operation_inside_finalize(name, span))
1205        } else if !matches!(self.scope_state.variant, Some(Variant::AsyncFunction) | Some(Variant::Script))
1206            && finalize_op
1207        {
1208            self.state.handler.emit_err(TypeCheckerError::invalid_operation_outside_finalize(name, span))
1209        }
1210    }

1212    pub fn is_external_record(&self, ty: &Type) -> bool {
1213        if let Type::Composite(typ) = &ty {
1214            let this_program = self.scope_state.program_name.unwrap();
1215            let program = typ.program.unwrap_or(this_program);
1216            program != this_program
1217                && self.state.symbol_table.lookup_record(Location::new(program, typ.id.name)).is_some()
1218        } else {
1219            false
1220        }
1221    }

1238    // Given a `Literal` and its type, if the literal is a numeric `Unsuffixed` literal, ensure it's a valid literal
1239    // given the type. E.g., a `256` is not a valid `u8`.
1240    pub fn check_numeric_literal(&self, input: &Literal, ty: &Type) {
1241        if let Literal { variant: LiteralVariant::Unsuffixed(s), .. } = input {
1242            let span = input.span();
1243            let has_nondecimal_prefix =
1244                |s: &str| ["0x", "0o", "0b", "-0x", "-0o", "-0b"].iter().any(|p| s.starts_with(p));
1245
1246            macro_rules! parse_int {
1247                ($t:ty, $name:expr) => {
1248                    self.parse_integer_literal::<$t>(s, span, $name)
1249                };
1250            }
1251
1252            match ty {
1253                Type::Integer(kind) => match kind {
1254                    IntegerType::U8 => parse_int!(u8, "u8"),
1255                    IntegerType::U16 => parse_int!(u16, "u16"),
1256                    IntegerType::U32 => parse_int!(u32, "u32"),
1257                    IntegerType::U64 => parse_int!(u64, "u64"),
1258                    IntegerType::U128 => parse_int!(u128, "u128"),
1259                    IntegerType::I8 => parse_int!(i8, "i8"),
1260                    IntegerType::I16 => parse_int!(i16, "i16"),
1261                    IntegerType::I32 => parse_int!(i32, "i32"),
1262                    IntegerType::I64 => parse_int!(i64, "i64"),
1263                    IntegerType::I128 => parse_int!(i128, "i128"),
1264                },
1265                Type::Group => {
1266                    if has_nondecimal_prefix(s) {
1267                        // This is not checked in the parser for unsuffixed numerals. So do that here.
1268                        self.emit_err(TypeCheckerError::hexbin_literal_nonintegers(span));
1269                    } else {
1270                        let trimmed = s.trim_start_matches('-').trim_start_matches('0');
1271                        if !trimmed.is_empty()
1272                            && format!("{trimmed}group")
1273                                .parse::<snarkvm::prelude::Group<snarkvm::prelude::TestnetV0>>()
1274                                .is_err()
1275                        {
1276                            self.emit_err(TypeCheckerError::invalid_int_value(trimmed, "group", span));
1277                        }
1278                    }
1279                }
1280                Type::Field | Type::Scalar => {
1281                    if has_nondecimal_prefix(s) {
1282                        // This is not checked in the parser for unsuffixed numerals. So do that here.
1283                        self.emit_err(TypeCheckerError::hexbin_literal_nonintegers(span));
1284                    }
1285                }
1286                _ => {
1287                    // Other types aren't expected here
1288                }
1289            }
1290        }
1291    }
