26impl TypeCheckingVisitor<'_> {
27    pub fn visit_expression_assign(&mut self, input: &Expression) -> Type {
28        let ty = match input {
29            Expression::ArrayAccess(array_access) => self.visit_array_access_general(array_access, true, &None),
30            Expression::Identifier(ident) => self.visit_identifier_assign(ident),
31            Expression::MemberAccess(member_access) => self.visit_member_access_general(member_access, true, &None),
32            Expression::TupleAccess(tuple_access) => self.visit_tuple_access_general(tuple_access, true, &None),
33            _ => {
34                self.emit_err(TypeCheckerError::invalid_assignment_target(input, input.span()));
35                Type::Err
36            }
37        };
38
39        // Prohibit reassignment of futures.
40        if let Type::Future(..) = ty {
41            self.emit_err(TypeCheckerError::cannot_reassign_future_variable(input, input.span()));
42        }
43
44        // Prohibit reassignment of mappings.
45        if let Type::Mapping(_) = ty {
46            self.emit_err(TypeCheckerError::cannot_reassign_mapping(input, input.span()));
47        }
48
49        // Add the expression and its associated type to the type table.
50        self.state.type_table.insert(input.id(), ty.clone());
51        ty
52    }
53
54    fn visit_array_access_general(&mut self, input: &ArrayAccess, assign: bool, expected: &Option<Type>) -> Type {
55        // Check that the expression is an array.
56        let this_type = if assign {
57            self.visit_expression_assign(&input.array)
58        } else {
59            self.visit_expression(&input.array, &None)
60        };
61        self.assert_array_type(&this_type, input.array.span());
62
63        // Check that the index is an integer type.
64        let index_type = self.visit_expression(&input.index, &None);
65        self.assert_int_type(&index_type, input.index.span());
66
67        // Get the element type of the array.
68        let Type::Array(array_type) = this_type else {
69            // We must have already reported an error above, in our type assertion.
70            return Type::Err;
71        };
72
73        let element_type = array_type.element_type();
74
75        // If the expected type is known, then check that the element type is the same as the expected type.
76        self.maybe_assert_type(element_type, expected, input.span());
77
78        // Return the element type of the array.
79        element_type.clone()
80    }

82    fn visit_member_access_general(&mut self, input: &MemberAccess, assign: bool, expected: &Option<Type>) -> Type {
83        // Handler member access expressions that correspond to valid operands in AVM code.
84        if !assign {
85            match input.inner {
86                // If the access expression is of the form `self.<name>`, then check the <name> is valid.
87                Expression::Identifier(identifier) if identifier.name == sym::SelfLower => {
88                    match input.name.name {
89                        sym::caller => {
90                            // Check that the operation is not invoked in a `finalize` block.
91                            self.check_access_allowed("self.caller", false, input.name.span());
92                            return Type::Address;
93                        }
94                        sym::signer => {
95                            // Check that operation is not invoked in a `finalize` block.
96                            self.check_access_allowed("self.signer", false, input.name.span());
97                            return Type::Address;
98                        }
99                        _ => {
100                            self.emit_err(TypeCheckerError::invalid_self_access(input.name.span()));
101                            return Type::Err;
102                        }
103                    }
104                }
105                // If the access expression is of the form `block.<name>`, then check the <name> is valid.
106                Expression::Identifier(identifier) if identifier.name == sym::block => match input.name.name {
107                    sym::height => {
108                        // Check that the operation is invoked in a `finalize` block.
109                        self.check_access_allowed("block.height", true, input.name.span());
110                        let ty = Type::Integer(IntegerType::U32);
111                        self.maybe_assert_type(&ty, expected, input.span());
112                        return ty;
113                    }
114                    _ => {
115                        self.emit_err(TypeCheckerError::invalid_block_access(input.name.span()));
116                        return Type::Err;
117                    }
118                },
119                // If the access expression is of the form `network.<name>`, then check that the <name> is valid.
120                Expression::Identifier(identifier) if identifier.name == sym::network => match input.name.name {
121                    sym::id => {
122                        // Check that the operation is not invoked outside a `finalize` block.
123                        self.check_access_allowed("network.id", true, input.name.span());
124                        let ty = Type::Integer(IntegerType::U16);
125                        self.maybe_assert_type(&ty, expected, input.span());
126                        return ty;
127                    }
128                    _ => {
129                        self.emit_err(TypeCheckerError::invalid_block_access(input.name.span()));
130                        return Type::Err;
131                    }
132                },
133                _ => {}
134            }
135        }
136
137        let ty = if assign {
138            self.visit_expression_assign(&input.inner)
139        } else {
140            self.visit_expression(&input.inner, &None)
141        };
142
143        // Check that the type of `inner` in `inner.name` is a struct.
144        match ty {
145            Type::Err => Type::Err,
146            Type::Composite(struct_) => {
147                // Retrieve the struct definition associated with `identifier`.
148                let Some(struct_) =
149                    self.lookup_struct(struct_.program.or(self.scope_state.program_name), struct_.id.name)
150                else {
151                    self.emit_err(TypeCheckerError::undefined_type(ty, input.inner.span()));
152                    return Type::Err;
153                };
154                // Check that `input.name` is a member of the struct.
155                match struct_.members.iter().find(|member| member.name() == input.name.name) {
156                    // Case where `input.name` is a member of the struct.
157                    Some(Member { type_, .. }) => {
158                        // Check that the type of `input.name` is the same as `expected`.
159                        self.maybe_assert_type(type_, expected, input.span());
160                        type_.clone()
161                    }
162                    // Case where `input.name` is not a member of the struct.
163                    None => {
164                        self.emit_err(TypeCheckerError::invalid_struct_variable(
165                            input.name,
166                            &struct_,
167                            input.name.span(),
168                        ));
169                        Type::Err
170                    }
171                }
172            }
173            type_ => {
174                self.emit_err(TypeCheckerError::type_should_be2(type_, "a struct or record", input.inner.span()));
175                Type::Err
176            }
177        }
178    }

180    fn visit_tuple_access_general(&mut self, input: &TupleAccess, assign: bool, expected: &Option<Type>) -> Type {
181        let this_type = if assign {
182            self.visit_expression_assign(&input.tuple)
183        } else {
184            self.visit_expression(&input.tuple, &None)
185        };
186        match this_type {
187            Type::Err => Type::Err,
188            Type::Tuple(tuple) => {
189                // Check out of range input.
190                let index = input.index.value();
191                let Some(actual) = tuple.elements().get(index) else {
192                    self.emit_err(TypeCheckerError::tuple_out_of_range(index, tuple.length(), input.span()));
193                    return Type::Err;
194                };
195
196                self.maybe_assert_type(actual, expected, input.span());
197
198                actual.clone()
199            }
200            Type::Future(_) => {
201                // Get the fully inferred type.
202                let Some(Type::Future(inferred_f)) = self.state.type_table.get(&input.tuple.id()) else {
203                    // If a future type was not inferred, we will have already reported an error.
204                    return Type::Err;
205                };
206
207                let Some(actual) = inferred_f.inputs().get(input.index.value()) else {
208                    self.emit_err(TypeCheckerError::invalid_future_access(
209                        input.index.value(),
210                        inferred_f.inputs().len(),
211                        input.span(),
212                    ));
213                    return Type::Err;
214                };
215
216                // If all inferred types weren't the same, the member will be of type `Type::Err`.
217                if let Type::Err = actual {
218                    self.emit_err(TypeCheckerError::future_error_member(input.index.value(), input.span()));
219                    return Type::Err;
220                }
221
222                self.maybe_assert_type(actual, expected, input.span());
223
224                actual.clone()
225            }
226            type_ => {
227                self.emit_err(TypeCheckerError::type_should_be2(type_, "a tuple or future", input.span()));
228                Type::Err
229            }
230        }
231    }

233    fn visit_identifier_assign(&mut self, input: &Identifier) -> Type {
234        // Lookup the variable in the symbol table and retrieve its type.
235        let Some(var) = self.state.symbol_table.lookup_variable(self.scope_state.program_name.unwrap(), input.name)
236        else {
237            self.emit_err(TypeCheckerError::unknown_sym("variable", input.name, input.span));
238            return Type::Err;
239        };
240
241        // If the variable exists, then check that it is not a constant.
242        match &var.declaration {
243            VariableType::Const => self.emit_err(TypeCheckerError::cannot_assign_to_const_var(input, var.span)),
244            VariableType::Input(Mode::Constant) => {
245                self.emit_err(TypeCheckerError::cannot_assign_to_const_input(input, var.span))
246            }
247            VariableType::Mut | VariableType::Input(_) => {}
248        }
249
250        // If the variable exists and it's in an async function, then check that it is in the current conditional scope.
251        if self.scope_state.variant.unwrap().is_async_function() && !self.symbol_in_conditional_scope(input.name) {
252            self.emit_err(TypeCheckerError::async_cannot_assign_outside_conditional(input, var.span));
253        }
254
255        var.type_.clone()
256    }
257}

259impl ExpressionVisitor for TypeCheckingVisitor<'_> {
260    type AdditionalInput = Option<Type>;
261    type Output = Type;
262
263    fn visit_expression(&mut self, input: &Expression, additional: &Self::AdditionalInput) -> Self::Output {
264        let output = match input {
265            Expression::Array(array) => self.visit_array(array, additional),
266            Expression::ArrayAccess(access) => self.visit_array_access_general(access, false, additional),
267            Expression::AssociatedConstant(constant) => self.visit_associated_constant(constant, additional),
268            Expression::AssociatedFunction(function) => self.visit_associated_function(function, additional),
269            Expression::Binary(binary) => self.visit_binary(binary, additional),
270            Expression::Call(call) => self.visit_call(call, additional),
271            Expression::Cast(cast) => self.visit_cast(cast, additional),
272            Expression::Struct(struct_) => self.visit_struct_init(struct_, additional),
273            Expression::Err(err) => self.visit_err(err, additional),
274            Expression::Identifier(identifier) => self.visit_identifier(identifier, additional),
275            Expression::Literal(literal) => self.visit_literal(literal, additional),
276            Expression::Locator(locator) => self.visit_locator(locator, additional),
277            Expression::MemberAccess(access) => self.visit_member_access_general(access, false, additional),
278            Expression::Ternary(ternary) => self.visit_ternary(ternary, additional),
279            Expression::Tuple(tuple) => self.visit_tuple(tuple, additional),
280            Expression::TupleAccess(access) => self.visit_tuple_access_general(access, false, additional),
281            Expression::Unary(unary) => self.visit_unary(unary, additional),
282            Expression::Unit(unit) => self.visit_unit(unit, additional),
283        };
284        // Add the expression and its associated type to the symbol table.
285        self.state.type_table.insert(input.id(), output.clone());
286        output
287    }

289    fn visit_array_access(&mut self, _input: &ArrayAccess, _additional: &Self::AdditionalInput) -> Self::Output {
290        panic!("Should not be called.");

298        panic!("Should not be called.");
299    }

301    fn visit_array(&mut self, input: &ArrayExpression, additional: &Self::AdditionalInput) -> Self::Output {
302        if input.elements.is_empty() {
303            self.emit_err(TypeCheckerError::array_empty(input.span()));
304            return Type::Err;
305        }
306
307        let element_type = self.visit_expression(&input.elements[0], &None);
308
309        if input.elements.len() > self.limits.max_array_elements {
310            self.emit_err(TypeCheckerError::array_too_large(
311                input.elements.len(),
312                self.limits.max_array_elements,
313                input.span(),
314            ));
315        }
316
317        if element_type == Type::Err {
318            return Type::Err;
319        }
320
321        for expression in input.elements[1..].iter() {
322            let next_type = self.visit_expression(expression, &None);
323
324            if next_type == Type::Err {
325                return Type::Err;
326            }
327
328            self.assert_type(&next_type, &element_type, expression.span());
329        }
330
331        let type_ = Type::Array(ArrayType::new(element_type, NonNegativeNumber::from(input.elements.len())));
332
333        self.maybe_assert_type(&type_, additional, input.span());
334
335        type_
336    }
337
338    fn visit_associated_constant(
339        &mut self,
340        input: &AssociatedConstantExpression,
341        expected: &Self::AdditionalInput,
342    ) -> Self::Output {
343        // Check associated constant type and constant name
344        let Some(core_constant) = self.get_core_constant(&input.ty, &input.name) else {
345            self.emit_err(TypeCheckerError::invalid_associated_constant(input, input.span));
346            return Type::Err;
347        };
348        let type_ = core_constant.to_type();
349        self.maybe_assert_type(&type_, expected, input.span());
350        type_
351    }
352
353    fn visit_associated_function(
354        &mut self,
355        input: &AssociatedFunctionExpression,
356        expected: &Self::AdditionalInput,
357    ) -> Self::Output {
358        // Check core struct name and function.
359        let Some(core_instruction) = self.get_core_function_call(&input.variant, &input.name) else {
360            self.emit_err(TypeCheckerError::invalid_core_function_call(input, input.span()));
361            return Type::Err;
362        };
363        // Check that operation is not restricted to finalize blocks.
364        if self.scope_state.variant != Some(Variant::AsyncFunction) && core_instruction.is_finalize_command() {
365            self.emit_err(TypeCheckerError::operation_must_be_in_finalize_block(input.span()));
366        }
367
368        // Get the types of the arguments.
369        let argument_types =
370            input.arguments.iter().map(|arg| (self.visit_expression(arg, &None), arg.span())).collect::<Vec<_>>();
371
372        // Check that the types of the arguments are valid.
373        let return_type = self.check_core_function_call(core_instruction.clone(), &argument_types, input.span());
374
375        // Check return type if the expected type is known.
376        self.maybe_assert_type(&return_type, expected, input.span());
377
378        // Await futures here so that can use the argument variable names to lookup.
379        if core_instruction == CoreFunction::FutureAwait && input.arguments.len() != 1 {
380            self.emit_err(TypeCheckerError::can_only_await_one_future_at_a_time(input.span));
381        }
382
383        return_type
384    }

386    fn visit_binary(&mut self, input: &BinaryExpression, destination: &Self::AdditionalInput) -> Self::Output {
387        let assert_same_type = |slf: &Self, t1: &Type, t2: &Type| -> Type {
388            if t1 == &Type::Err || t2 == &Type::Err {
389                Type::Err
390            } else if !slf.eq_user(t1, t2) {
391                slf.emit_err(TypeCheckerError::operation_types_mismatch(input.op, t1, t2, input.span()));
392                Type::Err
393            } else {
394                t1.clone()
395            }
396        };
397
398        match input.op {
399            BinaryOperation::And | BinaryOperation::Or | BinaryOperation::Nand | BinaryOperation::Nor => {
400                self.maybe_assert_type(&Type::Boolean, destination, input.span());
401                self.visit_expression(&input.left, &Some(Type::Boolean));
402                self.visit_expression(&input.right, &Some(Type::Boolean));
403                Type::Boolean
404            }
405            BinaryOperation::BitwiseAnd | BinaryOperation::BitwiseOr | BinaryOperation::Xor => {
406                let t1 = self.visit_expression(&input.left, &None);
407                self.assert_bool_int_type(&t1, input.left.span());
408                let t2 = self.visit_expression(&input.right, &None);
409                self.assert_bool_int_type(&t2, input.right.span());
410                let result_t = assert_same_type(self, &t1, &t2);
411                self.maybe_assert_type(&result_t, destination, input.span());
412                result_t
413            }
414            BinaryOperation::Add => {
415                let t1 = self.visit_expression(&input.left, &None);
416                let t2 = self.visit_expression(&input.right, &None);
417                let assert_add_type = |type_: &Type, span: Span| {
418                    if !matches!(type_, Type::Err | Type::Field | Type::Group | Type::Scalar | Type::Integer(_)) {
419                        self.emit_err(TypeCheckerError::type_should_be2(
420                            type_,
421                            "a field, group, scalar, or integer",
422                            span,
423                        ));
424                    }
425                };
426
427                assert_add_type(&t1, input.left.span());
428                assert_add_type(&t2, input.right.span());
429
430                let result_t = assert_same_type(self, &t1, &t2);
431
432                self.maybe_assert_type(&result_t, destination, input.span());
433
434                result_t
435            }
436            BinaryOperation::Sub => {
437                let t1 = self.visit_expression(&input.left, &None);
438                let t2 = self.visit_expression(&input.right, &None);
439
440                self.assert_field_group_int_type(&t1, input.left.span());
441                self.assert_field_group_int_type(&t2, input.right.span());
442
443                let result_t = assert_same_type(self, &t1, &t2);
444
445                self.maybe_assert_type(&result_t, destination, input.span());
446
447                result_t
448            }
449            BinaryOperation::Mul => {
450                let t1 = self.visit_expression(&input.left, &None);
451                let t2 = self.visit_expression(&input.right, &None);
452
453                let result_t = match (&t1, &t2) {
454                    (Type::Err, _) | (_, Type::Err) => Type::Err,
455                    (Type::Group, Type::Scalar) | (Type::Scalar, Type::Group) => Type::Group,
456                    (Type::Field, Type::Field) => Type::Field,
457                    (Type::Integer(integer_type1), Type::Integer(integer_type2)) if integer_type1 == integer_type2 => {
458                        t1.clone()
459                    }
460                    _ => {
461                        self.emit_err(TypeCheckerError::mul_types_mismatch(t1, t2, input.span()));
462                        Type::Err
463                    }
464                };
465
466                self.maybe_assert_type(&result_t, destination, input.span());
467
468                result_t
469            }
470            BinaryOperation::Div => {
471                let t1 = self.visit_expression(&input.left, &None);
472                let t2 = self.visit_expression(&input.right, &None);
473
474                self.assert_field_int_type(&t1, input.left.span());
475                self.assert_field_int_type(&t2, input.right.span());
476
477                let result_t = assert_same_type(self, &t1, &t2);
478
479                self.maybe_assert_type(&result_t, destination, input.span());
480
481                result_t
482            }
483            BinaryOperation::Rem | BinaryOperation::RemWrapped => {
484                let t1 = self.visit_expression(&input.left, &None);
485                let t2 = self.visit_expression(&input.right, &None);
486
487                self.assert_int_type(&t1, input.left.span());
488                self.assert_int_type(&t2, input.right.span());
489
490                let result_t = assert_same_type(self, &t1, &t2);
491
492                self.maybe_assert_type(&result_t, destination, input.span());
493
494                result_t
495            }
496            BinaryOperation::Mod => {
497                let t1 = self.visit_expression(&input.left, &None);
498                let t2 = self.visit_expression(&input.right, &None);
499
500                self.assert_unsigned_type(&t1, input.left.span());
501                self.assert_unsigned_type(&t1, input.right.span());
502
503                let result_t = assert_same_type(self, &t1, &t2);
504
505                self.maybe_assert_type(&result_t, destination, input.span());
506
507                result_t
508            }
509            BinaryOperation::Pow => {
510                let t1 = self.visit_expression(&input.left, &None);
511                let t2 = self.visit_expression(&input.right, &None);
512
513                let ty = match (&t1, &t2) {
514                    (Type::Err, _) | (_, Type::Err) => Type::Err,
515                    (Type::Field, Type::Field) => Type::Field,
516                    (base @ Type::Integer(_), t2) => {
517                        if !matches!(
518                            t2,
519                            Type::Integer(IntegerType::U8)
520                                | Type::Integer(IntegerType::U16)
521                                | Type::Integer(IntegerType::U32)
522                        ) {
523                            self.emit_err(TypeCheckerError::pow_types_mismatch(base, t2, input.span()));
524                        }
525                        base.clone()
526                    }
527                    _ => {
528                        self.emit_err(TypeCheckerError::pow_types_mismatch(t1, t2, input.span()));
529                        Type::Err
530                    }
531                };
532
533                self.maybe_assert_type(&ty, destination, input.span());
534
535                ty
536            }
537            BinaryOperation::Eq | BinaryOperation::Neq => {
538                let t1 = self.visit_expression(&input.left, &None);
539                let t2 = self.visit_expression(&input.right, &None);
540
541                let _ = assert_same_type(self, &t1, &t2);
542
543                self.maybe_assert_type(&Type::Boolean, destination, input.span());
544
545                Type::Boolean
546            }
547            BinaryOperation::Lt | BinaryOperation::Gt | BinaryOperation::Lte | BinaryOperation::Gte => {
548                // Assert left and right are equal field, scalar, or integer types.
549                let t1 = self.visit_expression(&input.left, &None);
550                let t2 = self.visit_expression(&input.right, &None);
551
552                let assert_compare_type = |type_: &Type, span: Span| {
553                    if !matches!(type_, Type::Err | Type::Field | Type::Scalar | Type::Integer(_)) {
554                        self.emit_err(TypeCheckerError::type_should_be2(type_, "a field, scalar, or integer", span));
555                    }
556                };
557
558                assert_compare_type(&t1, input.left.span());
559                assert_compare_type(&t2, input.right.span());
560
561                let _ = assert_same_type(self, &t1, &t2);
562
563                self.maybe_assert_type(&Type::Boolean, destination, input.span());
564
565                Type::Boolean
566            }
567            BinaryOperation::AddWrapped
568            | BinaryOperation::SubWrapped
569            | BinaryOperation::DivWrapped
570            | BinaryOperation::MulWrapped => {
571                let t1 = self.visit_expression(&input.left, &None);
572                let t2 = self.visit_expression(&input.right, &None);
573
574                self.assert_int_type(&t1, input.left.span());
575                self.assert_int_type(&t2, input.right.span());
576
577                let result_t = assert_same_type(self, &t1, &t2);
578
579                self.maybe_assert_type(&result_t, destination, input.span());
580
581                result_t
582            }
583            BinaryOperation::Shl
584            | BinaryOperation::ShlWrapped
585            | BinaryOperation::Shr
586            | BinaryOperation::ShrWrapped
587            | BinaryOperation::PowWrapped => {
588                let t1 = self.visit_expression(&input.left, &None);
589                let t2 = self.visit_expression(&input.right, &None);
590
591                self.assert_int_type(&t1, input.left.span());
592
593                if !matches!(
594                    &t2,
595                    Type::Err
596                        | Type::Integer(IntegerType::U8)
597                        | Type::Integer(IntegerType::U16)
598                        | Type::Integer(IntegerType::U32)
599                ) {
600                    self.emit_err(TypeCheckerError::shift_type_magnitude(input.op, t2, input.right.span()));
601                }
602
603                t1
604            }
605        }
606    }

608    fn visit_call(&mut self, input: &CallExpression, expected: &Self::AdditionalInput) -> Self::Output {
609        // Get the function symbol.
610        let Expression::Identifier(ident) = &input.function else {
611            panic!("Parsing guarantees that a function name is always an identifier.");
612        };
613
614        let callee_program = input.program.or(self.scope_state.program_name).unwrap();
615
616        let Some(func_symbol) = self.state.symbol_table.lookup_function(Location::new(callee_program, ident.name))
617        else {
618            self.emit_err(TypeCheckerError::unknown_sym("function", ident.name, ident.span()));
619            return Type::Err;
620        };
621
622        let func = func_symbol.function.clone();
623
624        // Check that the call is valid.
625        // We always set the variant before entering the body of a function, so this unwrap works.
626        match self.scope_state.variant.unwrap() {
627            Variant::AsyncFunction | Variant::Function if !matches!(func.variant, Variant::Inline) => {
628                self.emit_err(TypeCheckerError::can_only_call_inline_function(input.span))
629            }
630            Variant::Transition | Variant::AsyncTransition
631                if matches!(func.variant, Variant::Transition)
632                    && input.program.unwrap() == self.scope_state.program_name.unwrap() =>
633            {
634                self.emit_err(TypeCheckerError::cannot_invoke_call_to_local_transition_function(input.span))
635            }
636            _ => {}
637        }
638
639        // Check that the call is not to an external `inline` function.
640        if func.variant == Variant::Inline && input.program.unwrap() != self.scope_state.program_name.unwrap() {
641            self.emit_err(TypeCheckerError::cannot_call_external_inline_function(input.span));
642        }
643
644        // Async functions return a single future.
645        let mut ret = if func.variant == Variant::AsyncFunction {
646            // Type check after resolving the input types.
647            if let Some(Type::Future(_)) = expected {
648                Type::Future(FutureType::new(Vec::new(), Some(Location::new(callee_program, ident.name)), false))
649            } else {
650                self.emit_err(TypeCheckerError::return_type_of_finalize_function_is_future(input.span));
651                Type::Unit
652            }
653        } else if func.variant == Variant::AsyncTransition {
654            // Fully infer future type.
655            let Some(inputs) = self
656                .async_function_input_types
657                .get(&Location::new(callee_program, Symbol::intern(&format!("finalize/{}", ident.name))))
658            else {
659                self.emit_err(TypeCheckerError::async_function_not_found(ident.name, input.span));
660                return Type::Future(FutureType::new(
661                    Vec::new(),
662                    Some(Location::new(callee_program, ident.name)),
663                    false,
664                ));
665            };
666
667            let future_type =
668                Type::Future(FutureType::new(inputs.clone(), Some(Location::new(callee_program, ident.name)), true));
669            let fully_inferred_type = match &func.output_type {
670                Type::Tuple(tup) => Type::Tuple(TupleType::new(
671                    tup.elements()
672                        .iter()
673                        .map(|t| if matches!(t, Type::Future(_)) { future_type.clone() } else { t.clone() })
674                        .collect::<Vec<Type>>(),
675                )),
676                Type::Future(_) => future_type,
677                _ => panic!("Invalid output type for async transition."),
678            };
679            self.assert_and_return_type(fully_inferred_type, expected, input.span())
680        } else {
681            self.assert_and_return_type(func.output_type, expected, input.span())
682        };
683
684        // Check number of function arguments.
685        if func.input.len() != input.arguments.len() {
686            self.emit_err(TypeCheckerError::incorrect_num_args_to_call(
687                func.input.len(),
688                input.arguments.len(),
689                input.span(),
690            ));
691        }
692
693        // Check function argument types.
694        self.scope_state.is_call = true;
695        let (mut input_futures, mut inferred_finalize_inputs) = (Vec::new(), Vec::new());
696        for (expected, argument) in func.input.iter().zip(input.arguments.iter()) {
697            // Get the type of the expression. If the type is not known, do not attempt to attempt any further inference.
698            let ty = self.visit_expression(argument, &Some(expected.type_().clone()));
699            if ty == Type::Err {
700                return Type::Err;
701            }
702            // Extract information about futures that are being consumed.
703            if func.variant == Variant::AsyncFunction && matches!(expected.type_(), Type::Future(_)) {
704                // Consume the future.
705                let option_name =
706                    match argument {
707                        Expression::Identifier(id) => Some(id.name),
708                        Expression::TupleAccess(tuple_access) => {
709                            if let Expression::Identifier(id) = &tuple_access.tuple { Some(id.name) } else { None }
710                        }
711                        _ => None,
712                    };
713
714                if let Some(name) = option_name {
715                    match self.scope_state.futures.shift_remove(&name) {
716                        Some(future) => {
717                            self.scope_state.call_location = Some(future);
718                        }
719                        None => {
720                            self.emit_err(TypeCheckerError::unknown_future_consumed(name, argument.span()));
721                        }
722                    }
723                }
724
725                match argument {
726                    Expression::Identifier(_) | Expression::Call(_) | Expression::TupleAccess(_) => {
727                        match self.scope_state.call_location {
728                            Some(location) => {
729                                // Get the external program and function name.
730                                input_futures.push(location);
731                                // Get the full inferred type.
732                                inferred_finalize_inputs.push(ty);
733                            }
734                            None => {
735                                self.emit_err(TypeCheckerError::unknown_future_consumed(argument, argument.span()));
736                            }
737                        }
738                    }
739                    _ => {
740                        self.emit_err(TypeCheckerError::unknown_future_consumed("unknown", argument.span()));
741                    }
742                }
743            } else {
744                inferred_finalize_inputs.push(ty);
745            }
746        }
747        self.scope_state.is_call = false;
748
749        // Add the call to the call graph.
750        let Some(caller_name) = self.scope_state.function else {
751            panic!("`self.function` is set every time a function is visited.");
752        };
753
754        // Don't add external functions to call graph. Since imports are acyclic, these can never produce a cycle.
755        if input.program.unwrap() == self.scope_state.program_name.unwrap() {
756            self.state.call_graph.add_edge(caller_name, ident.name);
757        }
758
759        if func.variant.is_transition()
760            && self.scope_state.variant == Some(Variant::AsyncTransition)
761            && self.scope_state.has_called_finalize
762        {
763            // Cannot call an external async transition after having called the async function.
764            self.emit_err(TypeCheckerError::external_transition_call_must_be_before_finalize(input.span));
765        }
766
767        // Propagate futures from async functions and transitions.
768        if func.variant.is_async_function() {
769            // Cannot have async calls in a conditional block.
770            if self.scope_state.is_conditional {
771                self.emit_err(TypeCheckerError::async_call_in_conditional(input.span));
772            }
773
774            // Can only call async functions and external async transitions from an async transition body.
775            if self.scope_state.variant != Some(Variant::AsyncTransition) {
776                self.emit_err(TypeCheckerError::async_call_can_only_be_done_from_async_transition(input.span));
777            }
778
779            // Can only call an async function once in a transition function body.
780            if self.scope_state.has_called_finalize {
781                self.emit_err(TypeCheckerError::must_call_async_function_once(input.span));
782            }
783            // Check that all futures consumed.
784            if !self.scope_state.futures.is_empty() {
785                self.emit_err(TypeCheckerError::not_all_futures_consumed(
786                    self.scope_state.futures.iter().map(|(f, _)| f).join(", "),
787                    input.span,
788                ));
789            }
790            self.state
791                .symbol_table
792                .attach_finalizer(
793                    Location::new(callee_program, caller_name),
794                    Location::new(callee_program, ident.name),
795                    input_futures,
796                    inferred_finalize_inputs.clone(),
797                )
798                .expect("Failed to attach finalizer");
799            // Create expectation for finalize inputs that will be checked when checking corresponding finalize function signature.
800            self.async_function_callers
801                .entry(Location::new(self.scope_state.program_name.unwrap(), ident.name))
802                .or_default()
803                .insert(self.scope_state.location());
804
805            // Set scope state flag.
806            self.scope_state.has_called_finalize = true;
807
808            // Update ret to reflect fully inferred future type.
809            ret = Type::Future(FutureType::new(
810                inferred_finalize_inputs,
811                Some(Location::new(callee_program, ident.name)),
812                true,
813            ));
814
815            // Type check in case the expected type is known.
816            self.assert_and_return_type(ret.clone(), expected, input.span());
817        }
818
819        // Set call location so that definition statement knows where future comes from.
820        self.scope_state.call_location = Some(Location::new(callee_program, ident.name));
821
822        ret
823    }

825    fn visit_cast(&mut self, input: &CastExpression, expected: &Self::AdditionalInput) -> Self::Output {
826        let expression_type = self.visit_expression(&input.expression, &None);
827
828        let assert_castable_type = |actual: &Type, span: Span| {
829            if !matches!(
830                actual,
831                Type::Integer(_) | Type::Boolean | Type::Field | Type::Group | Type::Scalar | Type::Address | Type::Err,
832            ) {
833                self.emit_err(TypeCheckerError::type_should_be2(
834                    actual,
835                    "an integer, bool, field, group, scalar, or address",
836                    span,
837                ));
838            }
839        };
840
841        assert_castable_type(&input.type_, input.span());
842
843        assert_castable_type(&expression_type, input.expression.span());
844
845        self.maybe_assert_type(&input.type_, expected, input.span());
846
847        input.type_.clone()
848    }

850    fn visit_struct_init(&mut self, input: &StructExpression, additional: &Self::AdditionalInput) -> Self::Output {
851        let struct_ = self.lookup_struct(self.scope_state.program_name, input.name.name).clone();
852        let Some(struct_) = struct_ else {
853            self.emit_err(TypeCheckerError::unknown_sym("struct or record", input.name.name, input.name.span()));
854            return Type::Err;
855        };
856
857        // Note that it is sufficient for the `program` to be `None` as composite types can only be initialized
858        // in the program in which they are defined.
859        let type_ = Type::Composite(CompositeType { id: input.name, program: None });
860        self.maybe_assert_type(&type_, additional, input.name.span());
861
862        // Check number of struct members.
863        if struct_.members.len() != input.members.len() {
864            self.emit_err(TypeCheckerError::incorrect_num_struct_members(
865                struct_.members.len(),
866                input.members.len(),
867                input.span(),
868            ));
869        }
870
871        for Member { identifier, type_, .. } in struct_.members.iter() {
872            if let Some(actual) = input.members.iter().find(|member| member.identifier.name == identifier.name) {
873                match &actual.expression {
874                    None => {
875                        // If `expression` is None, then the member uses the identifier shorthand, e.g. `Foo { a }`
876                        // We visit it as an expression rather than just calling `visit_identifer` so it will get
877                        // put into the type table.
878                        self.visit_expression(&actual.identifier.into(), &Some(type_.clone()));
879                    }
880                    Some(expr) => {
881                        // Otherwise, visit the associated expression.
882                        self.visit_expression(expr, &Some(type_.clone()));
883                    }
884                };
885            } else {
886                self.emit_err(TypeCheckerError::missing_struct_member(struct_.identifier, identifier, input.span()));
887            };
888        }
889
890        type_
891    }

909    fn visit_literal(&mut self, input: &Literal, expected: &Self::AdditionalInput) -> Self::Output {
910        fn parse_integer_literal<I: FromStrRadix>(handler: &Handler, raw_string: &str, span: Span, type_string: &str) {
911            let string = raw_string.replace('_', "");
912            if I::from_str_by_radix(&string).is_err() {
913                handler.emit_err(TypeCheckerError::invalid_int_value(string, type_string, span));
914            }
915        }
916
917        let type_ = match &input.variant {
918            LiteralVariant::Address(..) => Type::Address,
919            LiteralVariant::Boolean(..) => Type::Boolean,
920            LiteralVariant::Field(..) => Type::Field,
921            LiteralVariant::Integer(IntegerType::U8, string) => {
922                parse_integer_literal::<u8>(&self.state.handler, string, input.span(), "u8");
923                Type::Integer(IntegerType::U8)
924            }
925            LiteralVariant::Integer(IntegerType::U16, string) => {
926                parse_integer_literal::<u16>(&self.state.handler, string, input.span(), "u16");
927                Type::Integer(IntegerType::U16)
928            }
929            LiteralVariant::Integer(IntegerType::U32, string) => {
930                parse_integer_literal::<u32>(&self.state.handler, string, input.span(), "u32");
931                Type::Integer(IntegerType::U32)
932            }
933            LiteralVariant::Integer(IntegerType::U64, string) => {
934                parse_integer_literal::<u64>(&self.state.handler, string, input.span(), "u64");
935                Type::Integer(IntegerType::U64)
936            }
937            LiteralVariant::Integer(IntegerType::U128, string) => {
938                parse_integer_literal::<u128>(&self.state.handler, string, input.span(), "u128");
939                Type::Integer(IntegerType::U128)
940            }
941            LiteralVariant::Integer(IntegerType::I8, string) => {
942                parse_integer_literal::<i8>(&self.state.handler, string, input.span(), "i8");
943                Type::Integer(IntegerType::I8)
944            }
945            LiteralVariant::Integer(IntegerType::I16, string) => {
946                parse_integer_literal::<i16>(&self.state.handler, string, input.span(), "i16");
947                Type::Integer(IntegerType::I16)
948            }
949            LiteralVariant::Integer(IntegerType::I32, string) => {
950                parse_integer_literal::<i32>(&self.state.handler, string, input.span(), "i32");
951                Type::Integer(IntegerType::I32)
952            }
953            LiteralVariant::Integer(IntegerType::I64, string) => {
954                parse_integer_literal::<i64>(&self.state.handler, string, input.span(), "i64");
955                Type::Integer(IntegerType::I64)
956            }
957            LiteralVariant::Integer(IntegerType::I128, string) => {
958                parse_integer_literal::<i128>(&self.state.handler, string, input.span(), "i128");
959                Type::Integer(IntegerType::I128)
960            }
961            LiteralVariant::Group(s) => {
962                // Get rid of leading - and 0 and see if it parses
963                let s = s.trim_start_matches('-').trim_start_matches('0');
964                if !s.is_empty()
965                    && format!("{s}group").parse::<snarkvm::prelude::Group<snarkvm::prelude::TestnetV0>>().is_err()
966                {
967                    self.emit_err(TypeCheckerError::invalid_int_value(s, "group", input.span()));
968                }
969                Type::Group
970            }
971            LiteralVariant::Scalar(..) => Type::Scalar,
972            LiteralVariant::String(..) => {
973                self.emit_err(TypeCheckerError::strings_are_not_supported(input.span()));
974                Type::String
975            }
976        };
977
978        self.maybe_assert_type(&type_, expected, input.span());
979
980        type_
981    }

995    fn visit_ternary(&mut self, input: &TernaryExpression, expected: &Self::AdditionalInput) -> Self::Output {
996        self.visit_expression(&input.condition, &Some(Type::Boolean));
997
998        let t1 = self.visit_expression(&input.if_true, expected);
999        let t2 = self.visit_expression(&input.if_false, expected);
1000
1001        if t1 == Type::Err || t2 == Type::Err {
1002            Type::Err
1003        } else if !self.eq_user(&t1, &t2) {
1004            self.emit_err(TypeCheckerError::ternary_branch_mismatch(t1, t2, input.span()));
1005            Type::Err
1006        } else {
1007            t1
1008        }
1009    }

1011    fn visit_tuple(&mut self, input: &TupleExpression, expected: &Self::AdditionalInput) -> Self::Output {
1012        // Check the expected tuple types if they are known.
1013        let Some(Type::Tuple(expected_types)) = expected else {
1014            self.emit_err(TypeCheckerError::invalid_tuple(input.span()));
1015            return Type::Err;
1016        };
1017
1018        // Check actual length is equal to expected length.
1019        if expected_types.length() != input.elements.len() {
1020            self.emit_err(TypeCheckerError::incorrect_tuple_length(
1021                expected_types.length(),
1022                input.elements.len(),
1023                input.span(),
1024            ));
1025        }
1026
1027        for (expr, expected) in input.elements.iter().zip(expected_types.elements().iter()) {
1028            if matches!(expr, Expression::Tuple(_)) {
1029                self.emit_err(TypeCheckerError::nested_tuple_expression(expr.span()))
1030            }
1031
1032            self.visit_expression(expr, &Some(expected.clone()));
1033        }
1034
1035        Type::Tuple(expected_types.clone())
1036    }

1038    fn visit_unary(&mut self, input: &UnaryExpression, destination: &Self::AdditionalInput) -> Self::Output {
1039        let assert_signed_int = |slf: &mut Self, type_: &Type| {
1040            if !matches!(
1041                type_,
1042                Type::Err
1043                    | Type::Integer(IntegerType::I8)
1044                    | Type::Integer(IntegerType::I16)
1045                    | Type::Integer(IntegerType::I32)
1046                    | Type::Integer(IntegerType::I64)
1047                    | Type::Integer(IntegerType::I128)
1048            ) {
1049                slf.emit_err(TypeCheckerError::type_should_be2(type_, "a signed integer", input.span()));
1050            }
1051        };
1052
1053        let ty = match input.op {
1054            UnaryOperation::Abs => {
1055                let type_ = self.visit_expression(&input.receiver, &None);
1056                assert_signed_int(self, &type_);
1057                type_
1058            }
1059            UnaryOperation::AbsWrapped => {
1060                let type_ = self.visit_expression(&input.receiver, &None);
1061                assert_signed_int(self, &type_);
1062                type_
1063            }
1064            UnaryOperation::Double => {
1065                let type_ = self.visit_expression(&input.receiver, &None);
1066                if !matches!(&type_, Type::Err | Type::Field | Type::Group) {
1067                    self.emit_err(TypeCheckerError::type_should_be2(&type_, "a field or group", input.span()));
1068                }
1069                type_
1070            }
1071            UnaryOperation::Inverse => {
1072                let type_ = self.visit_expression(&input.receiver, &None);
1073                self.assert_type(&type_, &Type::Field, input.span());
1074                type_
1075            }
1076            UnaryOperation::Negate => {
1077                let type_ = self.visit_expression(&input.receiver, &None);
1078                if !matches!(
1079                    &type_,
1080                    Type::Err
1081                        | Type::Integer(IntegerType::I8)
1082                        | Type::Integer(IntegerType::I16)
1083                        | Type::Integer(IntegerType::I32)
1084                        | Type::Integer(IntegerType::I64)
1085                        | Type::Integer(IntegerType::I128)
1086                        | Type::Group
1087                        | Type::Field
1088                ) {
1089                    self.emit_err(TypeCheckerError::type_should_be2(
1090                        &type_,
1091                        "a signed integer, group, or field",
1092                        input.receiver.span(),
1093                    ));
1094                }
1095                type_
1096            }
1097            UnaryOperation::Not => {
1098                let type_ = self.visit_expression(&input.receiver, &None);
1099                if !matches!(&type_, Type::Err | Type::Boolean | Type::Integer(_)) {
1100                    self.emit_err(TypeCheckerError::type_should_be2(&type_, "a bool or integer", input.span()));
1101                }
1102                type_
1103            }
1104            UnaryOperation::Square => {
1105                let type_ = self.visit_expression(&input.receiver, &None);
1106                self.assert_type(&type_, &Type::Field, input.span());
1107                type_
1108            }
1109            UnaryOperation::SquareRoot => {
1110                let type_ = self.visit_expression(&input.receiver, &None);
1111                self.assert_type(&type_, &Type::Field, input.span());
1112                type_
1113            }
1114            UnaryOperation::ToXCoordinate | UnaryOperation::ToYCoordinate => {
1115                let _operand_type = self.visit_expression(&input.receiver, &Some(Type::Group));
1116                self.maybe_assert_type(&Type::Field, destination, input.span());
1117                Type::Field
1118            }
1119        };
1120
1121        self.maybe_assert_type(&ty, destination, input.span());
1122
1123        ty
1124    }
