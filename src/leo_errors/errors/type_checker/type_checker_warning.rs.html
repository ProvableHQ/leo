17use crate::create_messages;
18
19use std::fmt::Display;
20
21create_messages!(
22    /// ParserWarning enum that represents all the warnings for the `leo-parser` crate.
23    #[derive(Hash, PartialEq, Eq)]
24    TypeCheckerWarning,
25    code_mask: 2000i32,
26    code_prefix: "TYC",
27
28    @formatted
29    some_paths_do_not_await_all_futures {
30        args: (num_total_paths: impl Display, num_unawaited_paths: impl Display),
31        msg: format!("Not all paths through the function await all futures. {num_unawaited_paths}/{num_total_paths} paths contain at least one future that is never awaited."),
32        help: Some("Ex: `f.await()` to await a future. Remove this warning by including the `--disable-conditional-branch-type-checking` flag.".to_string()),
33    }
34
35    @formatted
36    some_paths_contain_duplicate_future_awaits {
37        args: (num_total_paths: impl Display, num_duplicate_await_paths: impl Display),
38        msg: format!("Some paths through the function contain duplicate future awaits. {num_duplicate_await_paths}/{num_total_paths} paths contain at least one future that is awaited more than once."),
39        help: Some("Look at the times `.await()` is called, and try to reduce redundancies. Remove this warning by including the `--disable-conditional-branch-type-checking` flag.".to_string()),
40    }
41
42    // TODO: This warning is deprecated, remove it in the future.
43    @formatted
44    async_function_is_never_called_by_transition_function {
45        args: (name: impl Display),
46        msg: format!("The async function `{name}` is never called by an async transition."),
47        help: None,
48    }
49
50    // TODO: This warning is unused, remove it in the future.
51    @formatted
52    max_conditional_block_depth_exceeded {
53        args: (max: impl Display),
54        msg: format!("The type checker has exceeded the max depth of nested conditional blocks: {max}."),
55        help: Some("Re-run with a larger maximum depth using the `--conditional_block_max_depth` build option. Ex: `leo run main --conditional_block_max_depth 25`.".to_string()),
56    }
57
58    @formatted
59    caller_as_record_owner {
60        args: (record_name: impl Display),
61        msg: format!("`self.caller` used as the owner of record `{record_name}`"),
62        help: Some("`self.caller` may refer to a program address, which cannot spend records.".to_string()),
63    }
64);
