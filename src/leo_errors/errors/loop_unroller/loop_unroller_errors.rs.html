17use crate::create_messages;
18use std::fmt::Debug;
19
20create_messages!(
21    /// LoopUnrollerError enum that represents all the errors for the loop unrolling errors in the `leo-loop_unroller` crate.
22    LoopUnrollerError,
23    code_mask: 9000i32,
24    code_prefix: "LUN",
25
26    // TODO This error is unused. Remove it in a future version.
27    @formatted
28    loop_range_decreasing {
29        args: (),
30        msg: format!("The loop range must be increasing."),
31        help: None,
32    }
33
34    // TODO This error is unused. Remove it in a future version.
35    @formatted
36    variable_array_access {
37        args: (),
38        msg: format!("The array index must be constant."),
39        help: None,
40    }
41
42    @formatted
43    value_out_of_i128_bounds {
44        args: (value: impl std::fmt::Display),
45        msg: format!("The loop bound {value} does not fit into an i128."),
46        help: Some("All loop bounds must fit into `i128`.".to_string()),
47    }
48);
